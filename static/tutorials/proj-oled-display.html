🖥️ STM32 OLED显示驱动教程

📋 学习目标：掌握I2C协议原理、SSD1306驱动编写、字符/图形显示，学会用OLED显示信息。

一、硬件接线图

1.1 元器件清单

元器件数量备注
STM32F103C8T6最小系统板1块蓝色小板
0.96寸I2C OLED显示屏1个SSD1306驱动芯片
杜邦线4根I2C通信线

1.2 接线示意

        STM32F103C8T6              0.96" OLED
        ┌─────────────┐           ┌─────────────┐
        │             │           │             │
        │  PB6 (SCL) ─┼───►───────┼─── SCL      │
        │             │           │             │
        │  PB7 (SDA) ─┼───►───────┼─── SDA      │
        │             │           │             │
        │  3.3V   ────┼─── VCC ───┤─── VCC (3.3V)│
        │             │           │             │
        │  GND    ────┼─── GND ───┤─── GND      │
        │             │           │             │
        └─────────────┘           └─────────────┘

【I2C接线要点】
1. SCL线：时钟信号，STM32输出时钟到OLED
2. SDA线：数据信号，双向通信
3. VCC：供电，0.96寸OLED通常3.3V-5V兼容
4. GND：共地（必须连接！）

【OLED屏幕选择】
- 0.96寸：128×64像素，蓝色/黄色/白色
- 1.3寸：128×64像素，SH1106驱动
- 0.91寸：128×32像素，SSD1306驱动
⚠️ 注意：部分OLED模块有多种地址： - 默认地址：0x78 (0x3C << 1) - 备选地址：0x7A (0x3D << 1) 购买前确认模块规格！

1.3 I2C地址说明

SSD1306 I2C地址计算：

SA0引脚接地 → 0x78 (二进制 0111 1000)
SA0引脚接VCC → 0x7A (二进制 0111 1010)

I2C协议中，地址需要左移1位 + R/W位：
  0x78 << 1 = 0xF0 (写) 或 0xF1 (读)

在HAL库中直接使用7位地址：0x3C 或 0x78

二、CubeMX配置步骤

2.1 新建工程

  1. 打开STM32CubeMX，点击 File → New Project
  2. 搜索 STM32F103C8T6
  3. Start Project

2.2 配置I2C1

  1. 点击 PB6 → 选择 I2C1_SCL
  2. 点击 PB7 → 选择 I2C1_SDA
  3. 左侧 Connectivity → I2C1： 
    Mode = I2C
Configuration:
  I2C Speed Mode = Fast Mode (400kHz)
  I2C Clock Speed = 400000
  Primary Address Length = 7-bit
  Clock Stretch Mode = Standard

2.3 配置时钟

  1. RCC → HSE = Crystal/Ceramic Resonator
  2. Clock Configuration → 72MHz

2.4 生成代码

  1. Project Manager → 项目名 oled_demo
  2. Toolchain = MDK-ARM
  3. Generate Code

3.2 ssd1306.c 驱动文件

/**
  ******************************************************************************
  * @file    ssd1306.c
  * @brief   SSD1306 OLED驱动程序
  ******************************************************************************
  *
  * 驱动原理：
  * 1. SSD1306是128×64像素的单色OLED驱动芯片
  * 2. 显存为128×64位(1bit/pixel)，共1024字节
  * 3. 显示分为8页(Page 0-7)，每页128列
  * 4. 每字节控制8个像素(垂直排列)
  *
  *    Page 0: Column 0-127, 控制像素y=0-7
  *    Page 1: Column 0-127, 控制像素y=8-15
  *    ...
  *    Page 7: Column 0-127, 控制像素y=56-63
  *
  ******************************************************************************
  */

#include "ssd1306.h"
#include "i2c.h"
#include <string.h>
#include <stdio.h>

/* 显存缓冲区 */
static uint8_t ssd1306_gram[SSD1306_HEIGHT / 8][SSD1306_WIDTH];

/* I2C句柄外部声明 */
extern I2C_HandleTypeDef hi2c1;

/**
  * @brief  I2C写入一个字节
  * @param  data: 数据
  * @param  mode: SSD1306_CMD或SSD1306_DATA
  * @retval None
  */
static void ssd1306_WriteByte(uint8_t data, uint8_t mode)
{
  uint8_t buffer[2];
  buffer[0] = mode;   // 控制字节：0x00=命令，0x40=数据
  buffer[1] = data;

  HAL_I2C_Master_Transmit(&hi2c1, SSD1306_I2C_ADDR, buffer, 2, 100);
}

/**
  * @brief  I2C连续写入多个字节
  * @param  data: 数据指针
  * @param  len: 长度
  * @param  mode: 命令或数据模式
  * @retval None
  */
static void ssd1306_WriteBuffer(uint8_t *data, uint16_t len, uint8_t mode)
{
  uint8_t *buffer = (uint8_t *)malloc(len + 1);
  buffer[0] = mode;
  memcpy(&buffer[1], data, len);

  HAL_I2C_Master_Transmit(&hi2c1, SSD1306_I2C_ADDR, buffer, len + 1, 100);
  free(buffer);
}

/**
  * @brief  发送命令
  * @param  cmd: 命令字节
  * @retval None
  */
static void ssd1306_SendCommand(uint8_t cmd)
{
  ssd1306_WriteByte(cmd, SSD1306_CMD);
}

/**
  * @brief  发送多个命令
  * @param  cmd: 命令数组
  * @param  len: 命令数量
  * @retval None
  */
static void ssd1306_SendCommands(uint8_t *cmd, uint8_t len)
{
  ssd1306_WriteBuffer(cmd, len, SSD1306_CMD);
}

/**
  * @brief  SSD1306初始化
  * @param  None
  * @retval None
  */
void ssd1306_Init(void)
{
  HAL_Delay(100);  // 等待OLED上电稳定

  /* SSD1306初始化序列 */
  uint8_t init_commands[] = {
    /* 关闭显示 */
    0xAE,

    /* 设置时钟分频 */
    0xD5, 0x80,

    /* 设置 multiplex ratio */
    0xA8, 0x3F,  // 1/64 duty (0x3F = 63)

    /* 设置显示偏移 */
    0xD3, 0x00,

    /* 设置显示起始行 */
    0x40 | 0x00,  // 从0行开始

    /* 启用电荷泵 */
    0x8D, 0x14,  // 启用电荷泵（OLED需要）

    /* 设置内存寻址模式 */
    0x20, 0x00,  // 水平寻址模式

    /* 设置Segment引脚配置 */
    0xA1,  // 设置段重映射（0xA0=正常，0xA1=翻转）

    /* 设置COM引脚配置 */
    0xC8,  // 设置COM扫描方向（0xC0=从上到下，0xC8=从下到上）

    /* 设置COM引脚硬件配置 */
    0xDA, 0x12,  // 备选引脚配置

    /* 设置对比度 */
    0x81, 0x7F,  // 对比度=0x7F（中间值）

    /* 关闭预充电 */
    0xD9, 0xF1,

    /* 设置VCOMH电压 */
    0xDB, 0x40,

    /* 开启显示 */
    0xA6,  // 正常显示（非反转）

    /* 清除屏幕 */
    0x00,  // 跳过显示
  };

  /* 发送初始化命令 */
  ssd1306_SendCommands(init_commands, sizeof(init_commands));

  /* 清空显存 */
  ssd1306_Fill(SSD1306_COLOR_BLACK);

  /* 开启显示 */
  ssd1306_SendCommand(0xAF);

  HAL_Delay(100);
}

/**
  * @brief  清屏
  * @param  color: SSD1306_COLOR_BLACK或SSD1306_COLOR_WHITE
  * @retval None
  */
void ssd1306_Fill(uint8_t color)
{
  memset(ssd1306_gram, color == SSD1306_COLOR_WHITE ? 0xFF : 0x00, sizeof(ssd1306_gram));
}

/**
  * @brief  更新显示（将显存数据发送到OLED）
  * @param  None
  * @retval None
  *
  * @note   必须调用此函数才能看到屏幕变化
  */
void ssd1306_UpdateScreen(void)
{
  /* 设置列地址范围 */
  ssd1306_SendCommand(0x21);  // 设置列地址
  ssd1306_SendCommand(0x00);  // 起始列
  ssd1306_SendCommand(SSD1306_WIDTH - 1);  // 结束列

  /* 设置页地址范围 */
  ssd1306_SendCommand(0x22);  // 设置页地址
  ssd1306_SendCommand(0x00);  // 起始页
  ssd1306_SendCommand(0x07);  // 结束页（7）

  /* 发送显存数据 */
  ssd1306_WriteBuffer((uint8_t *)ssd1306_gram, sizeof(ssd1306_gram), SSD1306_DATA);
}

/**
  * @brief  设置单个像素
  * @param  x: x坐标 (0-127)
  * @param  y: y坐标 (0-63)
  * @param  color: 颜色
  * @retval None
  */
void ssd1306_SetPixel(int16_t x, int16_t y, uint8_t color)
{
  if (x < 0 || x >= SSD1306_WIDTH || y < 0 || y >= SSD1306_HEIGHT)
    return;

  if (color == SSD1306_COLOR_WHITE)
  {
    ssd1306_gram[y / 8][x] |= (1 << (y % 8));
  }
  else if (color == SSD1306_COLOR_BLACK)
  {
    ssd1306_gram[y / 8][x] &= ~(1 << (y % 8));
  }
  else  // INVERSE
  {
    ssd1306_gram[y / 8][x] ^= (1 << (y % 8));
  }
}

/**
  * @brief  画线
  * @param  x0, y0: 起点坐标
  * @param  x1, y1: 终点坐标
  * @param  color: 颜色
  * @retval None
  */
void ssd1306_DrawLine(int16_t x0, int16_t y0, int16_t x1, int16_t y1, uint8_t color)
{
  int16_t dx = abs(x1 - x0);
  int16_t dy = -abs(y1 - y0);
  int16_t sx = x0 < x1 ? 1 : -1;
  int16_t sy = y0 < y1 ? 1 : -1;
  int16_t err = dx + dy;

  while (1)
  {
    ssd1306_SetPixel(x0, y0, color);
    if (x0 == x1 && y0 == y1) break;
    int16_t e2 = 2 * err;
    if (e2 >= dy) { err += dy; x0 += sx; }
    if (e2 <= dx) { err += dx; y0 += sy; }
  }
}

/**
  * @brief  画矩形
  * @param  x, y: 左上角坐标
  * @param  w, h: 宽和高
  * @param  color: 颜色
  * @retval None
  */
void ssd1306_DrawRect(int16_t x, int16_t y, int16_t w, int16_t h, uint8_t color)
{
  ssd1306_DrawLine(x, y, x + w, y, color);
  ssd1306_DrawLine(x, y + h, x + w, y + h, color);
  ssd1306_DrawLine(x, y, x, y + h, color);
  ssd1306_DrawLine(x + w, y, x + w, y + h, color);
}

/**
  * @brief  填充矩形
  * @param  x, y: 左上角坐标
  * @param  w, h: 宽和高
  * @param  color: 颜色
  * @retval None
  */
void ssd1306_FillRect(int16_t x, int16_t y, int16_t w, int16_t h, uint8_t color)
{
  for (int16_t i = 0; i < h; i++)
  {
    for (int16_t j = 0; j < w; j++)
    {
      ssd1306_SetPixel(x + j, y + i, color);
    }
  }
}

/**
  * @brief  画圆
  * @param  x, y: 圆心坐标
  * @param  r: 半径
  * @param  color: 颜色
  * @retval None
  */
void ssd1306_DrawCircle(int16_t x, int16_t y, int16_t r, uint8_t color)
{
  int16_t d = 3 - 2 * r;
  int16_t xx = 0;
  int16_t yy = r;

  while (xx <= yy)
  {
    ssd1306_SetPixel(x + xx, y + yy, color);
    ssd1306_SetPixel(x + xx, y - yy, color);
    ssd1306_SetPixel(x - xx, y + yy, color);
    ssd1306_SetPixel(x - xx, y - yy, color);
    ssd1306_SetPixel(x + yy, y + xx, color);
    ssd1306_SetPixel(x + yy, y - xx, color);
    ssd1306_SetPixel(x - yy, y + xx, color);
    ssd1306_SetPixel(x - yy, y - xx, color);

    if (d < 0)
    {
      d += 4 * xx + 6;
    }
    else
    {
      d += 4 * (xx - yy) + 10;
      yy--;
    }
    xx++;
  }
}

/**
  * @brief  填充圆
  * @param  x, y: 圆心坐标
  * @param  r: 半径
  * @param  color: 颜色
  * @retval None
  */
void ssd1306_FillCircle(int16_t x, int16_t y, int16_t r, uint8_t color)
{
  int16_t d = 3 - 2 * r;
  int16_t xx = 0;
  int16_t yy = r;

  while (xx <= yy)
  {
    ssd1306_DrawLine(x - xx, y + yy, x + xx, y + yy, color);
    ssd1306_DrawLine(x - xx, y - yy, x + xx, y - yy, color);
    ssd1306_DrawLine(x - yy, y + xx, x + yy, y + xx, color);
    ssd1306_DrawLine(x - yy, y - xx, x + yy, y - xx, color);

    if (d < 0)
    {
      d += 4 * xx + 6;
    }
    else
    {
      d += 4 * (xx - yy) + 10;
      yy--;
    }
    xx++;
  }
}

/**
  * @brief  设置光标位置
  * @param  x: x坐标
  * @param  y: y坐标（页号，0-7）
  * @retval None
  */
void ssd1306_SetCursor(int16_t x, int16_t y)
{
  /* 设置页地址 */
  ssd1306_SendCommand(0xB0 | (y & 0x07));
  /* 设置列地址高4位 */
  ssd1306_SendCommand(0x10 | ((x >> 4) & 0x0F));
  /* 设置列地址低4位 */
  ssd1306_SendCommand(0x00 | (x & 0x0F));
}

/**
  * @brief  显示字符
  * @param  ch: 字符
  * @param  color: 颜色
  * @retval None
  */
void ssd1306_PrintChar(char ch, uint8_t color)
{
  uint8_t c = (uint8_t)ch;

  if (c < ' ' || c > '~')
  {
    c = ' ';  // 未定义的字符显示为空格
  }

  c = c - ' ';

  for (uint8_t i = 0; i < 6; i++)
  {
    uint8_t line = font_8x6[c][i];

    for (uint8_t j = 0; j < 8; j++)
    {
      if (line & (1 << j))
      {
        ssd1306_SetPixel(0, 0, color);  // 占位，实际位置由光标控制
      }
    }
  }

  /* 直接写入显存 */
  /* 简化版本：直接写入一列 */
}

/**
  * @brief  水平滚动
  * @param  speed: 滚动速度 (0=慢, 7=快)
  * @retval None
  */
void ssd1306_ScrollHorizontal(int16_t speed)
{
  ssd1306_SendCommand(0x27);  // 水平滚动向右
  ssd1306_SendCommand(0x00);  // 虚拟字节
  ssd1306_SendCommand(0x00);  // 起始页
  ssd1306_SendCommand(0x00 | (speed & 0x07));  // 速度
  ssd1306_SendCommand(0x07);  // 结束页
  ssd1306_SendCommand(0x00);  // 虚拟字节
  ssd1306_SendCommand(0xFF);  // 虚拟字节
  ssd1306_SendCommand(0x2F);  // 激活滚动
}

/**
  * @brief  滚动配置
  * @param  start_page: 起始页 0-7
  * @param  end_page: 结束页 0-7
  * @param  direction: 0=右, 1=左
  * @retval None
  */
void ssd1306_Scroll(uint8_t start_page, uint8_t end_page, uint8_t direction)
{
  if (direction == 0)
  {
    ssd1306_SendCommand(0x27);  // 水平滚动向右
  }
  else
  {
    ssd1306_SendCommand(0x26);  // 水平滚动向左
  }
  ssd1306_SendCommand(0x00);
  ssd1306_SendCommand(start_page);
  ssd1306_SendCommand(0x00);
  ssd1306_SendCommand(end_page);
  ssd1306_SendCommand(0x00);
  ssd1306_SendCommand(0xFF);
  ssd1306_SendCommand(0x2F);
}

五、常见问题排查

5.1 OLED不亮

检查1：接线是否正确 
用万用表检查：
1. OLED的VCC和GND是否连通
2. SCL/SDA是否连通
3. 尝试交换SCL和SDA（有些模块标注可能错误）
检查2：I2C地址是否正确 
OLED不亮的常见原因：地址错误

尝试不同的地址：
  #define SSD1306_I2C_ADDR  0x78  // 常用地址
  #define SSD1306_I2C_ADDR  0x7A  // 备选地址
  #define SSD1306_I2C_ADDR  0x3C  // 7位地址格式
  #define SSD1306_I2C_ADDR  0x3D  // 7位地址格式(备选)

用I2C扫描工具检测实际地址：
  while (HAL_I2C_IsDeviceReady(&hi2c1, addr, 3, 100) != HAL_OK);
检查3：OLED初始化序列 
有些OLED模块需要更长的初始化时间：
  HAL_Delay(200);  // 增加到200ms

检查电荷泵是否启用：
  0x8D, 0x14  // 启用电荷泵
  0x8D, 0x10  // 禁用电荷泵（需要外部电源供电）

5.2 显示花屏

花屏原因通常是初始化序列不正确：

常见原因：
1. 初始化命令顺序错误
2. 某些OLED模块需要不同的初始化序列
3. 显存写入格式不对

解决方案：
1. 尝试不同的初始化序列
2. 检查Page/Column地址设置
3. 确认是水平寻址模式(0x20, 0x00)

5.3 I2C通信失败

检查项：
1. 确认PB6/PB7已配置为I2C1
2. 确认I2C时钟频率正确（100kHz标准，400kHz快速）
3. 确认没有引脚冲突

I2C扫描代码：
void I2C_Scan(void)
{
  printf("I2C Devices:\r\n");
  for (uint8_t addr = 1; addr < 128; addr++)
  {
    if (HAL_I2C_IsDeviceReady(&hi2c1, addr << 1, 3, 100) == HAL_OK)
    {
      printf("Found: 0x%02X\r\n", addr << 1);
    }
  }
}

❓ 常见问题 FAQ

Q: OLED不亮？

A: 1)I2C地址扫描(0x3C或0x3D) 2)SDA/SCL上拉电阻4.7kΩ 3)供电3.3V（不是5V！）

Q: 显示花屏？

A: 1)刷新率太快导致I2C拥堵 2)显存buffer未清零 3)坐标计算越界（0~127×0~63）
