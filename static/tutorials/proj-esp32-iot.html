教程14: ESP32 WiFi物联网

📋 项目概述

本项目使用ESP32开发板构建完整的物联网(IoT)解决方案。包含WiFi联网、HTTP客户端请求、MQTT协议发布订阅、ThingSpeak云端数据记录以及OTA远程升级。学会将传感器数据上云、通过手机APP远程监控、以及后续的远程固件升级。

📝 接线文字描述：
ESP32开发板通过Micro USB线连接到电脑，既供电又提供串口通信和固件下载。DHT11传感器插在面包板上，VCC接ESP32的3.3V引脚，GND接GND。DATA引脚接GPIO4，同时在DATA和VCC之间接一个4.7KΩ上拉电阻（可选，部分模块已内置）。LED指示灯的长脚（阳极）串联220Ω电阻后接GPIO2，短脚（阴极）接GND，用于指示WiFi连接状态。注意ESP32的GPIO2在烧录时需要保持低电平（不接LED），所以LED可以用跳线帽或开关控制，烧录完成后接通。

📚 技术背景

ESP32简介

ESP32是乐鑫科技推出的双核WiFi+蓝牙MCU，主频240MHz，内置520KB SRAM和4MB Flash，支持802.11 b/g/n WiFi和经典/低功耗蓝牙。Arduino框架下编程非常友好。

MQTT协议

MQTT(Message Queuing Telemetry Transport)是轻量级的发布/订阅模式物联网协议，基于TCP/IP。核心概念：Broker(服务器)、Topic(主题)、Publish(发布)、Subscribe(订阅)。支持QoS三个等级。

ThingSpeak云平台

ThingSpeak是MathWorks旗下的IoT分析平台，通过HTTP GET/POST上传数据，提供数据可视化图表和MATLAB分析。

OTA升级

OTA(Over-The-Air)升级允许ESP32通过WiFi网络远程更新固件，无需物理连接USB。ArduinoOTA库内置了安全鉴权。

💻 完整代码

esp32_iot.ino — 主程序

#include <WiFi.h>
#include <HTTPClient.h>
#include <WiFiClient.h>
#include <PubSubClient.h>
#include <DHT.h>
#include <ArduinoOTA.h>
#include <ArduinoJson.h>

/* =========== WiFi配置 =========== */
const char* ssid     = "YourWiFiName";       // 替换为你的WiFi SSID
const char* password = "YourWiFiPassword";    // 替换为你的WiFi密码

/* =========== MQTT配置 =========== */
const char* mqtt_server = "broker.emqx.io";  // 公共MQTT broker
const int   mqtt_port   = 1883;
const char* mqtt_client_id = "esp32_dht11_001";
const char* mqtt_pub_topic  = "esp32/sensor/dht11";   // 发布主题
const char* mqtt_sub_topic  = "esp32/cmd/led";         // 订阅主题

/* =========== ThingSpeak配置 =========== */
const char* thingspeak_server = "api.thingspeak.com";
String      thingspeak_api_key = "YOUR_WRITE_API_KEY_HERE"; // 替换为你的Write API Key

/* =========== DHT11配置 =========== */
#define DHTPIN 4        // GPIO4
#define DHTTYPE DHT11   // DHT11传感器
DHT dht(DHTPIN, DHTTYPE);

/* =========== LED配置 =========== */
#define LED_PIN 2       // GPIO2 (内置LED)
#define WIFI_LED_PIN 22 // 外部LED(可选)

/* =========== 全局对象 =========== */
WiFiClient wifi_client;
PubSubClient mqtt_client(wifi_client);

/* =========== 状态变量 =========== */
unsigned long last_sensor_read = 0;
unsigned long last_mqtt_pub    = 0;
unsigned long last_thingspeak  = 0;
const unsigned long sensor_interval   = 2000;  // 传感器采集间隔 ms
const unsigned long mqtt_interval     = 10000; // MQTT发布间隔 ms
const unsigned long thingspeak_interval = 20000; // ThingSpeak上传间隔 ms

float temperature = 0.0f;
float humidity    = 0.0f;
bool  led_state   = false;

/* =========== WiFi连接 =========== */
void setup_wifi()
{
    Serial.print("Connecting to WiFi");
    WiFi.begin(ssid, password);

    int retry = 0;
    while (WiFi.status() != WL_CONNECTED && retry < 30) {
        delay(500);
        Serial.print(".");
        retry++;
        // 闪烁LED指示连接状态
        digitalWrite(LED_PIN, retry % 2);
    }

    if (WiFi.status() == WL_CONNECTED) {
        Serial.println("\r\nWiFi connected!");
        Serial.print("IP: ");
        Serial.println(WiFi.localIP());
        digitalWrite(LED_PIN, HIGH); // 连接成功LED常亮
    } else {
        Serial.println("\r\nWiFi connection failed!");
    }
}

/* =========== MQTT回调处理 =========== */
void mqtt_callback(char* topic, byte* payload, unsigned int length)
{
    Serial.print("MQTT message arrived [");
    Serial.print(topic);
    Serial.print("] ");

    char msg[length + 1];
    memcpy(msg, payload, length);
    msg[length] = '\0';
    Serial.println(msg);

    /* 处理LED控制命令 */
    if (strcmp(topic, mqtt_sub_topic) == 0) {
        if (strcmp(msg, "ON") == 0) {
            led_state = true;
            digitalWrite(LED_PIN, HIGH);
            Serial.println("LED ON");
        } else if (strcmp(msg, "OFF") == 0) {
            led_state = false;
            digitalWrite(LED_PIN, LOW);
            Serial.println("LED OFF");
        } else if (strcmp(msg, "TOGGLE") == 0) {
            led_state = !led_state;
            digitalWrite(LED_PIN, led_state ? HIGH : LOW);
            Serial.printf("LED TOGGLE: %s\r\n", led_state ? "ON" : "OFF");
        }
    }
}

/* =========== MQTT重连 =========== */
void mqtt_reconnect()
{
    while (!mqtt_client.connected()) {
        Serial.print("Attempting MQTT connection...");
        if (mqtt_client.connect(mqtt_client_id)) {
            Serial.println("connected");
            mqtt_client.subscribe(mqtt_sub_topic);
            Serial.print("Subscribed to: ");
            Serial.println(mqtt_sub_topic);

            // 发布上线通知
            mqtt_client.publish(mqtt_pub_topic, "ESP32 online");
        } else {
            Serial.print("failed, rc=");
            Serial.print(mqtt_client.state());
            Serial.println(" retry in 5s");
            delay(5000);
        }
    }
}

/* =========== 发布传感器数据到MQTT（JSON格式）=========== */
void publish_sensor_mqtt()
{
    StaticJsonDocument<200> doc;
    doc["temp"]     = temperature;
    doc["humid"]    = humidity;
    doc["timestamp"] = millis();

    char buffer[256];
    size_t n = serializeJson(doc, buffer);
    mqtt_client.publish(mqtt_pub_topic, buffer, n);

    Serial.print("MQTT publish: ");
    Serial.println(buffer);
}

/* =========== 上传到ThingSpeak（HTTP GET）=========== */
void upload_thingspeak()
{
    HTTPClient http;
    String url = String("http://") + thingspeak_server +
                 "/update?api_key=" + thingspeak_api_key +
                 "&field1=" + String(temperature) +
                 "&field2=" + String(humidity);

    http.begin(url);
    int http_code = http.GET();

    if (http_code == HTTP_CODE_OK) {
        String response = http.getString();
        Serial.print("ThingSpeak OK, entry: ");
        Serial.println(response);
    } else {
        Serial.print("ThingSpeak error: ");
        Serial.println(http_code);
    }

    http.end();
}

/* =========== 获取网络时间（HTTP GET JSON）=========== */
String get_ntp_time()
{
    HTTPClient http;
    http.begin("http://worldtimeapi.org/api/ip");
    int code = http.GET();

    if (code == HTTP_CODE_OK) {
        DynamicJsonDocument doc(1024);
        deserializeJson(doc, http.getString());
        String datetime = doc["datetime"].as<String>();
        http.end();
        return datetime.substring(0, 19); // "2026-06-16T18:45:00"
    }
    http.end();
    return "N/A";
}

/* =========== DHT11传感器读取 =========== */
void read_sensor()
{
    float h = dht.readHumidity();
    float t = dht.readTemperature();

    if (isnan(h) || isnan(t)) {
        Serial.println("DHT11 read failed!");
        return;
    }

    temperature = t;
    humidity = h;

    Serial.printf("Sensor: T=%.1f°C H=%.1f%%\r\n", t, h);
}

/* =========== ArduinoOTA配置 =========== */
void setup_ota()
{
    ArduinoOTA.setHostname("esp32-sensor-001");
    ArduinoOTA.setPassword("esp32ota123");  // 设置升级密码

    ArduinoOTA
    .onStart([]() {
        String type = (ArduinoOTA.getCommand() == U_FLASH) ?
                       "sketch" : "filesystem";
        Serial.println("OTA Start: " + type);
    })
    .onEnd([]() {
        Serial.println("\r\nOTA End");
    })
    .onProgress([](unsigned int progress, unsigned int total) {
        Serial.printf("OTA: %u%%\r\n", (progress / (total / 100)));
    })
    .onError([](ota_error_t error) {
        Serial.printf("OTA Error[%u]: ", error);
        if (error == OTA_AUTH_ERROR)       Serial.println("Auth Failed");
        else if (error == OTA_BEGIN_ERROR)  Serial.println("Begin Failed");
        else if (error == OTA_CONNECT_ERROR) Serial.println("Connect Failed");
        else if (error == OTA_RECEIVE_ERROR) Serial.println("Receive Failed");
        else if (error == OTA_END_ERROR)    Serial.println("End Failed");
    });

    ArduinoOTA.begin();
    Serial.println("OTA ready");
}

/* =========== setup =========== */
void setup()
{
    Serial.begin(115200);
    Serial.println("\r\n=== ESP32 IoT Sensor Station ===");

    pinMode(LED_PIN, OUTPUT);
    digitalWrite(LED_PIN, LOW);

    dht.begin();
    delay(1000);

    setup_wifi();

    /* 初始化MQTT */
    mqtt_client.setServer(mqtt_server, mqtt_port);
    mqtt_client.setCallback(mqtt_callback);

    /* 初始化OTA */
    setup_ota();
}

/* =========== loop =========== */
void loop()
{
    unsigned long now = millis();

    /* ===== 1. OTA处理 ===== */
    ArduinoOTA.handle();

    /* ===== 2. WiFi状态检查 ===== */
    if (WiFi.status() != WL_CONNECTED) {
        digitalWrite(LED_PIN, (now / 500) % 2);
        Serial.println("WiFi disconnected, reconnecting...");
        WiFi.reconnect();
        delay(1000);
        return;
    }

    /* ===== 3. MQTT连接维护 ===== */
    if (!mqtt_client.connected()) {
        mqtt_reconnect();
    }
    mqtt_client.loop();  // 处理MQTT消息

    /* ===== 4. 传感器采集 ===== */
    if (now - last_sensor_read >= sensor_interval) {
        read_sensor();
        last_sensor_read = now;
    }

    /* ===== 5. MQTT发布 ===== */
    if (now - last_mqtt_pub >= mqtt_interval) {
        publish_sensor_mqtt();
        last_mqtt_pub = now;
    }

    /* ===== 6. ThingSpeak上传 ===== */
    if (now - last_thingspeak >= thingspeak_interval) {
        upload_thingspeak();
        last_thingspeak = now;
    }

    /* ===== 7. 串口命令处理 ===== */
    if (Serial.available()) {
        String cmd = Serial.readStringUntil('\n');
        cmd.trim();
        if (cmd == "time") {
            String ntp = get_ntp_time();
            Serial.println("NTP time: " + ntp);
        } else if (cmd == "ping") {
            Serial.println("pong");
        } else if (cmd.startsWith("led")) {
            String state = cmd.substring(3);
            state.trim();
            mqtt_client.publish(mqtt_sub_topic, state.c_str());
        } else if (cmd == "status") {
            Serial.printf("WiFi: %s (%ddBm)\r\n",
                          WiFi.SSID(), WiFi.RSSI());
            Serial.printf("MQTT: %s\r\n",
                          mqtt_client.connected() ? "OK" : "Disconnected");
            Serial.printf("Temp: %.1f  Humi: %.1f\r\n",
                          temperature, humidity);
        }
    }

    /* 主循环不做长延时，保证OTA和MQTT及时响应 */
    delay(10);
}

/* WiFi事件处理（可选）*/
void WiFiEvent(WiFiEvent_t event)
{
    switch (event) {
    case ARDUINO_EVENT_WIFI_STA_DISCONNECTED:
        Serial.println("WiFi disconnected");
        break;
    case ARDUINO_EVENT_WIFI_STA_GOT_IP:
        Serial.print("Got IP: ");
        Serial.println(WiFi.localIP());
        break;
    }
}

❓ 常见问题排查

⚠️ 问题1：WiFi连接超时
现象：WiFi始终无法连接，返回WL_CONNECT_FAILED
原因：WiFi密码错误、信号太弱、2.4G/5G兼容性
解决：
1. 确认SSID和密码无误（特别注意大小写）
2. ESP32只支持2.4GHz WiFi，确认路由器开启了2.4G频段
3. 将ESP32靠近路由器（2米以内测试）
4. 使用WiFiManager库，手机端连接ESP32的热点配置WiFi
5. 检查路由器是否开启了MAC地址过滤
⚠️ 问题2：MQTT断线重连
现象：ESP32运行一段时间后MQTT断开无法恢复
原因：MQTT心跳超时、网络抖动、Broker主动断开
解决：
1. 设置长KeepAlive：mqtt_client.setKeepAlive(30)
2. 主循环中持续调用 mqtt_client.loop()
3. 断线后自动重连：mqtt_reconnect()
4. 使用本地MQTT Broker（如Mosquitto）代替公共Broker
5. 检查路由器是否阻止了1883端口
⚠️ 问题3：OTA升级失败
现象：OTA进度条到100%后报错，ESP32死机
原因：Flash分区不足、WiFi信号弱、固件过大
解决：
1. 确保分区表选了OTA方案（工具 → Partition Scheme）
2. OTA固件大小必须 < app分区大小（OTA预留1MB）
3. 升级时靠近路由器保证信号强度
4. 使用有线烧录(USB)恢复：按住BOOT键再按EN键进入下载模式
5. 前期开发建议先省略OTA，有线烧录稳定后再启用
⚠️ 问题4：DHT11采集与WiFi冲突
现象：WiFi连接期间DHT11读取失败率升高
原因：ESP32的WiFi射频干扰单总线时序
解决：
1. 在WiFi不活跃时（如刚连接完）再采集，避开扫描/重连
2. 采集时降低WiFi功耗模式：WiFi.setSleep(WIFI_PS_NONE)
3. 使用DHT22（精度更高，抗干扰更好）
4. 考虑用I2C接口的传感器替代DHT11
⚠️ 问题5：ThingSpeak上传限制
现象：数据上传失败或收到HTTP 403
原因：ThingSpeak限制免费账户每15秒最多1次上传
解决：
1. 设置上传间隔 ≥ 20秒
2. 确认Write API Key正确
3. ThingSpeak免费账户每天最多8190条数据
4. 超过限制后，使用MQTT作为补充的数据通道

💡 ESP32调试技巧：
1. 串口监视器波特率：115200
2. 使用 Serial.printf() 格式化输出（Arduino默认支持）
3. 开启Debug输出：core_debug_level = 5（platformio.ini）
4. 使用ESP32内置的FreeRTOS：vTaskDelay(pdMS_TO_TICKS(1000))
5. 内存状态监控：Serial.printf("Free heap: %u\r\n", ESP.getFreeHeap())
6. 可访问 http://ip 查看ESP32内置Web服务器内容（需实现WebServer库）
