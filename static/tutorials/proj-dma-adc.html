📊 教程17: DMA双缓冲ADC采集

📋 项目概述

本教程将带你实现基于DMA双缓冲的ADC连续采集系统。通过双缓冲技术，CPU可以在处理一个缓冲区的数据时，DMA同时在填充另一个缓冲区，实现"零丢点"的不间断采集，并通过串口输出简单的波形数据。

为什么要双缓冲？单缓冲模式下，DMA写完整个缓冲区后需要CPU立即处理，否则下一轮DMA传输会覆盖数据。双缓冲让CPU有足够的时间处理数据，同时DMA无间断地继续采集，真正实现"边采边处理"。

📖 知识讲解

DMA双缓冲工作原理

时间轴:
─────────────────────────────────────────────→

DMA: [===写Buffer A===][===写Buffer B===][===写Buffer A===]...
CPU:                  [处理Buffer A    ][处理Buffer B    ]...
                      ↑                ↑
                 HalfCplt回调    Cplt回调

关键回调:
- HAL_ADC_ConvHalfCpltCallback: DMA写满前半段时触发
  → CPU可以安全处理前半段数据
- HAL_ADC_ConvCpltCallback: DMA写满整个缓冲区时触发
  → CPU可以安全处理后半段数据

这种方式下，CPU和DMA永远不会同时访问同一段数据！

采样率计算

STM32F103 ADC时钟和采样率：
- ADC时钟最大14MHz(APB2的2分频=36MHz → 再2分频=18MHz或4分频=9MHz)
- 采样时间：1.5+12.5=14个ADC时钟周期(最小)
- 采样率 = ADC时钟 / 采样周期
- 例：ADC时钟12MHz，采样周期14 → 采样率 = 12M/14 ≈ 857Ksps
- 但实际上受DMA和总线带宽限制，连续采样建议不超过1Msps

💻 完整代码

DMA双缓冲ADC采集 (main.c)

/* ============================================
 * DMA双缓冲ADC采集 - 完整示例
 * MCU: STM32F103C8T6
 * 功能: ADC连续采样 + DMA双缓冲 + 串口波形输出
 * ============================================ */

#include "main.h"
#include <stdio.h>
#include <string.h>

/* 配置常量 */
#define ADC_BUF_LEN       256      /* 总缓冲区长度(半缓冲=128) */
#define SAMPLE_RATE_HZ    10000    /* 目标采样率10KHz */
#define WAVEFORM_SEND_EN  1        /* 1=串口发送波形, 0=不发送 */

/* 私有变量 */
ADC_HandleTypeDef hadc1;
DMA_HandleTypeDef hdma_adc1;
UART_HandleTypeDef huart1;
TIM_HandleTypeDef htim3;

/* 双缓冲区 - DMA循环模式使用一个大数组的前半和后半 */
ALIGN_32BYTES(uint16_t adc_buffer[ADC_BUF_LEN]);

/* 处理缓冲区 */
uint16_t process_buffer[ADC_BUF_LEN / 2];
volatile uint8_t data_ready = 0;      /* 数据就绪标志 */
volatile uint8_t process_half = 0;    /* 0=前半, 1=后半 */
volatile uint32_t sample_count = 0;   /* 采样计数 */

/* 串口发送缓冲区 */
char uart_buf[256];

/* 函数声明 */
void SystemClock_Config(void);
static void MX_GPIO_Init(void);
static void MX_DMA_Init(void);
static void MX_ADC1_Init(void);
static void MX_USART1_UART_Init(void);
static void MX_TIM3_Init(void);
static void Process_ADC_Data(uint16_t *buf, uint16_t len);
static void Send_Waveform(uint16_t *buf, uint16_t len);
static void Print_Statistics(uint16_t *buf, uint16_t len);

int main(void)
{
    /* HAL初始化 */
    HAL_Init();
    SystemClock_Config();

    /* 外设初始化 - 注意DMA必须先于ADC初始化! */
    MX_GPIO_Init();
    MX_DMA_Init();     /* DMA必须先初始化 */
    MX_USART1_UART_Init();
    MX_TIM3_Init();
    MX_ADC1_Init();

    printf("\r\n=== DMA双缓冲ADC采集 ===\r\n");
    printf("采样率: %d Hz\r\n", SAMPLE_RATE_HZ);
    printf("缓冲区: %d 样本(半缓冲 %d)\r\n", ADC_BUF_LEN, ADC_BUF_LEN/2);
    printf("ADC分辨率: 12位 (0-4095)\r\n\r\n");

    /* 启动ADC DMA连续采集 */
    /* DMA_CIRCULAR模式: DMA写到末尾后自动回到开头继续写 */
    if (HAL_ADC_Start_DMA(&hadc1, (uint32_t*)adc_buffer, ADC_BUF_LEN) != HAL_OK)
    {
        printf("ADC DMA启动失败!\r\n");
        Error_Handler();
    }

    /* 启动定时器(如果使用定时器触发) */
    HAL_TIM_Base_Start(&htim3);

    while (1)
    {
        if (data_ready)
        {
            data_ready = 0;

            /* 复制数据到处理缓冲区(避免DMA在处理时覆盖) */
            uint16_t *src = process_half ?
                            &adc_buffer[ADC_BUF_LEN / 2] :
                            &adc_buffer[0];
            memcpy(process_buffer, src, (ADC_BUF_LEN / 2) * sizeof(uint16_t));

            /* 处理数据 */
            Process_ADC_Data(process_buffer, ADC_BUF_LEN / 2);
        }

        /* 主循环可以做其他事情 */
        HAL_Delay(100);
    }
}

/* ============ ADC数据处理 ============ */
static void Process_ADC_Data(uint16_t *buf, uint16_t len)
{
    sample_count += len;

#if WAVEFORM_SEND_EN
    /* 发送波形数据到串口(供上位机显示) */
    Send_Waveform(buf, len);
#endif

    /* 每10次处理打印一次统计信息 */
    static uint8_t print_counter = 0;
    print_counter++;
    if (print_counter >= 10)
    {
        print_counter = 0;
        Print_Statistics(buf, len);
    }
}

/* ============ 发送波形数据(简单文本格式) ============ */
static void Send_Waveform(uint16_t *buf, uint16_t len)
{
    /* 格式: "W:val1,val2,val3,...\r\n"
     * 上位机收到后可以绘制波形图 */

    uint16_t i;
    uint16_t pos = 0;

    uart_buf[pos++] = 'W';
    uart_buf[pos++] = ':';

    for (i = 0; i < len && pos < sizeof(uart_buf) - 10; i++)
    {
        /* 降采样: 每4个点取1个，减少串口传输量 */
        if (i % 4 == 0)
        {
            pos += snprintf(&uart_buf[pos], sizeof(uart_buf) - pos,
                           "%u,", buf[i]);
        }
    }

    /* 替换最后的逗号为换行 */
    if (pos > 0 && uart_buf[pos-1] == ',')
    {
        uart_buf[pos-1] = '\r';
        uart_buf[pos] = '\n';
        pos++;
    }

    HAL_UART_Transmit(&huart1, (uint8_t*)uart_buf, pos, 100);
}

/* ============ 打印统计信息 ============ */
static void Print_Statistics(uint16_t *buf, uint16_t len)
{
    uint32_t sum = 0;
    uint16_t min_val = 0xFFFF;
    uint16_t max_val = 0;
    float voltage;

    for (uint16_t i = 0; i < len; i++)
    {
        sum += buf[i];
        if (buf[i] < min_val) min_val = buf[i];
        if (buf[i] > max_val) max_val = buf[i];
    }

    uint16_t avg = sum / len;
    voltage = (avg * 3.3f) / 4096.0f;

    printf("ADC统计: Avg=%u Min=%u Max=%u V=%.2fV Samples=%lu\r\n",
           avg, min_val, max_val, voltage, sample_count);
}

/* ============ DMA半传输完成回调 ============ */
/* DMA写完前半段时触发，此时后半段正在被写入
 * 前半段数据安全，可以处理 */
void HAL_ADC_ConvHalfCpltCallback(ADC_HandleTypeDef *hadc)
{
    process_half = 0;  /* 标记：前半段数据就绪 */
    data_ready = 1;
}

/* ============ DMA传输完成回调 ============ */
/* DMA写完整整个缓冲区时触发，即将从头开始写
 * 后半段数据安全，可以处理 */
void HAL_ADC_ConvCpltCallback(ADC_HandleTypeDef *hadc)
{
    process_half = 1;  /* 标记：后半段数据就绪 */
    data_ready = 1;
}

/* ============ ADC初始化 ============ */
static void MX_ADC1_Init(void)
{
    ADC_ChannelConfTypeDef sConfig = {0};

    hadc1.Instance = ADC1;
    hadc1.Init.ScanConvMode = ADC_SCAN_DISABLE;         /* 单通道，不扫描 */
    hadc1.Init.ContinuousConvMode = ENABLE;              /* 连续转换 */
    hadc1.Init.DiscontinuousConvMode = DISABLE;
    hadc1.Init.ExternalTrigConv = ADC_SOFTWARE_START;    /* 软件触发(或定时器) */
    hadc1.Init.DataAlign = ADC_DATAALIGN_RIGHT;
    hadc1.Init.NbrOfConversion = 1;

    if (HAL_ADC_Init(&hadc1) != HAL_OK)
    {
        Error_Handler();
    }

    /* 配置ADC通道0(PA0) */
    sConfig.Channel = ADC_CHANNEL_0;
    sConfig.Rank = ADC_REGULAR_RANK_1;
    sConfig.SamplingTime = ADC_SAMPLETIME_55CYCLES_5;   /* 55.5个周期 */

    if (HAL_ADC_ConfigChannel(&hadc1, &sConfig) != HAL_OK)
    {
        Error_Handler();
    }
}

/* ============ DMA初始化(必须在ADC之前!) ============ */
static void MX_DMA_Init(void)
{
    /* DMA控制器初始化 */
    __HAL_RCC_DMA1_CLK_ENABLE();

    /* DMA中断配置 */
    HAL_NVIC_SetPriority(DMA1_Channel1_IRQn, 0, 0);
    HAL_NVIC_EnableIRQ(DMA1_Channel1_IRQn);
}

/* ============ 定时器初始化(用于精确采样率) ============ */
static void MX_TIM3_Init(void)
{
    htim3.Instance = TIM3;
    /* 定时器频率 = 72MHz / (PSC+1) / (ARR+1)
     * 目标采样率: 10KHz
     * 72MHz / 72 / 100 = 10KHz */
    htim3.Init.Prescaler = 72 - 1;     /* 72分频 → 1MHz */
    htim3.Init.CounterMode = TIM_COUNTERMODE_UP;
    htim3.Init.Period = 100 - 1;        /* 100分频 → 10KHz */
    htim3.Init.ClockDivision = TIM_CLOCKDIVISION_DIV1;

    if (HAL_TIM_Base_Init(&htim3) != HAL_OK)
    {
        Error_Handler();
    }
}

/* ============ ADC HAL MSP初始化(回调) ============ */
void HAL_ADC_MspInit(ADC_HandleTypeDef *hadc)
{
    GPIO_InitTypeDef GPIO_InitStruct = {0};

    if (hadc->Instance == ADC1)
    {
        /* 使能时钟 */
        __HAL_RCC_ADC1_CLK_ENABLE();
        __HAL_RCC_GPIOA_CLK_ENABLE();

        /* PA0: ADC1_IN0 模拟输入 */
        GPIO_InitStruct.Pin = GPIO_PIN_0;
        GPIO_InitStruct.Mode = GPIO_MODE_ANALOG;
        HAL_GPIO_Init(GPIOA, &GPIO_InitStruct);

        /* 配置DMA */
        hdma_adc1.Instance = DMA1_Channel1;
        hdma_adc1.Init.Direction = DMA_PERIPH_TO_MEMORY;
        hdma_adc1.Init.PeriphInc = DMA_PINC_DISABLE;
        hdma_adc1.Init.MemInc = DMA_MINC_ENABLE;
        hdma_adc1.Init.PeriphDataAlignment = DMA_PDATAALIGN_HALFWORD;  /* 16bit */
        hdma_adc1.Init.MemDataAlignment = DMA_MDATAALIGN_HALFWORD;     /* 16bit */
        hdma_adc1.Init.Mode = DMA_CIRCULAR;       /* 循环模式: 双缓冲的关键! */
        hdma_adc1.Init.Priority = DMA_PRIORITY_HIGH;

        if (HAL_DMA_Init(&hdma_adc1) != HAL_OK)
        {
            Error_Handler();
        }

        __HAL_LINKDMA(hadc, DMA_Handle, hdma_adc1);
    }
}

/* ============ DMA中断处理 ============ */
void DMA1_Channel1_IRQHandler(void)
{
    HAL_DMA_IRQHandler(&hdma_adc1);
}

/* ============ USART初始化 ============ */

    }

}

    RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_HSE;

    RCC_OscInitStruct.PLL.PLLMUL = RCC_PLLMUL_9;

    RCC_ClkInitStruct.ClockType = RCC_CLOCKTYPE_HCLK | RCC_CLOCKTYPE_SYSCLK |

    if (HAL_RCC_ClockConfig(&RCC_ClkInitStruct, FLASH_LATENCY_2) != HAL_OK)
        Error_Handler();

❓ 常见问题排查

问题1：DMA半传输回调不触发

原因：DMA中断未正确配置或未使能半传输中断。

排查步骤：

  1. 确认DMA中断在NVIC中已使能
  2. 确认DMA模式为DMA_CIRCULAR(循环模式)
  3. 检查CubeMX是否正确生成了DMA初始化代码
  4. 手动使能半传输中断：__HAL_DMA_ENABLE_IT(&hdma_adc1, DMA_IT_HT);

问题2：采样率计算

/* 采样率计算方法 */

/* 方法1: 连续转换模式(软件触发) */
/* 采样率 = ADC时钟 / (采样周期 + 转换周期) */
/* F103: ADC时钟 = APB2/分频 = 72MHz/6 = 12MHz */
/* 采样周期 = 55.5, 转换周期 = 12.5 → 总共68周期 */
/* 采样率 = 12MHz / 68 ≈ 176KHz */

/* 方法2: 定时器触发(精确控制) */
/* 采样率 = 定时器频率 */
/* TIM3: 72MHz / (PSC+1) / (ARR+1) */
/* 例: PSC=71, ARR=99 → 72M/72/100 = 10KHz */
/* 注意: 需要设置 ADC ExternalTrigConv = ADC_EXTERNALTRIGCONV_T3_TRGO */

问题3：双缓冲切换时机

关键原则：CPU只处理DMA当前不写入的那一半。

ConvHalfCplt → DMA正在写后半段，前半段安全 → 处理前半段
ConvCplt → DMA正在写前半段，后半段安全 → 处理后半段

注意：处理速度必须快于DMA填充速度，否则下次回调时数据已被覆盖！如果处理太慢，降低采样率或增大缓冲区。

问题4：数据被覆盖/丢点

现象原因解决
波形有毛刺处理速度跟不上DMA增大缓冲区或降低采样率
数据全为零ADC未启动或DMA未配置检查HAL_ADC_Start_DMA返回值
前后半数据混乱处理了错误的一半确认process_half标志逻辑
只有前半段有数据DMA未配置循环模式确认Mode = DMA_CIRCULAR

问题5：DMA初始化顺序

⚠️ DMA必须在ADC之前初始化！
CubeMX生成的代码可能顺序不对，需要手动调整。
在main()中确保：MX_DMA_Init() 在 MX_ADC1_Init() 之前调用。
否则DMA无法正常工作，ADC数据不会被传输到内存。
💡 进阶技巧：
1. 多通道扫描模式：可以同时采集多个ADC通道，设置ScanConvMode=ENABLE
2. 注入通道：在规则通道采集中插入高优先级采样(如过压检测)
3. 使用F407的DMA双缓冲模式(FIFO+Double Buffer)更高效
4. 信号处理：在Process_ADC_Data中添加FFT、滤波等算法
5. 使用DMA的M2M模式实现数据的快速搬移

❓ 常见问题 FAQ

Q: ADC数据跳变？

A: 1)模拟输入阻抗太高（加运放缓冲） 2)采样时间太短（增大ADC采样周期） 3)参考电压不稳（加去耦电容） 4)数字噪声干扰（模拟地数字地分开）

Q: DMA传输完成中断不触发？

A: 1)检查DMA模式和通道配置 2)NVIC使能DMA中断 3)ADC连续采样模式 4)缓冲区大小是DMA传输数据量的整数倍
