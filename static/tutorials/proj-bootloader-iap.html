🚀 教程16: Bootloader串口升级IAP

📋 项目概述

本教程将带你实现STM32的串口IAP(In-Application Programming) bootloader，支持通过串口接收bin文件并写入Flash，实现固件空中升级。

什么是IAP？IAP即"应用内编程"，允许在应用程序运行时更新固件。相比于通过ST-Link烧录，IAP可以实现远程无线升级，是物联网设备的标准功能。

🔧 硬件准备

元件数量说明
STM32F103C8T6 最小系统板164KB Flash，分区后Bootloader 16KB，APP 48KB
USB转TTL串口模块1CH340/FT232，用于连接电脑
杜邦线若干PA9(TX) ↔ USB转TTL RX, PA10(RX) ↔ USB转TTL TX

中断向量表重定位原理

为什么要重定位中断向量表？
ARM Cortex-M的中断向量表默认在Flash起始地址(0x08000000)。当Bootloader和APP分别位于不同地址时，APP的中断向量表需要重定位到自己的起始地址(0x08004000)，否则中断将跳转到Bootloader的中断处理函数。 
SCB->VTOR = 0x08004000;  // 告诉CPU，APP的中断向量表在这里

// APP的中断向量表现在从0x08004000开始:
// 0x08004000: 初始SP
// 0x08004004: 复位向量 → main()
// 0x08004008: 硬fault
// ...

跳转函数原理

typedef void (*pFunction)(void);

pFunction Jump_To_Application;
uint32_t JumpAddress;

JumpAddress = *(__IO uint32_t*)(APP_ADDR + 4);  // 获取复位向量地址
Jump_To_Application = (pFunction)JumpAddress;

__set_MSP(*(__IO uint32_t*)APP_ADDR);  // 设置主堆栈指针

Jump_To_Application();  // 跳转到APP的复位向量 → 执行main()

/* ============================================
 * Bootloader - 串口IAP升级程序
 * MCU: STM32F103C8T6
 * Flash: 64KB, 分区: Bootloader 16KB(0x08000000), APP 48KB(0x08004000)
 * 通信: USART1 (PA9/PA10), 115200bps
 * 协议: YMODEM (简单版)
 * ============================================ */

#include "main.h"
#include <string.h>
#include <stdio.h>

/* 定义常量 */
#define APP_ADDR         0x08004000   /* APP起始地址 */
#define APP_SIZE_MAX     0xC000      /* APP最大48KB */
#define BOOTLOADER_SIZE  0x4000      /* Bootloader 16KB */
#define PAGE_SIZE        0x400       /* Flash页2KB */
#define BAUD_RATE        115200

/* 私有变量 */
UART_HandleTypeDef huart1;
uint8_t uart_rx_buf[1024];
uint8_t uart_tx_buf[128];
volatile uint8_t rx_done = 0;
volatile uint16_t rx_len = 0;

/* YMODEM协议定义 */
#define SOH              0x01        /* 128字节数据块 */
#define STX              0x02        /* 1024字节数据块 */
#define EOT              0x04        /* 传输结束 */
#define ACK              0x06        /* 确认 */
#define NAK              0x15        /* 否定应答 */
#define CAN              0x18        /* 取消 */
#define CRC16            0x43        /* 'C'，请求CRC */

static void Ymodem_Receive(void);
static void Bootloader_Init(void);

int main(void)
{
    HAL_Init();
    SystemClock_Config();
    MX_GPIO_Init();
    MX_USART1_UART_Init();

    /* 打印启动信息 */
    printf("\r\n\r\n========== STM32 IAP Bootloader ==========\r\n");
    printf("Bootloader Version: 1.0\r\n");
    printf("Build Date: %s %s\r\n", __DATE__, __TIME__);
    printf("Flash Size: %lu KB\r\n", *(uint16_t*)0x1FFFF7E0);
    printf("==========================================\r\n\r\n");

    /* 初始化外设(如果需要) */
    Bootloader_Init();

    /* 等待用户选择：进入升级模式 或 跳转APP */
    printf("请选择模式:\r\n");
    printf("  1. 按 'U' 进入升级模式\r\n");
    printf("  2. 按 'G' 跳转到APP\r\n");
    printf("  3. 5秒后自动跳转APP\r\n");
    printf("> ");

    /* 等待串口输入，最多5秒 */
    uint8_t mode_select = 0;
    uint32_t start_tick = HAL_GetTick();
    while ((HAL_GetTick() - start_tick) < 5000)
    {
        if (rx_done)
        {
            rx_done = 0;
            if (uart_rx_buf[0] == 'U' || uart_rx_buf[0] == 'u')
            {
                mode_select = 1; /* 进入升级模式 */
                break;
            }
            else if (uart_rx_buf[0] == 'G' || uart_rx_buf[0] == 'g')
            {
                mode_select = 2; /* 跳转APP */
                break;
            }
        }
    }

    if (mode_select == 1)
    {
        /* 进入YMODEM升级模式 */
        printf("\r\n进入升级模式，准备接收固件...\r\n");
        Ymodem_Receive();
    }
    else
    {
        /* 跳转APP或超时自动跳转 */
        printf("\r\n准备跳转APP...\r\n");
        JumpToApp();
    }

    /* 如果JumpToApp失败(没有有效APP)，停在这里 */
    while (1)
    {
        HAL_Delay(1000);
        printf("No valid APP found. Please upload firmware.\r\n");
    }
}

/* ============ 跳转到APP ============ */
static void JumpToApp(void)
{
    uint32_t app_reset_handler;

    /* 检查APP地址是否有有效程序(检查栈指针和复位向量) */
    if ((*(volatile uint32_t*)APP_ADDR) != 0xFFFFFFFF)
    {
        /* 读取APP的复位向量地址(向量表第2项) */
        app_reset_handler = *(volatile uint32_t*)(APP_ADDR + 4);

        printf("APP Reset Handler: 0x%08lX\r\n", app_reset_handler);

        /* 检查是否是有效的Thumb指令地址 */
        if (((*(volatile uint32_t*)APP_ADDR) & 0x2FFE0000) == 0x20000000)
        {
            /* 关闭所有外设 */
            HAL_UART_DeInit(&huart1);
            __HAL_RCC_USART1_CLK_DISABLE();

            /* 禁用在Bootloader中使能的中断 */
            /* SysTick中断 */
            HAL_NVIC_DisableIRQ(SysTick_IRQn);

            /* 重置所有外设时钟(可选) */
            __HAL_RCC_APB1_FORCE_RESET();
            __HAL_RCC_APB2_FORCE_RESET();
            HAL_Delay(10);

            /* 设置APP的中断向量表地址 */
            SCB->VTOR = APP_ADDR;

            /* 设置主堆栈指针(MSP) */
            __asm volatile("msr msp, %0" : : "r"(*(volatile uint32_t*)APP_ADDR));

            /* 跳转到APP的复位向量 */
            typedef void (*pFunction)(void);
            pFunction Jump_To_App = (pFunction)app_reset_handler;
            Jump_To_App();

            /* 不应该执行到这里 */
        }
        else
        {
            printf("ERROR: Invalid APP reset handler\r\n");
        }
    }
    else
    {
        printf("ERROR: APP address not programmed\r\n");
    }
}

/* ============ YMODEM接收(简化版) ============ */
static void Ymodem_Receive(void)
{
    uint32_t firmware_size = 0;
    uint32_t flash_addr = APP_ADDR;
    uint8_t block_num = 1;
    uint8_t received;
    uint16_t crc;
    uint8_t *rx_buf = uart_rx_buf;

    /* 发送'C'请求第一次传输 */
    Send_Response(CRC16);
    rx_done = 0;

    while (1)
    {
        /* 等待接收数据 */
        uint32_t tick_start = HAL_GetTick();
        while (!rx_done)
        {
            if (HAL_GetTick() - tick_start > 30000) /* 30秒超时 */
            {
                printf("\r\nTimeout! Upgrade cancelled.\r\n");
                return;
            }
        }

        rx_done = 0;
        received = uart_rx_buf[0];

        if (received == SOH || received == STX)
        {
            /* 数据块开始 */
            uint8_t block = uart_rx_buf[1];
            uint8_t block_neg = uart_rx_buf[2];
            uint8_t data_len = (received == SOH) ? 128 : 1024;

            /* 简单校验 */
            if (block + block_neg != 0xFF)
            {
                Send_Response(NAK);
                continue;
            }

            /* 计算CRC16(最后2字节是CRC) */
            crc = CRC16_Calc(&uart_rx_buf[3], data_len);
            uint16_t recv_crc = (uart_rx_buf[3 + data_len] << 8) | uart_rx_buf[3 + data_len + 1];

            if (crc != recv_crc)
            {
                printf("CRC Error! Block %d\r\n", block);
                Send_Response(NAK);
                continue;
            }

            /* 第一块：文件名信息 */
            if (block_num == 1)
            {
                /* 解析文件名和文件大小 */
                char filename[128] = {0};
                sscanf((char*)&uart_rx_buf[3], "%s %lu", filename, &firmware_size);

                printf("Receiving: %s, Size: %lu bytes\r\n", filename, firmware_size);

                if (firmware_size == 0 || firmware_size > APP_SIZE_MAX)
                {
                    printf("Invalid firmware size!\r\n");
                    Send_Response(CAN);
                    return;
                }

                /* 擦除Flash */
                printf("Erasing flash...\r\n");
                Flash_Erase(APP_ADDR, firmware_size);
                printf("Flash erased.\r\n");

                block_num = 2;
                flash_addr = APP_ADDR;
                Send_Response(ACK);
                Send_Response(CRC16); /* 请求下一个块 */
            }
            else if (block == (block_num & 0xFF))
            {
                /* 数据块 */
                uint8_t *data = &uart_rx_buf[3];

                /* 写入Flash */
                Flash_Write(flash_addr, data, data_len);
                flash_addr += data_len;

                block_num++;
                Send_Response(ACK);
            }
            else if (block == 0) /* 结束块 */
            {
                Send_Response(ACK);

                /* 校验整个固件 */
                printf("Firmware received. Size: %lu bytes\r\n", firmware_size);

                /* 跳转APP */
                printf("Jumping to APP...\r\n");
                HAL_Delay(100);
                JumpToApp();
                return;
            }
        }
        else if (received == EOT)
        {
            /* 传输结束 */
            Send_Response(ACK);
            printf("Transfer complete!\r\n");
            JumpToApp();
            return;
        }
        else if (received == CAN)
        {
            printf("Transfer cancelled by sender.\r\n");
            return;
        }
        else
        {
            printf("Unknown: 0x%02X\r\n", received);
        }
    }
}

/* ============ 发送响应 ============ */
static void Send_Response(uint8_t c)
{
    HAL_UART_Transmit(&huart1, &c, 1, 100);
}

/* ============ Flash擦除 ============ */
static void Flash_Erase(uint32_t addr, uint32_t len)
{
    FLASH_EraseInitTypeDef erase;
    uint32_t page_error;

    /* 解锁Flash */
    HAL_FLASH_Unlock();

    erase.TypeErase = FLASH_TYPEERASE_PAGES;
    erase.PageAddress = addr;
    erase.NbPages = (len + PAGE_SIZE - 1) / PAGE_SIZE;

    HAL_FLASH_Erase(&erase, &page_error);

    HAL_FLASH_Lock();
}

/* ============ Flash写入 ============ */
static void Flash_Write(uint32_t addr, uint8_t *data, uint32_t len)
{
    uint32_t i;

    HAL_FLASH_Unlock();

    for (i = 0; i < len; i += 4)
    {
        uint32_t word = data[i] | (data[i+1] << 8) | (data[i+2] << 16) | (data[i+3] << 24);
        HAL_FLASH_Program(FLASH_TYPEPROGRAM_WORD, addr + i, word);
    }

    HAL_FLASH_Lock();
}

/* ============ CRC16计算 ============ */
static uint16_t CRC16_Calc(uint8_t *data, uint32_t len)
{
    uint16_t crc = 0;
    uint32_t i, j;

    for (i = 0; i < len; i++)
    {
        crc ^= data[i];
        for (j = 0; j < 8; j++)
        {
            if (crc & 0x0001)
                crc = (crc >> 1) ^ 0xA001;
            else
                crc = crc >> 1;
        }
    }
    return crc;
}

/* ============ USART回调 ============ */
void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart)
{
    static uint16_t pos = 0;

    if (huart->Instance == USART1)
    {
        rx_buf[pos++] = uart_rx_buf[0];

        if (uart_rx_buf[0] == '\n' || pos >= sizeof(uart_rx_buf))
        {
            rx_len = pos;
            pos = 0;
            rx_done = 1;
        }

        /* 继续接收 */
        HAL_UART_Receive_IT(&huart1, uart_rx_buf, 1);
    }
}

❓ 常见问题排查

问题1：跳转后APP不运行

最常见原因：中断向量表未重定位。
Bootloader位于0x08000000时，CPU默认中断向量表也在0x08000000。
APP位于0x08004000，如果APP中没有重定位向量表，则中断发生后CPU会到0x08000000+offset处查找中断向量，那里是Bootloader的中断处理函数，而Bootloader已经跑完了，所以程序就跑飞了。

解决：在APP的SystemInit()或main()开头添加：SCB->VTOR = 0x08004000;

问题2：Flash写入失败

原因：STM32的Flash写入前必须先解锁，否则写操作被忽略。
写入前：HAL_FLASH_Unlock();
写入后：HAL_FLASH_Lock();

另外，每次写入必须是4字节对齐(Word)，不能单字节写入。
