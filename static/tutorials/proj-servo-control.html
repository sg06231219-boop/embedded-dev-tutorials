🔬 舵机控制原理

参数数值

舵机PWM信号格式

舵机通过PWM信号的脉冲宽度来控制角度，而不是占空比：

┌──────────────────────────────────────────────────────────┐
│                   20ms 周期 (50Hz)                        │
│                                                          │
│  ┌────┐                                                   │
│  │0.5ms│   ← 0°位置                                       │
│  └────┘                                                   │
│                    ┌────────┐                             │
│                    │ 1.5ms  │   ← 90°中间位置             │
│                    └────────┘                             │
│                                   ┌──────────┐            │
│                                   │  2.5ms   │  ← 180°位置│
│                                   └──────────┘            │
│                                                          │
│  ▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓▓│
│  (低电平区间)                                             │
└──────────────────────────────────────────────────────────┘

角度-脉冲对照表：
  0°   → 0.5ms 脉冲
  45°  → 1.0ms 脉冲
  90°  → 1.5ms 脉冲
  135° → 2.0ms 脉冲
  180° → 2.5ms 脉冲
关键区别：舵机关心的是脉冲宽度（ms），而不是占空比百分比！20ms周期中，0.5ms只占2.5%，但这就是0°位置。

CCR值计算公式

假设 TIM3 时钟 = 72MHz, PSC = 72, ARR = 19999 (20ms)

计数频率 = 72000000 / 72 = 1MHz = 1计数/μs

CCR = 脉冲宽度(μs)
  0°   → CCR = 500
  90°  → CCR = 1500
  180° → CCR = 2500

通用公式：
  CCR(angle) = 500 + (angle / 180) × 2000
  CCR(angle) = 500 + angle × 11.111...

例如 angle = 90°:
  CCR = 500 + 90 × 11.111 = 500 + 1000 = 1500 ✓

⚠ 供电警告：
- 强烈建议使用独立5V电源（如手机充电头5V2A或稳压模块），不要用STM32开发板的3.3V或5V给舵机供电！
- STM32F103C8T6的5V引脚从USB取电，电流能力有限，带不动舵机。
- 多个舵机同时运行需要更大电流（每个约200mA，两个就是400mA+）。
- 共地！STM32 GND 和 外部电源 GND 必须连接。

CubeMX TIM2 配置

TIM2 配置（32位定时器，适合长周期）：
  Prescaler (PSC): 72-1    → 计数频率 1MHz (1μs/计数)
  Counter Period (ARR): 20000-1 → 周期 20000μs = 20ms = 50Hz

Channel 1, 2, 3 均配置为 PWM Generation

💻 完整代码

main.c — 舵机控制核心代码

/* USER CODE BEGIN Includes */
#include <stdio.h>
#include <stdint.h>
/* USER CODE END Includes */

/* USER CODE BEGIN PV */
#define SERVO_MIN   500   // 0° 对应 CCR
#define SERVO_MID  1500   // 90° 对应 CCR
#define SERVO_MAX  2500   // 180° 对应 CCR

// 运动序列：每个动作 {舵机1角度, 舵机2角度, 延时ms}
typedef struct {
    uint16_t angle1;   // 舵机1 (底座) 0~180
    uint16_t angle2;   // 舵机2 (臂) 0~180
    uint16_t delay_ms; // 保持时间
} ServoAction_t;

// 演示动作序列
const ServoAction_t g_action_seq[] = {
    { 90,  90, 1000},  // 初始位置
    { 30,  60,  500},  // 左转+抬起
    { 90,  30,  500},  // 中间+下压
    { 90, 120,  500},  // 中间+抬起
    {150,  60,  500},  // 右转+抬起
    {150,  30,  500},  // 右转+下压
    { 90,  90, 1000},  // 回到初始
};
/* USER CODE END PV */

/* USER CODE BEGIN 0 */

// 角度转CCR值
// angle: 0~180度
// 返回: CCR值 500~2500
static uint32_t Angle_To_CCR(uint16_t angle)
{
    if (angle > 180) angle = 180;
    return SERVO_MIN + (angle * (SERVO_MAX - SERVO_MIN) / 180);
}

// 设置舵机角度
// htim: 定时器句柄
// channel: TIM_CHANNEL_1 / TIM_CHANNEL_2 等
// angle: 0~180度
void Servo_SetAngle(TIM_HandleTypeDef *htim, uint32_t channel, uint16_t angle)
{
    uint32_t ccr = Angle_To_CCR(angle);
    __HAL_TIM_SET_COMPARE(htim, channel, ccr);
}

// 平滑移动到目标角度（步进方式）
void Servo_SmoothMove(TIM_HandleTypeDef *htim, uint32_t channel,
                      uint16_t target, uint16_t step_delay_ms)
{
    static uint16_t current_angle = 0; // 上次位置（需要在函数外声明持久变量）
    
    // 简化版：直接跳到目标位置
    // 完整版需要记录current_angle并逐步逼近
    Servo_SetAngle(htim, channel, target);
    HAL_Delay(step_delay_ms);
}

// 执行动作序列
void Servo_RunSequence(void)
{
    uint8_t seq_len = sizeof(g_action_seq) / sizeof(ServoAction_t);
    
    for (uint8_t i = 0; i < seq_len; i++) {
        Servo_SetAngle(&htim2, TIM_CHANNEL_1, g_action_seq[i].angle1);
        Servo_SetAngle(&htim2, TIM_CHANNEL_2, g_action_seq[i].angle2);
        printf("Action %d: S1=%d°, S2=%d°\r\n", 
               i, g_action_seq[i].angle1, g_action_seq[i].angle2);
        HAL_Delay(g_action_seq[i].delay_ms);
    }
}

// 串口打印帮助
void Print_Help(void)
{
    printf("\r\n=== Servo Control Help ===\r\n");
    printf("S1:90  - Set Servo1 angle (0~180)\r\n");
    printf("S2:90  - Set Servo2 angle (0~180)\r\n");
    printf("RUN    - Run action sequence\r\n");
    printf("HELP   - Show this help\r\n");
    printf("=========================\r\n");
}

// 解析串口命令
uint8_t g_rx_buf[64];
volatile uint8_t g_rx_len = 0;

void Parse_Command(uint8_t *cmd)
{
    if (strncmp((char*)cmd, "S1:", 3) == 0) {
        uint16_t ang = atoi((char*)(cmd + 3));
        Servo_SetAngle(&htim2, TIM_CHANNEL_1, ang);
        printf("Servo1 → %d°\r\n", ang);
    }
    else if (strncmp((char*)cmd, "S2:", 3) == 0) {
        uint16_t ang = atoi((char*)(cmd + 3));
        Servo_SetAngle(&htim2, TIM_CHANNEL_2, ang);
        printf("Servo2 → %d°\r\n", ang);
    }
    else if (strncmp((char*)cmd, "RUN", 3) == 0) {
        Servo_RunSequence();
    }
    else if (strncmp((char*)cmd, "HELP", 4) == 0) {
        Print_Help();
    }
    else {
        printf("Unknown command: %s\r\n", cmd);
    }
}

/* USER CODE END 0 */

int main(void)
{
    HAL_Init();
    SystemClock_Config();
    MX_GPIO_Init();
    MX_TIM2_Init();
    MX_USART1_UART_Init();

    /* USER CODE BEGIN 2 */
    // 启动两路PWM输出
    HAL_TIM_PWM_Start(&htim2, TIM_CHANNEL_1);
    HAL_TIM_PWM_Start(&htim2, TIM_CHANNEL_2);
    
    // 设置到中位 (90°)
    Servo_SetAngle(&htim2, TIM_CHANNEL_1, 90);
    Servo_SetAngle(&htim2, TIM_CHANNEL_2, 90);
    
    printf("Servo Control Ready!\r\n");
    printf("Initial: Both at 90°\r\n");
    Print_Help();
    
    // 启动串口接收
    HAL_UART_Receive_IT(&huart1, (uint8_t*)g_rx_buf, 32);
    /* USER CODE END 2 */

    /* USER CODE BEGIN WHILE */
    while (1)
    {
        // 处理串口命令（需要在USART中断或主循环中调用）
        // 这里简化处理，实际项目建议用中断
        
        // 测试：每3秒自动运行一次序列
        static uint32_t last_run = 0;
        if (HAL_GetTick() - last_run > 10000) {
            last_run = HAL_GetTick();
            Servo_RunSequence();
        }
        
        HAL_Delay(100);
    }
    /* USER CODE END WHILE */
}

// USART1 中断回调（需要添加此函数到stm32f1xx_it.c或main.c）
void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart)
{
    if (huart->Instance == USART1) {
        Parse_Command(g_rx_buf);
        memset(g_rx_buf, 0, sizeof(g_rx_buf));
        HAL_UART_Receive_IT(&huart1, (uint8_t*)g_rx_buf, 32);
    }
}

// CubeMX自动生成，但手动配置时参考：
void MX_TIM2_Init(void)
{
    TIM_ClockConfigTypeDef sClockSourceConfig = {0};
    TIM_MasterConfigTypeDef sMasterConfig = {0};
    TIM_OC_InitTypeDef sConfigOC = {0};

    htim2.Instance = TIM2;
    htim2.Init.Prescaler = 72 - 1;        // 72分频 → 1MHz
    htim2.Init.CounterMode = TIM_COUNTERMODE_UP;
    htim2.Init.Period = 20000 - 1;        // 20ms周期
    htim2.Init.ClockDivision = TIM_CLOCKDIVISION_DIV1;
    htim2.Init.AutoReloadPreload = TIM_AUTORELOAD_PRELOAD_ENABLE;
    
    // PWM模式配置
    sConfigOC.OCMode = TIM_OCMODE_PWM1;
    sConfigOC.Pulse = 1500;               // 90°位置
    sConfigOC.OCPolarity = TIM_OCPOLARITY_HIGH;
    sConfigOC.OCFastMode = TIM_OCFAST_DISABLE;
    
    // 配置通道1和2
    HAL_TIM_PWM_Init(&htim2);
    HAL_TIM_PWM_ConfigChannel(&htim2, &sConfigOC, TIM_CHANNEL_1);
    
    sConfigOC.Pulse = 1500;
    HAL_TIM_PWM_ConfigChannel(&htim2, &sConfigOC, TIM_CHANNEL_2);
    
    HAL_TIM_MspPostInit(&htim2);
}

❓ 常见问题排查

问题1：舵机持续抖动

最常见原因：
1. 供电不足 — STM32开发板的5V只能提供~100mA，舵机需要200mA+，必须用独立电源。
2. PWM频率不对 — 舵机需要50Hz（20ms周期），不是1000Hz！如果PSC/ARR设置成1kHz，舵机会疯狂抖动。
3. 脉冲宽度不在范围内 — 确保CCR值在500~2500之间，不要超过边界太多。
4. 舵机自身问题 — 测试多个舵机确认是硬件问题还是接线问题。

问题2：角度范围校准

不同舵机的实际角度范围可能略有差异。如果SG90达不到0°或180°：
// 修改CCR边界值适配你的舵机
#define SERVO_MIN   600   // 原来500，调大让0°更靠边
#define SERVO_MAX  2400   // 原来2500，调小让180°更靠边
#define SERVO_MID  1500   // 中间通常比较准
建议先打印实际角度，再逐步微调。

问题3：多舵机供电功率计算

功率预算：
SG90舵机电流：
  空闲时: ~50mA
  正常运转: ~100~200mA
  堵转(受阻): ~500~600mA (⚠ 不要让舵机长时间堵转!)

多舵机电源选择：
  1个舵机 → 5V 0.5A 足矣
  2个舵机 → 5V 1A 推荐
  4个舵机 → 5V 2A 必须
  
推荐方案：
  LM2596降压模块 (12V输入) → 5V 3A输出
  或 USB 5V 3A电源适配器
  或 手机充电宝 (5V 2.1A)

问题4：舵机反应迟钝/卡顿

1. 检查是否有其他中断干扰定时器PWM输出
2. 降低主循环中HAL_Delay的频率
3. 确保PWM引脚没有被复用为普通GPIO
4. SG90舵机本身响应速度约0.12s/60°，不是瞬时的，这是正常现象

问题5：想让舵机旋转360°怎么办？

SG90是180°角度舵机，无法360°旋转。
如果需要连续旋转的舵机（也叫"360°舵机"或"连续旋转舵机"），控制逻辑完全不同：
- 0.5ms脉冲 = 全速顺时针
- 1.5ms脉冲 = 停止
- 2.5ms脉冲 = 全速逆时针
连续旋转舵机没有位置反馈，无法知道当前角度！
