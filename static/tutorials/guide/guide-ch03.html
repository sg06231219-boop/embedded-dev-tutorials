# ====== STM32F103 Makefile ======
CC = arm-none-eabi-gcc
OBJCOPY = arm-none-eabi-objcopy
SIZE = arm-none-eabi-size
TARGET = blink
MCU = cortex-m3

C_SOURCES = $(wildcard src/*.c)
INCLUDES = -Iinc -Idrivers/CMSIS

CFLAGS  = -mcpu=$(MCU) -mthumb -Wall -O2 -g3
CFLAGS += -ffunction-sections -fdata-sections $(INCLUDES)
CFLAGS += -DSTM32F103C8T6

LDFLAGS  = -mcpu=$(MCU) -mthumb -Wl,--gc-sections
LDFLAGS += -TSTM32F103C8Tx.ld -Wl,-Map=$(TARGET).map
LDFLAGS += --specs=nano.specs --specs=nosys.specs

OBJS = $(C_SOURCES:.c=.o)

.PHONY: all clean flash

all: $(TARGET).bin $(TARGET).hex
	@echo "Build complete!"
	$(SIZE) $(TARGET).elf

$(TARGET).elf: $(OBJS)
	$(CC) $(LDFLAGS) $^ -o $@

$(TARGET).bin: $(TARGET).elf
	$(OBJCOPY) -O binary $< $@

$(TARGET).hex: $(TARGET).elf
	$(OBJCOPY) -O ihex $< $@

%.o: %.c
	$(CC) $(CFLAGS) -c $< -o $@

clean:
	powershell -Command "Remove-Item src/*.o $(TARGET).elf $(TARGET).bin $(TARGET).hex $(TARGET).map -ErrorAction SilentlyContinue"
# жіЁпјҡеөҢе…ҘејҸMakefileйҖҡеёёеңЁWindows PowerShellзҺҜеўғиҝҗиЎҢ
# еҰӮжһңдҪ еңЁLinux/WSLдёӢпјҢж”№з”Ёпјҡ
# 	rm -f $(OBJS) $(TARGET).elf $(TARGET).bin $(TARGET).hex $(TARGET).map

flash: $(TARGET).bin
	st-flash write $(TARGET).bin 0x08000000
