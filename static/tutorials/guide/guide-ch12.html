做一个完整可运行的项目：DHT11温湿度传感器 → STM32读取 → 串口输出 → OLED显示屏显示。

💡 类比：这个项目就像做一道完整的菜——前面的章节是学切菜、调味、火候，现在要把这些技能组合起来，做一道真正的菜上桌。

🛠️ 项目创建（二选一）

⚠️ 重要：下面的代码依赖STM32 HAL库。你需要先创建一个HAL库项目，以下是两种方式：
方式1：STM32CubeMX（推荐）自动生成
CubeMX自动生成初始化代码，你只需写业务逻辑。
步骤1：打开STM32CubeMX → New Project → 搜索 STM32F103C8 → 双击选择
步骤2：配置引脚：PA1=GPIO_Output（DHT11）、PB6=I2C1_SCL、PB7=I2C1_SDA、PA9=USART1_TX、PA10=USART1_RX
步骤3：左侧配置：Connectivity → USART1 → Mode: Asynchronous → Baud: 115200
步骤4：Connectivity → I2C1 → 勾选 Enabled
步骤5：Project Manager → Toolchain: CMake → Generate Code
步骤6：生成的工程中，在 /* USER CODE BEGIN */ 区域内写你的代码
方式2：EIDE/PlatformIO插件（免CubeMX）VS Code内
已安装的VS Code插件，可以在VS Code内直接创建STM32项目。
// EIDE方式（推荐）
// 1. VS Code → Ctrl+Shift+P → "EIDE: New Project"
// 2. 选择 "Cortex-M Project" → STM32F103C8
// 3. 项目自动包含HAL库和启动文件

// PlatformIO方式
// 1. VS Code → PlatformIO → New Project
// 2. Board: BluePill F103C8 → Framework: STM32Cube
// 3. 自动下载HAL库和编译工具链
💡 还没装CubeMX？项目实战可以先跳过，把前面章节的纯寄存器代码练熟。CubeMX是效率工具，不是必需品。

DHT11驱动代码

// dht11.h
#ifndef DHT11_H
#define DHT11_H
#include "stm32f1xx_hal.h"

typedef struct {
    uint8_t humidity_int;    // 湿度整数部分
    uint8_t humidity_dec;    // 湿度小数部分
    uint8_t temp_int;        // 温度整数部分
    uint8_t temp_dec;        // 温度小数部分
} DHT11_Data_t;

uint8_t DHT11_Read(DHT11_Data_t *data, GPIO_TypeDef *port, uint16_t pin);

#endif
// dht11.c
#include "dht11.h"

static void delay_us(uint32_t us) {
    uint32_t ticks = us * (SystemCoreClock / 1000000) / 5;
    while (ticks--);
}

uint8_t DHT11_Read(DHT11_Data_t *data, GPIO_TypeDef *port, uint16_t pin) {
    uint8_t buf[5] = {0};
    GPIO_InitTypeDef GPIO_InitStruct = {0};

    // 主机发送起始信号（拉低18ms）
    GPIO_InitStruct.Pin = pin;
    GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
    GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_HIGH;
    HAL_GPIO_Init(port, &GPIO_InitStruct);
    HAL_GPIO_WritePin(port, pin, GPIO_PIN_RESET);
    HAL_Delay(18);
    HAL_GPIO_WritePin(port, pin, GPIO_PIN_SET);
    delay_us(30);

    // 切换为输入
    GPIO_InitStruct.Mode = GPIO_MODE_INPUT;
    GPIO_InitStruct.Pull = GPIO_PULLUP;
    HAL_GPIO_Init(port, &GPIO_InitStruct);

    // 等待DHT11响应
    uint32_t timeout = 100;
    while (HAL_GPIO_ReadPin(port, pin) && timeout--) delay_us(1);
    if (timeout == 0) return 1;  // 无响应

    // 接收40位数据
    for (int i = 0; i < 5; i++) {
        for (int j = 0; j < 8; j++) {
            timeout = 100;
            while (!HAL_GPIO_ReadPin(port, pin) && timeout--) delay_us(1);
            delay_us(40);
            if (HAL_GPIO_ReadPin(port, pin)) {
                buf[i] |= (1 << (7 - j));
                timeout = 100;
                while (HAL_GPIO_ReadPin(port, pin) && timeout--) delay_us(1);
            }
        }
    }

    // 校验
    if (buf[4] != (uint8_t)(buf[0] + buf[1] + buf[2] + buf[3]))
        return 2;  // 校验失败

    data->humidity_int = buf[0];
    data->humidity_dec = buf[1];
    data->temp_int     = buf[2];
    data->temp_dec     = buf[3];
    return 0;  // 成功
}

主程序

// main.c
#include "stm32f1xx_hal.h"
#include "dht11.h"
#include <stdio.h>

DHT11_Data_t dht;
extern UART_HandleTypeDef huart1;
extern I2C_HandleTypeDef hi2c1;

int fputc(int ch, FILE *f) {
    HAL_UART_Transmit(&huart1, (uint8_t *)&ch, 1, HAL_MAX_DELAY);
    return ch;
}

int main(void) {
    HAL_Init();
    SystemClock_Config();
    MX_GPIO_Init();
    MX_USART1_UART_Init();
    MX_I2C1_Init();

    printf("
=== Environment Monitor ===
");

    while (1) {
        uint8_t ret = DHT11_Read(&dht, GPIOA, GPIO_PIN_1);
        if (ret == 0) {
            printf("Temp: %d.%d C, Humi: %d.%d %%\n",
                dht.temp_int, dht.temp_dec,
                dht.humidity_int, dht.humidity_dec);
        } else {
            printf("DHT11 error: %d\n", ret);
        }
        HAL_Delay(2000);  // DHT11最小间隔2秒
    }
}
OLED显示屏扩展SSD1306 I2C可选
想让温湿度数据显示在OLED屏上？按以下步骤操作：
步骤1：下载ssd1306驱动：github.com/afiskon/stm32-ssd1306（GitHub可能被墙，替代：淘宝搜"SSD1306 STM32驱动代码"）
步骤2：将 ssd1306.c、ssd1306.h、ssd1306_fonts.c 复制到你的 src/ 和 inc/ 目录
步骤3：在main.c顶部添加：#include "ssd1306.h"
步骤4：在while(1)的DHT11读取成功分支里加入OLED显示代码：
// OLED显示代码（在DHT11读取成功后添加）
ssd1306_Init();  // 只需调用一次，建议放在main()初始化区
char buf[32];
snprintf(buf, sizeof(buf), "Temp:%d.%dC", dht.temp_int, dht.temp_dec);
ssd1306_SetCursor(0, 0);
ssd1306_WriteString(buf, Font_7x10, White);
snprintf(buf, sizeof(buf), "Humi:%d.%d%%", dht.humidity_int, dht.humidity_dec);
ssd1306_SetCursor(0, 16);
ssd1306_WriteString(buf, Font_7x10, White);
ssd1306_UpdateScreen(&hi2c1);
步骤5：CMakeLists.txt的SOURCES中添加：src/ssd1306.c src/ssd1306_fonts.c
💡 暂时没有OLED？没关系！只用串口输出也完全OK，先跑通DHT11传感器再说。OLED是锦上添花。

CMakeLists.txt 完整版

cmake_minimum_required(VERSION 3.20)
project(env_monitor C ASM)

set(SOURCES
    src/main.c src/dht11.c src/stm32f1xx_it.c
    src/system_stm32f1xx.c
    # 如果要启用OLED显示，取消下行注释：
    # src/ssd1306.c
)

include_directories(inc CMSIS)

add_compile_options(
    -mcpu=cortex-m3 -mthumb -Wall -O2 -g3
    -ffunction-sections -fdata-sections -DSTM32F103C8T6
)

add_link_options(
    -mcpu=cortex-m3 -mthumb -Wl,--gc-sections
    -T${CMAKE_SOURCE_DIR}/STM32F103C8Tx.ld
    -Wl,-Map=${PROJECT_NAME}.map
    --specs=nano.specs --specs=nosys.specs
)

add_executable(${PROJECT_NAME}.elf ${SOURCES})

add_custom_command(TARGET ${PROJECT_NAME}.elf POST_BUILD
    COMMAND ${CMAKE_OBJCOPY} -O binary ${PROJECT_NAME}.elf ${PROJECT_NAME}.bin
    COMMAND ${CMAKE_OBJCOPY} -O ihex ${PROJECT_NAME}.elf ${PROJECT_NAME}.hex
    COMMAND ${CMAKE_SIZE} ${PROJECT_NAME}.elf
    COMMAND echo "Flash: st-flash write ${PROJECT_NAME}.bin 0x08000000"
)
