// dht11.h —— 简洁接口
#ifndef DHT11_H
#define DHT11_H
#include "stm32f1xx_hal.h"

typedef struct {
    uint8_t humidity_int;    // 湿度整数部分 (%)
    uint8_t humidity_dec;    // 湿度小数部分
    uint8_t temp_int;        // 温度整数部分 (°C)
    uint8_t temp_dec;        // 温度小数部分
    uint8_t checksum;        // 校验和
} DHT11_Data_t;

typedef enum {
    DHT11_OK = 0,
    DHT11_ERR_TIMEOUT,       // 传感器无响应
    DHT11_ERR_CHECKSUM,      // 校验失败
    DHT11_ERR_BUSY           // 距上次读取不足2秒
} DHT11_Status_t;

DHT11_Status_t DHT11_Read(DHT11_Data_t *data, GPIO_TypeDef *port, uint16_t pin);

#endif
// dht11.c —— 完整实现
#include "dht11.h"

static uint32_t last_read_tick = 0;

// 微秒级延时（基于系统时钟72MHz估算）
static void delay_us(uint32_t us) {
    uint32_t ticks = us * (SystemCoreClock / 1000000UL) / 5;
    while (ticks--) __NOP();
}

DHT11_Status_t DHT11_Read(DHT11_Data_t *data, GPIO_TypeDef *port, uint16_t pin) {
    uint8_t buf[5] = {0};
    uint32_t timeout;
    GPIO_InitTypeDef GPIO_InitStruct = {0};
    
    // 防止读取间隔过短（DHT11要求>1秒，保守取2秒）
    if (HAL_GetTick() - last_read_tick < 2000) {
        return DHT11_ERR_BUSY;
    }
    last_read_tick = HAL_GetTick();
    
    // === 步骤1：主机发送起始信号（拉低18ms） ===
    GPIO_InitStruct.Pin = pin;
    GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_OD;  // 开漏输出（兼容5V）
    GPIO_InitStruct.Pull = GPIO_NOPULL;
    GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_HIGH;
    HAL_GPIO_Init(port, &GPIO_InitStruct);
    
    // 拉低至少18ms
    HAL_GPIO_WritePin(port, pin, GPIO_PIN_RESET);
    HAL_Delay(18);
    
    // 拉高20-40μs，通知DHT11"我准备好了"
    HAL_GPIO_WritePin(port, pin, GPIO_PIN_SET);
    delay_us(30);
    
    // === 步骤2：切换为输入，等待DHT11响应 ===
    GPIO_InitStruct.Mode = GPIO_MODE_INPUT;
    GPIO_InitStruct.Pull = GPIO_PULLUP;
    HAL_GPIO_Init(port, &GPIO_InitStruct);
    delay_us(5);  // 让IO稳定
    
    // DHT11应该拉低80μs再拉高80μs作为响应
    // 等待DHT11拉低
    timeout = 100;
    while (HAL_GPIO_ReadPin(port, pin) == GPIO_PIN_SET) {
        if (--timeout == 0) return DHT11_ERR_TIMEOUT;
        delay_us(1);
    }
    
    // 等待DHT11拉高
    timeout = 100;
    while (HAL_GPIO_ReadPin(port, pin) == GPIO_PIN_RESET) {
        if (--timeout == 0) return DHT11_ERR_TIMEOUT;
        delay_us(1);
    }
    // 等待DHT11再次拉低（准备发送数据）
    timeout = 100;
    while (HAL_GPIO_ReadPin(port, pin) == GPIO_PIN_SET) {
        if (--timeout == 0) return DHT11_ERR_TIMEOUT;
        delay_us(1);
    }
    
    // === 步骤3：读取40位数据 ===
    for (int i = 0; i < 5; i++) {
        for (int j = 0; j < 8; j++) {
            // 等待低电平结束
            timeout = 100;
            while (HAL_GPIO_ReadPin(port, pin) == GPIO_PIN_RESET) {
                if (--timeout == 0) return DHT11_ERR_TIMEOUT;
                delay_us(1);
            }
            
            // 测量高电平持续时间：26-28μs=0, 70μs=1
            delay_us(35);  // 延时到关键判断点
            if (HAL_GPIO_ReadPin(port, pin) == GPIO_PIN_SET) {
                buf[i] |= (1 << (7 - j));  // MSB先发
            }
            
            // 等待高电平结束
            timeout = 100;
            while (HAL_GPIO_ReadPin(port, pin) == GPIO_PIN_SET) {
                if (--timeout == 0) return DHT11_ERR_TIMEOUT;
                delay_us(1);
            }
        }
    }
    
    // === 步骤4：校验 ===
    if (buf[4] != (uint8_t)(buf[0] + buf[1] + buf[2] + buf[3])) {
        return DHT11_ERR_CHECKSUM;
    }
    
    // 填充数据
    data->humidity_int = buf[0];
    data->humidity_dec = buf[1];
    data->temp_int     = buf[2];
    data->temp_dec     = buf[3];
    data->checksum     = buf[4];
    
    return DHT11_OK;
}

            ssd1306_WriteString(buf, Font_7x10, White);

            ssd1306_SetCursor(0, 36);
            ssd1306_WriteString(buf, Font_7x10, White);

    }

📡 项目二：HC-SR04超声波测距

💡 原理：超声波像蝙蝠回声定位——发射40KHz超声波脉冲，遇到障碍物反射回来，测量往返时间，乘以声速÷2就得到距离。
// 超声波测距 - 基于定时器输入捕获
// HC-SR04引脚：VCC(5V)、GND、TRIG(触发)、ECHO(回响)

#define TRIG_PORT GPIOB
#define TRIG_PIN  GPIO_PIN_0
#define ECHO_PORT GPIOB
#define ECHO_PIN  GPIO_PIN_1
#define SOUND_SPEED_CM_US 0.0343f  // 声速 = 343m/s = 0.0343cm/μs

// 发送10μs触发脉冲
void hcsr04_trigger(void) {
    HAL_GPIO_WritePin(TRIG_PORT, TRIG_PIN, GPIO_PIN_RESET);
    delay_us(2);
    HAL_GPIO_WritePin(TRIG_PORT, TRIG_PIN, GPIO_PIN_SET);
    delay_us(10);  // 至少10μs
    HAL_GPIO_WritePin(TRIG_PORT, TRIG_PIN, GPIO_PIN_RESET);
}

// 使用输入捕获测量ECHO高电平时间
float hcsr04_get_distance_cm(void) {
    uint32_t pulse_width_us = 0;
    uint32_t timeout;
    
    // 发送触发脉冲
    hcsr04_trigger();
    
    // 等待ECHO变高（超时=无回波）
    timeout = 100000;  // ~100ms超时
    while (HAL_GPIO_ReadPin(ECHO_PORT, ECHO_PIN) == GPIO_PIN_RESET) {
        if (--timeout == 0) return -1.0f;  // 超时=超出范围
    }
    
    // 测量高电平宽度（μs级精度）
    uint32_t start = DWT->CYCCNT;  // DWT计数器（需先使能）
    timeout = 100000;
    while (HAL_GPIO_ReadPin(ECHO_PORT, ECHO_PIN) == GPIO_PIN_SET) {
        if (--timeout == 0) return -1.0f;
    }
    uint32_t end = DWT->CYCCNT;
    
    // 计算时间（假设72MHz）
    pulse_width_us = (end - start) / (SystemCoreClock / 1000000);
    
    // 距离 = 声速 × 时间 / 2（往返）
    return pulse_width_us * SOUND_SPEED_CM_US / 2.0f;
}

// 使能DWT周期计数器（用于精确计时）
void dwt_enable(void) {
    CoreDebug->DEMCR |= CoreDebug_DEMCR_TRCENA_Msk;
    DWT->CYCCNT = 0;
    DWT->CTRL |= DWT_CTRL_CYCCNTENA_Msk;
}

// 更稳定的版本：TIM输入捕获+中断（推荐）
// 配置TIM2_CH1为输入捕获，检测ECHO引脚
volatile uint32_t ic_val1 = 0, ic_val2 = 0;
volatile uint8_t ic_captured = 0;

void HAL_TIM_IC_CaptureCallback(TIM_HandleTypeDef *htim) {
    if (htim->Instance == TIM2 && htim->Channel == HAL_TIM_ACTIVE_CHANNEL_1) {
        if (ic_captured == 0) {
            ic_val1 = HAL_TIM_ReadCapturedValue(htim, TIM_CHANNEL_1);
            __HAL_TIM_SET_CAPTUREPOLARITY(htim, TIM_CHANNEL_1, TIM_INPUTCHANNELPOLARITY_FALLING);
            ic_captured = 1;
        } else {
            ic_val2 = HAL_TIM_ReadCapturedValue(htim, TIM_CHANNEL_1);
            __HAL_TIM_SET_CAPTUREPOLARITY(htim, TIM_CHANNEL_1, TIM_INPUTCHANNELPOLARITY_RISING);
            ic_captured = 2;
            __HAL_TIM_DISABLE_IT(htim, TIM_IT_CC1);  // 停止捕获
        }
    }
}

float hcsr04_get_distance_tim(void) {
    // 1. 发送触发脉冲
    hcsr04_trigger();
    
    // 2. 启动输入捕获，等待中断处理
    ic_captured = 0;
    __HAL_TIM_SET_COUNTER(&htim2, 0);
    HAL_TIM_IC_Start_IT(&htim2, TIM_CHANNEL_1);
    
    // 3. 等待捕获完成（最多100ms）
    uint32_t timeout = 100000;
    while (ic_captured < 2 && timeout--) {
        delay_us(1);
    }
    HAL_TIM_IC_Stop_IT(&htim2, TIM_CHANNEL_1);
    
    if (timeout == 0) return -1.0f;
    
    // 4. 计算距离
    uint32_t pulse = (ic_val2 > ic_val1) ? (ic_val2 - ic_val1) 
        : (0xFFFF - ic_val1 + ic_val2);
    float time_us = pulse * (1000000.0f / (float)(HAL_RCC_GetPCLK1Freq() * 2 / (htim2.Init.Prescaler + 1)));
    
    return time_us * SOUND_SPEED_CM_US / 2.0f;
}

🔬 项目三：MPU6050姿态传感器

MPU6050集成了3轴加速度计+3轴陀螺仪，是姿态检测的首选传感器。

// MPU6050 6轴传感器 I2C驱动
#define MPU6050_ADDR    0x68  // AD0接地=0x68，接VCC=0x69
#define MPU6050_WHO_AM_I 0x75
#define MPU6050_PWR_MGMT1 0x6B
#define MPU6050_ACCEL_XOUT_H 0x3B

typedef struct {
    int16_t ax, ay, az;   // 加速度原始值
    int16_t gx, gy, gz;   // 陀螺仪原始值
    int16_t temp;         // 温度原始值
} MPU6050_Raw_t;

typedef struct {
    float ax, ay, az;     // 加速度(g)
    float gx, gy, gz;     // 角速度(°/s)
    float pitch, roll;    // 姿态角(°)
    float temp;           // 温度(°C)
} MPU6050_Data_t;

// 初始化
uint8_t mpu6050_init(void) {
    uint8_t whoami;
    HAL_I2C_Mem_Read(&hi2c1, MPU6050_ADDR<<1, MPU6050_WHO_AM_I, 1, &whoami, 1, 100);
    if (whoami != 0x68) return 0;  // MPU6050 ID恒为0x68
    
    // 唤醒MPU6050（清除睡眠位）
    uint8_t wake_cmd = 0x00;
    HAL_I2C_Mem_Write(&hi2c1, MPU6050_ADDR<<1, MPU6050_PWR_MGMT1, 1, &wake_cmd, 1, 100);
    
    // 配置量程：加速度±2g, 陀螺仪±250°/s
    uint8_t accel_cfg = 0x00;  // ±2g
    uint8_t gyro_cfg  = 0x00;  // ±250°/s
    HAL_I2C_Mem_Write(&hi2c1, MPU6050_ADDR<<1, 0x1C, 1, &accel_cfg, 1, 100);
    HAL_I2C_Mem_Write(&hi2c1, MPU6050_ADDR<<1, 0x1B, 1, &gyro_cfg, 1, 100);
    
    return 1;
}

// 读取原始数据
void mpu6050_read_raw(MPU6050_Raw_t *raw) {
    uint8_t buf[14];
    HAL_I2C_Mem_Read(&hi2c1, MPU6050_ADDR<<1, MPU6050_ACCEL_XOUT_H, 1, buf, 14, 100);
    
    raw->ax = (int16_t)((buf[0] << 8) | buf[1]);
    raw->ay = (int16_t)((buf[2] << 8) | buf[3]);
    raw->az = (int16_t)((buf[4] << 8) | buf[5]);
    raw->temp = (int16_t)((buf[6] << 8) | buf[7]);
    raw->gx = (int16_t)((buf[8] << 8) | buf[9]);
    raw->gy = (int16_t)((buf[10] << 8) | buf[11]);
    raw->gz = (int16_t)((buf[12] << 8) | buf[13]);
}

// 转换为工程单位
void mpu6050_convert(MPU6050_Raw_t *raw, MPU6050_Data_t *data) {
    // ±2g量程 → 16384 LSB/g
    data->ax = raw->ax / 16384.0f;
    data->ay = raw->ay / 16384.0f;
    data->az = raw->az / 16384.0f;
    
    // ±250°/s量程 → 131 LSB/(°/s)
    data->gx = raw->gx / 131.0f;
    data->gy = raw->gy / 131.0f;
    data->gz = raw->gz / 131.0f;
    
    // 温度
    data->temp = raw->temp / 340.0f + 36.53f;
    
    // 加速度计计算初始姿态角
    data->pitch = atan2(data->ay, sqrt(data->ax*data->ax + data->az*data->az)) * 180/M_PI;
    data->roll  = atan2(-data->ax, data->az) * 180/M_PI;
}

🎯 传感器融合：互补滤波器

单个传感器有局限——加速度计有重力干扰和高频振动噪声，陀螺仪有零漂。融合两者才能得到稳定姿态。

// 互补滤波：陀螺仪积分(短期准) + 加速度计(长期准)
// angle = α × (angle + gyro×dt) + (1-α) × accel_angle
// α典型值 0.96~0.98

#define ALPHA 0.98f
float pitch = 0.0f, roll = 0.0f;
float last_time = 0.0f;

void complementary_filter(MPU6050_Data_t *data, float dt) {
    // 加速度计计算角度（低频准，高频不准）
    float acc_pitch = atan2(data->ay, sqrt(data->ax*data->ax + data->az*data->az)) * 180/M_PI;
    float acc_roll  = atan2(-data->ax, data->az) * 180/M_PI;
    
    // 首次初始化
    if (pitch == 0.0f && roll == 0.0f) {
        pitch = acc_pitch;
        roll  = acc_roll;
        return;
    }
    
    // 互补滤波融合
    pitch = ALPHA * (pitch + data->gx * dt) + (1.0f - ALPHA) * acc_pitch;
    roll  = ALPHA * (roll  + data->gy * dt) + (1.0f - ALPHA) * acc_roll;
    
    data->pitch = pitch;
    data->roll  = roll;
}

📐 I2C总线扫描器

不确定I2C设备地址？先用扫描器排查。这是嵌入式开发最常用的调试工具之一。

// I2C总线扫描器 —— 必备调试工具
void i2c_scan(I2C_HandleTypeDef *hi2c) {
    printf("\r\n=== I2C Bus Scan ===\r\n");
    printf("    ");
    for (int col = 0; col < 16; col++) {
        printf(" %01X ", col);
    }
    printf("\r\n");
    
    uint8_t found = 0;
    for (uint8_t addr = 1; addr < 128; addr++) {
        if (addr % 16 == 0) printf("%02X: ", addr);
        
        if (HAL_I2C_IsDeviceReady(hi2c, addr << 1, 2, 5) == HAL_OK) {
            uint8_t reg0;
            // 尝试读寄存器0确认设备确实存在
            if (HAL_I2C_Mem_Read(hi2c, addr << 1, 0x00, 1, ®0, 1, 10) == HAL_OK ||
                HAL_I2C_Master_Receive(hi2c, addr << 1, ®0, 1, 10) == HAL_OK) {
                printf("%02X ", addr);
                found++;
            } else {
                printf("?? ");
            }
        } else {
            printf("-- ");
        }
        
        if ((addr + 1) % 16 == 0) printf("\r\n");
    }
    
    printf("Found %d device(s) on I2C bus\r\n\r\n", found);
}

🎛 卡尔曼滤波器实战

// 一维卡尔曼滤波：传感器数据平滑
// 适用于：温度、气压、距离等单一测量值

typedef struct {
    float x;  // 状态估计值
    float P;  // 估计误差协方差
    float Q;  // 过程噪声协方差（调大→更信任预测/平滑）
    float R;  // 测量噪声协方差（调大→更不信任传感器）
    float K;  // 卡尔曼增益（自动计算）
} Kalman1D_t;

// 初始化
void kalman_init(Kalman1D_t *kf, float initial_value, float Q, float R) {
    kf->x = initial_value;
    kf->P = 1.0f;
    kf->Q = Q;
    kf->R = R;
    kf->K = 0.0f;
}

// 每次新测量值到来时调用
float kalman_update(Kalman1D_t *kf, float measurement) {
    // 预测步骤
    kf->P = kf->P + kf->Q;
    
    // 更新步骤
    kf->K = kf->P / (kf->P + kf->R);
    kf->x = kf->x + kf->K * (measurement - kf->x);
    kf->P = (1.0f - kf->K) * kf->P;
    
    return kf->x;
}

// 使用示例
Kalman1D_t temp_kf;
kalman_init(&temp_kf, 25.0f, 0.01f, 1.0f);

// 在循环中使用
for (;;) {
    float raw_temp = read_temperature();  // 带噪声的原始值
    float smooth_temp = kalman_update(&temp_kf, raw_temp);
    printf("Raw: %.2f → Smooth: %.2f\r\n", raw_temp, smooth_temp);
    HAL_Delay(100);
}

// 参数调优指南：
// Q小(0.001) + R大(10) → 超平滑但响应慢（适合温度气压）
// Q大(0.1) + R小(0.1) → 响应快但噪声多（适合快速变化量）
// 典型起点：Q=0.01, R=1.0，然后根据效果微调

🪡 滑动平均滤波器

比卡尔曼更简单直观，适合资源受限的场景。

// 滑动平均滤波器（Moving Average）
// 适合：ADC噪声平滑、传感器稳定读数

#define MA_WINDOW 10

typedef struct {
    float buffer[MA_WINDOW];
    uint8_t index;
    uint8_t count;
    float sum;
} MovingAvg_t;

void ma_init(MovingAvg_t *ma) {
    memset(ma->buffer, 0, sizeof(ma->buffer));
    ma->index = 0;
    ma->count = 0;
    ma->sum   = 0;
}

float ma_filter(MovingAvg_t *ma, float new_val) {
    // 减去最旧值
    ma->sum -= ma->buffer[ma->index];
    // 加入新值
    ma->buffer[ma->index] = new_val;
    ma->sum += new_val;
    // 移动指针
    ma->index = (ma->index + 1) % MA_WINDOW;
    if (ma->count < MA_WINDOW) ma->count++;
    // 返回平均值
    return ma->sum / (float)ma->count;
}

// 使用
MovingAvg_t adc_filter;
ma_init(&adc_filter);
float stable_adc = ma_filter(&adc_filter, raw_adc_value);

🛠 常见问题排查

🔍 问题1：DHT11一直读出错/返回TIMEOUT
❌ 症状：串口一直输出"DHT11 Error: No response!"
✅ 排查：
1. 接线不对：VCC/GND/DATA三线缺一不可
2. DATA线没接上拉电阻：DHT11 DATA需要4.7KΩ上拉到VCC
3. GPIO配置错误：初始化为输出推挽(OUTPUT_PP)发送信号，再切为输入(INPUT)接收
4. 延时时间不准：delay_us基于的SystemCoreClock不对
✅ 修复：
1. 用万用表测VCC-GND电压是否为3.3V
2. 加4.7KΩ上拉电阻
3. 确保时序正确：拉低18ms→拉高30μs→等响应
4. 用逻辑分析仪看DATA线波形最直观
🔍 问题2：DHT11读数全是0或255
❌ 症状：读取成功(返回OK)，但温度和湿度全是0或255。
✅ 排查：
1. 读取间隔太短：DHT11要求≥1秒（安全取2秒）
2. 位读取逻辑有误：DHT11的"0"是高电平26-28μs，"1"是70μs
3. 校验和计算有误：checksum = hum_int + hum_dec + temp_int + temp_dec
4. 干扰导致数据跳变
✅ 修复：
1. 加读取间隔检查
2. 在拉高后延时30-40μs再判断MCU引脚电平
3. 连续读3次取中位数
🔍 问题3：I2C通信失败(HAL_TIMEOUT)
❌ 症状：HAL_I2C_Master_Transmit返回HAL_TIMEOUT。
✅ 排查：
1. 用i2c_scan()确认设备地址是否正确（7位地址 << 1）
2. SCL/SDA没有上拉电阻：I2C是开漏输出，必须外接上拉（一般4.7KΩ）
3. 设备没上电：检查VCC和GND
4. 总线冲突：多个设备地址相同？
✅ 修复：
1. 扫描总线确认地址
2. 加4.7KΩ上拉电阻到3.3V
3. 降低I2C频率：从400K降到100K
4. 用逻辑分析仪看SCL/SDA波形
🔍 问题4：HC-SR04测距不准/跳动大
❌ 症状：距离读数忽大忽小，或固定值不变。
✅ 排查：
1. 供电不足：HC-SR04需要5V，3.3V供电会不稳定
2. 测量表面：软质/倾斜表面回波弱
3. 最小盲区：HC-SR04有2cm盲区，<2cm的测量不准
4. Echo引脚电平：5V输出直接接3.3V容忍的MCU引脚（STM32大部分5V容忍）
✅ 修复：
1. 用5V供电（注意MCU的Echo引脚是否5V容忍）
2. 多次测量取中位数（如测5次排序取中间值）
3. 加滑动平均滤波
4. 测量间隔>60ms（超声波脉冲衰减时间）
🔍 问题5：MPU6050静止时角度漂移
❌ 症状：传感器不动但角度缓慢漂移(几秒就偏了10°)。
✅ 排查：
1. 只用陀螺仪积分？陀螺仪有零偏(Zero-G offset)，随时间积分会漂
2. 互补滤波系数不对：ALPHA太小(太信加速度计)或有震动干扰
3. 采样率太低
✅ 修复：
1. 用互补滤波或Mahony/Madgwick融合算法
2. 上电后先静止0.5s测陀螺仪零偏，后续减去零偏
3. 提高采样率到100Hz以上
4. Yaw角(航向)必须加磁力计才能长期稳定

🔧 MPU6050 姿态解算深度实战

MPU6050是最常见的6轴IMU（3轴加速度+3轴陀螺仪）。但原始数据不能直接用——加速度计噪声大、陀螺仪有漂移。姿态融合算法是核心。

// ★ MPU6050 初始化（避开最常见配置陷阱）
void mpu6050_init_correct(void) {
    // ① 唤醒MPU6050（默认Sleep模式！新人第一坑）
    i2c_write_reg(MPU6050_ADDR, 0x6B, 0x00);  // 清除SLEEP位
    HAL_Delay(100);  // ★ 必须等100ms！时钟稳定需要时间
    
    // ② 设置陀螺仪量程（±250/500/1000/2000 °/s）
    // 250°/s: 精度0.0076°/LSB → 适合姿态检测
    // 2000°/s: 精度0.061°/LSB → 适合无人机快速翻转
    i2c_write_reg(MPU6050_ADDR, 0x1B, 0x00);  // ±250°/s
    
    // ③ 设置加速度计量程（±2/4/8/16g）
    i2c_write_reg(MPU6050_ADDR, 0x1C, 0x00);  // ±2g
    
    // ④ 设置数字低通滤波器（DLPF）
    // DLPF_CFG=0: 260Hz(加速度)/256Hz(陀螺仪) → 延迟0.98ms
    // DLPF_CFG=3: 44Hz → 延迟4.9ms → 更平滑但延迟大
    // DLPF_CFG=6: 5Hz → 延迟19ms → 仅适合静态
    // 推荐：DLPF_CFG=2(94Hz) 用于姿态控制
    i2c_write_reg(MPU6050_ADDR, 0x1A, 0x02);  // DLPF=94Hz
    
    // ⑤ 设置采样率分频 → 采样率 = 8kHz / (1 + SMPLRT_DIV)
    // SMPLRT_DIV=7 → 1kHz采样率（陀螺仪输出频率）
    i2c_write_reg(MPU6050_ADDR, 0x19, 0x07);  // 1KHz
}

// ★ 读取原始数据并转换（注意：先读加速度高位！）
void mpu6050_read_all(float *accel, float *gyro) {
    uint8_t raw[14];
    // 从0x3B开始连续读14字节：
    // ACCEL_XOUT_H/L, ACCEL_YOUT_H/L, ACCEL_ZOUT_H/L,
    // TEMP_OUT_H/L,
    // GYRO_XOUT_H/L, GYRO_YOUT_H/L, GYRO_ZOUT_H/L
    i2c_read_regs(MPU6050_ADDR, 0x3B, raw, 14);
    
    // 加速度转换（±2g量程: 16384 LSB/g）
    accel[0] = (int16_t)(raw[0]<<8 | raw[1]) / 16384.0f;  // X
    accel[1] = (int16_t)(raw[2]<<8 | raw[3]) / 16384.0f;  // Y
    accel[2] = (int16_t)(raw[4]<<8 | raw[5]) / 16384.0f;  // Z
    
    // 陀螺仪转换（±250°/s量程: 131 LSB/°/s）
    gyro[0] = (int16_t)(raw[8]<<8  | raw[9])  / 131.0f;   // X
    gyro[1] = (int16_t)(raw[10]<<8 | raw[11]) / 131.0f;   // Y
    gyro[2] = (int16_t)(raw[12]<<8 | raw[13]) / 131.0f;   // Z
    // ★ 常见错误：raw[8]和raw[9]顺序反了！高位在前
}
🔍 MPU6050踩坑TOP3：
1. 忘记唤醒：默认SLEEP=1，直接读数据全是0。必须先写0x6B寄存器为0x00，等100ms再读。
2. I2C地址混淆：AD0引脚接GND→地址0x68，接VDD→地址0x69。HAL库地址要左移1位→0xD0/0xD2。很多人右移了。
3. 数据跳变剧烈：DLPF没设对。默认DLPF=0(260Hz)，加速度含大量振动噪声。运动平稳应用设DLPF=3~4(44~21Hz)。

📡 传感器总线拓扑实战

实际项目中很少只有一个传感器。多个传感器共享I2C总线时，电源解耦、上拉电阻、地址冲突都是实际问题。

// ★ I2C多传感器共享总线的电源设计要点
// 每个传感器VCC-GND之间必须有去耦电容：
// ① 100nF陶瓷电容（贴片0603）——滤高频噪声
// ② 10μF钽电容（或大容量MLCC）——滤低频纹波
// ③ 两个电容紧贴传感器VCC引脚（距离<5mm！）
//
// ★ 上拉电阻计算：
// 标准I2C: Rp = (VDD - VOL) / IOL
// 3.3V系统: Rp ≈ (3.3 - 0.4) / 0.003 = 967Ω → 选1KΩ~4.7KΩ
// 高速(400KHz): 用1KΩ（边沿更陡）
// 省电: 用10KΩ（电流小）
// 多个设备共享I2C时，上拉电阻只需一组（不要每个设备都加！）

// ★ 磁力计/气压计/IMU共同使用时I2C总线布局：
// MCU ── SDA ──┬── BMP280 ──┬── MPU6050 ──┬── OLED
//      ── SCL ──┘            │             │
//      ── 3.3V ─┴── 各芯片独立去耦 ──────┘
// 上拉电阻（4.7KΩ×2）：放在MCU侧第一个节点处

🌐 传感器数据格式标准化

// 标准化数据输出格式（方便上位机/ChatGPT解析）
// 格式：$TYPE,val1,val2,val3*CK\r\n

void send_nmea_style_data(float temp, float hum, float pressure) {
    uint8_t checksum = 0;
    char buf[100];
    
    // 构造消息
    snprintf(buf, sizeof(buf), "$ENV,%.1f,%.1f,%.1f", temp, hum, pressure);
    
    // 计算校验和（$和*之间所有字节的异或）
    for (char *p = buf + 1; *p != '\0'; p++) {
        checksum ^= (uint8_t)*p;
    }
    
    // 发送（注意\r\n不是\n！）
    printf("%s*%02X\r\n", buf, checksum);
}
// 输出示例：$ENV,25.3,68.5,1013.2*5A

🔬 传感器校准深入

传感器出厂时都有误差，不校准的数据基本等于"仅供参考"。校准是做产品的分水岭——Demo读到值就完事了，产品必须保证精度。

校准三步法必学
所有传感器通用校准流程。以BMP280气压计为例，输出原始值→补偿计算→得到真实值。

第一步：读取出厂校准参数

// BMP280出厂校准参数（每颗芯片都不一样的）
// 这些值在芯片出厂时写入NVM，必须读出来参与计算！
typedef struct {
    uint16_t dig_T1;  // 温度补偿参数1
    int16_t  dig_T2;  // 温度补偿参数2
    int16_t  dig_T3;  // 温度补偿参数3
    uint16_t dig_P1;  // 气压补偿参数1
    int16_t  dig_P2;  // 气压补偿参数2
    int16_t  dig_P3;  // 气压补偿参数3
    int16_t  dig_P4;  // 气压补偿参数4
    int16_t  dig_P5;  // 气压补偿参数5
    int16_t  dig_P6;  // 气压补偿参数6
    int16_t  dig_P7;  // 气压补偿参数7
    int16_t  dig_P8;  // 气压补偿参数8
    int16_t  dig_P9;  // 气压补偿参数9
} bmp280_calib_t;

void bmp280_read_calib(bmp280_calib_t *cal) {
    uint8_t buf[24];
    // 校准参数存储在0x88~0xA1（共24字节）
    HAL_I2C_Mem_Read(&hi2c1, BMP280_ADDR<<1, 0x88, 1, buf, 24, 100);
    // 解析（注意LSB在前）
    cal->dig_T1 = buf[0]  | (buf[1] << 8);
    cal->dig_T2 = buf[2]  | (buf[3] << 8);
    // ... 剩余参数类似
}

第二步：软件补偿算法

// BMP280温度补偿公式（来自Bosch数据手册§3.11.3）
// 这是32位有符号定点运算，中间变量必须用int32_t！
float bmp280_compensate_temp(int32_t adc_T, bmp280_calib_t *cal) {
    int32_t var1, var2, T;
    // var1 = (adc_T/16384 - dig_T1/1024) * dig_T2
    var1 = ((((adc_T >> 3) - ((int32_t)cal->dig_T1 << 1))) * 
            ((int32_t)cal->dig_T2)) >> 11;
    // var2 = (adc_T/131072 - dig_T1/8192)^2 * dig_T3
    var2 = (((((adc_T >> 4) - ((int32_t)cal->dig_T1)) * 
              ((adc_T >> 4) - ((int32_t)cal->dig_T1))) >> 12) *
            ((int32_t)cal->dig_T3)) >> 14;
    T = (var1 + var2);
    // 返回 °C（BMP280内部精度0.01°C）
    return (float)((T * 5 + 128) >> 8) / 100.0f;
}
// ★ 关键：这些公式必须严格按数据手册实现，差一个移位精度全废！

第三步：二次校准（零偏+比例）

// 出厂校准 ≠ 最终精度！实际使用时还要做二次校准：
// 真实值 = (原始值 - 偏移量) * 比例因子
//
// 例如温度传感器：
// 在冰水混合物中测得 2.3°C → 偏移量 = 2.3
// 在沸水中测得 98.5°C → 比例因子 = 100/(98.5-2.3) = 1.0395
// 校准后：温度 = (raw - 2.3) * 1.0395

typedef struct {
    float offset;  // 零偏
    float scale;   // 比例系数
} sensor_cal_t;

float apply_calibration(float raw, sensor_cal_t *cal) {
    return (raw - cal->offset) * cal->scale;
}

🔌 I2C总线深入：多设备仲裁与时钟拉伸

时钟拉伸(Clock Stretching)容易踩坑
I2C规范允许从设备在来不及处理时拉住SCL为低，暂停通信。这是一个"隐藏"特性，很多初学者遇到I2C卡死都是因为这个。
// ★ 时钟拉伸的典型场景：
// 1. 从设备（如BMP280）正在进行ADC转换，主设备发起读取
//    → 从设备拉低SCL，等ADC完成（最长几ms）再释放
// 2. EEPROM正在写操作，主设备试图读
//    → 从设备拉低SCL直到写入完成（最长5ms！）

// STM32 HAL库默认开启了时钟拉伸支持，但可以配置超时：
// 在CubeMX → I2C Configuration → Clock Stretching → Enable
// HAL_I2C_Master_Transmit(&hi2c1, addr, data, len, timeout);
// timeout 必须大于最坏情况的时钟拉伸时间！
// BMP280测量时最长8ms → 超时设20ms比较安全

// ★ 调试技巧：逻辑分析仪抓SCL线，如果看到某段SCL持续低电平
// 而SDA为高 → 多半是从设备在时钟拉伸
I2C多设备地址冲突常见问题
多个相同型号传感器挂在同一I2C总线上时地址冲突。比如两个BMP280默认地址都是0x76。
// 解决方案1：利用芯片地址选择引脚
// BMP280的SDO引脚：接GND=地址0x76, 接VDD=地址0x77
// MPU6050的AD0引脚：接GND=地址0x68, 接VDD=地址0x69
// → 一块板最多挂2个同型号传感器

// 解决方案2：I2C多路复用器（TCA9548A）
// 1个I2C地址→8个独立I2C通道，每条通道可以有重叠地址
// TCA9548A默认地址0x70-0x77（A0/A1/A2引脚选择）
void tca_select_channel(uint8_t ch) {
    uint8_t cmd = 1 << ch;  // 通道0→0x01, 通道1→0x02, ...
    HAL_I2C_Master_Transmit(&hi2c1, 0x70<<1, &cmd, 1, 10);
}
// 切换通道后要加1ms等待信号稳定
// 使用：tca_select_channel(0); bmp280_read_ch0();
//       tca_select_channel(1); bmp280_read_ch1();

⚡ 传感器低功耗采样策略

电池供电的设备（纽扣电池/锂电池），传感器的功耗决定了续航时间。一味的"每秒读一次"会耗光电。以下是实战中常用的低功耗采样模式：

// 模式1：间歇采样（Duty Cycling）—— 最常用
// 原理：大部分时间MCU和传感器都休眠，定时醒来采样
// 计算：DHT11每次读数20ms @ 2.5mA = 0.05mAs
//       STM32在Stop模式 = 5μA
//       如果每10秒采样一次，平均功耗 ≈ (0.05mAs + 9.98s*5μA)/10s ≈ 10μA
//       → 200mAh纽扣电池可用 ~20000小时 = 2年！

void low_power_sample_loop(void) {
    while (1) {
        // 1. 唤醒传感器（如果传感器有Sleep模式）
        bmp280_set_mode(BMP280_MODE_FORCED);  // 单次转换模式
        HAL_Delay(10);  // 等待转换完成（BMP280最长8ms）
        
        // 2. 快速读取
        float temp = bmp280_read_temp();
        float press = bmp280_read_press();
        
        // 3. 传感器休眠
        bmp280_set_mode(BMP280_MODE_SLEEP);   // 功耗从720μA→0.1μA
        
        // 4. 数据缓存到RAM，攒够一批再通过无线发送
        store_reading(temp, press);
        if (buffer_full()) {
            wifi_wake_and_send();  // 集中发送省WiFi功耗
        }
        
        // 5. MCU深度休眠
        enter_stop_mode(10000);  // RTC唤醒，10秒后回来
    }
}

// 模式2：阈值触发采样
// 原理：平时休眠，传感器内置比较器检测到超过阈值才唤醒MCU
// 典型场景：振动检测（加速度计运动中断）、门磁、水位
void threshold_trigger_init(void) {
    // MPU6050运动唤醒：加速度超过阈值→INT引脚拉低→唤醒MCU
    mpu6050_set_motion_threshold(50);  // 50mg阈值
    mpu6050_enable_motion_interrupt(); // 使能运动中断
    // MCU进入Stop模式，MPU6050的INT引脚接WKUP
    HAL_PWR_EnterSTOPMode(PWR_LOWPOWERREGULATOR_ON, PWR_STOPENTRY_WFI);
}

🎯 实战练习

📝 练习1：温湿度监测站（入门）
DHT11 + STM32 + 串口打印。先跑通最基本的数据读取。加LED指示：温度>30°C红灯闪烁。

📝 练习2：多传感器融合站（进阶）
DHT11 + BMP280 + OLED + STM32。I2C总线上挂2个传感器+1个OLED。用互斥量保护I2C。对比卡尔曼滤波前后的数据稳定性。

📝 练习3：超声波倒车雷达（实战）
HC-SR04 + 蜂鸣器 + LED。距离<50cm→蜂鸣器慢响，<20cm→快速响，<10cm→长响。用状态机管理。

📝 练习4：姿态显示仪（进阶）
MPU6050互补滤波 → 串口100Hz输出pitch/roll → 上位机绘图。加分：OLED实时显示角度+水平仪图标。

📝 练习5：I2C总线多设备联调
一跟I2C总线上挂BMP280+MPU6050+OLED，I2C Scan确认地址后，3个设备轮询读取。用逻辑分析仪抓I2C通信波形。

📋 知识点回顾

✅ 传感器开发核心技能：
• I2C共享总线：最多128个设备，靠7位地址区分。先扫描再写驱动
• 传感器融合：互补滤波(简单α=0.98) → Mahony(中等) → Madgwick(精度高)
• 卡尔曼滤波：数学最优估计，核心是调Q(过程噪声)和R(测量噪声)
• 滑动平均：最简单去噪，窗口大小=平滑度vs响应速度的权衡
• 校准参数：传感器出厂预存的修正系数，必须读取并用于补偿计算
🛠 调试技巧：
• 先用I2C Scanner确认设备地址
• 用逻辑分析仪/示波器看通信波形（比猜代码快100倍）
• 串口以固定频率输出，用SerialPlot看波形
• 传感器读数跳变 → 检查电源去耦电容(100nF+10μF紧贴VCC-GND)
• 通信超时 → 检查上拉电阻(4.7KΩ)和地址(7位<<1)
📏 性能优化：
• DHT11: ≥2秒间隔，~20ms读取时间
• HC-SR04: ≥60ms间隔，2cm-400cm量程
• MPU6050: ≥100Hz采样率做姿态融合，≤400KHz I2C总线
• 卡尔曼: Q=0.01, R=1.0起步，根据响应/噪声权衡调参
