第6章：STM32开发实战

📖 小白完全指南 · 6/14 | 阅读时间约3分钟

STM32是ARM Cortex-M系列最流行的单片机，市场占有率超过50%。学会STM32，其他MCU都是触类旁通。

💡 类比：STM32就像一辆手动挡汽车——GPIO是方向盘（控制方向），UART是对讲机（通信），TIM是转速表（计时），ADC是油量表（测模拟量）。HAL库就是自动挡模式，LL库是手动挡模式。

HAL库 vs LL库

特性HAL库LL库
抽象层级高（封装完善）低（接近寄存器）
代码量大小
执行效率中等高
学习难度低高
适用场景快速开发/初学者极致优化/专业项目
💡 建议：入门先用HAL库，做产品时关键代码再换LL库优化。两者可以混用！

GPIO点灯 — 嵌入式的"Hello World"

⚠️ 两种方式二选一：

方式1：纯寄存器（推荐新手，无需任何工具，立即可用）
不依赖CubeMX和HAL库，直接操作寄存器。适合快速跑通第一个程序。

方式2：CubeMX + HAL库（正式项目推荐）
① 下载安装 STM32CubeMX（需注册ST账号，免费）
② 新建项目→选择MCU型号→配置引脚→生成代码
③ 生成的工程已包含所有初始化函数，你只需在while(1)中写业务逻辑
④ 也可以用EIDE/PlatformIO插件创建项目，免去CubeMX

💡 建议：先用方式1跑通，有经验了再学CubeMX。

方式1：纯寄存器版（无需CubeMX，5分钟跑通）

// main.c - STM32F103 点亮LED（PC13）— 纯寄存器版
// 最小依赖：只需要启动文件+链接脚本（见第2章"5分钟快速上手"）

// 寄存器地址定义
#define RCC_APB2ENR  (*(volatile uint32_t*)0x40021018)
#define GPIOC_CRH    (*(volatile uint32_t*)0x40011004)
#define GPIOC_ODR    (*(volatile uint32_t*)0x4001100C)

void delay(volatile uint32_t count) {
    while(count--);
}

int main(void) {
    // 步骤1：使能GPIOC时钟（必须先开时钟，否则引脚不工作！）
    RCC_APB2ENR |= (1 << 4);  // bit4 = IOPCEN
    
    // 步骤2：配置PC13为推挽输出（50MHz）
    // CRH控制PC8-PC15，PC13对应bits [23:20]
    GPIOC_CRH = (GPIOC_CRH & ~0x00F00000) | 0x00300000;
    
    // 步骤3：循环翻转LED
    while (1) {
        GPIOC_ODR ^= (1 << 13);  // 翻转PC13（XOR操作）
        delay(500000);            // 约500ms延时（72MHz下）
    }
}

方式2：CubeMX + HAL库版

如果你已经安装了STM32CubeMX并生成了工程，把下面的代码放在 /* USER CODE BEGIN */ 区域内：

// main.c - STM32F103 点亮LED（PC13）— HAL库版
// 前提：CubeMX已启用GPIOC Pin13为GPIO_Output
#include "stm32f1xx_hal.h"

// SystemClock_Config() 等初始化函数由CubeMX自动生成，不要手写！

int main(void) {
    HAL_Init();
    SystemClock_Config();
    // MX_GPIO_Init() 等初始化由CubeMX生成，自动调用

    while (1) {
        HAL_GPIO_TogglePin(GPIOC, GPIO_PIN_13);  // 翻转LED
        HAL_Delay(500);                            // 延时500ms
    }
}

UART发送接收

⚠️ 前提：以下UART代码基于HAL库，需用CubeMX启用USART1并配置为Asynchronous模式。EIDE/PlatformIO创建的项目也包含HAL库。纯寄存器方式见第2章"5分钟快速上手"。
// UART初始化 + printf重定向
#include <stdio.h>

UART_HandleTypeDef huart1;

// printf重定向到UART1
int fputc(int ch, FILE *f) {
    HAL_UART_Transmit(&huart1, (uint8_t *)&ch, 1, HAL_MAX_DELAY);
    return ch;
}

// 主循环
while (1) {
    printf("Temperature: %.1f C
", read_temp());
    HAL_Delay(1000);
}

// 接收中断
uint8_t rx_buf[1];
HAL_UART_Receive_IT(&huart1, rx_buf, 1);

// 中断回调
void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart) {
    if (huart->Instance == USART1) {
        // 处理接收到的字节 rx_buf[0]
        HAL_UART_Receive_IT(&huart1, rx_buf, 1);  // 重新开启接收
    }
}

TIM定时器中断

// 1ms定时器中断
TIM_HandleTypeDef htim2;

void MX_TIM2_Init(void) {
    htim2.Instance = TIM2;
    htim2.Init.Prescaler = 72 - 1;       // 72MHz/72 = 1MHz
    htim2.Init.CounterMode = TIM_COUNTERMODE_UP;
    htim2.Init.Period = 1000 - 1;         // 1MHz/1000 = 1kHz = 1ms
    HAL_TIM_Base_Init(&htim2);
    HAL_TIM_Base_Start_IT(&htim2);
}

// 中断回调（每1ms执行一次）
void HAL_TIM_PeriodElapsedCallback(TIM_HandleTypeDef *htim) {
    if (htim->Instance == TIM2) {
        static uint32_t ms_count = 0;
        ms_count++;
    }
}

ADC读取

// 读取PA0引脚的模拟电压
ADC_HandleTypeDef hadc1;

uint16_t read_adc(void) {
    HAL_ADC_Start(&hadc1);
    HAL_ADC_PollForConversion(&hadc1, HAL_MAX_DELAY);
    return HAL_ADC_GetValue(&hadc1);  // 0-4095 (12位ADC)
}

float read_voltage(void) {
    uint16_t raw = read_adc();
    return raw * 3.3f / 4096.0f;  // 转换为电压值
}

中断优先级 NVIC

// 设置优先级分组（在main开头调用一次）
HAL_NVIC_SetPriorityGrouping(NVIC_PRIORITYGROUP_4);  // 4位抢占优先级

// 设置具体中断优先级（数字越小优先级越高）
HAL_NVIC_SetPriority(USART1_IRQn, 0, 0);   // UART最高优先级
HAL_NVIC_SetPriority(TIM2_IRQn, 1, 0);      // 定时器次之
HAL_NVIC_SetPriority(ADC1_IRQn, 2, 0);       // ADC最低

// 使能中断
HAL_NVIC_EnableIRQ(USART1_IRQn);
🔍 常见问题：HAL_Delay卡死
HAL_Delay依赖SysTick中断。如果在其他中断里调用HAL_Delay且优先级高于SysTick，就会死锁。解决：确保调用HAL_Delay的中断优先级数字 > SysTick优先级数字（数字越大优先级越低）。
