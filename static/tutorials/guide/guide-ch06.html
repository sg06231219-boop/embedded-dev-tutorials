第6章：STM32开发实战

📖 小白完全指南 · 6/14 | 阅读时间约20分钟

STM32是ARM Cortex-M系列最流行的单片机，市场占有率超过50%。学会STM32，其他MCU都是触类旁通。这一章从寄存器到HAL库，从GPIO到DMA，带你彻底掌握STM32开发。

💡 类比：STM32就像一辆手动挡汽车——GPIO是方向盘（控制方向），UART是对讲机（通信），TIM是转速表（计时），ADC是油量表（测模拟量）。HAL库就是自动挡模式，LL库是手动挡模式。

📊 HAL库 vs LL库 vs 寄存器

特性HAL库LL库直接寄存器
抽象层级高（封装完善）低（接近寄存器）无（裸寄存器）
代码量大中小
执行效率中等高最高
学习难度低中高
可移植性好（跨系列）一般（同系列）差
适用场景快速开发/初学者极致优化/专业项目教学/极小Flash
💡 建议：入门先用HAL库，做产品时关键代码换LL库优化。两者可以混用——例如用HAL做初始化，LL做中断处理。

🎛 时钟系统：理解STM32的心脏

STM32F103时钟树进阶
时钟是STM32最容易被忽视但最关键的配置。时钟不对，所有外设（UART波特率、TIM定时、ADC采样率）都会出错。
时钟源频率精度用途
HSE（外部高速）4-16MHz（常用8MHz）±10ppm主时钟源（推荐）
HSI（内部高速）8MHz±1%（校准后）备用/低功耗
LSE（外部低速）32.768kHz±5ppmRTC时钟
LSI（内部低速）~40kHz差（温度漂移大）看门狗/低功耗唤醒
PLL（锁相环）HSI或HSE倍频取决于输入产生72MHz系统时钟
// 手动配置时钟（寄存器版——理解底层原理）
// 目标：HSE(8MHz) → PLL×9 → SYSCLK=72MHz

// 1. 等待HSE稳定
RCC->CR |= RCC_CR_HSEON;            // 开启HSE
while (!(RCC->CR & RCC_CR_HSERDY)); // 等待HSE就绪

// 2. 配置Flash延时（72MHz需要2个等待周期）
FLASH->ACR |= FLASH_ACR_LATENCY_2;

// 3. 配置PLL: HSE作为输入，倍频9 → 8MHz×9=72MHz
RCC->CFGR |= RCC_CFGR_PLLSRC;       // HSE作为PLL输入
RCC->CFGR |= RCC_CFGR_PLLMULL9;     // 9倍频

// 4. 配置AHB/APB1/APB2预分频
RCC->CFGR |= RCC_CFGR_HPRE_DIV1;    // AHB=72MHz
RCC->CFGR |= RCC_CFGR_PPRE1_DIV2;   // APB1=36MHz（最大36MHz！）
RCC->CFGR |= RCC_CFGR_PPRE2_DIV1;   // APB2=72MHz

// 5. 使能PLL，等待锁定
RCC->CR |= RCC_CR_PLLON;
while (!(RCC->CR & RCC_CR_PLLRDY));

// 6. 切换到PLL作为系统时钟
RCC->CFGR |= RCC_CFGR_SW_PLL;
while ((RCC->CFGR & RCC_CFGR_SWS) != RCC_CFGR_SWS_PLL);

// 完成！SYSCLK=72MHz, AHB=72MHz, APB1=36MHz, APB2=72MHz
⚠️ 重要限制：APB1最大频率是36MHz！如果你把APB1也配成72MHz，芯片不会烧（有保护电路），但外设可能工作异常。UART/TIM2-7挂在APB1上，注意时钟来源。
🔍 常见错误：时钟配置不对，程序跑飞或外设不工作

❌ 现象：程序下载后无反应、UART输出乱码、定时器不准
✅ 排查：
1. 检查HSE_VALUE宏是否等于你的晶振频率（常见8MHz/25MHz）
2. APB1最大36MHz(F103)/42MHz(F4)，APB2最大72MHz(F103)/84MHz(F4)
3. USB要用48MHz，需要精确的PLL配置（F103: PLL×6 if HSE=8MHz→仅HSI可用）
4. 用CubeMX的Clock Configuration页面验证所有外设时钟是否正确

💡 GPIO点灯 — 嵌入式的"Hello World"

方式1：纯寄存器版（无需CubeMX，5分钟跑通）

// main.c - STM32F103 点亮LED（PC13）— 纯寄存器版
// 最小依赖：只需要启动文件+链接脚本

// 寄存器地址定义
#define RCC_APB2ENR  (*(volatile uint32_t*)0x40021018)
#define GPIOC_CRH    (*(volatile uint32_t*)0x40011004)
#define GPIOC_ODR    (*(volatile uint32_t*)0x4001100C)
#define GPIOC_BSRR   (*(volatile uint32_t*)0x40011010)

void delay(volatile uint32_t count) {
    while(count--);
}

int main(void) {
    // 步骤1：使能GPIOC时钟（必须先开时钟，否则引脚不工作！）
    RCC_APB2ENR |= (1 << 4);  // bit4 = IOPCEN
    
    // 步骤2：配置PC13为推挽输出（50MHz）
    // CRH控制PC8-PC15，PC13对应bits [23:20]
    // MODE13[1:0]=11(50MHz) CNF13[1:0]=00(通用推挽)
    GPIOC_CRH = (GPIOC_CRH & ~0x00F00000) | 0x00300000;
    
    // 步骤3：循环翻转LED
    while (1) {
        // 方法A：XOR翻转（最简洁）
        GPIOC_ODR ^= (1 << 13);
        
        // 方法B：BSRR原子操作（推荐，不会误改其他引脚）
        // GPIOC_BSRR = (1 << 13);      // 置高
        // delay(500000);
        // GPIOC_BSRR = (1 << (13+16)); // 置低（高16位对应复位）
        
        delay(500000);  // 约500ms延时（72MHz下）
    }
}

GPIO速度与上下拉

速度等级最大频率适用场景
GPIO_SPEED_FREQ_LOW2 MHzLED、按钮、低速信号
GPIO_SPEED_FREQ_MEDIUM25 MHz普通IO、UART、I2C
GPIO_SPEED_FREQ_HIGH50 MHzSPI、PWM、高速信号
GPIO_SPEED_FREQ_VERY_HIGH100 MHzFMC/SDRAM、高速并行总线
💊 EMI提示：不要把不必要的高速IO设太高——速度越快，电磁干扰(EMI)越大，功耗也更高。LED用Low就够了。

// 输入捕获中断
void HAL_TIM_IC_CaptureCallback(TIM_HandleTypeDef *htim) {
    static uint32_t last_capture = 0;
    uint32_t now = HAL_TIM_ReadCapturedValue(htim, TIM_CHANNEL_1);
    
    if (now > last_capture) {
        uint32_t period = now - last_capture;
        freq_hz = 1000000 / period;  // 1MHz / 计数值 = 频率
    }
    last_capture = now;
}

I2C是传感器和存储芯片最常用的接口。以AT24C02（256字节EEPROM）为例。

// I2C1读写AT24C02 EEPROM
I2C_HandleTypeDef hi2c1;

void I2C_Init(void) {
    hi2c1.Instance = I2C1;
    hi2c1.Init.ClockSpeed = 100000;       // 100KHz（标准模式）
    hi2c1.Init.DutyCycle = I2C_DUTYCYCLE_2;
    hi2c1.Init.OwnAddress1 = 0;
    hi2c1.Init.AddressingMode = I2C_ADDRESSINGMODE_7BIT;
    HAL_I2C_Init(&hi2c1);
}

// 写入一个字节到指定地址
HAL_StatusTypeDef EEPROM_WriteByte(uint8_t addr, uint8_t data) {
    // AT24C02地址=0xA0，器件地址<<1 + R/W位
    uint8_t buf[2] = {addr, data};
    return HAL_I2C_Master_Transmit(&hi2c1, 0xA0, buf, 2, 100);
}

// 从指定地址读取一个字节
uint8_t EEPROM_ReadByte(uint8_t addr) {
    uint8_t data = 0xFF;
    // 先发送要读的地址
    HAL_I2C_Master_Transmit(&hi2c1, 0xA0, &addr, 1, 100);
    // 再读取数据
    HAL_I2C_Master_Receive(&hi2c1, 0xA0, &data, 1, 100);
    return data;
}

// 页写入（AT24C02页大小=8字节）
HAL_StatusTypeDef EEPROM_WritePage(uint8_t addr, uint8_t *data, uint8_t len) {
    uint8_t buf[9];  // 1字节地址 + 最多8字节数据
    buf[0] = addr;
    memcpy(&buf[1], data, len);
    return HAL_I2C_Master_Transmit(&hi2c1, 0xA0, buf, len + 1, 100);
}

// 测试：写入字符串并读出验证
void EEPROM_Test(void) {
    const char *str = "Hello!";
    EEPROM_WritePage(0, (uint8_t*)str, 6);
    HAL_Delay(10);  // EEPROM写入需要时间（内部写周期）
    
    char buf[7] = {0};
    for (int i = 0; i < 6; i++) {
        buf[i] = EEPROM_ReadByte(i);
    }
    printf("Read from EEPROM: %s\r\n", buf);  // 期望: Hello!
}

🔗 SPI通信（Flash读写）

SPI是高速外设（Flash、显示屏、SD卡）的首选。以W25Q16（2MB Flash）为例。

// SPI1读写W25Q16 Flash
SPI_HandleTypeDef hspi1;

// W25Q16指令
#define W25Q_READ_ID    0x9F
#define W25Q_READ_DATA  0x03
#define W25Q_WRITE_EN   0x06
#define W25Q_PAGE_PROG  0x02
#define W25Q_READ_SR1   0x05
#define W25Q_SECTOR_ERASE 0x20

void SPI_Init(void) {
    hspi1.Instance = SPI1;
    hspi1.Init.Mode = SPI_MODE_MASTER;
    hspi1.Init.Direction = SPI_DIRECTION_2LINES;
    hspi1.Init.DataSize = SPI_DATASIZE_8BIT;
    hspi1.Init.CLKPolarity = SPI_POLARITY_LOW;
    hspi1.Init.CLKPhase = SPI_PHASE_1EDGE;
    hspi1.Init.NSS = SPI_NSS_SOFT;
    hspi1.Init.BaudRatePrescaler = SPI_BAUDRATEPRESCALER_4; // 18MHz
    hspi1.Init.FirstBit = SPI_FIRSTBIT_MSB;
    HAL_SPI_Init(&hspi1);
}

// 读芯片ID（确认SPI通信正常）
uint16_t W25Q_ReadID(void) {
    uint8_t tx[4] = {W25Q_READ_ID, 0, 0, 0};
    uint8_t rx[4] = {0};
    HAL_GPIO_WritePin(GPIOA, GPIO_PIN_4, GPIO_PIN_RESET); // CS低
    HAL_SPI_TransmitReceive(&hspi1, tx, rx, 4, 100);
    HAL_GPIO_WritePin(GPIOA, GPIO_PIN_4, GPIO_PIN_SET);   // CS高
    return (rx[1] << 8) | rx[2];  // 期望: 0xEF15(W25Q16)
}

// 从指定地址读取数据
void W25Q_Read(uint32_t addr, uint8_t *buf, uint16_t len) {
    uint8_t cmd[4] = {
        W25Q_READ_DATA,
        (uint8_t)(addr >> 16),
        (uint8_t)(addr >> 8),
        (uint8_t)(addr)
    };
    HAL_GPIO_WritePin(GPIOA, GPIO_PIN_4, GPIO_PIN_RESET);
    HAL_SPI_Transmit(&hspi1, cmd, 4, 100);
    HAL_SPI_Receive(&hspi1, buf, len, 1000);
    HAL_GPIO_WritePin(GPIOA, GPIO_PIN_4, GPIO_PIN_SET);
}

// 写使能（任何写操作前必须执行）
void W25Q_WriteEnable(void) {
    uint8_t cmd = W25Q_WRITE_EN;
    HAL_GPIO_WritePin(GPIOA, GPIO_PIN_4, GPIO_PIN_RESET);
    HAL_SPI_Transmit(&hspi1, &cmd, 1, 100);
    HAL_GPIO_WritePin(GPIOA, GPIO_PIN_4, GPIO_PIN_SET);
}

📡 DMA（直接存储器访问）实战

DMA是STM32最强大的特性之一。理解它，你的程序效率能提升10倍——CPU可以完全不参与数据传输。

// ADC + DMA 连续采样：CPU零干预采集2048个样本
#define ADC_BUF_SIZE 2048
uint16_t adc_buffer[ADC_BUF_SIZE];
DMA_HandleTypeDef hdma_adc;

void ADC_DMA_Init(void) {
    // 1. 使能DMA时钟
    __HAL_RCC_DMA1_CLK_ENABLE();
    
    // 2. 配置DMA
    hdma_adc.Instance = DMA1_Channel1;
    hdma_adc.Init.Direction = DMA_PERIPH_TO_MEMORY;       // 外设→内存
    hdma_adc.Init.PeriphInc = DMA_PINC_DISABLE;           // 外设地址不变(ADC_DR)
    hdma_adc.Init.MemInc = DMA_MINC_ENABLE;               // 内存地址递增
    hdma_adc.Init.PeriphDataAlignment = DMA_PDATAALIGN_HALFWORD; // 16位
    hdma_adc.Init.MemDataAlignment = DMA_MDATAALIGN_HALFWORD;
    hdma_adc.Init.Mode = DMA_CIRCULAR;                    // 循环模式(不停)
    hdma_adc.Init.Priority = DMA_PRIORITY_HIGH;
    HAL_DMA_Init(&hdma_adc);
    
    // 3. 关联ADC和DMA
    __HAL_LINKDMA(&hadc1, DMA_Handle, hdma_adc);
    
    // 4. 启动ADC DMA传输
    HAL_ADC_Start_DMA(&hadc1, (uint32_t*)adc_buffer, ADC_BUF_SIZE);
    // 此后CPU完全不用管，DMA会自动把每次ADC结果写到adc_buffer[]
}

// DMA半传输完成回调（前1024个样本就绪）
void HAL_ADC_ConvHalfCpltCallback(ADC_HandleTypeDef* hadc) {
    // 处理前半段数据（adc_buffer[0..1023]）
    // 此时DMA正在填充后半段，互不干扰！（双缓冲技巧）
}

// DMA全部传输完成回调（后1024个样本就绪）
void HAL_ADC_ConvCpltCallback(ADC_HandleTypeDef* hadc) {
    // 处理后半段数据（adc_buffer[1024..2047]）
    float sum = 0;
    for (int i = 1024; i < ADC_BUF_SIZE; i++) {
        sum += adc_buffer[i];
    }
    float avg = sum / 1024.0f;
    printf("ADC average (2nd half): %.2f mV\r\n", avg * 3300 / 4096);
}

// DMA传输方向总结
// PERIPH_TO_MEMORY:  ADC采样, UART接收, SPI接收
// MEMORY_TO_PERIPH:  UART发送大段, SPI发送显存, DAC输出波形
// MEMORY_TO_MEMORY:  快速memcpy (不占用CPU循环)
DMA双缓冲技巧进阶
利用半传输和全传输中断实现"零拷贝"数据处理：DMA在填充后半段时，CPU处理前半段；DMA填满后半段回到开头时，CPU处理后半段。这在示波器、音频处理中至关重要。

关键API：
• HAL_ADC_ConvHalfCpltCallback() — 前半段OK→CPU处理前半段
• HAL_ADC_ConvCpltCallback() — 后半段OK→CPU处理后半段
• __HAL_DMA_GET_COUNTER(&hdma) — 查询DMA剩余传输次数

🔧 HAL库深度解析

HAL库状态机

// 每个HAL函数返回HAL_StatusTypeDef
typedef enum {
  HAL_OK       = 0x00,  // 成功
  HAL_ERROR    = 0x01,  // 一般错误
  HAL_BUSY     = 0x02,  // 资源忙（DMA/外设正在使用）
  HAL_TIMEOUT  = 0x03   // 超时（通常外设没响应）
} HAL_StatusTypeDef;

// 黄金法则：永远检查返回值！
HAL_StatusTypeDef ret = HAL_UART_Transmit(&huart2, data, len, 100);
if (ret != HAL_OK) {
    // 记录错误类型
    switch (ret) {
    case HAL_BUSY:
        printf("UART busy, retrying...\r\n");
        break;
    case HAL_TIMEOUT:
        printf("UART timeout! Check wiring.\r\n");
        break;
    default:
        Error_Handler();
    }
}

Tick定时器与HAL_Delay的陷阱

⚠ 关键坑：HAL_Delay依赖SysTick中断。如果你在中断服务函数（尤其是优先级高于SysTick的中断）里调用HAL_Delay，会死循环——因为SysTick中断被阻塞了，计数器永远不会更新！
// 正确做法1：在中断中用非阻塞轮询
void process_in_isr(void) {
    uint32_t tickstart = HAL_GetTick();
    while ((HAL_GetTick() - tickstart) < 100) {  // 最多等100ms
        if (HAL_UART_Transmit(&huart2, data, 1, 1) == HAL_OK) break;
    }
}

// 正确做法2：实现独立的微秒延时（不依赖SysTick）
void delay_us(uint32_t us) {
    // DWT（Data Watchpoint）计数器，不受中断影响
    uint32_t start = DWT->CYCCNT;
    uint32_t ticks = us * (SystemCoreClock / 1000000);
    while ((DWT->CYCCNT - start) < ticks);
}

// DWT初始化（main开头调用一次）
void DWT_Init(void) {
    CoreDebug->DEMCR |= CoreDebug_DEMCR_TRCENA_Msk;
    DWT->CTRL |= DWT_CTRL_CYCCNTENA_Msk;
}

// 正确做法3：用软件定时器代替HAL_Delay
volatile uint32_t g_ms_ticks = 0;  // 在SysTick或TIM中断中递增

void nonblock_delay(uint32_t ms) {
    uint32_t start = g_ms_ticks;
    while ((g_ms_ticks - start) < ms) {
        __WFI();  // 等待中断（省电）
    }
}

void enter_sleep(void) {

    HAL_PWR_EnterSTOPMode(PWR_LOWPOWERREGULATOR_ON, PWR_STOPENTRY_WFI);

    SystemClock_Config();
}
// 用法：配置PA0为EXTI唤醒源，按键唤醒

// Standby模式 — 最省电，唤醒=复位
void enter_standby(void) {
    __HAL_RCC_PWR_CLK_ENABLE();
    HAL_PWR_EnableWakeUpPin(PWR_WAKEUP_PIN1);  // PA0作为WKUP
    __HAL_PWR_CLEAR_FLAG(PWR_FLAG_WU);
    HAL_PWR_EnterSTANDBYMode();
    // 永远不会执行到这里！唤醒后从main开头重跑
}

🐛 调试技巧

HardFault排查

🔍 程序跑飞进HardFault_Handler

❌ 最常见原因：
1. 访问了未初始化的外设（没开时钟就操作寄存器）
2. 数组越界/栈溢出（局部变量太多或递归太深）
3. 函数指针为NULL（忘了注册回调就调用）
4. 除零错误（Cortex-M3/4支持，需使能）
5. 非对齐访问（Cortex-M0不支持非对齐）

✅ 排查方法：
1. 在HardFault_Handler中设断点
2. 查看SP（栈指针）指向的栈帧
3. 从栈帧中取出PC（程序计数器）→ 定位出错的指令地址
4. 在.map文件中查找该地址对应的函数
5. 用GDB命令行：arm-none-eabi-addr2line -e app.elf 0x0800xxxx
// HardFault调试代码：保存现场信息
void HardFault_Handler(void) {
    __asm volatile (
        "TST LR, #4               \n"
        "ITE EQ                   \n"
        "MRSEQ R0, MSP            \n"  // 主栈
        "MRSNE R0, PSP            \n"  // 进程栈
        "B hard_fault_handler_c   \n"
    );
}

// C语言处理函数（可以通过串口或LED输出调试信息）
void hard_fault_handler_c(uint32_t *stack) {
    printf("!!! HARDFAULT !!!\r\n");
    printf("  R0  = 0x%08X\r\n", stack[0]);
    printf("  R1  = 0x%08X\r\n", stack[1]);
    printf("  R2  = 0x%08X\r\n", stack[2]);
    printf("  R3  = 0x%08X\r\n", stack[3]);
    printf("  R12 = 0x%08X\r\n", stack[4]);
    printf("  LR  = 0x%08X\r\n", stack[5]);
    printf("  PC  = 0x%08X\r\n", stack[6]);  // ← 出错的指令地址！
    printf("  xPSR= 0x%08X\r\n", stack[7]);
    
    // 用addr2line定位：
    // arm-none-eabi-addr2line -e firmware.elf 0x08000xxx
    
    while(1);  // 停在这里方便调试
}

🎯 实战项目：多传感器数据采集系统

// 综合项目：定时采集温度(ADC)+存储到EEPROM+串口上报
// 硬件：STM32F103C8T6 + NTC热敏电阻(PA0) + AT24C02 + USART1

#define LOG_INTERVAL_MS  5000   // 每5秒采集一次
#define EEPROM_BASE_ADDR 0x00
#define MAX_RECORDS      32     // 最多存32条记录

typedef struct {
    uint32_t timestamp;  // 系统启动后的秒数
    uint16_t adc_raw;    // ADC原始值
    float voltage;       // 电压值
    float temp_c;        // 摄氏温度
} sensor_record_t;

sensor_record_t g_records[MAX_RECORDS];
uint8_t g_record_count = 0;

// NTC温度转换（10K NTC, B=3950, 串联10K上拉电阻到3.3V）
float ntc_to_temp(uint16_t adc) {
    float v = adc * 3.3f / 4096.0f;           // 电压
    float r = v * 10000.0f / (3.3f - v);      // NTC电阻
    // Steinhart-Hart简化公式
    float temp = 1.0f / (1.0f/298.15f + log(r/10000.0f)/3950.0f) - 273.15f;
    return temp;
}

void collect_and_store(void) {
    if (g_record_count >= MAX_RECORDS) {
        // 环形覆盖最早的记录
        g_record_count = 0;
    }
    
    sensor_record_t *rec = &g_records[g_record_count];
    rec->timestamp = HAL_GetTick() / 1000;
    rec->adc_raw = read_adc();
    rec->voltage = rec->adc_raw * 3.3f / 4096.0f;
    rec->temp_c = ntc_to_temp(rec->adc_raw);
    
    // 保存到EEPROM
    EEPROM_WritePage(EEPROM_BASE_ADDR + g_record_count * sizeof(sensor_record_t),
                     (uint8_t*)rec, sizeof(sensor_record_t));
    g_record_count++;
    
    // 串口上报
    printf("[%05lu] ADC=%4d V=%.3fV Temp=%.1f°C (record %d/%d)\r\n",
           rec->timestamp, rec->adc_raw, rec->voltage, rec->temp_c,
           g_record_count, MAX_RECORDS);
}

// 主循环
int main(void) {
    HAL_Init();
    SystemClock_Config();
    MX_GPIO_Init();
    MX_USART1_UART_Init();
    MX_ADC1_Init();
    MX_I2C1_Init();
    MX_TIM2_Init();
    
    printf("=== Multi-Sensor Logger ===\r\n");
    printf("Interval: %dms, Max records: %d\r\n", LOG_INTERVAL_MS, MAX_RECORDS);
    
    uint32_t last_log = 0;
    while (1) {
        uint32_t now = HAL_GetTick();
        if (now - last_log >= LOG_INTERVAL_MS) {
            last_log = now;
            collect_and_store();
        }
        // 其余时间可以处理串口命令等
    }
}

⚡ NVIC中断优先级组与嵌套配置

STM32的NVIC有16个优先级（4位），但实际可用的优先级数量取决于优先级分组（Priority Group）。这个设置一旦搞错，中断嵌套就乱了。

// ★ NVIC优先级分组（系统级，整个程序只能设一次！）
// 4位优先级 = 抢占优先级位 + 子优先级位
// 
// HAL_NVIC_SetPriorityGrouping(NVIC_PRIORITYGROUP_X);
// 
// 分组0: 0位抢占 + 4位子优先级 → 无嵌套！所有中断平等
// 分组1: 1位抢占(0-1)  + 3位子 → 2级抢占
// 分组2: 2位抢占(0-3)  + 2位子 → 4级抢占  ← 最常用！
// 分组3: 3位抢占(0-7)  + 1位子 → 8级抢占
// 分组4: 4位抢占(0-15) + 0位子 → 16级抢占  ← 全部用于嵌套

// ★ 实战配置（HAL库默认分组4，但推荐分组2有更多灵活性）
HAL_NVIC_SetPriorityGrouping(NVIC_PRIORITYGROUP_2);  // 4级抢占

// 中断优先级配置（数值越大优先级越低！）
// 抢占优先级: 0-3  (低数值=高优先级，可以打断高数值)
// 子优先级:   0-3  (同一抢占优先级内的排队顺序)

HAL_NVIC_SetPriority(SysTick_IRQn,   3, 0);  // 最低抢占（系统时钟）
HAL_NVIC_SetPriority(USART1_IRQn,    1, 0);  // 高抢占（通信不可丢）
HAL_NVIC_SetPriority(EXTI0_IRQn,     0, 0);  // 最高抢占（按键/紧急）
HAL_NVIC_SetPriority(TIM2_IRQn,      2, 1);  // 中抢占（定时器）

// ★ 优先级设置原则：
// 1. 硬实时中断设最高优先级（电机控制、急停按钮→0）
// 2. 通信中断设高优先级（UART RX→1，丢数据恢复代价大）
// 3. 定时器设中优先级（TIM→2，稍晚几μs没关系）
// 4. SysTick和DMA设为最低（3，晚点无妨）
// 5. ISR内部绝对不调阻塞函数（HAL_Delay/Semaphore等）

🔋 低功耗模式深入

STM32F103低功耗模式对比必学
STM32F103有三种低功耗模式。选择的关键：唤醒后能多快恢复、哪些外设还在工作、电流能到多低。
模式电流(典型)唤醒方式唤醒时间外设状态典型场景
Sleep~5mA任意中断/事件无延迟全部运行等串口数据
Stop~20μAEXTI / RTC / IWDG~5μs(HSI起振)仅LSE/LSI运行电池设备间歇采样

// ★ Stop模式实战（唤醒后SRAM数据保留）
void enter_stop_mode(uint32_t wakeup_ms) {
    // ① 关掉耗电大户
    HAL_ADC_DeInit(&hadc1);          // ADC: ~1mA
    HAL_UART_DeInit(&huart1);        // UART: ~300μA
    __HAL_RCC_GPIOA_CLK_DISABLE();   // 关GPIO时钟（能省几十μA）
    __HAL_RCC_GPIOB_CLK_DISABLE();
    
    // ② 配置唤醒源（RTC闹钟）
    HAL_RTCEx_DeactivateWakeUpTimer(&hrtc);
    HAL_RTCEx_SetWakeUpTimer_IT(&hrtc, wakeup_ms * 10, RTC_WAKEUPCLOCK_RTCCLK_DIV16);
    
    // ③ 进入Stop模式
    HAL_PWR_EnterSTOPMode(PWR_LOWPOWERREGULATOR_ON, PWR_STOPENTRY_WFI);
    // === CPU停在这里，直到RTC唤醒 ===
    
    // ④ 唤醒后恢复（时钟配置丢失，需重新配置！）
    SystemClock_Config();  // 重新配置HSE+PLL
    HAL_GPIO_Init(GPIOA, &gpio_init);  // 重新初始化用到的GPIO
    // ★ 全局变量和SRAM内容完好无损！
⚠ Stop模式陷阱：唤醒后HSE/PLL停了！在 SystemClock_Config() 完成前所有依赖系统时钟的外设（UART/SPI/TIM）都不工作。这就是为什么中断服务函数里不能直接进Stop然后继续——时钟还没恢复。

🔄 DMA双缓冲与乒乓传输

ADC连续采样+串口发送时，如果没有双缓冲，数据处理和传输会互相等待。双缓冲的核心思想：一个缓冲区在"生产"（ADC填充），另一个在"消费"（CPU处理），两个交替切换。

// ★ DMA双缓冲模式（仅STM32F4/F7支持硬件双缓冲，F1用软件模拟）
// 这里展示软件双缓冲——在DMA传输完成中断中切换

#define ADC_BUF_SIZE  512  // 每个缓冲区512个采样点
uint16_t adc_buf_a[ADC_BUF_SIZE];  // 缓冲区A
uint16_t adc_buf_b[ADC_BUF_SIZE];  // 缓冲区B
volatile uint8_t active_buf = 0;   // 当前在用的缓冲区(0=A, 1=B)
volatile uint8_t buf_ready_a = 0;  // A有数据待处理
volatile uint8_t buf_ready_b = 0;  // B有数据待处理

// 启动第一个DMA传输到buf_a
void dma_double_buf_start(void) {
    HAL_ADC_Start_DMA(&hadc1, (uint32_t*)adc_buf_a, ADC_BUF_SIZE);
}

// DMA传输完成中断
void HAL_ADC_ConvCpltCallback(ADC_HandleTypeDef* hadc) {
    if (active_buf == 0) {
        // A满了，通知主循环处理A，同时启动B
        buf_ready_a = 1;
        active_buf = 1;
        HAL_ADC_Start_DMA(&hadc1, (uint32_t*)adc_buf_b, ADC_BUF_SIZE);
    } else {
        // B满了，通知主循环处理B，同时启动A
        buf_ready_b = 1;
        active_buf = 0;
        HAL_ADC_Start_DMA(&hadc1, (uint32_t*)adc_buf_a, ADC_BUF_SIZE);
    }
}

// 主循环处理（无等待！只管消费ready的缓冲区）
void main_loop(void) {
    while (1) {
        if (buf_ready_a) {
            process_samples(adc_buf_a, ADC_BUF_SIZE);  // 处理A数据
            buf_ready_a = 0;
        }
        if (buf_ready_b) {
            process_samples(adc_buf_b, ADC_BUF_SIZE);  // 处理B数据
            buf_ready_b = 0;
        }
        __WFI();  // 没数据就休眠等中断
    }
}
// ★ 关键优势：ADC在不停采样，CPU在处理上一批数据，互不等待！
// 单缓冲方案会在 process_samples() 期间丢ADC数据
// 双缓冲方案永远不会丢（只要process比采样快）
🔍 常见问题：HAL_Delay卡死
❌ 现象：调用HAL_Delay后程序卡住不返回
✅ 原因：HAL_Delay依赖SysTick中断。如果在中断里调用且中断优先级高于SysTick，SysTick无法触发导致死锁
✅ 解决：确保调用HAL_Delay的中断优先级数字 > SysTick优先级数字（数字越大优先级越低）
🔍 常见问题：外设不工作（已开时钟）
❌ 现象：RCC时钟已使能，但外设寄存器读写无效
✅ 排查：
1. 确认使用了正确的总线时钟（APB1/APB2）
2. 检查外设是否在APB总线上被禁用了（某些F4/F7有额外门控）
3. GPIO的AF（复用功能）是否正确配置
4. 如果有重映射（Remap），是否已使能AFIO时钟并配置重映射
🔍 常见问题：编译OK但烧录后不运行
❌ 现象：程序编译下载成功，但LED不闪、串口无输出
✅ 排查清单：
1. 启动文件(startup_xxx.s)是否匹配MCU型号
2. 链接脚本中的Flash/RAM起始地址和大小是否正确
3. BOOT0引脚是否接GND（拉高会进Bootloader不跑用户程序）
4. 复位引脚是否被意外拉低
5. 用调试器单步看是否卡在HardFault_Handler
6. 检查晶振是否起振（示波器测HSE引脚）

📋 知识点回顾

✅ STM32开发核心：
• 三层次开发：寄存器(教学) → HAL库(快速开发) → LL库(性能优化)
• 时钟树：HSE→PLL→AHB→APB1(≤36MHz)/APB2(≤72MHz)→外设
• GPIO：速度等级匹配场景(LED用Low, SPI用High)，BSRR原子操作更安全
• UART：阻塞发送(调试)、中断接收(单字节)、DMA+IDLE(不定长数据)
• TIM：基本定时(中断)、PWM输出(呼吸灯/舵机)、输入捕获(测频率)
• ADC：轮询读取(简单)、DMA连续采样(高效)、双缓冲(示波器)
• I2C/SPI：标准外设总线，EEPROM/Flash/传感器都走它们
• DMA：P→M(ADC)、M→P(UART发送)、M→M(快拷)
• 看门狗：IWDG(防死机)、WWDG(检测时序异常)

🛠 调试利器：
• HardFault_Handler改造 → 获取PC/LR定位崩溃点
• 条件编译日志宏 → 开发时全开，发布时一键关闭
• HAL返回值检查 → 每个API调用检查HAL_OK
• ISR中不用HAL_Delay → 用非阻塞轮询或独立us延时

📖 进阶方向：
• FreeRTOS移植 → 任务调度让多任务变成简单函数调用
• CMSIS-DSP → 数字信号处理库(FFT/FIR/矩阵运算)
• TouchGFX → 图形界面开发
• USB协议栈 → CDC/HID/MSC设备开发
