⚡ 5分钟快速上手（Arduino/PlatformIO）

⚠️ 最快上手方式：用PlatformIO创建ESP32 Arduino项目，无需单独装ESP-IDF（约3GB）！
步骤1：VS Code → PlatformIO → New Project → Board: ESP32 Dev Module → Framework: Arduino
步骤2：把下面的代码粘贴到 src/main.cpp
// ESP32 LED闪烁 + 串口输出（Arduino框架，5分钟跑通）
// 大多数ESP32开发板的LED在GPIO2（内置LED）

void setup() {
    Serial.begin(115200);  // 初始化串口
    pinMode(2, OUTPUT);     // GPIO2设为输出
    Serial.println("ESP32 Started! Chip: " + String(ESP.getChipModel()));
    Serial.printf("Free heap: %d bytes\n", ESP.getFreeHeap());
}

void loop() {
    digitalWrite(2, HIGH);   // LED亮
    Serial.println("LED ON");
    delay(500);
    digitalWrite(2, LOW);    // LED灭
    Serial.println("LED OFF");
    delay(500);
}
步骤3：连接ESP32 → 点击PlatformIO的→箭头按钮（Upload） → 自动编译+烧录
步骤4：打开串口监视器（PlatformIO插头图标），看到"LED ON/OFF"交替输出就成功了！
🔍 常见问题：esptool连接失败
❌ 现象：A fatal error occurred: Failed to connect to ESP32
✅ 解决步骤：
1. 按住BOOT键不放，再按一下EN(RST)键，松开BOOT键 → 进入下载模式
2. 检查COM口号：设备管理器 → 端口 → 找"Silicon Labs CP210x"或"CH340"
3. USB驱动确认：CP2102驱动 → Silabs官网下载；CH340驱动 → 搜"CH340 driver"
4. 换一根数据线（某些充电线没有数据功能！）
5. 降低波特率：esptool.py -b 115200

🔄 Arduino vs ESP-IDF：选哪个？

特性Arduino框架ESP-IDF
上手难度极低（setup/loop模型）中等（需要理解FreeRTOS）
灵活性受限（API封装固定）完全控制
性能优化一般极致（IRAM加速、cache控制）
FreeRTOS隐藏使用（单核loop）显式使用（多任务、多核）
menuconfig不支持完整的Kconfig配置系统
适合原型验证/初学者/快速demo产品开发/性能要求/进阶用户
PlatformIO✅ 支持✅ 支持（framework=espidf）
💡 建议：Arduino快速验证想法（1小时出原型），ESP-IDF做正式产品（功能全、性能好）。PlatformIO支持两种框架一键切换，非常方便。

📶 WiFi开发实战

WiFi Station模式（连接路由器）

// ESP-IDF WiFi Station 完整示例
#include "esp_wifi.h"
#include "esp_event.h"
#include "nvs_flash.h"
#include "esp_log.h"

#define WIFI_SSID "YourWiFi"
#define WIFI_PASS "YourPassword"
#define TAG "WiFi"

static void wifi_event_handler(void* arg, esp_event_base_t base,
    int32_t id, void* data) {
    if (base == WIFI_EVENT && id == WIFI_EVENT_STA_START) {
        esp_wifi_connect();
    } else if (base == WIFI_EVENT && id == WIFI_EVENT_STA_DISCONNECTED) {
        ESP_LOGW(TAG, "WiFi断开，尝试重连...");
        esp_wifi_connect();
    } else if (base == IP_EVENT && id == IP_EVENT_STA_GOT_IP) {
        ip_event_got_ip_t* ev = (ip_event_got_ip_t*)data;
        ESP_LOGI(TAG, "获取IP: " IPSTR, IP2STR(&ev->ip_info.ip));
    }
}

void wifi_init_sta(void) {
    // 1. 初始化NVS（WiFi配置存储）
    nvs_flash_init();
    
    // 2. 初始化TCP/IP协议栈
    esp_netif_init();
    esp_event_loop_create_default();
    esp_netif_create_default_wifi_sta();
    
    // 3. 初始化WiFi驱动
    wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
    esp_wifi_init(&cfg);
    
    // 4. 注册事件处理
    esp_event_handler_register(WIFI_EVENT, ESP_EVENT_ANY_ID,
        &wifi_event_handler, NULL);
    esp_event_handler_register(IP_EVENT, IP_EVENT_STA_GOT_IP,
        &wifi_event_handler, NULL);
    
    // 5. 配置并连接
    wifi_config_t wifi_cfg = {
        .sta = {
            .ssid = WIFI_SSID,
            .password = WIFI_PASS,
            .threshold.authmode = WIFI_AUTH_WPA2_PSK,  // 最低安全要求
        }
    };
    esp_wifi_set_mode(WIFI_MODE_STA);
    esp_wifi_set_config(WIFI_IF_STA, &wifi_cfg);
    esp_wifi_start();
}

// 扫描周围WiFi热点
void wifi_scan(void) {
    esp_wifi_set_mode(WIFI_MODE_STA);
    esp_wifi_start();
    
    wifi_scan_config_t scan_cfg = {
        .ssid = NULL,
        .bssid = NULL,
        .channel = 0,              // 全信道扫描
        .show_hidden = true,       // 显示隐藏网络
        .scan_type = WIFI_SCAN_TYPE_ACTIVE,
    };
    
    esp_wifi_scan_start(&scan_cfg, true);  // true=阻塞等待完成
    
    uint16_t ap_count = 0;
    esp_wifi_scan_get_ap_num(&ap_count);
    
    wifi_ap_record_t *ap_list = malloc(sizeof(wifi_ap_record_t) * ap_count);
    esp_wifi_scan_get_ap_records(&ap_count, ap_list);
    
    printf("\n=== WiFi Scan Results (%d APs) ===\n", ap_count);
    printf("%-32s %6s %3s %s\n", "SSID", "RSSI", "Ch", "Auth");
    printf("--------------------------------------\n");
    
    for (int i = 0; i < ap_count; i++) {
        printf("%-32s %6d %3d %s\n",
            ap_list[i].ssid,
            ap_list[i].rssi,
            ap_list[i].primary,
            ap_list[i].authmode == WIFI_AUTH_OPEN ? "Open" : "****");
    }
    free(ap_list);
}

// 用手机APP一键配网（免输入SSID/密码）

    esp_wifi_set_mode(WIFI_MODE_STA);
    esp_wifi_start();
    esp_smartconfig_set_type(SC_TYPE_ESPTOUCH);
    esp_smartconfig_start(smartconfig_callback);
    printf("SmartConfig started. Use Esptouch APP to configure.\n");
}

static void smartconfig_callback(smartconfig_status_t status, void *pdata) {
    switch (status) {
    case SC_STATUS_LINK:
        printf("SmartConfig: Got SSID/Password, connecting...\n");
        break;
    case SC_STATUS_LINK_OVER:
        printf("SmartConfig: WiFi connected!\n");
        // 保存WiFi凭据到NVS
        wifi_config_t cfg;
        esp_wifi_get_config(WIFI_IF_STA, &cfg);
        // nvs_save_wifi((char*)cfg.sta.ssid, (char*)cfg.sta.password);
        esp_smartconfig_stop();
        break;
    case SC_STATUS_FIND_CHANNEL:
        printf("SmartConfig: Found channel.\n");
        break;
    }
}

HTTP GET请求

// ESP-IDF HTTP客户端
#include "esp_http_client.h"

void http_get_example(void) {
    char response[1024] = {0};
    
    esp_http_client_config_t cfg = {
        .url = "http://api.openweathermap.org/data/2.5/weather?q=Beijing&appid=YOUR_KEY",
        .method = HTTP_METHOD_GET,
        .timeout_ms = 5000,
        .user_agent = "ESP32/1.0",
    };
    
    esp_http_client_handle_t client = esp_http_client_init(&cfg);
    esp_err_t err = esp_http_client_perform(client);
    
    if (err == ESP_OK) {
        int status = esp_http_client_get_status_code(client);
        int read_len = esp_http_client_read(client, response, sizeof(response)-1);
        response[read_len] = 0;
        printf("HTTP %d, Response: %s\n", status, response);
    } else {
        printf("HTTP GET failed: %s\n", esp_err_to_name(err));
    }
    esp_http_client_cleanup(client);
}

static esp_err_t root_handler(httpd_req_t *req) {
    httpd_resp_set_type(req, "text/html");

    return ESP_OK;
}

static esp_err_t save_handler(httpd_req_t *req) {
    char buf[128];
    int len = httpd_req_recv(req, buf, sizeof(buf)-1);
    buf[len] = 0;
    
    // 解析POST数据：ssid=XXX&pass=YYY
    char ssid[33] = {0}, pass[65] = {0};
    // 简化解析（生产代码用URL decode）
    sscanf(buf, "ssid=%32[^&]&pass=%64s", ssid, pass);
    
    printf("WiFi配置: SSID=%s PASS=%s\n", ssid, pass);
    // TODO: 保存到NVS，然后连接WiFi + 重启
    
    httpd_resp_sendstr(req, "OK! Rebooting...");
    vTaskDelay(pdMS_TO_TICKS(1000));
    esp_restart();
    return ESP_OK;
}

void start_webserver(void) {
    httpd_config_t cfg = HTTPD_DEFAULT_CONFIG();
    httpd_handle_t server = NULL;
    httpd_start(&server, &cfg);
    
    httpd_uri_t root = {.uri="/", .method=HTTP_GET, .handler=root_handler};
    httpd_uri_t save = {.uri="/save", .method=HTTP_POST, .handler=save_handler};
    httpd_register_uri_handler(server, &root);
    httpd_register_uri_handler(server, &save);
    
    printf("WebServer started on http://%s\n", "192.168.4.1");
}

🌐 MQTT物联网通信

MQTT是IoT领域的HTTP——所有云平台都支持它。比HTTP轻100倍，适合MCU。

// ESP-IDF MQTT客户端完整示例
#include "mqtt_client.h"

static esp_mqtt_client_handle_t mqtt_client;

static void mqtt_event_handler(void* arg, esp_event_base_t base,
    int32_t id, void* event_data) {
    esp_mqtt_event_handle_t ev = event_data;
    
    switch (ev->event_id) {
    case MQTT_EVENT_CONNECTED:
        ESP_LOGI("MQTT", "已连接!");
        esp_mqtt_client_subscribe(mqtt_client, "device/cmd", 1);
        esp_mqtt_client_publish(mqtt_client, "device/status", 
            "online", 0, 1, 0);
        break;
    
    case MQTT_EVENT_DATA:
        ESP_LOGI("MQTT", "收到: topic=%.*s data=%.*s",
            ev->topic_len, ev->topic, ev->data_len, ev->data);
        // 处理云端命令
        if (strncmp(ev->data, "reboot", 6) == 0) {
            esp_restart();
        } else if (strncmp(ev->data, "status", 6) == 0) {
            char buf[64];
            snprintf(buf, sizeof(buf),
                "{\"uptime\":%lld,\"free_heap\":%d}",
                esp_timer_get_time()/1000000, esp_get_free_heap_size());
            esp_mqtt_client_publish(mqtt_client, "device/status", 
                buf, 0, 1, 0);
        }
        break;
    
    case MQTT_EVENT_DISCONNECTED:
        ESP_LOGW("MQTT", "断开，自动重连...");
        break;
    }
}

void mqtt_init(void) {
    esp_mqtt_client_config_t cfg = {
        .broker.address.uri = "mqtt://broker.emqx.io:1883",  // 免费测试
        .credentials = {
            .username = "esp32-001",
            .client_id = "esp32-sensor-001",
        },
        .session.keepalive = 60,
    };
    mqtt_client = esp_mqtt_client_init(&cfg);
    esp_mqtt_client_register_event(mqtt_client, ESP_EVENT_ANY_ID,
        mqtt_event_handler, NULL);
    esp_mqtt_client_start(mqtt_client);
}

// 定时上报传感器数据（JSON格式）
void publish_sensor_data(float temp, float hum) {
    char payload[128];
    snprintf(payload, sizeof(payload),
        "{\"temp\":%.1f,\"hum\":%.1f,\"ts\":%lld,\"rssi\":%d}",
        temp, hum, esp_timer_get_time()/1000000, get_wifi_rssi());
    esp_mqtt_client_publish(mqtt_client, "sensor/data",
        payload, 0, 1, 0);  // QoS 1 = 至少送达一次
}

⚡ 双核编程

ESP32有两个Xtensa LX6核心。默认情况下，Arduino的loop()只跑在Core1，ESP-IDF让你自由分配任务到两个核心。

// ESP-IDF双核任务
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"

// 任务1：传感器采集（Core 0）
void sensor_task(void *pv) {
    while (1) {
        float temp = read_temperature();
        float hum = read_humidity();
        ESP_LOGI("Sensor", "Temp=%.1f Hum=%.1f", temp, hum);
        
        // 通过队列发送给Core 1处理
        sensor_data_t data = {.temp=temp, .hum=hum};
        xQueueSend(sensor_queue, &data, portMAX_DELAY);
        
        vTaskDelay(pdMS_TO_TICKS(5000));
    }
}

// 任务2：网络上报（Core 1）
void network_task(void *pv) {
    sensor_data_t data;
    while (1) {
        if (xQueueReceive(sensor_queue, &data, portMAX_DELAY) == pdTRUE) {
            publish_sensor_data(data.temp, data.hum);
            ESP_LOGI("Net", "Published: %.1f, %.1f", data.temp, data.hum);
        }
    }
}

// 任务3：LED闪烁（Core 1）
void led_task(void *pv) {
    while (1) {
        gpio_set_level(GPIO_NUM_2, 1);
        vTaskDelay(pdMS_TO_TICKS(500));
        gpio_set_level(GPIO_NUM_2, 0);
        vTaskDelay(pdMS_TO_TICKS(500));
    }
}

void app_main(void) {
    sensor_queue = xQueueCreate(10, sizeof(sensor_data_t));
    
    // 创建任务并指定核心
    xTaskCreatePinnedToCore(sensor_task,  "sensor", 4096, NULL, 1, NULL, 0);  // Core 0
    xTaskCreatePinnedToCore(network_task, "net",    4096, NULL, 2, NULL, 1);  // Core 1
    xTaskCreatePinnedToCore(led_task,     "led",    2048, NULL, 1, NULL, 1);  // Core 1
    
    // Core 0 跑传感器，Core 1 跑网络+LED
    // 两核通过队列通信，共享内存互不干扰
}
双核任务分配建议最佳实践
Core 0（Protocol Core）: 协议栈（WiFi/BLE/PHY）默认在这，适合：
• 传感器采集（定时、低CPU占用）
• 协议处理（WiFi事件、BLE事件）
• 中断服务函数

Core 1（Application Core）: 应用逻辑跑这，适合：
• 数据处理与计算
• HTTP/MQTT客户端
• LED显示、UI更新
• 文件系统操作

⚠ 不指定核心时（xTaskCreate）：FreeRTOS会自动分配到负载较轻的核心

GPIO基础操作

// ESP-IDF GPIO API
#include "driver/gpio.h"

void gpio_example(void) {
    // 配置GPIO2为输出
    gpio_config_t io_conf = {
        .pin_bit_mask = (1ULL << GPIO_NUM_2),
        .mode = GPIO_MODE_OUTPUT,
        .pull_up_en = GPIO_PULLUP_DISABLE,
        .pull_down_en = GPIO_PULLDOWN_DISABLE,
        .intr_type = GPIO_INTR_DISABLE,
    };
    gpio_config(&io_conf);
    
    // 控制输出
    gpio_set_level(GPIO_NUM_2, 1);  // 高电平
    gpio_set_level(GPIO_NUM_2, 0);  // 低电平
    
    // 配置GPIO0为输入（带按键中断）
    io_conf.pin_bit_mask = (1ULL << GPIO_NUM_0);
    io_conf.mode = GPIO_MODE_INPUT;
    io_conf.pull_up_en = GPIO_PULLUP_ENABLE;  // 内部上拉
    io_conf.intr_type = GPIO_INTR_NEGEDGE;    // 下降沿触发
    gpio_config(&io_conf);
    
    // 安装GPIO中断服务
    gpio_install_isr_service(0);
    gpio_isr_handler_add(GPIO_NUM_0, button_isr_handler, NULL);
}

// 按键中断处理
static void IRAM_ATTR button_isr_handler(void *arg) {
    // ISR中不要做耗时操作，给任务发信号
    xTaskNotifyFromISR(task_handle, 0, eNoAction, NULL);
}

        vTaskDelay(pdMS_TO_TICKS(100));
        retry++;
    }
    return sntp_get_sync_status() == SNTP_SYNC_STATUS_COMPLETED;
}

static void ws_event_handler(void *arg, esp_event_base_t base,
    int32_t id, void *event_data) {
    esp_websocket_event_data_t *data = event_data;

    case WEBSOCKET_EVENT_CONNECTED:
        ESP_LOGI("WS", "已连接");
        esp_websocket_client_send_text(client, "Hello from ESP32!", 
            strlen("Hello from ESP32!"), pdMS_TO_TICKS(1000));

    case WEBSOCKET_EVENT_DATA:

            gpio_set_level(GPIO_NUM_2, 0);
        }
        break;
    
    case WEBSOCKET_EVENT_DISCONNECTED:
        ESP_LOGW("WS", "断开");
        break;
    }
}

}

🐛 常见问题排查

🔍 问题：编译报错 "section `.dram0.bss' will not fit in region `dram0_0_seg'"

❌ 现象：链接时RAM溢出
✅ 解决：
1. 减少全局变量（用动态分配替代）
2. 修改menuconfig：增大Main task stack → 但可能还是不够
3. 将const数据改为PROGMEM存储到Flash
4. 用PSRAM（ESP32-S3/WROVER模块支持外部PSRAM）
5. 检查是否有未释放的malloc
🔍 问题：WiFi连接后ADC读数不准

❌ 现象：开启WiFi后ADC2通道读数异常
✅ 原因：WiFi驱动占用了ADC2！这是ESP32的硬件设计限制
✅ 解决：WiFi使用时只用ADC1（GPIO32-39），ADC2留给不用WiFi时的场景
🔍 问题：OTA升级后设备变砖

❌ 现象：固件更新后无法启动
✅ 预防（工业级方案）：
1. 使用OTA双分区（ota_0/ota_1），升级失败自动回滚
2. 在bootloader中检查固件签名和CRC
3. 保留factory分区作为最后的回退
4. 上电后检查启动计数：超过N次启动失败自动回滚到factory
🔍 问题：编译太慢（每次几分钟）

✅ 加速方案：
1. 并行编译：idf.py -j8 build（设为CPU核心数）
2. 安装ccache：brew install ccache(Mac) / sudo apt install ccache(Linux)
3. 在menuconfig中 → Compiler options → 勾选"Enable Ccache"
4. 二次编译几乎瞬间完成！
🔍 问题：空闲时频繁重启（看门狗复位）

❌ 现象：设备运行一段时间后自动重启
✅ 原因：任务看门狗(TWDT)超时——某个任务长时间不释放CPU
✅ 排查：
1. 检查loop()或任务中是否有死循环或长时间阻塞
2. 用vTaskDelay()或delay()让出CPU
3. 在menuconfig中增大TWDT超时时间
4. 或者在长循环中手动喂狗: esp_task_wdt_reset()

📡 ESP-NOW对等通信协议

ESP-NOW是乐鑫自有的低功耗对等无线通信协议，不依赖WiFi路由器——两个ESP32可以直接通信，类似无线串口。延迟低至2ms，非常适合传感器网络和遥控场景。

// ESP-NOW发送端（传感器节点）
#include "esp_now.h"
#include "esp_wifi.h"

typedef struct {
    float temperature;
    float humidity;
    uint32_t timestamp;
} sensor_data_t;

void espnow_sender_init(void) {
    // ① 初始化WiFi（ESP-NOW基于WiFi，但不需要连路由）
    wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
    esp_wifi_init(&cfg);
    esp_wifi_set_mode(WIFI_MODE_STA);  // Station模式即可
    esp_wifi_start();
    
    // ② 初始化ESP-NOW
    esp_now_init();
    
    // ③ 注册对端（接收方的MAC地址）
    esp_now_peer_info_t peer = {
        .peer_addr = {0xAA, 0xBB, 0xCC, 0xDD, 0xEE, 0xFF},  // 目标MAC
        .channel = 0,  // 0=使用当前信道
        .encrypt = false,
    };
    esp_now_add_peer(&peer);
}

void send_sensor_data(float temp, float hum) {
    sensor_data_t data = {temp, hum, esp_log_timestamp()};
    esp_now_send(receiver_mac, (uint8_t*)&data, sizeof(data));
    // ★ 不需要连接、不需要握手、不需要ACK（可选）
    // 延迟 ~2ms，比建立TCP连接（~200ms）快100倍
}

💾 NVS非易失存储深入

NVS（Non-Volatile Storage）是ESP-IDF内置的键值存储系统，基于SPI Flash，提供磨损均衡和掉电保护。比裸Flash API好用得多。

// NVS API —— 比直接操作Flash简单100倍
#include "nvs_flash.h"
#include "nvs.h"

void nvs_example(void) {
    // ① 初始化NVS（只需一次）
    esp_err_t err = nvs_flash_init();
    if (err == ESP_ERR_NVS_NO_FREE_PAGES) {
        // Flash空间不够→擦除→重试
        nvs_flash_erase();
        nvs_flash_init();
    }
    
    // ② 打开命名空间（类似文件夹，不同模块不互相干扰）
    nvs_handle_t handle;
    nvs_open("config", NVS_READWRITE, &handle);
    
    // ③ 读写操作（key-value，自动类型检查）
    // 写
    nvs_set_i32(handle, "wifi_channel", 6);
    nvs_set_str(handle, "device_name", "sensor_01");
    nvs_set_blob(handle, "calib_data", cal_array, sizeof(cal_array));
    
    // 读（带默认值，不存在则返回默认值）
    int32_t channel = 6;  // 默认值
    nvs_get_i32(handle, "wifi_channel", &channel);
    
    char name[32] = {0};
    size_t len = sizeof(name);
    nvs_get_str(handle, "device_name", name, &len);
    
    // ④ 一定要commit！（不commit不落盘，断电丢失）
    nvs_commit(handle);
    
    // ⑤ 关闭
    nvs_close(handle);
}

// ★ NVS踩坑经验：
// 1. nvs_set_xxx后必须nvs_commit！否则断电丢数据
// 2. 命名空间长度≤15字符
// 3. key长度≤15字符
// 4. blob最大508000字节（但Flash空间有限）
// 5. 频繁写入注意磨损：NVS有磨损均衡，但写同一key太频繁会集中磨损
//    → 解决：用递增key（log_001, log_002...）分散写

🧠 ESP32-S3 PSRAM配置与使用

ESP32-S3支持外挂8MB/16MB PSRAM（Octal SPI接口），把可用内存从512KB扩展到8MB+。做图像处理、大缓存、多任务必须用它。

# ★ 启用PSRAM（menuconfig或sdkconfig）
# CONFIG_SPIRAM=y
# CONFIG_SPIRAM_MODE_OCT=y          # Octal模式（最高带宽）
# CONFIG_SPIRAM_SPEED_80M=y         # 80MHz时钟
# CONFIG_SPIRAM_USE_MALLOC=y        # ★ 关键！让malloc自动用PSRAM
# CONFIG_SPIRAM_ALLOW_BSS_SEG_EXTERNAL_MEMORY=y  # 大BSS段放PSRAM

// C代码中使用PSRAM（配置文件启用后，malloc自动用PSRAM！）
void psram_example(void) {
    // malloc 会自动从PSRAM分配（如果PSRAM可用且常规RAM不够）
    uint8_t *big_buf = malloc(2 * 1024 * 1024);  // 2MB缓冲区
    if (big_buf) {
        // 检查是否从PSRAM分配的
        if (esp_ptr_external_ram(big_buf)) {
            ESP_LOGI("MEM", "2MB allocated from PSRAM!");
        }
        free(big_buf);
    }
    
    // 强制从PSRAM分配（即使内部RAM够）
    uint8_t *psram_buf = heap_caps_malloc(512*1024, MALLOC_CAP_SPIRAM);
    
    // 强制从内部RAM（DMA会用到，PSRAM不能DMA）
    uint8_t *dma_buf = heap_caps_malloc(4096, MALLOC_CAP_DMA);
    // ★ 重要：PSRAM不能做DMA！DMA缓冲区必须用MALLOC_CAP_DMA
}

// ★ PSRAM性能影响：
// 访问延迟：内部RAM ~5ns,  PSRAM ~30-40ns（约6-8倍慢）
// 吞吐量：  内部RAM ~2GB/s, PSRAM ~200MB/s（80MHz Octal模式）
// 结论：热点数据（频繁访问）放内部RAM，冷数据（大缓冲）放PSRAM

⚡ WiFi省电与共存策略

WiFi电源管理模式重要
ESP32的WiFi功耗可达300mA，是整板耗电大户。合理配置省电模式能降到5mA以下。
// WiFi省电模式配置
// ① None: 不休眠，功耗~100-300mA，延迟最低
// ② Minimum Modem: 调制解调器间歇休眠，功耗~20-30mA
// ③ Maximum Modem: 调制解调器深度休眠，功耗~5-10mA，延迟+~100ms

esp_wifi_set_ps(WIFI_PS_MAX_MODEM);  // 最大省电模式

// ★ WiFi+BLE共存时的注意事项：
// ESP32只有一根天线，WiFi和BLE分时共享
// 如果同时用WiFi和BLE：
// 1. 协议栈自动时分复用（不需要手动切换）
// 2. BLE连接间隔影响WiFi速度（每30ms一个BLE窗口≈占用3% WiFi时间）
// 3. BLE广播频率高→WiFi延迟大→MQTT可能超时
// 4. 解决：设置BLE连接间隔 >100ms，降低WiFi影响
// 
// ESP32-S3有更好的共存设计（独立控制），影响更小

🔄 OTA抗变砖策略

OTA中最可怕的事：升级到一半断电→设备变砖。乐鑫的方案是双分区+回滚机制，即使升级失败也能自动退回到旧固件。

// ★ ESP-IDF OTA 双分区机制（默认启用）
// Flash分区表（partitions.csv）：
// factory,  app,  factory, 0x10000, 1M         ← 出厂固件（出厂时烧）
// ota_0,    app,  ota_0,   ,        1M         ← OTA分区0
// ota_1,    app,  ota_1,   ,        1M         ← OTA分区1
// ota_data, data, ota,     ,        0x2000     ← OTA状态存储
//
// 工作流程：
// ① 设备运行 ota_0
// ② 下载新固件 → 写入 ota_1
// ③ 校验通过 → 设置下次启动从 ota_1 启动 → 重启
// ④ 如果 ota_1 启动失败（连续N次）→ 自动回滚到 ota_0

// ★ OTA回滚配置（menuconfig）：
// CONFIG_BOOTLOADER_APP_ROLLBACK_ENABLE=y          # 启用回滚
// CONFIG_BOOTLOADER_APP_ANTI_ROLLBACK=y            # 防降级攻击
// CONFIG_BOOTLOADER_SKIP_VALIDATE_IN_DEEP_SLEEP=n  # 深度睡眠唤醒也要验证

void ota_with_rollback(void) {
    esp_ota_handle_t ota_handle;
    const esp_partition_t *update_partition = 
        esp_ota_get_next_update_partition(NULL);  // 获取空闲OTA分区
    
    // 开始OTA
    esp_ota_begin(update_partition, OTA_SIZE_UNKNOWN, &ota_handle);
    
    // 分块写入固件
    while (has_more_data()) {
        uint8_t chunk[4096];
        int len = download_chunk(chunk, sizeof(chunk));
        esp_ota_write(ota_handle, chunk, len);
    }
    
    // 完成OTA
    esp_ota_end(ota_handle);
    
    // 设置启动分区（但不马上升级，等APP确认运行正常后调用）
    esp_ota_set_boot_partition(update_partition);
    
    // ★ 重启前先记录"待验证"状态
    // APP启动后调用 esp_ota_mark_app_valid_cancel_rollback() 确认新固件OK
    // 如果APP崩溃/看门狗复位 且没调用该函数 → 回滚
    esp_restart();
}
⚠ OTA防变砖Checklist：
1. 永远保留 factory 分区（出厂固件，最后防线）
2. 启用 OTA 回滚（CONFIG_BOOTLOADER_APP_ROLLBACK_ENABLE=y）
3. 固件签名验证（防止恶意固件，启用 Secure Boot）
4. OTA下载过程加CRC/MD5校验（每块都要校验，不是整个文件）
5. 升级前检查电池电量（<20%拒绝升级）
6. 给用户"强制恢复出厂"的方式（物理按键组合进入factory分区）

📋 知识点回顾

✅ ESP32核心能力：
• WiFi双模：Station(连路由) + SoftAP(自建热点) + SmartConfig(APP配网)
• BLE：GAP(广播/扫描/连接) + GATT(Service/Characteristic/Descriptor)
• MQTT：IoT标准协议(QoS 0/1/2)，所有云平台通用
• HTTP Client/Server：Web配网页面 + REST API通信
• SPIFFS：轻量文件系统(存配置/证书/网页)
• OTA：远程固件升级，双分区保证升级安全
• 双核编程：Core0(协议)→Core1(应用) 通过队列通信
• 低功耗：Deep Sleep→Timer/GPIO唤醒→上报数据→继续睡

🛠 ESP-IDF核心命令：
• idf.py set-target esp32s3 — 切换芯片型号
• idf.py menuconfig — 图形化配置(必学!)
• idf.py build flash monitor — 一键编译烧录查看日志
• idf.py size-components — 查看各组件内存占用
• Ctrl+T Ctrl+E 在monitor中查看任务列表

⚠ 新手痛点TOP5：
• GPIO引脚因芯片型号而异，务必查datasheet
• WiFi和ADC2冲突——用WiFi时ADC只能用ADC1
• 编译慢？用ccache + idf.py -j8 并行编译
• ESP-IDF约3GB——硬盘预留空间
• 烧录失败先按BOOT键再按EN键进入下载模式
