第2章：CMake构建系统

📖 小白完全指南 · 2/14 | 阅读时间约4分钟

CMake不直接编译代码——它是"构建文件生成器"。你写CMakeLists.txt描述项目结构，CMake帮你生成Makefile，再用make编译。

💡 类比：CMake就像建筑设计师，你告诉他"我要3层楼、2个卧室"（CMakeLists.txt），他画出施工图纸（Makefile），施工队（make）按图纸盖房子（编译）。

CMakeLists.txt 最小模板

cmake_minimum_required(VERSION 3.20)
project(stm32_blink C ASM)

set(SOURCES src/main.c src/stm32f1xx_it.c src/system_stm32f1xx.c)

include_directories(inc CMSIS)

add_compile_options(-mcpu=cortex-m3 -mthumb -Wall -ffunction-sections -fdata-sections)

add_link_options(-mcpu=cortex-m3 -mthumb -Wl,--gc-sections
    -T${CMAKE_SOURCE_DIR}/STM32F103C8Tx.ld
    -Wl,-Map=${PROJECT_NAME}.map)

add_executable(${PROJECT_NAME}.elf ${SOURCES})

add_custom_command(TARGET ${PROJECT_NAME}.elf POST_BUILD
    COMMAND ${CMAKE_OBJCOPY} -O binary ${PROJECT_NAME}.elf ${PROJECT_NAME}.bin
    COMMAND ${CMAKE_OBJCOPY} -O ihex ${PROJECT_NAME}.elf ${PROJECT_NAME}.hex
    COMMAND ${CMAKE_SIZE} ${PROJECT_NAME}.elf)

CMake变量和作用域

CMake的核心是变量——几乎所有构建逻辑都靠变量传递。理解变量作用域是写复杂CMakeLists.txt的基础。

变量基础操作

# set() —— 设置变量
set(MY_VAR "hello")              # 普通变量
set(MY_LIST "a" "b" "c")        # 列表（空格或分号分隔）
set(MY_LIST a b c)               # 等价写法

# message() —— 打印变量（调试必备）
message(STATUS "MY_VAR = ${MY_VAR}")    # 状态信息
message(WARNING "检查这个值: ${MY_VAR}") # 警告（继续执行）
message(FATAL_ERROR "致命错误！")        # 终止配置

# list() —— 列表操作
list(APPEND MY_LIST "d" "e")     # 追加元素
list(LENGTH MY_LIST len)          # 获取长度 ${len}=5
list(GET MY_LIST 0 first)         # 获取第一个元素
list(REMOVE_ITEM MY_LIST "c")     # 删除元素

# string() —— 字符串操作
string(TIMESTAMP TODAY %Y-%m-%d)  # 当前日期
string(REPLACE "." "_" result ${CMAKE_SYSTEM_VERSION})

变量作用域陷阱

# CMake变量默认是函数作用域 —— 在函数里set的变量外面看不到！
function(my_func)
    set(LOCAL_VAR "只在函数内可见")
    # ${LOCAL_VAR} 在这里能用
endfunction()
my_func()
message(STATUS "${LOCAL_VAR}")  # ❌ 空的！

# 要让变量"穿透"出去，用 PARENT_SCOPE
function(my_func_out)
    set(OUT_VAR "父作用域可见" PARENT_SCOPE)
endfunction()
my_func_out()
message(STATUS "${OUT_VAR}")    # ✅ 能看到了

# macro() 没有独立作用域 —— 跟调用者共享变量
macro(my_macro)
    set(MACRO_VAR "随处可用")
endmacro()
my_macro()
message(STATUS "${MACRO_VAR}")  # ✅ macro没有作用域隔离
⚠️ 常见的坑：在 function() 里用 set() 修改变量，外面看不到——这是新手最容易困惑的CMake行为。解决：要么用 macro()，要么用 set(var value PARENT_SCOPE)。

多目录项目管理

真实嵌入式项目绝不会只有一个源文件目录。库、驱动、协议栈各有各的文件夹，用 add_subdirectory 组织：

实战项目结构

# 项目目录树
stm32-project/
├── CMakeLists.txt            # 根CMake（总指挥）
├── toolchain-arm.cmake
├── STM32F103C8Tx.ld
├── src/
│   ├── CMakeLists.txt        # 应用层
│   └── main.c
├── drivers/
│   ├── CMakeLists.txt        # 驱动层
│   ├── gpio/
│   │   ├── CMakeLists.txt    # GPIO驱动
│   │   ├── gpio.c
│   │   └── gpio.h
│   └── uart/
│       ├── CMakeLists.txt    # UART驱动
│       ├── uart.c
│       └── uart.h
└── lib/
    ├── CMakeLists.txt        # 第三方库
    └── tinyprintf/
        ├── CMakeLists.txt
        └── tinyprintf.c

各层CMakeLists.txt实现

# ── 根 CMakeLists.txt ──
cmake_minimum_required(VERSION 3.20)
project(stm32_demo C)

# 全局编译选项（子目录继承）
add_compile_options(
    -mcpu=cortex-m3 -mthumb -Wall -O2 -g3
    -ffunction-sections -fdata-sections
    -DSTM32F103C8T6
)

# 全局头文件路径
include_directories(inc)

# 递归加载子目录（顺序决定依赖关系）
add_subdirectory(lib)       # 先编译库
add_subdirectory(drivers)   # 再编译驱动（可能依赖库）
add_subdirectory(src)       # 最后编译主程序


# ── drivers/gpio/CMakeLists.txt ──
# 方式1：创建静态库（推荐——复用性强）
add_library(gpio STATIC
    gpio.c
)
target_include_directories(gpio PUBLIC .)  # gpio.h的路径
# PUBLIC = 链接gpio的目标也会获得这个头文件路径


# ── drivers/CMakeLists.txt ──
# 递归加载子驱动模块
add_subdirectory(gpio)
add_subdirectory(uart)


# ── lib/tinyprintf/CMakeLists.txt ──
add_library(tinyprintf STATIC
    tinyprintf.c
)
target_include_directories(tinyprintf PUBLIC .)


# ── lib/CMakeLists.txt ──
add_subdirectory(tinyprintf)


# ── src/CMakeLists.txt ──
add_executable(${PROJECT_NAME}.elf
    main.c
)

# 链接依赖库
target_link_libraries(${PROJECT_NAME}.elf
    gpio         # 来自 drivers/gpio
    uart         # 来自 drivers/uart
    tinyprintf   # 来自 lib/tinyprintf
)

# 链接选项（只给可执行文件加，不给库加）
target_link_options(${PROJECT_NAME}.elf PRIVATE
    -mcpu=cortex-m3 -mthumb
    -T${CMAKE_SOURCE_DIR}/STM32F103C8Tx.ld
    -Wl,--gc-sections
    -Wl,-Map=${PROJECT_NAME}.map
    --specs=nano.specs --specs=nosys.specs
)

# 后处理：生成.bin/.hex并显示大小
add_custom_command(TARGET ${PROJECT_NAME}.elf POST_BUILD
    COMMAND ${CMAKE_OBJCOPY} -O binary ${PROJECT_NAME}.elf ${PROJECT_NAME}.bin
    COMMAND ${CMAKE_OBJCOPY} -O ihex ${PROJECT_NAME}.elf ${PROJECT_NAME}.hex
    COMMAND ${CMAKE_SIZE} ${PROJECT_NAME}.elf
)
💡 关键概念：STATIC库 vs OBJECT库
• STATIC — 打包成 .a 文件，链接时只提取用到的符号（推荐）
• OBJECT — 编译成 .o 但不打包，跟项目一起链接（符号全部保留）
• INTERFACE — 只有头文件的库（header-only library）
嵌入式开发中，驱动层和库层用 STATIC 最合适。

CMake生成器表达式（Generator Expressions）

生成器表达式在生成构建文件时才求值——比普通变量更灵活，特别适合处理多配置场景（Debug/Release）和多平台构建。

# 基本语法：$<表达式>
# $<BOOL:expr>      → expr为真返回1，否则返回0
# $<IF:cond,true,false> → 条件表达式
# $<CONFIG>          → 当前构建类型（Debug/Release/MinSizeRel）
# $<TARGET_FILE:tgt> → 目标文件的完整路径

# === 嵌入式开发中的实际用法 ===

# ① 根据构建类型选择优化等级
target_compile_options(${TARGET}.elf PRIVATE
    $<$<CONFIG:Debug>:-O0>          # Debug=不优化
    $<$<CONFIG:Release>:-O2>        # Release=速度优化
    $<$<CONFIG:MinSizeRel>:-Os>     # MinSizeRel=体积优化
)

# ② 只在Debug模式加调试标志
target_compile_definitions(${TARGET}.elf PRIVATE
    $<$<CONFIG:Debug>:DEBUG=1>     # Debug时定义DEBUG宏
    $<$<CONFIG:Debug>:ENABLE_ASSERT> # Debug时开启断言
)

# ③ 根据目标名动态生成输出路径
add_custom_command(TARGET ${PROJECT_NAME}.elf POST_BUILD
    COMMAND ${CMAKE_OBJCOPY} -O binary
        $<TARGET_FILE:${PROJECT_NAME}.elf>
        $<TARGET_FILE_DIR:${PROJECT_NAME}.elf>/${PROJECT_NAME}.bin
)

自定义函数和宏

# 把重复逻辑封装成函数，提高可维护性

# 嵌入式源文件自动发现函数
function(collect_sources OUTPUT_VAR search_dir)
    file(GLOB_RECURSE sources
        ${search_dir}/*.c
        ${search_dir}/*.s
    )
    set(${OUTPUT_VAR} ${sources} PARENT_SCOPE)
endfunction()

# 使用
collect_sources(APP_SOURCES src/)
collect_sources(DRV_SOURCES drivers/)
add_executable(app.elf ${APP_SOURCES} ${DRV_SOURCES})


# 自动生成编译数据库（compile_commands.json）
function(enable_compile_commands)
    set(CMAKE_EXPORT_COMPILE_COMMANDS ON PARENT_SCOPE)
    message(STATUS "✓ 编译数据库已启用（用于clangd/IDE智能提示）")
endfunction()


# 快速创建嵌入式可执行目标
function(add_embedded_target name)
    add_executable(${name}.elf ${ARGN})
    target_compile_options(${name}.elf PRIVATE
        -mcpu=cortex-m3 -mthumb -Wall -O2 -g3
        -ffunction-sections -fdata-sections)
    target_link_options(${name}.elf PRIVATE
        -mcpu=cortex-m3 -mthumb
        -T${CMAKE_SOURCE_DIR}/STM32F103C8Tx.ld
        -Wl,--gc-sections -Wl,-Map=${name}.map
        --specs=nano.specs --specs=nosys.specs)
    add_custom_command(TARGET ${name}.elf POST_BUILD
        COMMAND ${CMAKE_OBJCOPY} -O binary ${name}.elf ${name}.bin
        COMMAND ${CMAKE_SIZE} ${name}.elf)
endfunction()

# 使用——一行代码搞定！（参数通过 ARGN 传入）
add_embedded_target(blink src/main.c src/stm32f1xx_it.c)

# src/main.c - 最小STM32程序（纯寄存器，无需HAL库）
// 定义寄存器地址
#define RCC_APB2ENR  (*(volatile uint32_t*)0x40021018)
#define GPIOC_CRH    (*(volatile uint32_t*)0x40011004)
#define GPIOC_ODR    (*(volatile uint32_t*)0x4001100C)

void delay(volatile uint32_t count) {
    while(count--);
}

int main(void) {
    // 使能GPIOC时钟
    RCC_APB2ENR |= (1 << 4);
    
    // 配置PC13为推挽输出（50MHz）
    GPIOC_CRH = (GPIOC_CRH & ~0x00F00000) | 0x00300000;
    
    while(1) {
        GPIOC_ODR ^= (1 << 13);  // 翻转PC13
        delay(500000);
    }
}

🐛 常见CMake错误排查

🔍 错误1：CMake Error: The source directory "..." does not appear to contain CMakeLists.txt
你在没有CMakeLists.txt的目录运行了cmake。确保当前目录（或 .. 指向的目录）有CMakeLists.txt。最常发生在你忘记了先 mkdir build && cd build。
🔍 错误2：add_library cannot create ALIAS target "xxx" because another target with the same name already exists
CMake目标名全局唯一。不同子目录的CMakeLists.txt里不能有同名的 add_library / add_executable。预防：给库名加模块前缀（如 drv_gpio 而非 gpio）。
🔍 错误3：Cannot specify link libraries for target "xxx" which is not built by this project
你在 target_link_libraries 中引用了不存在的target。检查：①拼写是否正确 ②add_subdirectory是否包含了定义该target的目录 ③顺序：先add_subdirectory再target_link_libraries。
🔍 错误4：arm-none-eabi-gcc: error: unrecognized command line option '-mthumb'
用了x86的gcc而非ARM GCC。确保：①指定了 toolchain-arm.cmake ②代码中写的是 arm-none-eabi-gcc 而非 gcc。先确认 arm-none-eabi-gcc --version 能执行。
🔍 错误5：CMake Warning: Manually-specified variables were not used by the project: CMAKE_TOOLCHAIN_FILE
CMake缓存中存了旧配置，-D参数被忽略。解决：清空build目录，从头重新cmake：
rm -r build && mkdir build && cd build && cmake -DCMAKE_TOOLCHAIN_FILE=../toolchain-arm.cmake ..

🏋️ 实战练习

💡 练习1：搭建多级目录CMake项目
创建一个至少3层目录的CMake嵌入式项目，要求：
① 根目录CMakeLists.txt定义全局编译选项
② drivers/ 下有至少2个子驱动（gpio, uart），每个编译为静态库
③ lib/ 下有一个第三方库
④ src/ 下的可执行文件链接所有库
⑤ 编译通过并生成 .bin 文件
验收标准：能在build目录下成功运行 cmake --build . 且无警告。
💡 练习2：CMake变量调试
在已有的CMakeLists.txt中加入 message() 输出以下变量值，确认你理解了每个变量的含义：
message(STATUS "源码目录: ${CMAKE_SOURCE_DIR}")
message(STATUS "构建目录: ${CMAKE_BINARY_DIR}")
message(STATUS "项目名: ${PROJECT_NAME}")
message(STATUS "C编译器: ${CMAKE_C_COMPILER}")
message(STATUS "构建类型: ${CMAKE_BUILD_TYPE}")
message(STATUS "工具链文件: ${CMAKE_TOOLCHAIN_FILE}")
运行 cmake 配置后观察输出，然后尝试修改 CMAKE_BUILD_TYPE 为 Release/MinSizeRel 看变化。
💡 练习3：Debug/Release双配置构建
用 CMake 生成器表达式实现：Debug模式编译时定义 DEBUG=1 宏并输出详细日志，Release模式编译时用 -O2 优化且不输出日志。分别构建两种模式并对比生成的 .elf 文件大小。

CMake Presets（预设配置）进阶
CMake 3.20+支持 Presets——把命令行参数写进JSON文件，彻底告别"复制粘贴cmake命令"的痛苦。VS Code的CMake Tools插件原生支持Presets，直接在UI里选配置。
// CMakePresets.json（放在项目根目录）
{
  "version": 6,
  "configurePresets": [
    {
      "name": "arm-debug",
      "displayName": "STM32F103 Debug",
      "generator": "Ninja",
      "binaryDir": "${sourceDir}/build/debug",
      "cacheVariables": {
        "CMAKE_TOOLCHAIN_FILE": "${sourceDir}/toolchain-arm.cmake",
        "CMAKE_BUILD_TYPE": "Debug"
      }
    },
    {
      "name": "arm-release",
      "displayName": "STM32F103 Release",

外部依赖管理：FetchContent vs CPM.cmake进阶
你的项目需要引用第三方库（如 FreeRTOS、LVGL）时，手动拷贝源码又累又难更新。CMake提供了几种自动化方案：
# 方案1：FetchContent（CMake 3.14+内置）
include(FetchContent)
FetchContent_Declare(
    FreeRTOS-Kernel
    GIT_REPOSITORY https://github.com/FreeRTOS/FreeRTOS-Kernel.git
    GIT_TAG V11.0.1
)
FetchContent_MakeAvailable(FreeRTOS-Kernel)
# 现在可以直接 target_link_libraries(app.elf FreeRTOS-Kernel)

# 方案2：CPM.cmake（更简洁，一个脚本文件搞定）
# 下载 CPM.cmake 到项目根目录，然后：
include(cmake/CPM.cmake)
CPMAddPackage(
    NAME tinyprintf
    GITHUB_REPOSITORY ccochard/tinyprintf
    VERSION 1.0
)

# 方案3：手动管理 git submodule（老派做法）
git submodule add https://github.com/FreeRTOS/FreeRTOS-Kernel.git lib/FreeRTOS
git submodule update --init --recursive
# 然后在CMakeLists.txt里 add_subdirectory(lib/FreeRTOS)
💡 推荐策略：小型项目用 FetchContent（零依赖），团队项目用 git submodule（版本锁定），复杂项目用 CPM.cmake（支持缓存、离线模式）。
CMake与Ninja：更快构建进阶
Ninja是比Make更快的构建工具——并行度更高、增量编译判断更精确。在Windows上配合CMake使用体验很好。
# 安装 Ninja（三选一）
winget install Ninja-build.Ninja    # winget
choco install ninja                  # Chocolatey
# 或从 github.com/ninja-build/ninja/releases 下载

# 使用 Ninja 构建（指定 -G 生成器）
cmake -G Ninja -DCMAKE_TOOLCHAIN_FILE=../toolchain-arm.cmake ..
cmake --build .          # Ninja自动并行，不需要 -j4

# 对比（实测4核8线程Windows PC，stm32-blink项目）
# Make:    3.2秒  (cmake --build . -j4)
# Ninja:   1.4秒  (ninja)
# 原因：Ninja原生并行+精确依赖追踪
💡 是否值得换Ninja：项目大于20个源文件时Ninja的优势就明显了；10个文件以下Make和Ninja差不多。建议开发期用Ninja，CI/CD用Make（兼容性好）。

⚙️ ARM GCC 编译选项完全手册

这些选项你在 toolchain-arm.cmake 里写过无数次，但你真的理解每个选项的含义和影响吗？选错一个可能导致程序静默崩溃。下面逐一拆解：

-mcpu：CPU核心选择（最重要！）

选项适用MCU系列ARM架构说明
-mcpu=cortex-m0STM32F0, STM32L0, nRF51822, RP2040ARMv6-M最小指令集，无硬件除法。选m0plus也可以
-mcpu=cortex-m3STM32F1, STM32F2, STM32L1ARMv7-M有硬件除法+位带操作，嵌入式最经典架构
-mcpu=cortex-m4STM32F3, STM32F4, nRF52832, TMC4C129ARMv7E-MM3基础上加DSP指令+可选FPU
-mcpu=cortex-m7STM32F7, STM32H7, i.MX RT, RP2350ARMv7E-M六级流水线+双精度FPU+CACHE，性能最强
-mcpu=cortex-m33STM32L5, STM32U5, nRF5340, LPC55S69ARMv8-MTrustZone安全扩展，M4的后继者
-mcpu=cortex-m0plusSTM32G0, STM32C0, RP2040(实际是M0+)ARMv6-MM0的增强版，单周期IO
⚠️ 选错 -mcpu 的后果：选了 cortex-m4 编译 STM32F103(M3内核)的代码有90%概率能跑（因为M4是M3的超集），但一旦用到M4专属指令（如 SMLAD），芯片立刻进 HardFault。反过来用 cortex-m3 编译STM32F407(M4)会浪费DSP指令集。永远以数据手册上的内核型号为准。

-mthumb：Thumb/Thumb2 指令集

Cortex-M 只支持Thumb指令集，不支持ARM 32-bit指令集。所以 -mthumb 是必须的，不是可选的。

有趣的历史：ARM 7/9/11时代（ARM7TDMI）有两种模式：ARM模式（32位指令，快但代码大）和Thumb模式（16位指令，省Flash但慢）。Cortex-M3开始的Thumb-2混合了16位和32位指令，性能接近ARM模式但代码密度远超ARM模式。这就是为什么Cortex-M完全抛弃了ARM模式。

-mfloat-abi 和 -mfpu：浮点选项（最容易配错！）

这三个设置必须保持一致，否则链接阶段会报一堆 uses VFP register arguments 错误：

MCU-mcpu-mfpu-mfloat-abi说明
STM32F103cortex-m3无（没有FPU）soft纯软件浮点，float运算编译成库函数调用
STM32F407cortex-m4fpv4-sp-d16hard单精度硬件FPU，float用硬件指令
STM32F407（软浮点兼容模式）cortex-m4fpv4-sp-d16softfp用硬件FPU但传参用整数寄存器，兼容无FPU的库
STM32F767cortex-m7fpv5-d16 / fpv5-sp-d16hard双精度+单精度FPU
STM32H743cortex-m7fpv5-sp-d16hard单精度FPU（H7系列部分型号无双精度）
⚡ 踩坑：链接时报 "uses VFP register arguments, output does not"
这是嵌入式编译最经典的浮点错误。根因是：你的 .c 文件用 -mfloat-abi=hard 编译，但链接的某个 .a 库是用 -mfloat-abi=soft 编译的。解决方案：
① 确认你的 toolchain 文件中 CMAKE_C_FLAGS、CMAKE_ASM_FLAGS、CMAKE_EXE_LINKER_FLAGS 的浮点选项完全一致
② 所有子目录库继承顶层 toolchain 的浮点设置（CMake 默认行为，但手动改过就可能不一致）
③ 如果用了第三方 .a 库且它是 soft 编译的，你也只能切到 softfp（用硬件指令但函数调用兼容 soft）

优化级别：-O0 / -Og / -O1 / -O2 / -Os / -O3 / -Ofast

选项编译时间代码体积运行速度调试体验适用场景
-O0最快最大最慢最好早期调试，单步时变量值可靠
-Og中等中等偏大中等好推荐：开发期调试，平衡调试和优化
-O1稍慢中等较快一般很少用，不上不下
-Os慢最小较快差Flash紧张时（如STM32F103C8只有64KB）
-O2慢较大快差推荐：Release版本，GCC优化最完整的级别
-O3最慢很大最快很差性能关键代码（DSP、图像处理）
-Ofast最慢不确定极快很差不推荐嵌入式：违反IEEE754，浮点结果可能不对
⚠️ -Os vs -O2 抉择：对64KB Flash的STM32F103，-Os 比 -O2 通常节省15-25%代码体积（~10KB），速度下降5-10%。如果你的代码快撑满Flash了，果断用 -Os。如果是256KB+的中高端单片机，优先用 -O2（更快、更少踩坑）。

📝 常用的 ARM GCC 编译选项清单

-ggdb                      # 生成GDB格式调试信息

# === 链接器选项 ===
-specs=nosys.specs         # 无系统调用（裸机必须）
-specs=nano.specs          # 使用newlib-nano（更小的C库）

-fdata-sections            # 每个数据一个section
-fno-common                # 禁止common段（避免重复定义静默放过）
-fomit-frame-pointer       # 省略帧指针（省一个寄存器，-O以上自动开）

-TSTM32F407VGTx_FLASH.ld   # 指定链接脚本
💡 -ffunction-sections + -Wl,--gc-sections 黄金组合：在.c文件里定义了10个函数但你只用了2个？加上这两个选项后，没用到的8个函数的代码不会出现在最终的.bin里。对于Flash紧张的MCU，这可能是省3-8KB的简单手段。HAL库不裁剪能占50KB+，裁剪后可能只要20KB。

📦 Post-Build：从 .elf 生成 .bin / .hex / .size

CMake编译产出的是 .elf 文件，但烧录器（ST-Link Utility、J-Flash）通常需要 .bin 或 .hex。用 add_custom_command 在编译后自动生成这些文件：

📝 完整的 Post-Build 配置（加到add_executable之后） 
# 生成 bin 文件（原始二进制，用于ST-Link/J-Link烧录）
add_custom_command(TARGET ${PROJECT_NAME}.elf POST_BUILD
    COMMAND ${CMAKE_OBJCOPY} -O binary $<TARGET_FILE:${PROJECT_NAME}.elf>
            ${CMAKE_BINARY_DIR}/${PROJECT_NAME}.bin
    COMMENT "生成 ${PROJECT_NAME}.bin"
)

# 生成 hex 文件（Intel HEX格式，用于串口烧录/兼容旧工具）
add_custom_command(TARGET ${PROJECT_NAME}.elf POST_BUILD
    COMMAND ${CMAKE_OBJCOPY} -O ihex $<TARGET_FILE:${PROJECT_NAME}.elf>
            ${CMAKE_BINARY_DIR}/${PROJECT_NAME}.hex
    COMMENT "生成 ${PROJECT_NAME}.hex"
)

# 打印内存占用（Flash/RAM）
add_custom_command(TARGET ${PROJECT_NAME}.elf POST_BUILD
    COMMAND ${CMAKE_SIZE} --format=berkeley $<TARGET_FILE:${PROJECT_NAME}.elf>
    COMMENT "内存占用："
)

# ===== 综合写法（推荐：用add_custom_target集中管理） =====
add_custom_target(bin ALL
    COMMAND ${CMAKE_OBJCOPY} -O binary $<TARGET_FILE:${PROJECT_NAME}.elf>
            ${CMAKE_BINARY_DIR}/${PROJECT_NAME}.bin
    COMMAND ${CMAKE_OBJCOPY} -O ihex $<TARGET_FILE:${PROJECT_NAME}.elf>
            ${CMAKE_BINARY_DIR}/${PROJECT_NAME}.hex
    COMMAND ${CMAKE_SIZE} --format=berkeley $<TARGET_FILE:${PROJECT_NAME}.elf>
    DEPENDS ${PROJECT_NAME}.elf
    COMMENT "生成 .bin .hex 并打印内存占用"
)

objcopy 常用参数：

命令说明输出
arm-none-eabi-objcopy -O binary input.elf output.bin生成原始二进制最小体积，无地址信息
arm-none-eabi-objcopy -O ihex input.elf output.hex生成Intel HEX格式含地址信息，文本格式可读
arm-none-eabi-objcopy -O srec input.elf output.s19生成Motorola S-Record汽车电子常用格式
arm-none-eabi-objdump -h input.elf查看所有section头调试用
arm-none-eabi-objdump -S input.elf反汇编（混合C源码）分析优化效果
arm-none-eabi-size -A input.elf按section打印详细大小精确定位哪个section大
💡 只生成 .bin 够不够？对于 STM32 开发，.bin 完全够用。用 ST-Link Utility 或 st-flash write firmware.bin 0x08000000 即可烧录。.hex 在以下场景仍然有用：① 用串口ISP烧录（FlyMCU等工具只支持.hex）；② 合并bootloader+app时需要hex的地址信息；③ 给工厂生产时部分烧录器只吃hex。

🐛 CMake 踩坑实录

🔍 坑1：cmake .. 后修改了CMakeLists.txt，make 没有重新生成Makefile

❌ 现象：修改 CMakeLists.txt（比如改了源文件列表），直接 cmake --build .，发现编译的还是旧的文件列表
✅ 根因：CMake 在配置阶段把 CMakeLists.txt 转换为 Makefile，Make 只看 Makefile 的时间戳。修改 CMakeLists.txt 后，CMake 知道需要重新配置，但 Make 不知道
✅ 解决：
① 每次改 CMakeLists.txt 后手动 cmake .. 重新配置
② 用 cmake --build . 而不是 make——cmake --build 会自动检测是否需要重新配置
③ 在 VS Code 中安装 CMake Tools 扩展，它会监听 CMakeLists.txt 变化自动重新配置
🔍 坑2：CMake 找不到交叉编译器

❌ 现象：cmake -DCMAKE_TOOLCHAIN_FILE=../toolchain-arm.cmake .. 报错：
The CMAKE_C_COMPILER: arm-none-eabi-gcc is not a full path
✅ 根因：CMake 的 toolchain 文件要求编译器路径是绝对路径，或者在 PATH 中可找到。如果ARM GCC刚装完还没加到PATH，CMake会报这个错
✅ 解决：
① Toolchain文件中用绝对路径（推荐）：
set(CMAKE_C_COMPILER "C:/Program Files (x86)/Arm GNU Toolchain arm-none-eabi/14.2/bin/arm-none-eabi-gcc.exe")
② 或者确保 arm-none-eabi-gcc 在PATH中（见第0章PATH配置）
③ 调试命令：在 build 目录执行 cmake -DCMAKE_TOOLCHAIN_FILE=../toolchain-arm.cmake .. --debug-trycompile 查看CMake尝试用哪个编译器
🔍 坑3：add_subdirectory 引入的库没有正确传递编译选项

❌ 现象：顶层 CMakeLists.txt 设置了 -Os -mcpu=cortex-m4 -mfloat-abi=hard，但子目录的 .c 文件似乎是用默认选项编译的
✅ 根因：子目录的 CMakeLists.txt 里重新设置了 CMAKE_C_FLAGS（覆盖而非追加），或者在子目录里没有使用 target_compile_options 而用了 set(CMAKE_C_FLAGS ...) 覆盖了顶层设置
✅ 解决：
① 子目录不要用 set(CMAKE_C_FLAGS ...)，改用 target_compile_options(my_lib PRIVATE -Wall)
② 用 add_library(my_lib STATIC ...) 而非直接 add_executable
③ 检查子目录是否正确继承了 toolchain 变量：在子目录 CMakeLists.txt 加 message(STATUS "C_FLAGS=${CMAKE_C_FLAGS}")
🔍 坑4：CMake 缓存了旧配置，改了 toolchain 文件不生效

❌ 现象：修改 toolchain-arm.cmake（比如改 -mcpu=cortex-m3 为 -mcpu=cortex-m4），再次 cmake .. 后编译选项没变
✅ 根因：CMakeCache.txt 缓存了首次配置时的编译器路径和选项。改 toolchain 文件不会自动清除缓存
✅ 解决：
① 简单粗暴：删除整个 build 目录，重新 mkdir build && cd build && cmake ..
② 只删缓存：删除 build/CMakeCache.txt 然后重新 cmake
③ CMake 3.24+ 支持：cmake --fresh .. —— 忽略所有缓存从头配置
④ 预防：在 CMakeLists.txt 顶部加 set(CMAKE_C_COMPILER_WORKS ON) 确保编译器检查通过
🔍 坑5：add_custom_command 在交叉编译时不运行

❌ 现象：Post-build 的 objcopy 命令在配置阶段就被执行了（而且是用的 x86 objcopy，不是 arm-none-eabi-objcopy），生成的 .bin 是空的
✅ 根因：用了 add_custom_command(OUTPUT ... COMMAND objcopy ...) 而不是 add_custom_command(TARGET ... POST_BUILD ...)。前者在配置阶段求值，后者在目标构建完成后才执行
✅ 解决：
① 使用 add_custom_command(TARGET xxx POST_BUILD ...)（注意关键字是 TARGET 不是 OUTPUT）
② POST_BUILD 命令要用 CMake 变量：
COMMAND ${CMAKE_OBJCOPY} ... 而不是 COMMAND arm-none-eabi-objcopy ...，CMake 会自动找到交叉编译器对应的 objcopy
③ 如果必须用 OUTPUT 形式，DEPENDS 要正确指定 .elf 文件

📝 调试 CMake 的五种方法

📦 CMake 与 STM32CubeMX 生成的代码集成

STM32CubeMX 默认生成 Makefile 项目，但也可以在生成器设置里选 CMake。如果你已经有一个Makefile版本的CubeMX项目，迁移到CMake可以这样做：

# STM32CubeMX项目迁移到CMake的标准结构
# CubeMX生成的文件放在 Core/ 和 Drivers/ 下
# 手动创建 CMakeLists.txt 在项目根目录

cmake_minimum_required(VERSION 3.20)
project(stm32-cubemx C ASM)

# === 交叉编译器 ===
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
# 假设 toolchain-arm.cmake 在项目根目录
# 构建命令: cmake -B build -DCMAKE_TOOLCHAIN_FILE=toolchain-arm.cmake

# === CubeMX 自动生成的文件 ===
set(CORE_SOURCES
    Core/Src/main.c
    Core/Src/stm32f1xx_it.c
    Core/Src/stm32f1xx_hal_msp.c
    Core/Src/system_stm32f1xx.c
)

set(STARTUP_SOURCE
    Core/Startup/startup_stm32f103c8tx.s
)

# === HAL 库源文件 ===
set(HAL_SOURCES
    Drivers/STM32F1xx_HAL_Driver/Src/stm32f1xx_hal.c
    Drivers/STM32F1xx_HAL_Driver/Src/stm32f1xx_hal_cortex.c
    Drivers/STM32F1xx_HAL_Driver/Src/stm32f1xx_hal_gpio.c
    Drivers/STM32F1xx_HAL_Driver/Src/stm32f1xx_hal_rcc.c
    Drivers/STM32F1xx_HAL_Driver/Src/stm32f1xx_hal_uart.c
    Drivers/STM32F1xx_HAL_Driver/Src/stm32f1xx_hal_dma.c
    # 需要什么外设就加什么（CubeMX会告诉你用了哪些）
)

# === 头文件路径（CubeMX在Inc目录下生成了.h文件） ===
set(INCLUDE_DIRS
    Core/Inc
    Drivers/STM32F1xx_HAL_Driver/Inc
    Drivers/CMSIS/Device/ST/STM32F1xx/Include
    Drivers/CMSIS/Include
)

# === 编译目标 ===
add_executable(${PROJECT_NAME}.elf
    ${CORE_SOURCES}
    ${STARTUP_SOURCE}
    ${HAL_SOURCES}
)

# CubeMX 在 stm32f1xx.h 中通过宏选择MCU型号
target_compile_definitions(${PROJECT_NAME}.elf PRIVATE
    STM32F103xB    # 对应 STM32F103C8T6（128KB Flash）
    USE_HAL_DRIVER
)

# === 链接脚本（CubeMX生成的.ld文件） ===
set(LINKER_SCRIPT ${CMAKE_SOURCE_DIR}/STM32F103C8Tx_FLASH.ld)
target_link_options(${PROJECT_NAME}.elf PRIVATE
    -T${LINKER_SCRIPT}
    -Wl,--gc-sections
    -Wl,-Map=${CMAKE_BINARY_DIR}/${PROJECT_NAME}.map
    --specs=nosys.specs --specs=nano.specs
)

# === Post-build ===
add_custom_command(TARGET ${PROJECT_NAME}.elf POST_BUILD
    COMMAND ${CMAKE_OBJCOPY} -O binary $<TARGET_FILE:${PROJECT_NAME}.elf>
            ${CMAKE_BINARY_DIR}/${PROJECT_NAME}.bin
    COMMAND ${CMAKE_SIZE} --format=berkeley $<TARGET_FILE:${PROJECT_NAME}.elf>
    COMMENT "生成固件 .bin 并打印内存占用"
)
💡 CubeMX + CMake 最佳工作流：
① 在 CubeMX 中配置外设 → 生成代码（选 Makefile 输出）
② 在 CubeMX 中改用 CMake 生成（Project Manager → Toolchain / IDE → CMake）也可以，但手动维护 CMakeLists.txt 更灵活
③ 关键：CubeMX 重新生成代码时只覆盖 Core/ 和 Drivers/ 目录，你的 CMakeLists.txt 在项目根目录，不会被覆盖
④ 新增外设时，CubeMX 会告诉你需要加哪些 HAL 源文件 → 手动加到 CMakeLists.txt 的 HAL_SOURCES 列表

🎯 多MCU变体管理：一份CMake支持多种芯片

产品线常需要同一个代码库支持不同的MCU型号（如STM32F103C8 vs STM32F103RC，只是Flash/RAM大小不同）。CMake可以通过变量来控制：

# 通过 -DMCU_MODEL=F103C8 参数选择目标MCU
# cmake -B build -DCMAKE_TOOLCHAIN_FILE=toolchain-arm.cmake -DMCU_MODEL=F103C8

if(NOT DEFINED MCU_MODEL)
    set(MCU_MODEL "F103C8" CACHE STRING "目标MCU型号: F103C8/F103RC/F103VE")
endif()

# === MCU型号 → 编译参数映射表 ===
if(MCU_MODEL STREQUAL "F103C8")
    set(MCU_CPU "cortex-m3")
    set(MCU_DEFINE "STM32F103xB")   # 中等密度 (64K Flash, 20K RAM)
    set(MCU_LD "STM32F103C8Tx_FLASH.ld")
    set(FLASH_SIZE "64K")
    set(RAM_SIZE "20K")
elseif(MCU_MODEL STREQUAL "F103RC")
    set(MCU_CPU "cortex-m3")
    set(MCU_DEFINE "STM32F103xC")   # 高密度 (256K Flash, 48K RAM)
    set(MCU_LD "STM32F103RCTx_FLASH.ld")
    set(FLASH_SIZE "256K")
    set(RAM_SIZE "48K")
elseif(MCU_MODEL STREQUAL "F103VE")
    set(MCU_CPU "cortex-m3")
    set(MCU_DEFINE "STM32F103xE")   # XL密度 (512K Flash, 64K RAM)
    set(MCU_LD "STM32F103VETx_FLASH.ld")
    set(FLASH_SIZE "512K")
    set(RAM_SIZE "64K")
else()
    message(FATAL_ERROR "不支持的MCU型号: ${MCU_MODEL}，支持: F103C8/F103RC/F103VE")
endif()

# === 应用到编译选项 ===
target_compile_options(${PROJECT_NAME}.elf PRIVATE
    -mcpu=${MCU_CPU}
    -mthumb
)
target_compile_definitions(${PROJECT_NAME}.elf PRIVATE ${MCU_DEFINE})

# 链接脚本也随MCU型号切换
set(LINKER_SCRIPT ${CMAKE_SOURCE_DIR}/ld/${MCU_LD})

# 连接器标志中加入断言：确保代码不超过Flash/RAM容量
target_link_options(${PROJECT_NAME}.elf PRIVATE
    -T${LINKER_SCRIPT}
    -Wl,--defsym=_FLASH_SIZE=${FLASH_SIZE}
    -Wl,--defsym=_RAM_SIZE=${RAM_SIZE}
)

message(STATUS "目标MCU: ${MCU_MODEL} (${FLASH_SIZE} Flash / ${RAM_SIZE} RAM)")
💡 CI/CD 集成：多MCU管理在CI流水线中尤为强大——同一个仓库，通过不同的 -DMCU_MODEL=xxx 参数，一次性编译出3个MCU型号的固件。配合 CMake Presets 可以进一步简化成 cmake --preset F103C8-release。

📋 知识点回顾

🎯 本章核心要点：

1. CMake本质：构建文件生成器——你写CMakeLists.txt，它生成Makefile/Ninja文件。

2. 最小项目三要素：
• cmake_minimum_required(VERSION 3.20) — CMake版本
• project(xxx) — 项目名
• add_executable(xxx.elf main.c) — 生成目标

3. 交叉编译关键：必须指定 toolchain-arm.cmake，告诉CMake用 arm-none-eabi-gcc 而非 x86 gcc。

4. 构建命令三步走：
• mkdir build && cd build — 隔离构建产物
• cmake -DCMAKE_TOOLCHAIN_FILE=../toolchain-arm.cmake .. — 配置
• cmake --build . -j4 — 编译

5. 多目录管理：用 add_subdirectory() 组织代码，驱动层做成 STATIC 库，主程序链接库。

6. 变量作用域：function()有独立作用域 → set()要加PARENT_SCOPE；macro()无独立作用域 → 更灵活但易污染。

7. 生成器表达式：$<$<CONFIG:Debug>:-O0> 在生成构建文件时求值，实现Debug/Release双配置。
