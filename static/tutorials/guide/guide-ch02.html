第2章：CMake构建系统

📖 小白完全指南 · 2/14 | 阅读时间约4分钟

CMake不直接编译代码——它是"构建文件生成器"。你写CMakeLists.txt描述项目结构，CMake帮你生成Makefile，再用make编译。

💡 类比：CMake就像建筑设计师，你告诉他"我要3层楼、2个卧室"（CMakeLists.txt），他画出施工图纸（Makefile），施工队（make）按图纸盖房子（编译）。

CMakeLists.txt 最小模板

cmake_minimum_required(VERSION 3.20)
project(stm32_blink C ASM)

set(SOURCES src/main.c src/stm32f1xx_it.c src/system_stm32f1xx.c)

include_directories(inc CMSIS)

add_compile_options(-mcpu=cortex-m3 -mthumb -Wall -ffunction-sections -fdata-sections)

add_link_options(-mcpu=cortex-m3 -mthumb -Wl,--gc-sections
    -T${CMAKE_SOURCE_DIR}/STM32F103C8Tx.ld
    -Wl,-Map=${PROJECT_NAME}.map)

add_executable(${PROJECT_NAME}.elf ${SOURCES})

add_custom_command(TARGET ${PROJECT_NAME}.elf POST_BUILD
    COMMAND ${CMAKE_OBJCOPY} -O binary ${PROJECT_NAME}.elf ${PROJECT_NAME}.bin
    COMMAND ${CMAKE_OBJCOPY} -O ihex ${PROJECT_NAME}.elf ${PROJECT_NAME}.hex
    COMMAND ${CMAKE_SIZE} ${PROJECT_NAME}.elf)

交叉编译工具链文件

# toolchain-arm.cmake
set(CMAKE_SYSTEM_NAME Generic)
set(CMAKE_SYSTEM_PROCESSOR ARM)

set(TOOLCHAIN_PREFIX arm-none-eabi-)
set(CMAKE_C_COMPILER ${TOOLCHAIN_PREFIX}gcc)
set(CMAKE_CXX_COMPILER ${TOOLCHAIN_PREFIX}g++)
set(CMAKE_ASM_COMPILER ${TOOLCHAIN_PREFIX}gcc)
set(CMAKE_AR ${TOOLCHAIN_PREFIX}ar)
set(CMAKE_OBJCOPY ${TOOLCHAIN_PREFIX}objcopy)
set(CMAKE_OBJDUMP ${TOOLCHAIN_PREFIX}objdump)
set(CMAKE_SIZE ${TOOLCHAIN_PREFIX}size)

set(CMAKE_TRY_COMPILE_TARGET_TYPE STATIC_LIBRARY)
set(CMAKE_FIND_ROOT_PATH_MODE_PROGRAM NEVER)
set(CMAKE_FIND_ROOT_PATH_MODE_LIBRARY ONLY)
set(CMAKE_FIND_ROOT_PATH_MODE_INCLUDE ONLY)

构建命令

# 创建build目录（保持源码干净）
mkdir build && cd build

# 配置（生成Makefile）
cmake -DCMAKE_TOOLCHAIN_FILE=../toolchain-arm.cmake -DCMAKE_BUILD_TYPE=Debug ..

# 编译（4线程并行）
cmake --build . -j4

# 清理
cmake --build . --target clean
🔍 常见问题：cmake找不到arm-none-eabi-gcc
工具链没加PATH。解决：把ARM GCC的bin目录加到PATH，或在toolchain文件写死路径：
set(CMAKE_C_COMPILER "C:/Program Files (x86)/Arm GNU Toolchain arm-none-eabi/14.2 rel1/bin/arm-none-eabi-gcc.exe")

🚀 5分钟快速上手：最小可编译项目

跟着做，从零到一个能编译的项目，只需5分钟：

步骤1：创建项目目录
mkdir stm32-blink && cd stm32-blink
mkdir src inc
步骤2：创建最小main.c（点灯程序，无需任何库）
# src/main.c - 最小STM32程序（纯寄存器，无需HAL库）
// 定义寄存器地址
#define RCC_APB2ENR  (*(volatile uint32_t*)0x40021018)
#define GPIOC_CRH    (*(volatile uint32_t*)0x40011004)
#define GPIOC_ODR    (*(volatile uint32_t*)0x4001100C)

void delay(volatile uint32_t count) {
    while(count--);
}

int main(void) {
    // 使能GPIOC时钟
    RCC_APB2ENR |= (1 << 4);
    
    // 配置PC13为推挽输出（50MHz）
    GPIOC_CRH = (GPIOC_CRH & ~0x00F00000) | 0x00300000;
    
    while(1) {
        GPIOC_ODR ^= (1 << 13);  // 翻转PC13
        delay(500000);
    }
}

ENTRY(Reset_Handler)

_estack = ORIGIN(RAM) + LENGTH(RAM);  /* 栈顶 = RAM末尾 */

MEMORY {
  FLASH (rx) : ORIGIN = 0x08000000, LENGTH = 64K
  RAM (rwx)  : ORIGIN = 0x20000000, LENGTH = 20K
}
SECTIONS {
  .isr_vector : {
    . = ALIGN(4);

  .text : { *(.text*) } > FLASH
  .rodata : { *(.rodata*) } > FLASH
  .data : { *(.data*) } > RAM AT > FLASH

}

步骤4：创建CMakeLists.txt
cmake_minimum_required(VERSION 3.20)
project(stm32_blink C)
add_compile_options(-mcpu=cortex-m3 -mthumb -Wall -O2 -g3 -ffunction-sections -fdata-sections)
add_link_options(-mcpu=cortex-m3 -mthumb -Wl,--gc-sections -T${CMAKE_SOURCE_DIR}/STM32F103C8Tx.ld --specs=nano.specs --specs=nosys.specs)
add_executable(blink.elf src/main.c)
add_custom_command(TARGET blink.elf POST_BUILD
    COMMAND ${CMAKE_OBJCOPY} -O binary blink.elf blink.bin
    COMMAND ${CMAKE_SIZE} blink.elf)
步骤4.5：创建工具链文件（保存为 toolchain-arm.cmake，放在项目根目录）
# toolchain-arm.cmake - 告诉CMake用ARM编译器
set(CMAKE_SYSTEM_NAME Generic)
set(CMAKE_SYSTEM_PROCESSOR ARM)
set(TOOLCHAIN_PREFIX arm-none-eabi-)
set(CMAKE_C_COMPILER ${TOOLCHAIN_PREFIX}gcc)
set(CMAKE_CXX_COMPILER ${TOOLCHAIN_PREFIX}g++)
set(CMAKE_ASM_COMPILER ${TOOLCHAIN_PREFIX}gcc)
set(CMAKE_AR ${TOOLCHAIN_PREFIX}ar)
set(CMAKE_OBJCOPY ${TOOLCHAIN_PREFIX}objcopy)
set(CMAKE_OBJDUMP ${TOOLCHAIN_PREFIX}objdump)
set(CMAKE_SIZE ${TOOLCHAIN_PREFIX}size)
set(CMAKE_TRY_COMPILE_TARGET_TYPE STATIC_LIBRARY)
set(CMAKE_FIND_ROOT_PATH_MODE_PROGRAM NEVER)
set(CMAKE_FIND_ROOT_PATH_MODE_LIBRARY ONLY)
set(CMAKE_FIND_ROOT_PATH_MODE_INCLUDE ONLY)
步骤5：构建！
mkdir build && cd build
cmake -DCMAKE_TOOLCHAIN_FILE=../toolchain-arm.cmake ..
cmake --build .
