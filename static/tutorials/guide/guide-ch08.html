第8章：FreeRTOS入门

📖 小白完全指南 · 8/14 | 阅读时间约3分钟

裸机程序（super loop）就像一个人干所有事——烧水、切菜、炒菜、摆盘，只能一件一件来。RTOS让多个任务同时运行（实际是快速切换），就像雇了多个厨师各司其职。

💡 类比：RTOS就像餐厅后厨——任务(Task)=厨师（各干各的活），队列(Queue)=传菜窗口（任务间传递数据），信号量(Semaphore)=叫号器（通知"饭好了"），互斥量(Mutex)=独占灶台（同一时间只有一个厨师用）。

任务创建与管理

#include "FreeRTOS.h"
#include "task.h"

// 任务1：LED闪烁
void vLEDTask(void *pvParameters) {
    while (1) {
        HAL_GPIO_TogglePin(GPIOC, GPIO_PIN_13);
        vTaskDelay(pdMS_TO_TICKS(500));  // 延时500ms
    }
}

// 任务2：串口打印
void vPrintTask(void *pvParameters) {
    while (1) {
        printf("System running, tick: %lu
", xTaskGetTickCount());
        vTaskDelay(pdMS_TO_TICKS(1000));
    }
}

int main(void) {
    HAL_Init();
    SystemClock_Config();

    // 创建任务（参数：函数、名称、栈大小、参数、优先级、句柄）
    xTaskCreate(vLEDTask,    "LED",   128, NULL, 1, NULL);
    xTaskCreate(vPrintTask,  "Print", 256, NULL, 2, NULL);

    // 启动调度器
    vTaskStartScheduler();

    while (1) {}  // 永远不会到这里
}

队列通信

// 队列：任务间传递数据的"管道"
QueueHandle_t xTempQueue;

void vSensorTask(void *p) {
    float temp;
    while (1) {
        temp = read_temperature();
        // 发送到队列（等待最多10ms）
        xQueueSend(xTempQueue, &temp, pdMS_TO_TICKS(10));
        vTaskDelay(pdMS_TO_TICKS(100));
    }
}

void vDisplayTask(void *p) {
    float temp;
    while (1) {
        // 从队列接收（等待最多100ms）
        if (xQueueReceive(xTempQueue, &temp, pdMS_TO_TICKS(100))) {
            printf("Temp: %.1f C
", temp);
        }
    }
}

// 创建队列（容量5，每项sizeof(float)字节）
xTempQueue = xQueueCreate(5, sizeof(float));

信号量与互斥量

// 二值信号量：通知"事件发生了"
SemaphoreHandle_t xButtonSem;

// ISR中释放信号量
void HAL_GPIO_EXTI_Callback(uint16_t pin) {
    if (pin == BUTTON_PIN) {
        BaseType_t xHigherPriorityTaskWoken = pdFALSE;
        xSemaphoreGiveFromISR(xButtonSem, &xHigherPriorityTaskWoken);
        portYIELD_FROM_ISR(xHigherPriorityTaskWoken);
    }
}

// 任务中等待信号量
void vButtonTask(void *p) {
    while (1) {
        if (xSemaphoreTake(xButtonSem, portMAX_DELAY)) {
            printf("Button pressed!\n");
        }
    }
}

// 互斥量：保护共享资源
SemaphoreHandle_t xUartMutex;

void task1(void *p) {
    while (1) {
        if (xSemaphoreTake(xUartMutex, pdMS_TO_TICKS(100))) {
            printf("Task1: data=%d
", sensor1_read());
            xSemaphoreGive(xUartMutex);
        }
        vTaskDelay(pdMS_TO_TICKS(200));
    }
}

// 创建
xButtonSem = xSemaphoreCreateBinary();
xUartMutex  = xSemaphoreCreateMutex();

软件定时器

// 周期性定时器（每5秒喂狗）
void vFeedDogCallback(TimerHandle_t xTimer) {
    HAL_GPIO_WritePin(GPIOB, WDI_PIN, GPIO_PIN_SET);
    HAL_Delay(1);
    HAL_GPIO_WritePin(GPIOB, WDI_PIN, GPIO_PIN_RESET);
}

TimerHandle_t xWdTimer = xTimerCreate(
    "Watchdog",              // 名称
    pdMS_TO_TICKS(5000),     // 周期5秒
    pdTRUE,                  // 自动重载
    NULL,                    // 参数
    vFeedDogCallback         // 回调函数
);
xTimerStart(xWdTimer, 0);
🔍 常见问题：任务栈溢出
症状：程序跑飞、HardFault、变量莫名其妙变化。
排查：uxTaskGetStackHighWaterMark(NULL) 检查剩余栈空间，如果接近0就增大栈大小。
预防：创建任务时给足栈空间（printf需要至少256字节），开启 configCHECK_FOR_STACK_OVERFLOW。

FreeRTOS移植到STM32

上面的代码假设FreeRTOS已经在项目中。实际需要以下步骤：

步骤1：下载FreeRTOS源码：github.com/FreeRTOS/FreeRTOS（或从 freertos.org 下载）
步骤2：拷贝核心文件到项目：
FreeRTOS/Source/      → 复制到你的项目
├── tasks.c           ← 必须
├── queue.c           ← 必须
├── list.c            ← 必须
├── timers.c          ← 可选（软件定时器）
├── event_groups.c    ← 可选（事件组）
├── include/          ← 头文件目录
├── portable/
│   ├── MemMang/heap_4.c  ← 必须（内存管理方案4）
│   └── GCC/ARM_CM3/      ← 必须（Cortex-M3移植文件）
│       ├── port.c
│       └── portmacro.h
步骤3：创建 FreeRTOSConfig.h（配置文件，关键参数）：
#define configUSE_PREEMPTION        1    // 抢占式调度
#define configCPU_CLOCK_HZ          72000000UL  // STM32F103主频72MHz
#define configTICK_RATE_HZ          1000  // Tick频率1kHz（1ms）
#define configMAX_PRIORITIES        5     // 最大优先级数
#define configMINIMAL_STACK_SIZE    128   // 最小栈大小（单位：word）
#define configTOTAL_HEAP_SIZE       (15 * 1024)  // 堆大小15KB
#define configSUPPORT_STATIC_ALLOCATION  0
#define configSUPPORT_DYNAMIC_ALLOCATION 1
#define configUSE_MUTEXES           1
#define configUSE_COUNTING_SEMAPHORES  1

// 钩子函数
#define configCHECK_FOR_STACK_OVERFLOW  2    // 栈溢出检测等级2
#define configUSE_IDLE_HOOK          0
#define configUSE_TICK_HOOK          0

// 中断优先级（Cortex-M特有）
#define configKERNEL_INTERRUPT_PRIORITY         (5 << 4)  // 优先级5
#define configMAX_SYSCALL_INTERRUPT_PRIORITY    (3 << 4)  // 优先级3

// 需要实现的函数
void vApplicationStackOverflowHook(TaskHandle_t, char*);
void vApplicationMallocFailedHook(void);
步骤4：修改 stm32f1xx_it.c，把SysTick中断改为调用FreeRTOS：
// 替换原有的 SysTick_Handler
void SysTick_Handler(void) {
    if (xTaskGetSchedulerState() != taskSCHEDULER_NOT_STARTED) {
        xPortSysTickHandler();  // FreeRTOS的Tick处理
    }
    HAL_IncTick();  // 保留HAL库的Tick
}
⚠️ HAL_Delay与FreeRTOS冲突：HAL_Delay用轮询方式等待SysTick，在FreeRTOS任务中调用会阻塞整个系统！解决方案：
① 在任务中用 vTaskDelay(pdMS_TO_TICKS(ms)) 代替 HAL_Delay(ms)
② 在中断中可以用HAL_Delay（但不推荐，中断应尽快返回）
③ 如果必须用HAL_Delay，确保调用它的代码不在FreeRTOS任务内（如SystemClock_Config等初始化函数）
步骤5：在CMakeLists.txt/Makefile中添加源文件，编译即可
💡 更简单的方式：CubeMX可以直接添加FreeRTOS！Middleware → FREERTOS → Interface: CMSIS_V1，生成代码已包含FreeRTOS。PlatformIO也支持 lib_deps = FreeRTOS。
