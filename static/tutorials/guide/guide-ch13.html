🧵 状态机设计

💡 类比：状态机就像红绿灯——红灯→绿灯→黄灯→红灯，每个状态有明确的进入条件和持续时间，永远不会乱。用状态机管理复杂系统行为，代码清晰100倍。
// 智能花盆控制器：有限状态机完整实现
typedef enum {
    STATE_INIT,           // 初始化
    STATE_IDLE,           // 空闲（等待定时）
    STATE_MEASURING,      // 正在测量土壤湿度
    STATE_WATERING,       // 浇水中
    STATE_LOW_POWER,      // 低功耗待机（夜间）
    STATE_FAULT,          // 故障状态
    STATE_OTA             // OTA升级模式
} SystemState_t;

typedef struct {
    SystemState_t current;
    SystemState_t previous;
    uint32_t enter_time;  // 进入当前状态的时间戳
    uint32_t duration;    // 当前状态持续时间
    uint8_t  error_count; // 连续错误计数
} StateMachine_t;

StateMachine_t sm = {STATE_INIT, STATE_INIT, 0, 0, 0};

void state_machine_init(void) {
    sm.current = STATE_INIT;
    sm.enter_time = HAL_GetTick();
}

void state_machine_run(void) {
    sm.duration = HAL_GetTick() - sm.enter_time;
    
    switch (sm.current) {
    // ========== INIT状态 ==========
    case STATE_INIT:
        if (sensors_init() && pump_init() && wifi_init()) {
            transition_to(STATE_IDLE);
        } else {
            transition_to(STATE_FAULT);
        }
        break;
        
    // ========== IDLE状态 ==========
    case STATE_IDLE:
        if (is_time_to_measure(sm.duration)) {
            transition_to(STATE_MEASURING);
        } else if (is_night_time()) {
            transition_to(STATE_LOW_POWER);
        } else if (wifi_check_update()) {
            transition_to(STATE_OTA);
        }
        break;
        
    // ========== MEASURING状态 ==========
    case STATE_MEASURING: {
        float moisture = read_soil_moisture();
        if (moisture < 0) {  // 传感器故障
            sm.error_count++;
            if (sm.error_count > 3) {
                transition_to(STATE_FAULT);
            } else {
                transition_to(STATE_IDLE);
            }
        } else {
            sm.error_count = 0;
            upload_data(moisture);
            if (moisture < MOISTURE_THRESHOLD_DRY) {
                transition_to(STATE_WATERING);
            } else {
                transition_to(STATE_IDLE);
            }
        }
        break;
    }
        
    // ========== WATERING状态 ==========
    case STATE_WATERING:
        pump_start();
        if (read_soil_moisture() >= MOISTURE_THRESHOLD_WET 
            || sm.duration > MAX_WATER_TIME_MS) {
            pump_stop();
            transition_to(STATE_IDLE);
        } else if (pump_fault_detected()) {
            pump_stop();
            transition_to(STATE_FAULT);
        }
        break;
        
    // ========== LOW_POWER状态 ==========
    case STATE_LOW_POWER:
        if (is_day_time() || is_time_to_measure(sm.duration)) {
            wake_from_low_power();
            transition_to(STATE_IDLE);
        } else {
            enter_stop_mode();  // 深度睡眠
        }
        break;
        
    // ========== FAULT状态 ==========
    case STATE_FAULT:
        led_set_mode(0, LED_MODE_BLINK_FAST);  // 红色LED快闪告警
        log_error("System fault, error_count=%d", sm.error_count);
        if (fault_cleared()) {
            sm.error_count = 0;
            transition_to(STATE_INIT);  // 从头初始化
        }
        break;
        
    // ========== OTA状态 ==========
    case STATE_OTA:
        led_set_mode(0, LED_MODE_BLINK_SLOW);
        if (ota_perform_update()) {
            NVIC_SystemReset();  // 更新完成，重启
        } else {
            transition_to(STATE_IDLE);  // 更新失败，回退
        }
        break;
    }
}

// 状态迁移（记录时间戳便于超时检测）
void transition_to(SystemState_t new_state) {
    sm.previous   = sm.current;
    sm.current    = new_state;
    sm.enter_time = HAL_GetTick();
    printf("State: %d → %d\r\n", sm.previous, sm.current);
}

📊 RTOS任务优先级设计

⚠ 优先级规划是RTOS项目成功的关键。优先级设置不当会导致：高优先级饿死低优先级 / 优先级反转 / 实时性不满足。
// 任务优先级分配原则（FreeRTOS中数字越大优先级越高）
// ┌──────┬──────┬───────────────────────────────────┐
// │ 优先级 │ 任务  │ 说明                              │
// ├──────┼──────┼───────────────────────────────────┤
// │  3   │ 控制  │ 电机控制、PID、紧急停机(硬实时)      │
// │  2   │ 通信  │ WiFi/BLE协议栈、MQTT收发            │
// │  2   │ 采集  │ ADC采样+DMA(1KHz)、传感器中断        │
// │  1   │ 显示  │ OLED/LCD刷新(10-30Hz)、GUI          │
// │  1   │ 存储  │ SD卡写入、FATFS、数据日志           │
// │  0   │ 监控  │ 心跳LED、看门狗喂狗、温控监控       │
// │  —   │ IDLE  │ 系统空闲任务（永远就绪、永不阻塞）  │
// └──────┴──────┴───────────────────────────────────┘

// 关键规则：
// 1. 硬实时任务给最高优先级（控制/安全）
// 2. 计算密集型任务给低优先级（避免抢占交互性任务）
// 3. "运行→阻塞→运行"的周期性任务给中高优先级
// 4. 永不阻塞的纯CPU任务给最低优先级
// 5. 同功能模块任务给相同或相邻优先级，避免内部互相抢占

// 编译优化：发布版设置 CURRENT_LOG_LEVEL = LOG_WARN
// LOG_D和LOG_I全部编译为 ((void)0)，零开销！
// 带时间戳的日志（更专业的版本）
#define LOG_TS(fmt, ...) \
    printf("[%08lu] " fmt "\r\n", HAL_GetTick(), ##__VA_ARGS__)

// 环形日志缓冲区（出问题后回溯）
#define LOG_RING_SIZE 64
typedef struct {
    char entries[LOG_RING_SIZE][128];
    uint8_t head;
    uint8_t count;
} LogRing_t;

LogRing_t log_ring;

void log_ring_push(const char *fmt, ...) {
    va_list args;
    va_start(args, fmt);
    vsnprintf(log_ring.entries[log_ring.head], 128, fmt, args);
    va_end(args);
    
    log_ring.head = (log_ring.head + 1) % LOG_RING_SIZE;
    if (log_ring.count < LOG_RING_SIZE) log_ring.count++;
}

void log_ring_dump(void) {
    printf("=== Log Ring Buffer (last %d entries) ===\r\n", log_ring.count);
    for (int i = 0; i < log_ring.count; i++) {
        uint8_t idx = (log_ring.head - log_ring.count + i + LOG_RING_SIZE) % LOG_RING_SIZE;
        printf("%s\r\n", log_ring.entries[idx]);
    }
}
// HardFault时调用log_ring_dump()，查看崩溃前发生了什么

🧠 启动文件(startup)说明

启动文件用汇编编写，做三件事：①初始化堆栈指针 ②初始化中断向量表 ③调用SystemInit()和main()

// startup_stm32f103xx.s 关键部分
  .section .isr_vector,"a",%progbits
  .word _estack              // 初始栈指针（RAM末尾）
  .word Reset_Handler         // 复位向量
  .word NMI_Handler
  .word HardFault_Handler
  .word MemManage_Handler
  .word BusFault_Handler
  // ... 更多中断向量 ...

Reset_Handler:
  ldr sp, =_estack           // 设置栈指针
  bl SystemInit              // 初始化时钟
  bl __libc_init_array       // C库初始化（调用全局构造函数）
  bl main                    // 跳转到main()
  bx lr                      // main不应返回

Linker Script (STM32F103C8T6)

ENTRY(Reset_Handler)

MEMORY {

  RAM (rwx)  : ORIGIN = 0x20000000, LENGTH = 20K

📝 串口重定向printf（三种方法）

// 方法1：HAL_UART 阻塞发送（最常用、最简单）
int fputc(int ch, FILE *f) {
    HAL_UART_Transmit(&huart1, (uint8_t *)&ch, 1, HAL_MAX_DELAY);
    return ch;
}
// ⚠ 缺点：阻塞发送，大数据量时CPU空等。调试场景完全够用。

// 方法2：ITM/SWO 高速输出（不占串口、比UART快100倍）
int fputc(int ch, FILE *f) {
    ITM_SendChar(ch);
    return ch;
}
// 需要：SWO引脚接线 + OpenOCD/GDB Server启用ITM
// 查看：telnet localhost 4444 → monitor tpiu config internal /tmp/itm.log uart off 72000000
//       monitor itm port 0 on

// 方法3：DMA+环形缓冲区（最优雅、非阻塞、最快）
#define TX_BUF_SIZE 256
static uint8_t tx_buf[TX_BUF_SIZE];
static volatile uint16_t tx_head = 0;
static volatile uint16_t tx_tail = 0;

int fputc(int ch, FILE *f) {
    uint16_t next = (tx_head + 1) % TX_BUF_SIZE;
    while (next == tx_tail) { /* 缓冲区满，等待DMA发送 */ }
    tx_buf[tx_head] = ch;
    tx_head = next;
    // 启动或继续DMA传输（外部实现）
    uart_dma_trigger();
    return ch;
}
// 优点：printf立即返回不阻塞，DMA后台发送
// 缺点：实现复杂，需管理DMA半满/全满中断

🔢 常用位操作宏（嵌入式开发天天用）

// 位操作 — 嵌入式开发的基本功
#define BIT(x)           (1UL << (x))           // 第x位置1
#define SET_BIT(reg, x)  ((reg) |= BIT(x))      // 置位
#define CLR_BIT(reg, x)  ((reg) &= ~BIT(x))     // 清零
#define TOG_BIT(reg, x)  ((reg) ^= BIT(x))      // 翻转
#define GET_BIT(reg, x)  (((reg) >> (x)) & 1U)  // 读取

// 多位操作
#define SET_MASK(reg, mask)   ((reg) |= (mask))
#define CLR_MASK(reg, mask)   ((reg) &= ~(mask))
#define GET_MASK(reg, mask)   ((reg) & (mask))

// 寄存器写入（防止编译器优化掉）
#define REG32(addr)  (*(volatile uint32_t *)(addr))
#define REG16(addr)  (*(volatile uint16_t *)(addr))
#define REG8(addr)   (*(volatile uint8_t *)(addr))

// 实际使用示例：直接操作STM32寄存器（无需HAL库）
#define RCC_APB2ENR  REG32(0x40021018)
#define GPIOC_CRH    REG32(0x40011004)
#define GPIOC_ODR    REG32(0x4001100C)

// 使能GPIOC时钟
SET_BIT(RCC_APB2ENR, 4);    // RCC_APB2ENR bit4 = IOPCEN
// PC13推挽输出，50MHz
GPIOC_CRH = (GPIOC_CRH & ~0x00F00000) | 0x00300000;  // MODE13=11 CNF13=00
// PC13输出高
SET_BIT(GPIOC_ODR, 13);

寄存器地址计算规则

// 寄存器地址 = 总线基地址 + 外设偏移 + 寄存器偏移
//
// STM32F103总线基地址：
//   APB1: 0x40000000  (TIM2-TIM7, USART2-5, I2C, SPI2, CAN, USB...)
//   APB2: 0x40010000  (AFIO, GPIOA-GPIOE, ADC1-2, TIM1, USART1, SPI1...)
//   AHB:  0x40018000  (DMA, SDIO, FSMC...)
//
// 举例：
//   GPIOC 外设偏移 = 0x10000 (APB2基地址0x40010000 + 0x10000 = 0x40011000)
//   GPIOC_ODR = APB2基地址 + GPIOC偏移 + ODR偏移
//             = 0x40010000 + 0x10000 + 0x0C = 0x4001100C
//
//   GPIOA: 偏移 0x0800  → 基址 0x40010800
//   GPIOB: 偏移 0x0C00  → 基址 0x40010C00
//   GPIOC: 偏移 0x1000  → 基址 0x40011000
//
// 不同芯片查看对应Reference Manual第1章 "Memory Map" 表格

⚙ 实用代码模式

按键消抖

// 软件消抖：状态机+定时器，不阻塞CPU
typedef enum { BTN_UP, BTN_DEBOUNCE, BTN_DOWN } BtnState_t;

typedef struct {
    BtnState_t state;
    uint32_t debounce_tick;  // 消抖计时
    uint8_t press_count;      // 按键按下次数
} Button_t;

uint8_t button_debounce(Button_t *btn, uint8_t pin_level) {
    uint8_t event = 0;  // 0=无事件, 1=短按, 2=长按(>1秒)
    
    switch (btn->state) {
    case BTN_UP:
        if (pin_level == 0) {  // 检测到按下
            btn->debounce_tick = HAL_GetTick();
            btn->state = BTN_DEBOUNCE;
        }
        break;
    case BTN_DEBOUNCE:
        if (pin_level == 0) {
            if (HAL_GetTick() - btn->debounce_tick >= 20) {  // 20ms消抖
                btn->state = BTN_DOWN;
                btn->press_count++;
                btn->debounce_tick = HAL_GetTick();
            }
        } else {  // 是抖动，不算
            btn->state = BTN_UP;
        }
        break;
    case BTN_DOWN:
        if (pin_level == 1) {  // 释放
            if (HAL_GetTick() - btn->debounce_tick < 1000) {
                event = 1;  // 短按
            }
            btn->state = BTN_UP;
        } else if (HAL_GetTick() - btn->debounce_tick >= 1000) {
            event = 2;  // 长按>=1秒
            btn->state = BTN_UP;
        }
        break;
    }
    return event;
}

环形缓冲区

// 通用环形缓冲区（FIFO）
typedef struct {
    uint8_t *buffer;
    uint16_t size;
    volatile uint16_t head;  // 写指针
    volatile uint16_t tail;  // 读指针
    volatile uint16_t count; // 当前数据量
} RingBuffer_t;

void ringbuf_init(RingBuffer_t *rb, uint8_t *buf, uint16_t size) {
    rb->buffer = buf;
    rb->size = size;
    rb->head = rb->tail = rb->count = 0;
}

uint8_t ringbuf_put(RingBuffer_t *rb, uint8_t byte) {
    if (rb->count >= rb->size) return 0;  // 满
    rb->buffer[rb->head] = byte;
    rb->head = (rb->head + 1) % rb->size;
    rb->count++;
    return 1;
}

uint8_t ringbuf_get(RingBuffer_t *rb, uint8_t *byte) {
    if (rb->count == 0) return 0;  // 空
    *byte = rb->buffer[rb->tail];
    rb->tail = (rb->tail + 1) % rb->size;
    rb->count--;
    return 1;
}

uint16_t ringbuf_available(RingBuffer_t *rb) { return rb->count; }
uint16_t ringbuf_free(RingBuffer_t *rb) { return rb->size - rb->count; }

软件定时器（不依赖RTOS）

// 基于SysTick的轻量软件定时器（裸机可用）
#define MAX_SOFT_TIMERS 8

typedef struct {
    uint32_t interval;  // 定时间隔(ms)
    uint32_t last_tick; // 上次触发时间
    uint8_t  repeat;    // 1=重复, 0=一次性
    uint8_t  active;
    void (*callback)(void*);
    void *arg;
} SoftTimer_t;

SoftTimer_t timers[MAX_SOFT_TIMERS];

// 必须在SysTick_Handler(或定时器中断)中调用
void soft_timer_tick_handler(void) {
    uint32_t now = HAL_GetTick();
    for (int i = 0; i < MAX_SOFT_TIMERS; i++) {
        if (timers[i].active && (now - timers[i].last_tick >= timers[i].interval)) {
            timers[i].last_tick = now;
            if (timers[i].callback) {
                timers[i].callback(timers[i].arg);
            }
            if (!timers[i].repeat) {
                timers[i].active = 0;  // 一次性定时器自毁
            }
        }
    }
}

int soft_timer_create(uint32_t ms, uint8_t repeat, void (*cb)(void*), void *arg) {
    for (int i = 0; i < MAX_SOFT_TIMERS; i++) {
        if (!timers[i].active) {
            timers[i].interval = ms;
            timers[i].last_tick = HAL_GetTick();
            timers[i].repeat = repeat;
            timers[i].active = 1;
            timers[i].callback = cb;
            timers[i].arg = arg;
            return i;
        }
    }
    return -1;  // 定时器池满
}

CRC校验

// CRC8/CRC16常用校验（通信数据完整性）
uint8_t crc8(const uint8_t *data, uint16_t len) {
    uint8_t crc = 0xFF;
    while (len--) {
        crc ^= *data++;
        for (int i = 0; i < 8; i++) {
            crc = (crc & 0x80) ? ((crc << 1) ^ 0x31) : (crc << 1);
        }
    }
    return crc;
}

uint16_t crc16_modbus(const uint8_t *data, uint16_t len) {
    uint16_t crc = 0xFFFF;
    while (len--) {
        crc ^= (uint16_t)*data++;
        for (int i = 0; i < 8; i++) {
            crc = (crc & 0x0001) ? ((crc >> 1) ^ 0xA001) : (crc >> 1);
        }
    }
    return crc;  // Modbus CRC发送时低字节在前
}

// 使用示例：封包发送带CRC校验的数据
// 格式：[STX][LEN][DATA...][CRC16_L][CRC16_H][ETX]
void send_packet(uint8_t *data, uint16_t len) {
    uint16_t crc = crc16_modbus(data, len);
    printf("\x02");         // STX
    printf("%c%c", len>>8, len&0xFF);  // LEN(2字节)
    for (int i=0; i>8);  // CRC16(低字节在前)
    printf("\x03");         // ETX
}

🛡 看门狗策略与系统恢复

看门狗是嵌入式系统最后的防线——当一切软件保护都失效时，硬件看门狗把系统拉回来。但乱用看门狗反而会让系统陷入"死循环-复位-死循环"的无限循环。

IWDG vs WWDG 对比基础
STM32有两个看门狗：独立看门狗(IWDG)用内部40KHz LSI时钟，窗口看门狗(WWDG)用APB1时钟。选哪个取决于保护目标。
特性IWDGWWDG
时钟源LSI (40KHz, 独立于系统时钟)APB1 (受系统时钟影响)
能否被停止否（一旦启动只能复位才能停）调试时可停（DBGMCU控制）
超时范围~100μs ~ 26秒~50μs ~ 50ms
用途防系统死机（通用）防时序异常（精细保护）
典型场景主循环卡死、中断风暴控制循环超时、外设响应慢
⚠ 看门狗反模式：喂狗放在中断里！这是最常见的看门狗错误——即使主循环死锁了，中断还在喂狗，看门狗永远不会复位。正确的做法：主循环喂狗，ISR只做标记。
// ★ 正确的多任务看门狗策略
// 原理：每个任务设置一个"心跳"标志，主循环检查所有任务是否存活

typedef struct {
    volatile uint32_t last_beat;  // 最后心跳时间
    const char     *name;
    uint32_t        timeout_ms;   // 超时阈值
} task_heartbeat_t;

task_heartbeat_t heartbeats[] = {
    {0, "sensor_task",  5000},   // 传感器任务最多5秒不响应
    {0, "mqtt_task",   30000},   // MQTT允许30秒（网络波动）
    {0, "oled_task",    2000},   // OLED 2秒
    {0, "main_loop",    1000},   // 主循环 1秒
};
#define TASK_COUNT (sizeof(heartbeats)/sizeof(heartbeats[0]))

// 每个任务调用这个报告"我还活着"
void task_heartbeat(uint8_t task_id) {
    if (task_id < TASK_COUNT) {
        heartbeats[task_id].last_beat = HAL_GetTick();
    }
}

// 主循环中检查所有任务（放在喂狗之前！）
bool all_tasks_alive(void) {
    uint32_t now = HAL_GetTick();
    for (int i = 0; i < TASK_COUNT; i++) {
        if (now - heartbeats[i].last_beat > heartbeats[i].timeout_ms) {
            // ★ 记录死亡任务信息到备份寄存器（复位后可知谁死了）
            RTC->BKP0R = i;  // 备份寄存器0存死亡任务ID
            RTC->BKP1R = 0xDEAD0000 | (now & 0xFFFF);  // 存时间戳
            return false;  // 有任务死了，不要喂狗！
        }
    }
    return true;
}

// 主循环
void main_loop(void) {
    while (1) {
        task_heartbeat(3);  // 主循环自己的心跳
        
        // 执行各子任务（不阻塞！）
        sensor_poll();
        mqtt_poll();
        oled_update();
        
        // ★ 所有任务都活着才喂狗
        if (all_tasks_alive()) {
            HAL_IWDG_Refresh(&hiwdg);
        }
        // 如果有任务死了 → 不喂狗 → 看门狗超时复位 → 系统重启恢复
    }
}
🔍 看门狗踩坑经验：
1. 调试时忘记关IWDG：断点停下来→看门狗超时→复位→调试器断开。解决：调试脚本里加 DBGMCU->APB1FZ |= DBGMCU_APB1_FZ_IWDG_STOP; 让IWDG在调试时停住。
2. Flash写入时看门狗复位：Flash页擦除需要数十ms，期间CPU暂停→看门狗超时。解决：Flash操作前喂一次狗，操作完再喂；或用双看门狗（IWDG设长超时）。
3. 看门狗超时太短：设了100ms，但OTA期间Flash擦除可能200ms→频繁复位。IWDG一般设1-4秒，WWDG设20-50ms。

🚀 Bootloader设计模式

Bootloader是一个独立的程序，在复位后最先执行。它负责：检查是否需要更新固件、验证固件完整性、跳转到应用程序。

典型的Flash分区布局

/* STM32F103C8T6 Flash 64KB 分区设计 */
/* 0x08000000 ┌──────────────────────┐
              │ Bootloader (8KB)     │  ← 永远不动（除非出厂烧录）
   0x08002000 ├──────────────────────┤
              │ App 主程序 (28KB)    │  ← 正常运行在这里
   0x08009000 ├──────────────────────┤
              │ App 备份区 (28KB)    │  ← OTA升级先写到这里，验证完才切过来
   0x08010000 └──────────────────────┘ */

// 分区表定义
#define BOOTLOADER_ADDR   0x08000000
#define APP_ADDR          0x08002000  // 8KB偏移
#define APP_BACKUP_ADDR   0x08009000  // 36KB偏移
#define APP_MAX_SIZE      (28 * 1024)  // 28KB最大固件
#define FLASH_PAGE_SIZE   1024         // STM32F103 1KB/page

// 标志位（存放在Flash末页，不在bootloader/APP区域）
#define FLAG_OTA_PENDING  0xAA55AA55  // 有待升级固件
#define FLAG_APP_VALID    0x55AA55AA  // APP验证通过
#define FLAG_ADDR         (0x0800FC00) // Flash最后一页

Bootloader主流程

// Bootloader启动流程（STM32F103C8T6）
void bootloader_main(void) {
    // ① 系统时钟初始化（8MHz HSI即可，不需要PLL）
    SystemInit();
    HAL_Init();
    
    // ② 检查是否有新固件待OTA
    uint32_t ota_flag = *(volatile uint32_t*)FLAG_ADDR;
    if (ota_flag == FLAG_OTA_PENDING) {
        // ③ 从备份区复制到APP区
        flash_copy(APP_BACKUP_ADDR, APP_ADDR, APP_MAX_SIZE);
        
        // ④ 验证固件完整性（CRC32校验）
        uint32_t crc_calc = crc32((void*)APP_ADDR, APP_MAX_SIZE);
        uint32_t crc_stored = *(volatile uint32_t*)(FLAG_ADDR + 4);
        
        if (crc_calc == crc_stored) {
            // ⑤ 验证通过，清除OTA标志
            flash_erase_page(FLAG_ADDR);
            flash_write_word(FLAG_ADDR, FLAG_APP_VALID);
        } else {
            // CRC不匹配！不清标志，下次启动再试（或回退）
            // 可以在这里点亮红色LED指示升级失败
        }
    }
    
    // ⑥ 检查APP区是否有效（栈顶地址在RAM范围内）
    uint32_t app_sp = *(volatile uint32_t*)APP_ADDR;       // 栈顶地址
    uint32_t app_pc = *(volatile uint32_t*)(APP_ADDR + 4); // 复位向量
    
    if ((app_sp >= 0x20000000) && (app_sp <= 0x20005000) &&  // RAM范围
        (app_pc >= APP_ADDR) && (app_pc < APP_ADDR + APP_MAX_SIZE)) {
        
        // ⑦ 跳转前清理：关中断、恢复默认状态
        __disable_irq();
        SysTick->CTRL = 0;  // 关SysTick
        
        // ⑧ 设置向量表偏移到APP区
        SCB->VTOR = APP_ADDR;
        
        // ⑨ 跳转到APP！
        void (*app_reset_handler)(void) = (void (*)(void))app_pc;
        __set_MSP(app_sp);  // 设主栈指针
        app_reset_handler();  // 永远不返回
    }
    
    // APP无效，留在bootloader（可以进入下载模式）
    enter_dfu_mode();  // 等待固件通过串口/USB下载
}
🔍 Bootloader常见故障：
1. 跳转后HardFault：忘记设 SCB->VTOR = APP_ADDR→APP启动后中断向量表还在bootloader位置→一进中断就HardFault。
2. 跳转后SysTick不工作：APP重新初始化了SysTick，但bootloader的SysTick还在运行→计数冲突。解决：跳转前 SysTick->CTRL = 0;
3. APP用到的外设时钟不同：bootloader可能开了HSE+PLL，APP也开一遍→重复配置可能导致PLL失锁。解决：跳转前复位所有外设 RCC_DeInit()

// ★ 工厂测试固件（独立于产品固件）
// 生产流程：烧录测试固件→自动测试→烧录产品固件→打标出货

// 测试协议：上位机发命令，设备应答
// 命令格式：CMD:参数1,参数2\n

void factory_test_handler(uint8_t byte) {
    static char cmd_buf[64];
    static uint8_t idx = 0;
    
    if (byte == '\n') {
        cmd_buf[idx] = 0;
        idx = 0;

🔐 校验与错误检测速查

通信协议和存储系统离不开校验。不同的校验算法在速度、检错能力和代码大小之间有不同的权衡。

// ★ 常用校验算法实现（嵌入式优化版）

// 1. 异或校验（XOR Checksum）—— 最快，但最弱
// 适用：简单串口协议，配合超时重发
uint8_t xor_checksum(const uint8_t *data, int len) {
    uint8_t cks = 0;
    while (len--) cks ^= *data++;
    return cks;
}

// 2. 累加和校验（Sum Checksum）—— 比XOR稍强
uint8_t sum_checksum(const uint8_t *data, int len) {
    uint8_t sum = 0;
    while (len--) sum += *data++;
    return sum;
}

// 3. CRC16-Modbus（反向算法，查表优化）
// Modbus RTU标准CRC，不可用软件逐位算（太慢！）
static const uint16_t crc16_table[256] = {
    0x0000, 0xC0C1, 0xC181, 0x0140, 0xC301, 0x03C0, 0x0280, 0xC241,
    0xC601, 0x06C0, 0x0780, 0xC741, 0x0500, 0xC5C1, 0xC481, 0x0440,
    // ... 256个值预计算完毕（篇幅限制省略，实际项目复制完整表）
};

uint16_t crc16_modbus(const uint8_t *data, int len) {
    uint16_t crc = 0xFFFF;  // 初值
    while (len--) {
        crc = (crc >> 8) ^ crc16_table[(crc ^ *data++) & 0xFF];
    }
    return crc;  // Modbus发送时先发低字节
}
// ★ 验证：数据 {0x01,0x03,0x00,0x00,0x00,0x01} → CRC = 0x840A

// 4. CRC32（以太网/PNG/Flash校验标准）
// STM32F4+有硬件CRC32，用HAL_CRC_Calculate()
// 软件实现用查表法（256×4字节=1KB表）

// ★ 校验算法选型指南：
// 串口简单通信   → XOR (1行代码)
// Modbus/工业   → CRC16 (查表，~60字节表)
// Flash/文件/OTA → CRC32 (硬件加速或查表)
// 安全/防篡改    → SHA256/HMAC (需要加密库如mbedTLS)

📏 嵌入式数据压缩技巧

Flash和RAM都有限。了解几种简单实用的数据压缩技巧，省出宝贵的空间。

// ★ 技巧1：打包标志位（8个flag → 1字节）
// 不要这样：
bool led_state;      // 1字节
bool wifi_connected; // 1字节
bool alarm_enabled;  // 1字节
// ... 8个bool = 8字节

// 应该这样：
uint8_t flags;       // 1字节存8个标志！
#define FLAG_LED         (1 << 0)
#define FLAG_WIFI        (1 << 1)
#define FLAG_ALARM       (1 << 2)
#define FLAG_OTA_READY   (1 << 3)
// 使用：if (flags & FLAG_WIFI) { ... }
// 省了7字节！对于RAM只有20K的MCU，每字节都珍贵

// ★ 技巧2：枚举代替字符串
// 不要这样：
const char *state_str = "DISCONNECTED";  // 13字节+指针
// 应该这样：
enum { STATE_IDLE, STATE_CONNECTING, STATE_CONNECTED, STATE_ERROR };
uint8_t state = STATE_IDLE;  // 1字节！
// 需要打印时再用switch-case转字符串

// ★ 技巧3：结构体成员按大小降序排列（避免padding浪费）
// 乱序：
typedef struct {
    uint8_t  id;         // 1 + 3 padding
    uint32_t timestamp;  // 4
    uint16_t value;      // 2 + 2 padding
} bad_packed_t;          // 总共12字节

// 优化后：
typedef struct {
    uint32_t timestamp;  // 4
    uint16_t value;      // 2
    uint8_t  id;         // 1 + 1 padding
} good_packed_t;         // 总共8字节 → 省了33%！

// 或者用packed（不对齐但省空间，M0/M0+慎用）：
typedef struct __attribute__((packed)) {
    uint8_t  id;
    uint32_t timestamp;  // 非对齐访问
    uint16_t value;
} packed_t;  // 7字节，但M0访问timestamp会HardFault
