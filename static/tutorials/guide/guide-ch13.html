启动文件(startup)说明

启动文件用汇编编写，做三件事：①初始化堆栈指针 ②初始化中断向量表 ③调用SystemInit()和main()

// startup_stm32f103xx.s 关键部分
  .section .isr_vector,"a",%progbits
  .word _estack              // 初始栈指针（RAM末尾）
  .word Reset_Handler         // 复位向量
  .word NMI_Handler
  .word HardFault_Handler
  .word MemManage_Handler
  .word BusFault_Handler
  // ... 更多中断向量 ...

Reset_Handler:
  ldr sp, =_estack           // 设置栈指针
  bl SystemInit              // 初始化时钟
  bl __libc_init_array       // C库初始化
  bl main                    // 跳转到main()
  bx lr                      // main不应返回

Linker Script (STM32F103C8T6)

/* STM32F103C8Tx.ld - 完整版 */
ENTRY(Reset_Handler)

_estack = ORIGIN(RAM) + LENGTH(RAM);  /* 栈顶 = RAM末尾 */

MEMORY {
  FLASH (rx) : ORIGIN = 0x08000000, LENGTH = 64K
  RAM (rwx)  : ORIGIN = 0x20000000, LENGTH = 20K
}

SECTIONS {
  .isr_vector : {
    . = ALIGN(4);
    KEEP(*(.isr_vector))
    . = ALIGN(4);
  } > FLASH

  .text : {
    . = ALIGN(4);
    *(.text)
    *(.text*)
    *(.glue_7)
    *(.glue_7t)
    *(.eh_frame)
    KEEP(*(.init))
    KEEP(*(.fini))
    . = ALIGN(4);
    _etext = .;
  } > FLASH

  .rodata : {
    . = ALIGN(4);
    *(.rodata)
    *(.rodata*)
    . = ALIGN(4);
  } > FLASH

  .ARM.extab : { *(.ARM.extab* .gnu.linkonce.armextab.*) } > FLASH
  .ARM : {
    __exidx_start = .;
    *(.ARM.exidx*)
    __exidx_end = .;
  } > FLASH

  .data : {
    . = ALIGN(4);
    _sdata = .;
    *(.data)
    *(.data*)
    . = ALIGN(4);
    _edata = .;
  } > RAM AT > FLASH

  .bss : {
    . = ALIGN(4);
    _sbss = .;
    __bss_start__ = _sbss;
    *(.bss)
    *(.bss*)
    *(COMMON)
    . = ALIGN(4);
    _ebss = .;
    __bss_end__ = _ebss;
  } > RAM

  /DISCARD/ : { *(.ARM.attributes) }
}

串口重定向printf

// 方法1：HAL_UART（最常用）
int fputc(int ch, FILE *f) {
    HAL_UART_Transmit(&huart1, (uint8_t *)&ch, 1, HAL_MAX_DELAY);
    return ch;
}

// 方法2：ITM/SWO（更快，不占串口）
int fputc(int ch, FILE *f) {
    ITM_SendChar(ch);
    return ch;
}

// 方法3：DMA+环形缓冲区（最优雅，非阻塞）
#define TX_BUF_SIZE 256
static uint8_t tx_buf[TX_BUF_SIZE];
static volatile uint16_t tx_head = 0, tx_tail = 0;

int fputc(int ch, FILE *f) {
    tx_buf[tx_head] = ch;
    tx_head = (tx_head + 1) % TX_BUF_SIZE;
    // 触发DMA发送...
    return ch;
}

常用位操作宏

// 位操作 — 嵌入式开发天天用
#define BIT(x)          (1U << (x))           // 第x位置1
#define SET_BIT(reg, x)  ((reg) |= BIT(x))     // 置位
#define CLR_BIT(reg, x)  ((reg) &= ~BIT(x))   // 清零
#define TOG_BIT(reg, x)  ((reg) ^= BIT(x))     // 翻转
#define GET_BIT(reg, x)  (((reg) >> (x)) & 1)  // 读取

// 多位操作
#define SET_MASK(reg, mask)   ((reg) |= (mask))
#define CLR_MASK(reg, mask)   ((reg) &= ~(mask))

// 寄存器写入（防编译器优化）
#define REG32(addr)  (*(volatile uint32_t *)(addr))
#define REG16(addr)  (*(volatile uint16_t *)(addr))
#define REG8(addr)   (*(volatile uint8_t *)(addr))

// 示例：直接操作STM32寄存器
#define RCC_APB2ENR  REG32(0x40021018)
#define GPIOC_CRH    REG32(0x40011004)
#define GPIOC_ODR    REG32(0x4001100C)

// 使能GPIOC时钟
SET_BIT(RCC_APB2ENR, 4);    // RCC_APB2ENR bit4 = IOPCEN
// PC13推挽输出
GPIOC_CRH = (GPIOC_CRH & ~0x00F00000) | 0x00300000;
// PC13输出高
SET_BIT(GPIOC_ODR, 13);

💡 小技巧：用HAL库开发时一般不需要直接操作寄存器，但理解寄存器对排错很重要。当HAL库行为不符合预期时，查看寄存器值是最终手段。

寄存器地址计算规则

上面的地址表看起来很多，但其实有规律。记住这个方法，换任何芯片都能自己算：

// 寄存器地址 = 总线基地址 + 外设偏移 + 寄存器偏移
//
// STM32F103总线基地址：
//   APB1: 0x40000000  (TIM2-TIM7, USART2-5, I2C, SPI2, CAN, USB...)
//   APB2: 0x40010000  (AFIO, GPIOA-C, ADC1-2, TIM1, USART1, SPI1...)
//   AHB:  0x40018000  (DMA, SDIO, FSMC...)
//
// 举例：GPIOC_ODR = APB2基地址 + GPIOC偏移 + ODR偏移
//                   0x40010000 + 0x00010000 + 0x0C = 0x4001100C
//
// 不同芯片基地址不同，查看对应芯片的Reference Manual:
//   第1章 "Memory Map" 表格

常见错误码速查

来源错误码含义解决
HAL库HAL_OK (0)成功—
HAL库HAL_ERROR (1)一般错误检查参数和硬件连接
HAL库HAL_BUSY (2)外设忙等待或检查上一次操作是否完成
HAL库HAL_TIMEOUT (3)超时检查接线/波特率/时钟配置
