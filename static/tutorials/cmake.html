📦 CMake 构建系统 细致教程

CMake是一个跨平台构建系统生成器——你写CMakeLists.txt，它生成Makefile/Ninja文件，再编译你的项目。它是VS Code + STM32CubeMX的默认构建工具。

💡 类比：CMake就像装修公司的设计图——你描述"我要三室两厅、开放式厨房"，CMake生成具体的"先砌墙、再布水电、最后刷漆"的施工计划。Makefile就是那个按计划执行的人。

📦 安装 CMake

# 方法1：winget（推荐）
winget install Kitware.CMake

# 方法2：下载安装包
# cmake.org/download → Windows x64 Installer → cmake-*-windows-x86_64.msi

# 验证
cmake --version
# CMake 4.3.3（或你安装的版本）
💡 CMake只是生成器：CMake本身不编译，它生成Makefile（或Ninja），再由make/ninja执行编译。Windows上CMake默认用Ninja生成器，比Make快30%。

🤔 为什么要用 CMake

不用CMake也可以编译嵌入式项目（直接arm-none-eabi-gcc一条命令），但有以下问题：

不用CMake用CMake
手动敲所有编译命令（几十个）一条命令全部搞定
改一个.c要重新敲命令自动增量编译，只重编改过的
换电脑/系统要从头配CMakeLists.txt跨平台，拿到哪都能编译
只能单线程编译自动多核并行编译（-j$(nproc)）
VS Code插件支持差cmake-tools插件一键配置+编译+调试

🚀 最小 CMake 项目（5步跑通）

步骤1：创建项目目录
mkdir blink && cd blink
# 目录结构（最简）：
# blink/
# ├── CMakeLists.txt     ← 配置文件
# └── src/
#     └── main.c         ← 源代码
步骤2：写 main.c
// src/main.c
#include <stdint.h>

#define RCC_BASE    0x40021000
#define RCC_APB2ENR (*(volatile uint32_t *)(RCC_BASE + 0x18))
#define GPIOA_BASE  0x40010800
#define GPIOA_CRL   (*(volatile uint32_t *)(GPIOA_BASE + 0x00))
#define GPIOA_ODR   (*(volatile uint32_t *)(GPIOA_BASE + 0x0C))

int main(void) {
    // 使能GPIOA时钟（RCC_APB2ENR第2位=1）
    RCC_APB2ENR |= (1 << 2);

    // 配置PA1为推挽输出（MODE1=11, CNF1=00）
    GPIOA_CRL = (GPIOA_CRL & ~0xF0) | 0x30;

    while (1) {
        GPIOA_ODR ^= (1 << 1);  // 翻转PA1
        for (volatile int i = 0; i < 800000; i++);  // 延时
    }
}
步骤3：写 CMakeLists.txt（核心配置）
# CMakeLists.txt
cmake_minimum_required(VERSION 3.20)

# 项目名（任意）
project(blink C ASM)

# 设置C标准
set(CMAKE_C_STANDARD 99)
set(CMAKE_C_STANDARD_REQUIRED ON)

# ===== 工具链配置 =====
set(CMAKE_SYSTEM_NAME Generic)
set(CMAKE_SYSTEM_PROCESSOR ARM)

# 交叉编译工具链路径
set(TOOLCHAIN_PATH "C:/Program Files (x86)/Arm GNU Toolchain arm-none-eabi/14.2 rel1")
set(CMAKE_C_COMPILER   "${TOOLCHAIN_PATH}/bin/arm-none-eabi-gcc.exe")
set(CMAKE_ASM_COMPILER "${TOOLCHAIN_PATH}/bin/arm-none-eabi-gcc.exe")
set(CMAKE_OBJCOPY      "${TOOLCHAIN_PATH}/bin/arm-none-eabi-objcopy.exe")
set(CMAKE_SIZE         "${TOOLCHAIN_PATH}/bin/arm-none-eabi-size.exe")

# ===== 编译参数 =====
set(CPU_FLAGS "-mcpu=cortex-m3 -mthumb -mfpu=fpv4-sp-d16 -mfloat-abi=hard")

add_compile_options(
    ${CPU_FLAGS}
    -Os -Wall -Wextra -ffunction-sections -fdata-sections
    -fno-common -fno-strict-aliasing
)

add_link_options(
    ${CPU_FLAGS}
    -T"${CMAKE_SOURCE_DIR}/STM32F103C8.ld"
    -Wl,--gc-sections
    -Wl,-Map=${PROJECT_NAME}.map
    --specs=nosys.specs
    --specs=nano.specs
)

# ===== 源文件 =====
# 查找所有.c文件
aux_source_directory(src SRC_FILES)

# 生成可执行文件
add_executable(${PROJECT_NAME}.elf ${SRC_FILES})

# 生成.bin
add_custom_command(OUTPUT ${PROJECT_NAME}.bin
    COMMAND ${CMAKE_OBJCOPY} -O binary ${PROJECT_NAME}.elf ${PROJECT_NAME}.bin
    DEPENDS ${PROJECT_NAME}.elf
    COMMENT "Generating binary file"
)

# 生成.hex
add_custom_command(OUTPUT ${PROJECT_NAME}.hex
    COMMAND ${CMAKE_OBJCOPY} -O ihex ${PROJECT_NAME}.elf ${PROJECT_NAME}.hex
    DEPENDS ${PROJECT_NAME}.elf
    COMMENT "Generating HEX file"
)

# 显示大小
add_custom_command(TARGET ${PROJECT_NAME}.elf POST_BUILD
    COMMAND ${CMAKE_SIZE} --format=sysv ${PROJECT_NAME}.elf
)

# 打印完整路径
add_custom_command(TARGET ${PROJECT_NAME}.elf POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E echo "Output: ${PROJECT_BINARY_DIR}/${PROJECT_NAME}.bin"
)
步骤4：创建构建目录并编译
# 必须！CMake不允许在源码目录构建
mkdir build && cd build

# Windows PowerShell（生成Ninja构建文件）
cmake -G Ninja -DCMAKE_BUILD_TYPE=MinSizeRel ..

# 或者用Unix Makefiles
cmake -G "Unix Makefiles" -DCMAKE_BUILD_TYPE=MinSizeRel ..

# 编译（-j4用4核并行）
cmake --build . -j 4

# 查看输出
dir *.bin    # 应该看到 blink.bin（约几KB）
dir *.elf    # 应该看到 blink.elf
dir *.map    # 内存映射文件
步骤5：烧录到芯片
# OpenOCD烧录（如果装了OpenOCD）
openocd -f interface/stlink.cfg -f target/stm32f1x.cfg \
    -c "program build/blink.elf verify reset exit"

# st-flash烧录（如果装了stlink工具）
st-flash write build/blink.bin 0x08000000

🔧 工具链文件（可选最佳实践）

把工具链配置单独放到一个文件，CMakeLists.txt更整洁：

步骤1：创建 cmake/toolchain-arm.cmake
# cmake/toolchain-arm.cmake
set(CMAKE_SYSTEM_NAME Generic)
set(CMAKE_SYSTEM_PROCESSOR ARM)

# 工具链路径（根据你的实际安装位置修改）
set(TOOLCHAIN_DIR "C:/Program Files (x86)/Arm GNU Toolchain arm-none-eabi/14.2 rel1")
set(CMAKE_FIND_ROOT_PATH "${TOOLCHAIN_DIR}")

# 编译器
set(CMAKE_C_COMPILER   "${TOOLCHAIN_DIR}/bin/arm-none-eabi-gcc.exe")
set(CMAKE_CXX_COMPILER "${TOOLCHAIN_DIR}/bin/arm-none-eabi-g++.exe")
set(CMAKE_ASM_COMPILER "${TOOLCHAIN_DIR}/bin/arm-none-eabi-gcc.exe")
set(CMAKE_OBJCOPY      "${TOOLCHAIN_DIR}/bin/arm-none-eabi-objcopy.exe")
set(CMAKE_SIZE         "${TOOLCHAIN_DIR}/bin/arm-none-eabi-size.exe")
set(CMAKE_OBJDUMP      "${TOOLCHAIN_DIR}/bin/arm-none-eabi-objdump.exe")
set(CMAKE_GDB          "${TOOLCHAIN_DIR}/bin/arm-none-eabi-gdb.exe")

# 搜索模式（只在工具链目录找库/头文件）
set(CMAKE_FIND_ROOT_PATH_MODE_PROGRAM NEVER)
set(CMAKE_FIND_ROOT_PATH_MODE_LIBRARY ONLY)
set(CMAKE_FIND_ROOT_PATH_MODE_INCLUDE ONLY)

# CPU参数
set(MCPU "cortex-m3")
步骤2：根目录 CMakeLists.txt 引用它
# 在 project() 之前加载工具链文件
include(cmake/toolchain-arm.cmake)

project(blink C ASM)
步骤3：构建时指定工具链
cd build
cmake -G Ninja -DCMAKE_TOOLCHAIN_FILE=../cmake/toolchain-arm.cmake ..
cmake --build .

🎯 Target（目标）详解

CMake的核心概念是Target（目标）。每个Target代表一个可构建的东西：可执行文件、库、或自定义命令。

add_executable — 可执行文件

# 最基本
add_executable(myapp main.c utils.c)

# 指定编译选项（只影响这个目标）
add_executable(myapp main.c)
target_compile_options(myapp PRIVATE -Wall -Wextra)

# 添加include目录（只影响这个目标）
target_include_directories(myapp PRIVATE inc/)

# 链接库
target_link_libraries(myapp PRIVATE m)  # 链接数学库

add_library — 静态库/动态库

# 静态库（.a文件）
add_library(hal STATIC
    src/stm32f1xx_hal.c
    src/stm32f1xx_hal_gpio.c
    src/stm32f1xx_hal_uart.c
    # ...
)
target_include_directories(hal PUBLIC inc/Drivers/STM32F1xx_HAL_Driver/Inc)

# 可执行文件链接HAL库
add_executable(myapp main.c)
target_link_libraries(myapp PRIVATE hal)  # 自动携带hal的include目录

target_include_directories 作用域

关键字含义示例
PRIVATE只影响当前目标可执行文件用自己的include
PUBLIC影响当前目标+链接它的目标HAL库+它的使用者
INTERFACE不影响当前，只传给使用者头文件库（纯头文件的库）
⚠️ 规则：实现文件(.c)的库用PUBLIC/PRIVATE；纯头文件的库用INTERFACE。

⚙️ 编译参数配置详解

三种配置类型

# CMAKE_BUILD_TYPE 有三种：
# Debug      = 无优化 -g3，调试最友好
# RelWithDebInfo = 优化 -O2 + 调试信息
# MinSizeRel = -Os，体积优先（嵌入式最常用）
# Release    = -O3，速度优先

# 设置方法
cmake -G Ninja -DCMAKE_BUILD_TYPE=MinSizeRel ..

# 或在CMakeLists.txt中
set(CMAKE_BUILD_TYPE MinSizeRel CACHE STRING "" FORCE)

为不同目标设置不同参数

# 所有目标共享的基础参数
add_compile_options(-Wall -Wextra)

# ARM特定参数（所有C/C++目标）
if(CMAKE_SYSTEM_PROCESSOR MATCHES "ARM")
    add_compile_options(-mthumb -mfloat-abi=hard)
endif()

# HAL库：不用-fomit-frame-pointer（ARM不适用）
target_compile_options(hal PRIVATE -fno-omit-frame-pointer)

# 用户代码：严格检查
target_compile_options(user PRIVATE -Werror -pedantic)

条件编译（嵌入式常用）

# 根据MCU型号定义宏
if(MCPU STREQUAL "cortex-m3")
    add_compile_definitions(STM32F103x8 USE_HAL_DRIVER)
elseif(MCPU STREQUAL "cortex-m4")
    add_compile_definitions(STM32F407xx USE_HAL_DRIVER)
endif()

# 在代码中：
#ifdef STM32F103x8
    // STM32F1专用代码
#elif defined(STM32F407xx)
    // STM32F4专用代码
#endif

📂 多文件项目管理

方式1：aux_source_directory（简单，自动找所有.c）

# 自动收集src/下所有.c文件
aux_source_directory(src SRC_FILES)
aux_source_directory(drivers DRV_FILES)
add_executable(app ${SRC_FILES} ${DRV_FILES})

方式2：GLOB（需要手动维护，但不漏文件）

# 只收集src/下所有.c（不递归子目录）
file(GLOB APP_SOURCES "src/*.c")
# 递归收集
file(GLOB_RECURSE DRV_SOURCES "Drivers/**/*.c")
add_executable(app ${APP_SOURCES} ${DRV_SOURCES})

方式3：明确列出（最清晰，推荐）

# Core代码
set(CORE_SOURCES
    Core/Src/main.c
    Core/Src/stm32f1xx_it.c
    Core/Src/system_stm32f1xx.c
)

# HAL驱动（手动列出，防止垃圾文件混入）
set(HAL_SOURCES
    Drivers/STM32F1xx_HAL_Driver/Src/stm32f1xx_hal.c
    Drivers/STM32F1xx_HAL_Driver/Src/stm32f1xx_hal_gpio.c
    Drivers/STM32F1xx_HAL_Driver/Src/stm32f1xx_hal_uart.c
    Drivers/STM32F1xx_HAL_Driver/Src/stm32f1xx_hal_rcc.c
    # ... 其他需要的HAL驱动
)

# CMSIS
set(CMSIS_SOURCES
    Drivers/CMSIS/Device/ST/STM32F1xx/Source/system_stm32f1xx.c
)

add_executable(${PROJECT_NAME}.elf
    ${CORE_SOURCES} ${HAL_SOURCES} ${CMSIS_SOURCES}
)

🔌 STM32CubeMX + CMake 集成

CubeMX原生支持CMake，是最佳实践组合：

步骤1：CubeMX生成CMake项目
Project Manager → Project：Toolchain/IDE = CMake
Project Manager → Advanced：勾选 Generate peripheral initialization in separate files（推荐）
点击右上角：GENERATE CODE

CubeMX生成的CMake结构（不要改根CMakeLists.txt！）：

project/
├── CMakeLists.txt              ← CubeMX生成，勿改
├── STM32F103C8TX_FLASH.ld     ← CubeMX生成
├── cmake/
│   └── toolchain_arm_none_eabi_gcc.cmake  ← CubeMX工具链
├── Core/
│   ├── Inc/ & Src/            ← 你改这里
│   └── Startup/              ← 启动文件
├── Drivers/                   ← HAL+CMSIS
├── build/                     ← 你建
│   └── CMakeCache.txt         ← 构建缓存
└── CMakeLists_user.txt        ← 可选：你自己的额外配置
步骤2：添加你自己的代码
# 在Core/Src/main.c中，找"USER CODE BEGIN"区域写代码
# CubeMX重新生成时这些区域会被保留

/* USER CODE BEGIN 0 */
#include "dht11.h"
/* USER CODE END 0 */

/* USER CODE BEGIN PV */     ← 全局变量放这里
uint8_t temp = 0;
/* USER CODE END PV */

/* USER CODE BEGIN 0 */
void my_init(void) {
    // 初始化代码
}
/* USER CODE END 0 */

/* USER CODE BEGIN WHILE */
while (1) {
    /* USER CODE END WHILE */
    my_init();
    /* USER CODE BEGIN 3 */
    HAL_GPIO_TogglePin(GPIOA, GPIO_PIN_1);
    HAL_Delay(500);
    /* USER CODE END 3 */
}
/* USER CODE END WHILE */
步骤3：构建
cd build
cmake -G "Unix Makefiles" -DCMAKE_BUILD_TYPE=MinSizeRel ..
cmake --build . -j 4
⚠️ CubeMX重新生成代码后：不要改Core/Inc和Drivers/目录的内容，这些会被CubeMX覆盖。你自己的代码只写在USER CODE区域内。

🔍 CMake 常见问题

问题1：CMakeCache.txt 缓存了旧配置

# 每次改CMakeLists.txt前，先清缓存
Remove-Item -Recurse -Force build/

# 或者单独删一个变量的缓存
# 编辑 build/CMakeCache.txt，删掉对应行，再重新cmake

问题2：编译器找不到（路径含空格/中文）

# 错误：No CMAKE_C_COMPILER could be found
# 解决：路径用引号括起来，或者安装到无空格目录

# cmake/toolchain-arm.cmake中：
set(TOOLCHAIN_DIR "C:/Program Files (x86)/Arm GNU Toolchain arm-none-eabi/14.2 rel1")
# PowerShell中构建：
cmake -G Ninja "-DCMAKE_TOOLCHAIN_FILE=$PWD/cmake/toolchain-arm.cmake" ..

问题3：重复定义（.c被编译了两次）

# 错误：ld: ... multiple definition of ...
# 原因：同一个.c文件在两个target中被编译了
# 解决：检查aux_source_directory是否扫到了不该编译的.c

问题4：ninja.exe not found

# 解决：安装Ninja或改用Make
cmake -G "Unix Makefiles" ..   # 用make代替ninja
# 或安装Ninja后：
winget install Ninja-build.Ninja

问题5：链接脚本路径不对

# 错误：cannot open linker script file
# 解决：用绝对路径或正确引用
# ❌ 错：-TSTM32F103C8.ld         （相对路径，cmake构建时无效）
# ✅ 对：-T"${CMAKE_SOURCE_DIR}/STM32F103C8.ld"
# ✅ 对：-T"${CMAKE_CURRENT_SOURCE_DIR}/STM32F103C8.ld"

📋 CMake 速查表

操作命令
配置（生成Makefile）cmake -G "Unix Makefiles" -DCMAKE_BUILD_TYPE=MinSizeRel ..
配置（Ninja，更快）cmake -G Ninja -DCMAKE_BUILD_TYPE=MinSizeRel ..
配置（指定工具链）cmake -DCMAKE_TOOLCHAIN_FILE=../cmake/toolchain-arm.cmake ..
构建cmake --build . -j 4
清缓存重配Remove-Item -Recurse build; mkdir build; cd build; cmake ..
查看所有targetcmake --build . --target help
只生成.bincmake --build . --target app.bin
命令作用
cmake_minimum_required()最低CMake版本要求
project()定义项目名和语言
add_executable()生成可执行目标
add_library()生成库（STATIC/SHARED/MODULE）
add_subdirectory()包含子目录的CMakeLists
target_include_directories()设置include搜索路径
target_link_libraries()链接库
target_compile_options()设置编译选项
add_custom_command()添加自定义构建步骤
add_custom_target()添加自定义构建目标
aux_source_directory()自动收集目录中的源文件
file(GLOB)用通配符收集文件
option()定义可配置选项
set()定义变量
if/endif()条件判断
