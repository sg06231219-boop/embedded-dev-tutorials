步距角与角度计算

每步角度计算（八拍模式）：

1. 电机轴步距角 = 5.625°（每步）
2. 减速比 = 64:1（常见版本）
3. 输出轴每步角度 = 5.625° / 64 = 0.08789°
4. 每圈步数 = 360° / 0.08789° ≈ 4096步

角度 → 步数换算：
  steps = angle / 0.08789
  steps = angle × 11.378  （约）

示例：
  转90°  → 90 / 0.08789 ≈ 1024步
  转180° → 180 / 0.08789 ≈ 2048步
  转一圈 → 360 / 0.08789 ≈ 4096步

注意：如果实测减速比是1:6.1（约）：
  每圈步数 ≈ 2048步
  每步角度 ≈ 0.176°

main.c — 步进电机完整控制

/* USER CODE BEGIN Includes */
#include <stdio.h>
#include <stdint.h>
#include <math.h>
/* USER CODE END Includes */

/* USER CODE BEGIN PV */

// 步进电机相序表（四相八拍）
// 顺序: IN1(A) IN2(B) IN3(C) IN4(D)
// 对应PA0 PA1 PA2 PA3
static const uint8_t STEP_SEQ[8] = {
    0x01,  // 0b0001 - A相通电
    0x03,  // 0b0011 - A+B相通电
    0x02,  // 0b0010 - B相通电
    0x06,  // 0b0110 - B+C相通电
    0x04,  // 0b0100 - C相通电
    0x0C,  // 0b1100 - C+D相通电
    0x08,  // 0b1000 - D相通电
    0x09,  // 0b1001 - D+A相通电
};

// 电机状态
typedef enum {
    STEP_DIR_CW = 0,   // 顺时针
    STEP_DIR_CCW,       // 逆时针
} StepDir_t;

typedef struct {
    uint32_t current_step;    // 当前步数
    uint32_t target_step;      // 目标步数
    StepDir_t direction;       // 旋转方向
    uint8_t running;           // 运行状态
    uint16_t speed_delay_us;   // 每步延时(μs)，越小越快
} Stepper_t;

volatile Stepper_t g_stepper = {0};

// 28BYJ-48 参数（请根据实测调整）
#define STEPS_PER_REV   4096    // 每圈步数（减速比64:1 版本）
// #define STEPS_PER_REV   2038    // 每圈步数（实测减速比≈6.1 版本）

// 步进电机GPIO端口
#define STEP_PORT GPIOA
#define STEP_PIN_A GPIO_PIN_0
#define STEP_PIN_B GPIO_PIN_1
#define STEP_PIN_C GPIO_PIN_2
#define STEP_PIN_D GPIO_PIN_3

/* USER CODE END PV */

/* USER CODE BEGIN 0 */

// 设置相序输出
static void Stepper_SetPhase(uint8_t phase_idx)
{
    uint8_t seq = STEP_SEQ[phase_idx & 0x07];  // 取模8
    
    HAL_GPIO_WritePin(STEP_PORT, STEP_PIN_A, (seq & 0x01) ? GPIO_PIN_SET : GPIO_PIN_RESET);
    HAL_GPIO_WritePin(STEP_PORT, STEP_PIN_B, (seq & 0x02) ? GPIO_PIN_SET : GPIO_PIN_RESET);
    HAL_GPIO_WritePin(STEP_PORT, STEP_PIN_C, (seq & 0x04) ? GPIO_PIN_SET : GPIO_PIN_RESET);
    HAL_GPIO_WritePin(STEP_PORT, STEP_PIN_D, (seq & 0x08) ? GPIO_PIN_SET : GPIO_PIN_RESET);
}

// 微秒延时（使用定时器或简单循环）
static void Delay_us(uint16_t us)
{
    // 简单延时（不精确，适合步进电机）
    // 对于72MHz STM32，大约需要 72 个循环/μs
    uint32_t count = us * 72;
    while (count--) {
        __NOP();
    }
}

// 步进运行一步
// phase: 当前相序 (0~7)
// dir: 方向
// delay_us: 每步延时
// 返回: 下一个相序
static uint8_t Stepper_Step(uint8_t phase, StepDir_t dir, uint16_t delay_us)
{
    Stepper_SetPhase(phase);
    
    if (dir == STEP_DIR_CW) {
        phase = (phase + 1) & 0x07;   // 顺时针: 0→1→2→...→7→0
    } else {
        phase = (phase - 1) & 0x07;   // 逆时针: 0→7→6→...→1→0
    }
    
    Delay_us(delay_us);
    return phase;
}

// 转动指定步数
// steps: 步数
// dir: 方向
// speed: 速度等级 (1~10, 1最慢，10最快)
void Stepper_MoveSteps(uint32_t steps, StepDir_t dir, uint8_t speed)
{
    // 速度等级 → 每步延时 映射
    // 速度越大，延时越短
    uint16_t delay_table[11] = {
        0,      // 不使用
        12000,  // 1 - 很慢
        8000,   // 2
        5000,   // 3
        3000,   // 4
        2000,   // 5
        1500,   // 6
        1000,   // 7
        800,    // 8
        600,    // 9
        500,    // 10 - 很快（可能失步）
    };
    
    uint16_t delay_us = delay_table[speed > 10 ? 10 : (speed == 0 ? 1 : speed)];
    
    uint8_t phase = 0;
    for (uint32_t i = 0; i < steps; i++) {
        phase = Stepper_Step(phase, dir, delay_us);
        
        // 可在此处添加位置记录
        if (dir == STEP_DIR_CW) {
            g_stepper.current_step++;
        } else {
            g_stepper.current_step--;
        }
    }
    
    // 断电（可选，省电但失去保持力矩）
    // Stepper_SetPhase(8);  // 全部断电
}

// 转动指定角度（度）
void Stepper_MoveAngle(float angle_deg, StepDir_t dir, uint8_t speed)
{
    uint32_t steps = (uint32_t)(fabs(angle_deg) * STEPS_PER_REV / 360.0f);
    printf("Moving %.1f° = %lu steps [%s] speed=%d\r\n",
           angle_deg, steps,
           dir == STEP_DIR_CW ? "CW" : "CCW", speed);
    
    Stepper_MoveSteps(steps, dir, speed);
}

// 梯形加减速运行
// 实现平滑启动和停止，避免失步和振动
void Stepper_MoveAccel(uint32_t total_steps, StepDir_t dir)
{
    // 梯形加减速参数
    uint32_t accel_steps  = total_steps / 4;     // 加速段步数
    uint32_t decel_steps  = total_steps / 4;     // 减速段步数
    uint32_t const_steps  = total_steps - accel_steps - decel_steps;
    
    uint16_t min_delay_us = 500;   // 最高速延时
    uint16_t max_delay_us = 8000;  // 最低速延时
    
    uint8_t phase = 0;
    uint32_t step_count = 0;
    
    printf("Trapezoidal move: %lu steps\r\n", total_steps);
    
    while (step_count < total_steps) {
        uint16_t delay_us;
        
        if (step_count < accel_steps) {
            // 加速段：延时逐渐减小
            delay_us = max_delay_us - (step_count * (max_delay_us - min_delay_us) / accel_steps);
        }
        else if (step_count < accel_steps + const_steps) {
            // 匀速段
            delay_us = min_delay_us;
        }
        else {
            // 减速段：延时逐渐增大
            uint32_t decel_step = step_count - accel_steps - const_steps;
            delay_us = min_delay_us + (decel_step * (max_delay_us - min_delay_us) / decel_steps);
        }
        
        phase = Stepper_Step(phase, dir, delay_us);
        step_count++;
    }
}

// 精确位置控制（绝对位置）
// 将电机转到指定角度（相对0位的绝对角度）
static int32_t g_abs_position = 0;  // 当前绝对位置（步数）

void Stepper_MoveTo(int32_t target_angle_deg, uint8_t speed)
{
    int32_t target_steps = (int32_t)(target_angle_deg * STEPS_PER_REV / 360.0f);
    int32_t delta = target_steps - g_abs_position;
    
    StepDir_t dir = (delta >= 0) ? STEP_DIR_CW : STEP_DIR_CCW;
    uint32_t steps = (uint32_t)abs(delta);
    
    printf("MoveTo: %d° (from %d° to %d°)\r\n",
           target_angle_deg,
           (int)(g_abs_position * 360 / STEPS_PER_REV),
           target_angle_deg);
    
    Stepper_MoveSteps(steps, dir, speed);
    g_abs_position = target_steps;
}

❓ 常见问题排查

问题1：电机不转或抖动

最常见原因：相序错误！
1. 相序表错误 — 28BYJ-48的相序有多种版本，请对照表格逐一测试：
// 版本2（有些ULN2003板是反序的）
static const uint8_t STEP_SEQ_ALT[8] = {
    0x01,  // IN1
    0x09,  // IN1+IN4
    0x08,  // IN4
    0x0C,  // IN4+IN3
    0x04,  // IN3
    0x06,  // IN3+IN2
    0x02,  // IN2
    0x03,  // IN2+IN1
};
2. GPIO引脚配置错误 — 确认PA0~PA3是Output模式，不是Input。
3. 延时太短 — 速度太快电机会丢失步数，降低速度（增大delay_us）。
4. 电机未接触良好 — 用手转动电机轴，如果轻松转动说明线圈没通电。

问题2：失步（加速太快）

步进电机有最大启动力矩和最大响应频率，加速太快会失步：
- 启动速度：建议从2000μs/步开始，确认能启动
- 加速曲线：使用梯形加减速（代码中已实现）
- 负载过重：减轻负载或换用更大扭矩的步进电机
- 驱动电流不足：ULN2003最大500mA/路，28BYJ-48约200mA，理论上够用

问题3：28BYJ-48减速比到底是多少？

这是28BYJ-48项目中最大的坑！
实测方法：
// 让电机转2048步（半圈），观察实际转动
Stepper_MoveSteps(2048, STEP_DIR_CW, 3);

// 如果转了半圈 → 减速比约1:64，每圈4096步
// 如果转了整圈 → 减速比约1:128，每圈8192步
// 如果转了约1/4圈 → 减速比约1:6.1，每圈2038步
常见结论是：实测每圈约2038步（减速比≈1:6.1，不是标称的1:64）。
请根据实测结果修改 STEPS_PER_REV 宏定义。

问题4：电机发热严重

步进电机在静止时仍保持励磁，会持续发热（正常）。
解决方案：
1. 完成运动后调用 Stepper_SetPhase(8) 断电
2. 如果需要保持位置，使用半流保持（降低励磁电流）
3. 不要长时间全速运行

问题5：想用定时器中断驱动步进

上面的代码使用延时函数驱动，会阻塞主循环。
非阻塞驱动（推荐）：
// 用TIM3中断控制步进
// 中断中更新相序，主循环设置速度和目标位置

volatile uint32_t g_step_counter = 0;
volatile uint32_t g_step_target = 0;
volatile uint8_t  g_step_running = 0;

void HAL_TIM_PeriodElapsedCallback(TIM_HandleTypeDef *htim)
{
    if (htim->Instance == TIM3) {
        if (g_step_running && g_step_counter < g_step_target) {
            static uint8_t phase = 0;
            phase = Stepper_Step(phase, g_dir, 0);  // 0延时，由定时器周期控制速度
            g_step_counter++;
        } else {
            g_step_running = 0;
        }
    }
}

// 设置速度（定时器ARR值）
void Stepper_SetSpeed(uint16_t speed) {
    __HAL_TIM_SET_AUTORELOAD(&htim3, speed);
}

📐 步进电机微步细分原理

全步(full-step)时每一步1.8°(200步/转)，通过控制两相电流比例可以实现更小的步进角：

// 两相电流与微步位置的关系
// I_A = I_max × cos(θ), I_B = I_max × sin(θ)
// θ = (step / microsteps_per_step) × 90°

// 16微步的电流表(正弦表，DAC输出或PWM+滤波)
const uint16_t microstep_sine[16] = {

⚡ S曲线加减速算法

// S曲线加速 vs 梯形加速
// 梯形: 突然起停 → 机械振动、丢步
// S曲线: 加速度渐变 → 平滑、安静、高速

// Bresenham算法 + S曲线速度表
#define ACCEL_STEPS 200  // 加速步数

// 生成S曲线频率表（定时器重载值）
uint16_t s_curve_freq[ACCEL_STEPS];

void Generate_SCurve(uint32_t start_freq, uint32_t target_freq) {
    for (int i = 0; i < ACCEL_STEPS; i++) {
        float t = (float)i / ACCEL_STEPS;  // 0→1
        // S曲线公式: 使用sigmoid或三角函数的对称S
        // f(t) = start + (target-start) × (1-cos(π×t))/2
        float s = (1.0f - cosf(3.14159f * t)) / 2.0f;
        float freq = start_freq + (target_freq - start_freq) * s;
        s_curve_freq[i] = (uint16_t)(SystemCoreClock / freq);
    }
}

// 步进脉冲定时器中断
void TIMx_IRQHandler(void) {
    // 更新定时器ARR → 改变脉冲频率
    if (step_index < ACCEL_STEPS) {
        TIMx->ARR = s_curve_freq[step_index++];  // 加速
    } else if (step_index < total_steps - ACCEL_STEPS) {
        TIMx->ARR = s_curve_freq[ACCEL_STEPS-1];  // 匀速
    } else {
        TIMx->ARR = s_curve_freq[total_steps - step_index];  // 减速
    }
}

📏 共振与失步问题

步进电机共振区： 每台步进电机都有特定的共振频率，在此频率下振动幅度急剧增大，可能导致丢步。常见共振区在50-200RPM。

解决方案：
1. 加速时快速穿过共振区（不在此频率段做匀速）
2. 使用微步驱动（16细分以上，振动减小到1/16）
3. 机械阻尼（加装减震垫、柔性联轴器）
4. 使用TMC系列芯片的共振抑制算法
5. 电流闭环控制（通过反电动势检测丢步）

❓ 常见问题 FAQ

Q: 步进电机丢步？

A: 1)加速度太大，需要梯形/S形加减速曲线 2)负载过重 3)驱动电压不够 4)脉冲频率突变

Q: 28BYJ-48为什么转得慢？

A: 减速比1:64，实际步距角5.625°/64=0.088°。需要高速时换NEMA17+A4988驱动。
