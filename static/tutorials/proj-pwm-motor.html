🔬 PWM原理详解

什么是PWM？

PWM（Pulse Width Modulation，脉宽调制）是一种通过改变脉冲的占空比来等效改变输出电压的技术。

占空比（Duty Cycle） = 高电平时间 / 周期时间 × 100%
例如：周期10ms，高电平6ms → 占空比60% → 等效输出电压 = 3.6V（5V × 60%）

占空比与转速的关系

占空比CCR值(PSC=71,ARR=999)等效电压电机状态
0%00V停止
25%2501V低速
50%5002V中速
75%7503V高速
100%9994V最高速

定时器PWM频率计算

PWM频率 = TIM时钟 / ((PSC+1) × (ARR+1))

STM32F103 TIM3 时钟 = 72MHz
设置 PSC = 71, ARR = 999:
PWM频率 = 72000000 / (72 × 1000) = 1000Hz = 1kHz

这是一个适合电机控制的频率（通常1~20kHz）

💻 完整代码

main.c — 核心业务逻辑

/* USER CODE BEGIN Includes */
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
/* USER CODE END Includes */

/* USER CODE BEGIN PV */
// 电机方向枚举
typedef enum {
    MOTOR_STOP = 0,
    MOTOR_FWD,
    MOTOR_REV
} MotorDir_t;

// 全局变量
volatile uint16_t g_duty = 0;          // 当前占空比 (0~999)
volatile MotorDir_t g_dir = MOTOR_STOP; // 当前方向
volatile uint8_t g_speed_changed = 0;   // 速度变化标志

#define DUTY_STEP  50    // 每次调速步进值
#define DUTY_MAX   999   // 最大占空比
#define DUTY_MIN   0     // 最小占空比
/* USER CODE END PV */

/* USER CODE BEGIN 0 */

// 重定向printf到USART1
#ifdef __GNUC__
int __io_putchar(int ch)
#else
int fputc(int ch, FILE *f)
#endif
{
    HAL_UART_Transmit(&huart1, (uint8_t *)&ch, 1, HAL_MAX_DELAY);
    return ch;
}

// 设置电机方向
void Motor_SetDirection(MotorDir_t dir)
{
    switch(dir) {
        case MOTOR_FWD:
            HAL_GPIO_WritePin(GPIOA, GPIO_PIN_0, GPIO_PIN_SET);    // IN1 = 1
            HAL_GPIO_WritePin(GPIOA, GPIO_PIN_1, GPIO_PIN_RESET);  // IN2 = 0
            break;
        case MOTOR_REV:
            HAL_GPIO_WritePin(GPIOA, GPIO_PIN_0, GPIO_PIN_RESET);  // IN1 = 0
            HAL_GPIO_WritePin(GPIOA, GPIO_PIN_1, GPIO_PIN_SET);    // IN2 = 1
            break;
        case MOTOR_STOP:
        default:
            HAL_GPIO_WritePin(GPIOA, GPIO_PIN_0, GPIO_PIN_RESET);  // IN1 = 0
            HAL_GPIO_WritePin(GPIOA, GPIO_PIN_1, GPIO_PIN_RESET);  // IN2 = 0
            break;
    }
    g_dir = dir;
}

// 设置PWM占空比
void Motor_SetDuty(uint16_t duty)
{
    if (duty > DUTY_MAX) duty = DUTY_MAX;
    g_duty = duty;
    __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_1, duty);
    
    // 占空比为0时停止电机
    if (duty == 0) {
        Motor_SetDirection(MOTOR_STOP);
    } else if (g_dir == MOTOR_STOP) {
        // 有占空比但没方向时默认正转
        Motor_SetDirection(MOTOR_FWD);
    }
}

// 串口接收缓冲区
uint8_t g_rx_buf[64];
volatile uint8_t g_rx_flag = 0;

// 串口接收完成回调
void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart)
{
    if (huart->Instance == USART1) {
        g_rx_flag = 1;
        HAL_UART_Receive_IT(&huart1, g_rx_buf, 1);
    }
}

// 解析串口命令
void Parse_UART_Command(uint8_t *buf)
{
    // 格式: SPEED:500 或 DIR:REV 或 DIR:FWD 或 STOP
    if (strncmp((char*)buf, "SPEED:", 6) == 0) {
        int val = atoi((char*)(buf + 6));
        if (val >= 0 && val <= 999) {
            Motor_SetDuty((uint16_t)val);
            printf("Duty set to %d\r\n", val);
        }
    }
    else if (strncmp((char*)buf, "DIR:FWD", 7) == 0) {
        Motor_SetDirection(MOTOR_FWD);
        printf("Direction: Forward\r\n");
    }
    else if (strncmp((char*)buf, "DIR:REV", 7) == 0) {
        Motor_SetDirection(MOTOR_REV);
        printf("Direction: Reverse\r\n");
    }
    else if (strncmp((char*)buf, "STOP", 4) == 0) {
        Motor_SetDuty(0);
        printf("Motor stopped\r\n");
    }
}

// 外部中断回调 — 按键调速
void HAL_GPIO_EXTI_Callback(uint16_t GPIO_Pin)
{
    if (GPIO_Pin == GPIO_PIN_0) {  // PB0 加速
        if (g_duty + DUTY_STEP <= DUTY_MAX) {
            Motor_SetDuty(g_duty + DUTY_STEP);
        }
        g_speed_changed = 1;
    }
    else if (GPIO_Pin == GPIO_PIN_1) {  // PB1 减速
        if (g_duty >= DUTY_STEP) {
            Motor_SetDuty(g_duty - DUTY_STEP);
        } else {
            Motor_SetDuty(0);
        }
        g_speed_changed = 1;
    }
}
/* USER CODE END 0 */

int main(void)
{
    HAL_Init();
    SystemClock_Config();
    MX_GPIO_Init();
    MX_TIM3_Init();
    MX_USART1_UART_Init();

    /* USER CODE BEGIN 2 */
    // 启动PWM输出
    HAL_TIM_PWM_Start(&htim3, TIM_CHANNEL_1);
    
    // 初始占空比为0，电机静止
    Motor_SetDuty(0);
    Motor_SetDirection(MOTOR_STOP);
    
    // 启动串口接收中断
    HAL_UART_Receive_IT(&huart1, g_rx_buf, 1);
    
    printf("Motor PWM Control Ready!\r\n");
    printf("Commands: SPEED:0~999, DIR:FWD, DIR:REV, STOP\r\n");
    /* USER CODE END 2 */

    /* USER CODE BEGIN WHILE */
    while (1)
    {
        // 串口命令处理
        if (g_rx_flag) {
            g_rx_flag = 0;
            Parse_UART_Command(g_rx_buf);
        }
        
        // 按键调速后打印当前状态
        if (g_speed_changed) {
            g_speed_changed = 0;
            printf("Duty: %d/999 (%.1f%%)\r\n", g_duty, (float)g_duty / 9.99f);
        }
        
        HAL_Delay(10);
    }
    /* USER CODE END WHILE */
}

❓ 常见问题排查

问题1：电机不转

可能原因与解决方案：
1. ENA没有PWM信号 — 最常见！ENA必须接收PWM波才能调速，单纯接高电平只能全速运行。用示波器或逻辑分析仪检查PA6是否有PWM输出。
2. 忘记调用 HAL_TIM_PWM_Start() — 这是新手最常犯的错误！配置好定时器后必须手动启动。
3. 12V电源未连接 — L298N需要独立供电，仅靠STM32的3.3V无法驱动电机。
4. IN1/IN2都为0 — 两个方向引脚都为低时电机自由停止，需要设置一个为高。

问题2：转向反了

解决方案：交换IN1和IN2的接线，或者在代码中交换 GPIO_PIN_SET 和 GPIO_PIN_RESET。也可以直接交换电机接OUT1/OUT2的两根线。

问题3：转速不稳定

解决方案：
1. 确认电源供电稳定，12V电源功率足够
2. 在L298N的12V和GND之间并联一个100μF电解电容 + 0.1μF陶瓷电容
3. 如果需要精确转速控制，加入PID算法，使用编码器反馈实际转速：
// 简易PID结构
typedef struct {
    float Kp, Ki, Kd;
    float target;      // 目标转速
    float feedback;    // 实际转速(编码器)
    float error, last_error, integral;
    float output;      // PID输出 → CCR
} PID_t;

float PID_Calculate(PID_t *pid, float feedback)
{
    pid->feedback = feedback;
    pid->error = pid->target - feedback;
    pid->integral += pid->error;
    
    // 积分限幅
    if (pid->integral > 500) pid->integral = 500;
    if (pid->integral < -500) pid->integral = -500;
    
    pid->output = pid->Kp * pid->error
                + pid->Ki * pid->integral
                + pid->Kd * (pid->error - pid->last_error);
    
    pid->last_error = pid->error;
    return pid->output;
}

问题4：L298N发热严重

L298N效率较低（压降约2V），长时间大电流运行会发热。建议：
1. 加装散热片
2. 控制电流不超过1A（L298N单路最大2A）
3. 如需更高效率，可换用DRV8833或TB6612驱动模块

问题5：串口命令无响应

1. 检查波特率是否匹配（默认115200）
2. 确认TX/RX交叉连接（STM32 TX→TTL RX）
3. 检查是否调用了 HAL_UART_Receive_IT() 启动接收
4. 确认命令格式正确，如 SPEED:500（大写）

🧮 死区时间 (Dead-Time) 详解

H桥在同一半桥的两个MOSFET/BJT之间切换时，如果控制信号没有延时，会瞬间短路（直通/shoot-through），烧毁驱动芯片。

// STM32高级定时器死区插入 - TIM1/TIM8支持
// 死区时间 = DTG[7:0] × Tdtg
// Tdtg = tDTS (当 DTG[7:5]=000), tDTS = 1/72MHz = 13.89ns

void TIM1_DeadTime_Config(uint8_t deadtime_ns) {
    // 死区时间 = deadtime_ns / 13.89ns ≈ deadtime_ns * 72 / 1000
    uint32_t dtg = (deadtime_ns * 72) / 1000;
    if (dtg > 0x7F) dtg = 0x7F;  // 最大127 → ~1.76μs
    TIM1->BDTR = (TIM1->BDTR & ~TIM_BDTR_DTG) | (dtg & TIM_BDTR_DTG);
    TIM1->BDTR |= TIM_BDTR_MOE;  // 主输出使能
    TIM_CCxNCmd(TIM1, TIM_Channel_1, TIM_CCxN_Enable);  // 互补输出
}

// 典型值: 100-500ns 用于MOSFET, 1-2μs 用于BJT
// 太小→直通风险, 太大→输出失真
L298N需要死区吗？ L298N的BJT开关速度慢（μs级），自带"天然死区"，所以接IN1/IN2反向控制时不需要额外插入死区。但换用MOSFET驱动器时必须配置！

📐 电机电流采样与保护

// 使用ADC采样L298N的电流检测引脚(ISENA/ISENB)
// ISEN输出 = 电机电流 × Rsense × 增益
// 典型Rsense=0.5Ω, V_ISEN = I_motor × 0.5V/A

void Motor_CurrentMonitor(void) {
    HAL_ADC_Start(&hadc1);
    if (HAL_ADC_PollForConversion(&hadc1, 10) == HAL_OK) {
        uint32_t adc_val = HAL_ADC_GetValue(&hadc1);
        float voltage = adc_val * 3.3f / 4096.0f;
        float current = voltage / 0.5f;  // I = V / Rsense
        
        // 过流保护 - 超过2A立即停止
        if (current > 2.0f) {
            TIM3->CCR1 = 0;  // PWM=0
            HAL_GPIO_WritePin(IN1_GPIO_Port, IN1_Pin, GPIO_PIN_RESET);
            HAL_GPIO_WritePin(IN2_GPIO_Port, IN2_Pin, GPIO_PIN_RESET);
        }
    }
}

🔄 无刷直流电机(BLDC)入门

BLDC电机效率更高(85-95%)、寿命更长，是无人机/电动车的主流选择。控制方式分为：

控制方式复杂度效率噪音应用
六步换相(6-step)低中大(转矩波动)风扇/水泵
正弦波控制中高小家用电器
FOC (矢量控制)高最高最低机器人/电动车
从有刷到无刷的学习路径： 有刷PWM → 理解换相原理 → 六步换相(Hall传感器) → 反电动势检测(无传感器) → FOC(Clark/Park变换+电流环+速度环) → MTPA(最大转矩电流比)

🔬 用逻辑分析仪调试PWM信号

调试步骤:
1. 逻辑分析仪CH0接PWM输出引脚(如PA6=TIM3_CH1)
2. 设置采样率≥10MHz (1kHz PWM需要至少100x采样率)
3. 触发模式: 上升沿触发
4. 测量: 周期→验证频率, 高电平宽度/周期→验证占空比

常见异常:
- 占空比不随CCR变化 → 检查CCR是否更新到影子寄存器(TIM预装载)
- 频率不是期望值 → 检查PSC/ARR计算, 确认TIM时钟源(APB1=36MHz或APB2=72MHz?)
- 输出一直为高/低 → 检查GPIO复用功能配置, AFIO重映射

❓ 常见问题 FAQ

Q: 电机不转？

A: 1)PWM频率太高/太低（1~20kHz合适） 2)L298N使能脚未拉高 3)供电不足（电机需要独立电源）

Q: 电机啸叫？

A: PWM频率在人耳范围(20Hz~20kHz)，改到20kHz以上听不到。但太高开关损耗大，折中选16~25kHz。
