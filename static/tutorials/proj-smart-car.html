教程11: 智能小车综合项目

📋 项目概述

本项目构建一台多功能智能小车，搭载STM32F103C8T6主控，集成电机驱动、超声波避障、红外遥控、蓝牙APP控制和PID速度闭环。通过状态机设计统一协调各功能模块，是嵌入式开发的经典综合实战项目。

📝 接线文字描述：
STM32最小系统板安装在小车底盘中央。两块L298N分别固定在左右两侧：L298N-1驱动前方两个电机，L298N-2驱动后方两个电机。每块L298N的12V输入接7.4V电池正极，GND接电池负极并连到STM32的GND形成共地。L298N的5V输出可不接（跳线帽保持EN常开或用PWM控制）。HC-SR04超声波安装在小车前端，Trig接PA4，Echo经10K+20K分压后接PA5（将5V降到3.3V）。HX1838红外接收头装在车顶，OUT接PA6。HC-05蓝牙模块RXD接PA9，TXD接PA10，VCC接5V。OLED固定在车尾或车顶，SCL→PB6，SDA→PB7。电池组通过开关连接L298N和降压模块（降压到5V给STM32和模块供电）。

🛠️ 分步骤操作指引

步骤1：组装底盘与电机
1. 将4个TT减速电机安装到底盘四角
2. 安装轮子，确保转动顺畅无卡顿
3. 固定电池盒和电源开关
4. 理线，避免转动部件缠绕线缆
步骤2：CubeMX配置
1. SYS → Debug: Serial Wire
2. RCC → HSE: Crystal
3. Clock: 72MHz
4. TIM3 → CH3/CH4: PWM Generation, Prescaler=71, Period=999 → 1KHz PWM
5. TIM2 → 用于超声波Echo测量(输入捕获)
6. USART1 → 9600 baud (匹配HC-05默认)
7. I2C1 → Standard 100KHz
8. PA0-PA3, PB8/PB9/PB12/PB13: GPIO_Output (电机方向)
9. PA4: GPIO_Output (Trig), PA5: GPIO_Input (Echo)
10. PA6: GPIO_Input_EXTI (红外接收)
步骤3：电机驱动调试
1. 先只连一个电机测试
2. 设置IN1=1, IN2=0 → 正转；IN1=0, IN2=1 → 反转
3. 调节PWM占空比观察速度变化
4. 确认4个电机方向一致（安装可能镜像）
步骤4：超声波避障测试
1. 单独测试测距精度
2. 在20cm处放障碍物，验证读数
3. 设置安全距离阈值（如15cm）
步骤5：红外/蓝牙遥控调试
1. 用串口助手发命令测试蓝牙
2. 用遥控器测试红外解码
3. 确认各按键对应的键码
步骤6：PID速度闭环
1. 加装编码器测量电机转速
2. 先调P，再调I，最后调D
3. 目标：不同占空比下速度稳定
步骤7：状态机集成与联调
1. 整合所有模块到状态机
2. 测试模式切换
3. 测试紧急停止功能

motor.c — 电机驱动实现

#include "motor.h"
#include "tim.h"

/* 方向引脚映射 */
typedef struct {
    GPIO_TypeDef *in1_port; uint16_t in1_pin;
    GPIO_TypeDef *in2_port; uint16_t in2_pin;
    TIM_HandleTypeDef *pwm_tim; uint32_t pwm_channel;
} MotorConfig_t;

static const MotorConfig_t motors[4] = {
    {GPIOA, GPIO_PIN_0, GPIOA, GPIO_PIN_1, &htim3, TIM_CHANNEL_3},  // FL
    {GPIOA, GPIO_PIN_2, GPIOA, GPIO_PIN_3, &htim3, TIM_CHANNEL_4},  // FR
    {GPIOB, GPIO_PIN_8, GPIOB, GPIO_PIN_9, &htim12, TIM_CHANNEL_1}, // BL
    {GPIOB, GPIO_PIN_12,GPIOB, GPIO_PIN_13,&htim12, TIM_CHANNEL_2}, // BR
};

void Motor_Init(void)
{
    HAL_TIM_PWM_Start(&htim3, TIM_CHANNEL_3);
    HAL_TIM_PWM_Start(&htim3, TIM_CHANNEL_4);
    HAL_TIM_PWM_Start(&htim12, TIM_CHANNEL_1);
    HAL_TIM_PWM_Start(&htim12, TIM_CHANNEL_2);
    Car_Stop();
}

void Motor_SetSpeed(MotorID_t id, uint16_t speed)
{
    if (speed > 1000) speed = 1000;
    __HAL_TIM_SET_COMPARE(motors[id].pwm_tim, motors[id].pwm_channel, speed);
}

void Motor_SetDirection(MotorID_t id, MotorDir_t dir)
{
    switch (dir) {
    case DIR_FORWARD:
        HAL_GPIO_WritePin(motors[id].in1_port, motors[id].in1_pin, GPIO_PIN_SET);
        HAL_GPIO_WritePin(motors[id].in2_port, motors[id].in2_pin, GPIO_PIN_RESET);
        break;
    case DIR_BACKWARD:
        HAL_GPIO_WritePin(motors[id].in1_port, motors[id].in1_pin, GPIO_PIN_RESET);
        HAL_GPIO_WritePin(motors[id].in2_port, motors[id].in2_pin, GPIO_PIN_SET);
        break;
    case DIR_STOP:
        HAL_GPIO_WritePin(motors[id].in1_port, motors[id].in1_pin, GPIO_PIN_RESET);
        HAL_GPIO_WritePin(motors[id].in2_port, motors[id].in2_pin, GPIO_PIN_RESET);
        break;
    }
}

/* 全车运动控制 */
void Car_Forward(uint16_t speed)
{
    for (int i = 0; i < 4; i++) {
        Motor_SetDirection(i, DIR_FORWARD);
        Motor_SetSpeed(i, speed);
    }
}

void Car_Backward(uint16_t speed)
{
    for (int i = 0; i < 4; i++) {
        Motor_SetDirection(i, DIR_BACKWARD);
        Motor_SetSpeed(i, speed);
    }
}

void Car_TurnLeft(uint16_t speed)
{
    // 左侧后退，右侧前进
    Motor_SetDirection(MOTOR_FL, DIR_BACKWARD);
    Motor_SetDirection(MOTOR_BL, DIR_BACKWARD);
    Motor_SetDirection(MOTOR_FR, DIR_FORWARD);
    Motor_SetDirection(MOTOR_BR, DIR_FORWARD);
    for (int i = 0; i < 4; i++) Motor_SetSpeed(i, speed);
}

void Car_TurnRight(uint16_t speed)
{
    Motor_SetDirection(MOTOR_FL, DIR_FORWARD);
    Motor_SetDirection(MOTOR_BL, DIR_FORWARD);
    Motor_SetDirection(MOTOR_FR, DIR_BACKWARD);
    Motor_SetDirection(MOTOR_BR, DIR_BACKWARD);
    for (int i = 0; i < 4; i++) Motor_SetSpeed(i, speed);
}

void Car_Stop(void)
{
    for (int i = 0; i < 4; i++) {
        Motor_SetDirection(i, DIR_STOP);
        Motor_SetSpeed(i, 0);
    }
}

void Car_Brake(void)
{
    // 刹车：IN1=1, IN2=1 短接电机
    for (int i = 0; i < 4; i++) {
        HAL_GPIO_WritePin(motors[i].in1_port, motors[i].in1_pin, GPIO_PIN_SET);
        HAL_GPIO_WritePin(motors[i].in2_port, motors[i].in2_pin, GPIO_PIN_SET);
        Motor_SetSpeed(i, 0);
    }
}

ultrasonic.c — 超声波测距

#include "ultrasonic.h"
#include "gpio.h"
#include "tim.h"

#define TRIG_PORT GPIOA
#define TRIG_PIN  GPIO_PIN_4
#define ECHO_PORT GPIOA
#define ECHO_PIN  GPIO_PIN_5

static uint32_t echo_start = 0;
static uint32_t echo_end = 0;
static uint8_t  echo_captured = 0;

float HC_SR04_GetDistance(void)
{
    // 发送10us触发信号
    HAL_GPIO_WritePin(TRIG_PORT, TRIG_PIN, GPIO_PIN_SET);
    Delay_us(10);
    HAL_GPIO_WritePin(TRIG_PORT, TRIG_PIN, GPIO_PIN_RESET);

    // 等待Echo上升沿（超时30ms）
    uint32_t timeout = HAL_GetTick();
    while (HAL_GPIO_ReadPin(ECHO_PORT, ECHO_PIN) == GPIO_PIN_RESET) {
        if (HAL_GetTick() - timeout > 30) return -1.0f;
    }
    echo_start = TIM2->CNT;

    // 等待Echo下降沿
    timeout = HAL_GetTick();
    while (HAL_GPIO_ReadPin(ECHO_PORT, ECHO_PIN) == GPIO_PIN_SET) {
        if (HAL_GetTick() - timeout > 30) return -1.0f;
    }
    echo_end = TIM2->CNT;

    // 计算距离: 声速340m/s, 时间(us)*0.034/2
    uint32_t elapsed = echo_end - echo_start; // us
    float distance = elapsed * 0.034f / 2.0f;
    return distance;
}

void Delay_us(uint32_t us)
{
    uint32_t ticks = us * (SystemCoreClock / 1000000) / 5;
    while (ticks--);
}

ir_remote.c — 红外遥控解码

#include "ir_remote.h"
#include "gpio.h"

#define IR_PORT GPIOA
#define IR_PIN  GPIO_PIN_6

static uint32_t ir_data = 0;
static uint8_t  ir_bit_idx = 0;
static uint8_t  ir_ready = 0;
static uint32_t ir_last_fall = 0;

/* NEC协议解码(简化版) */
/* 在外部中断回调中调用 */
void IR_FallingEdgeHandler(void)
{
    uint32_t now = HAL_GetTick() * 1000 + SysTick->LOAD - SysTick->VAL;
    uint32_t pulse_width = now - ir_last_fall;
    ir_last_fall = now;

    if (pulse_width > 13000 && pulse_width < 14000) {
        // 引导码: 9ms低 + 4.5ms高
        ir_data = 0;
        ir_bit_idx = 0;
    } else if (pulse_width > 1000 && pulse_width < 1300) {
        // 逻辑0: 560us低 + 560us高
        ir_data >>= 1;
        ir_bit_idx++;
    } else if (pulse_width > 2000 && pulse_width < 2500) {
        // 逻辑1: 560us低 + 1.69ms高
        ir_data = (ir_data >> 1) | 0x80000000;
        ir_bit_idx++;
    }

    if (ir_bit_idx == 32) {
        ir_ready = 1;
    }
}

uint8_t IR_IsReady(void)
{
    return ir_ready;
}

uint32_t IR_GetData(void)
{
    ir_ready = 0;
    return ir_data;
}

uint8_t IR_GetKeyCode(void)
{
    if (!ir_ready) return 0xFF;
    uint32_t data = ir_data;
    ir_ready = 0;
    // NEC: 8位地址 + 8位地址反码 + 8位命令 + 8位命令反码
    uint8_t cmd = (data >> 8) & 0xFF;
    uint8_t cmd_inv = data & 0xFF;
    if ((cmd ^ cmd_inv) == 0xFF) {
        return cmd;
    }
    return 0xFF; // 解码失败
}

/* HAL GPIO外部中断回调 */
void HAL_GPIO_EXTI_Callback(uint16_t GPIO_Pin)
{
    if (GPIO_Pin == IR_PIN) {
        IR_FallingEdgeHandler();
    }
}

bluetooth.c — 蓝牙命令解析

#include "bluetooth.h"
#include "usart.h"
#include <string.h>
#include <stdlib.h>

#define BT_BUF_SIZE 64

static uint8_t bt_buf[BT_BUF_SIZE];
static uint8_t bt_idx = 0;

/* 蓝牙命令定义 */
/* 前进: F  后退: B  左转: L  右转: R  停止: S */
/* 速度: V<0-9>  模式: M<0-2>(0待机/1手动/2自动) */

typedef struct {
    char cmd;
    int16_t param;
} BT_Command_t;

static BT_Command_t last_cmd = {0, 0};

void BT_ProcessByte(uint8_t byte)
{
    if (byte == '\r' || byte == '\n') {
        if (bt_idx > 0) {
            bt_buf[bt_idx] = '\0';
            BT_ParseBuffer((char *)bt_buf);
            bt_idx = 0;
        }
    } else {
        if (bt_idx < BT_BUF_SIZE - 1) {
            bt_buf[bt_idx++] = byte;
        }
    }
}

BT_Command_t BT_GetLastCommand(void)
{
    BT_Command_t cmd = last_cmd;
    last_cmd.cmd = 0;  // 清除已读命令
    return cmd;
}

void BT_ParseBuffer(char *buf)
{
    if (strlen(buf) == 0) return;

    switch (buf[0]) {
    case 'F': last_cmd.cmd = 'F'; last_cmd.param = 0; break;
    case 'B': last_cmd.cmd = 'B'; last_cmd.param = 0; break;
    case 'L': last_cmd.cmd = 'L'; last_cmd.param = 0; break;
    case 'R': last_cmd.cmd = 'R'; last_cmd.param = 0; break;
    case 'S': last_cmd.cmd = 'S'; last_cmd.param = 0; break;
    case 'V':
        last_cmd.cmd = 'V';
        last_cmd.param = (strlen(buf) > 1) ? atoi(&buf[1]) : 5;
        if (last_cmd.param < 0) last_cmd.param = 0;
        if (last_cmd.param > 9) last_cmd.param = 9;
        break;
    case 'M':
        last_cmd.cmd = 'M';
        last_cmd.param = (strlen(buf) > 1) ? atoi(&buf[1]) : 0;
        break;
    default:
        break;
    }
}

/* USART接收中断回调 */
void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart)
{
    if (huart == &huart1) {
        static uint8_t rx_byte;
        BT_ProcessByte(rx_byte);
        HAL_UART_Receive_IT(&huart1, &rx_byte, 1);
    }
}

pid.c — PID速度控制器

#include "pid.h"

void PID_Init(PID_t *pid, float kp, float ki, float kd, float out_min, float out_max)
{
    pid->kp = kp;
    pid->ki = ki;
    pid->kd = kd;
    pid->setpoint = 0;
    pid->integral = 0;
    pid->last_error = 0;
    pid->out_min = out_min;
    pid->out_max = out_max;
}

float PID_Calculate(PID_t *pid, float feedback)
{
    float error = pid->setpoint - feedback;

    /* 积分限幅(抗积分饱和) */
    pid->integral += error;
    if (pid->integral > pid->out_max) pid->integral = pid->out_max;
    if (pid->integral < pid->out_min) pid->integral = pid->out_min;

    float derivative = error - pid->last_error;
    pid->last_error = error;

    float output = pid->kp * error + pid->ki * pid->integral + pid->kd * derivative;

    /* 输出限幅 */
    if (output > pid->out_max) output = pid->out_max;
    if (output < pid->out_min) output = pid->out_min;

    return output;
}

void PID_SetSetpoint(PID_t *pid, float setpoint)
{
    pid->setpoint = setpoint;
    pid->integral = 0;  // 切换目标时清积分
    pid->last_error = 0;
}

void StateMachine_Run(void)
{
    /* 1. 检测紧急条件(最高优先) */
    float dist = HC_SR04_GetDistance();
    if (dist > 0 && dist < safe_distance && car_state != CAR_IDLE) {
        Car_Brake();
        car_state = CAR_EMERGENCY;
        printf("EMERGENCY! Dist=%.1fcm\r\n", dist);
        return;
    }

    /* 2. 检查红外遥控 */
    if (IR_IsReady()) {
        uint8_t key = IR_GetKeyCode();
        switch (key) {
        case 0x45: car_state = CAR_MANUAL; printf("IR: Manual mode\r\n"); break;
        case 0x46: car_state = CAR_AUTO;   printf("IR: Auto mode\r\n");   break;
        case 0x47: car_state = CAR_IDLE; Car_Stop(); printf("IR: Idle\r\n"); break;
        case 0x40: Car_Forward(base_speed);  break; // OK
        case 0x19: Car_Backward(base_speed); break; // 1
        case 0x07: Car_TurnLeft(base_speed); break; // 2
        case 0x09: Car_TurnRight(base_speed);break; // 3
        case 0x15: Car_Stop();               break; // 5
        }
    }

    /* 3. 检查蓝牙命令 */
    BT_Command_t bt = BT_GetLastCommand();
    if (bt.cmd) {
        switch (bt.cmd) {
        case 'F': car_state = CAR_BT_CTRL; Car_Forward(base_speed); break;
        case 'B': car_state = CAR_BT_CTRL; Car_Backward(base_speed); break;
        case 'L': car_state = CAR_BT_CTRL; Car_TurnLeft(base_speed); break;
        case 'R': car_state = CAR_BT_CTRL; Car_TurnRight(base_speed); break;
        case 'S': Car_Stop();
                  car_state = CAR_BT_CTRL; break;
        case 'V': base_speed = bt.param * 100 + 100; break; // V0=100~V9=1000
        case 'M':
            if (bt.param == 0) { car_state = CAR_IDLE; Car_Stop(); }
            else if (bt.param == 1) car_state = CAR_MANUAL;
            else if (bt.param == 2) car_state = CAR_AUTO;
            break;
        }
    }

    /* 4. 状态机执行 */
    switch (car_state) {
    case CAR_IDLE:
        Car_Stop();
        break;

    case CAR_MANUAL:
        // 手动模式下由IR遥控器控制，此处无需额外逻辑
        break;

    case CAR_AUTO:
        AutoAvoid(dist);
        break;

    case CAR_BT_CTRL:
        // 蓝牙模式下由命令驱动，无需额外逻辑
        break;

    case CAR_EMERGENCY:
        // 等待距离恢复安全或切换模式
        if (dist > safe_distance + 10.0f) {
            car_state = CAR_AUTO;
            printf("Emergency cleared\r\n");
        }
        break;
    }

    /* 5. OLED状态显示 */
    OLED_ShowStatus(car_state, dist, base_speed);
}

void AutoAvoid(float distance)
{
    if (distance < 0) {
        // 超声波读取失败，慢速前进
        Car_Forward(base_speed / 3);
        return;
    }

    if (distance < safe_distance) {
        // 障碍物太近：先后退，再右转
        Car_Backward(base_speed / 2);
        HAL_Delay(500);
        Car_TurnRight(base_speed / 2);
        HAL_Delay(400);
    } else if (distance < safe_distance + 15.0f) {
        // 接近障碍物：右转避让
        Car_TurnRight(base_speed / 2);
    } else {
        // 安全距离：前进
        Car_Forward(base_speed);
    }
}

❓ 常见问题排查

⚠️ 问题1：电源功率不足
现象：电机启动时MCU复位、OLED闪烁、测距不稳定
原因：电机启动电流大(每路可达1A)，拉低供电电压
解决：
1. 电机供电与MCU供电分开：电池直供L298N，降压模块供MCU
2. 在MCU供电端加大电容(470μF电解+0.1μF陶瓷)
3. 电池容量不足时更换大容量电池组
4. 电机加速采用斜坡函数，避免全速启动
⚠️ 问题2：HC-05蓝牙默认波特率9600
现象：串口收发乱码
原因：CubeMX中USART1配置了115200，但HC-05默认9600
解决：
1. 将USART1波特率改为9600
2. 或通过AT命令修改HC-05波特率：AT+BAUD8→115200
3. 进入AT模式：按住HC-05按键上电，LED慢闪(2s间隔)表示进入
⚠️ 问题3：PID参数整定困难
现象：速度忽快忽慢、持续振荡、响应迟缓
原因：PID参数不匹配系统特性
解决：
1. 先开环控制(固定PWM)确认电机方向和编码器正常
2. 从小Kp开始逐步增大，找到产生持续振荡的临界Kp
3. 实际Kp取临界值的60%，Ki取Kp/(振荡周期×0.5)，Kd取Kp×振荡周期×0.125
4. 小车场景建议Kp偏小，防止轮子打滑
⚠️ 问题4：代码模块化不足
现象：改一个功能导致其他功能异常
原因：所有代码写在main.c，耦合严重
解决：
1. 严格模块化：每个外设独立.c/.h文件
2. 模块间通过函数接口通信，不直接操作其他模块的变量
3. 使用状态机统一调度，避免各模块互相调用
4. 全局变量用static限制作用域
⚠️ 问题5：HC-SR04 Echo引脚5V问题
现象：STM32 GPIO损坏或读数不准
原因：HC-SR04的Echo输出5V，STM32 GPIO耐受3.3V
解决：
1. Echo引脚串联10K+20K电阻分压，中间点接PA5
2. 或使用3.3V电平转换模块
3. 切勿直接连接5V信号到STM32 GPIO

❓ 常见问题 FAQ

Q: 小车跑偏？

A: 1)左右电机速度差（需单独校准PWM） 2)轮径不一致 3)电池电压下降导致速度变化 → 需要编码器+PID闭环

Q: 避障反应慢？

A: 1)超声波测距周期太长（降到50ms） 2)决策逻辑太复杂 3)电机响应延迟 → 提高控制频率到100Hz
