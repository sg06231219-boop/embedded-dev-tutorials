教程10: FreeRTOS多任务环境监测站

📋 项目概述

本项目使用 STM32F103C8T6 搭配 FreeRTOS 实时操作系统，构建一个多任务环境监测站。4个任务并发运行：DHT11温湿度采集、OLED显示刷新、串口命令解析、报警判断与蜂鸣器控制。通过队列(Queue)传递数据、互斥量(Mutex)保护I2C总线、信号量(Semaphore)同步任务，全面掌握 FreeRTOS 核心机制。

📚 FreeRTOS核心概念

任务(Task)

FreeRTOS的基本执行单元，每个任务拥有独立的栈空间。任务状态：运行→就绪→阻塞→挂起。

队列(Queue)

任务间通信的核心机制，FIFO方式传递数据。发送端 xQueueSend()，接收端 xQueueReceive()，支持阻塞等待。

互斥量(Mutex)

保护共享资源，同一时刻只有一个任务可以访问。自带优先级继承机制，防止优先级反转。

信号量(Semaphore)

用于任务同步，二值信号量像"标志位"，计数信号量管理多个相同资源。

任务优先级设计

任务优先级说明
报警判断任务4 (最高)安全相关，必须及时响应
DHT11采集任务3数据源，需及时采集
串口命令任务2人机交互，允许轻微延迟
OLED刷新任务1 (最低)显示刷新不需要高实时性

/* ============== DHT11驱动 ============== */
void DHT11_SetPin_Output(void)
{
    GPIO_InitTypeDef GPIO_InitStruct = {0};
    GPIO_InitStruct.Pin = DHT11_PIN;
    GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
    GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_HIGH;
    HAL_GPIO_Init(DHT11_PORT, &GPIO_InitStruct);
}

void DHT11_SetPin_Input(void)
{
    GPIO_InitTypeDef GPIO_InitStruct = {0};
    GPIO_InitStruct.Pin = DHT11_PIN;
    GPIO_InitStruct.Mode = GPIO_MODE_INPUT;
    GPIO_InitStruct.Pull = GPIO_PULLUP;
    HAL_GPIO_Init(DHT11_PORT, &GPIO_InitStruct);
}

void DHT11_Start(void)
{
    DHT11_SetPin_Output();
    HAL_GPIO_WritePin(DHT11_PORT, DHT11_PIN, GPIO_PIN_RESET);
    HAL_Delay(18);  // 拉低至少18ms
    HAL_GPIO_WritePin(DHT11_PORT, DHT11_PIN, GPIO_PIN_SET);
    Delay_us(30);   // 拉高20-40us
    DHT11_SetPin_Input();
}

uint8_t DHT11_Check_Response(void)
{
    uint8_t resp = 0;
    Delay_us(40);
    if (HAL_GPIO_ReadPin(DHT11_PORT, DHT11_PIN) == GPIO_PIN_RESET) {
        Delay_us(80);
        if (HAL_GPIO_ReadPin(DHT11_PORT, DHT11_PIN) == GPIO_PIN_SET) {
            resp = 1;
        }
    }
    while (HAL_GPIO_ReadPin(DHT11_PORT, DHT11_PIN) == GPIO_PIN_SET); // 等待变低
    return resp;
}

uint8_t DHT11_Read_Byte(void)
{
    uint8_t i, byte = 0;
    for (i = 0; i < 8; i++) {
        while (HAL_GPIO_ReadPin(DHT11_PORT, DHT11_PIN) == GPIO_PIN_RESET); // 等待50us低电平
        Delay_us(40); // 40us后判断
        if (HAL_GPIO_ReadPin(DHT11_PORT, DHT11_PIN) == GPIO_PIN_SET) {
            byte |= (1 << (7 - i)); // 1
            while (HAL_GPIO_ReadPin(DHT11_PORT, DHT11_PIN) == GPIO_PIN_SET);
        }
    }
    return byte;
}

uint8_t DHT11_Read(SensorData_t *data)
{
    uint8_t rh_int, rh_dec, temp_int, temp_dec, checksum;
    DHT11_Start();
    if (DHT11_Check_Response()) {
        rh_int   = DHT11_Read_Byte();
        rh_dec   = DHT11_Read_Byte();
        temp_int = DHT11_Read_Byte();
        temp_dec = DHT11_Read_Byte();
        checksum = DHT11_Read_Byte();
        if (checksum == (rh_int + rh_dec + temp_int + temp_dec)) {
            data->humidity    = rh_int + rh_dec * 0.1f;
            data->temperature = temp_int + temp_dec * 0.1f;
            data->timestamp   = HAL_GetTick();
            return 1; // 成功
        }
    }
    return 0; // 失败
}

/* ============== 微秒延时 ============== */
void Delay_us(uint32_t us)
{
    uint32_t ticks = us * (SystemCoreClock / 1000000) / 5;
    while (ticks--);
}

/* ============== OLED驱动(简化) ============== */
extern I2C_HandleTypeDef hi2c1;
#define OLED_ADDR 0x78

void OLED_WriteCmd(uint8_t cmd)
{
    if (xSemaphoreTake(xI2CMutex, portMAX_DELAY) == pdTRUE) {
        uint8_t buf[2] = {0x00, cmd};
        HAL_I2C_Master_Transmit(&hi2c1, OLED_ADDR, buf, 2, HAL_MAX_DELAY);
        xSemaphoreGive(xI2CMutex);
    }
}

void OLED_WriteData(uint8_t data)
{
    if (xSemaphoreTake(xI2CMutex, portMAX_DELAY) == pdTRUE) {
        uint8_t buf[2] = {0x40, data};
        HAL_I2C_Master_Transmit(&hi2c1, OLED_ADDR, buf, 2, HAL_MAX_DELAY);
        xSemaphoreGive(xI2CMutex);
    }
}

void OLED_Init(void)
{
    OLED_WriteCmd(0xAE); // Display Off
    OLED_WriteCmd(0xD5); OLED_WriteCmd(0x80);
    OLED_WriteCmd(0xA8); OLED_WriteCmd(0x3F);
    OLED_WriteCmd(0xD3); OLED_WriteCmd(0x00);
    OLED_WriteCmd(0x40);
    OLED_WriteCmd(0x8D); OLED_WriteCmd(0x14);
    OLED_WriteCmd(0x20); OLED_WriteCmd(0x02);
    OLED_WriteCmd(0xA1);
    OLED_WriteCmd(0xC8);
    OLED_WriteCmd(0xDA); OLED_WriteCmd(0x12);
    OLED_WriteCmd(0x81); OLED_WriteCmd(0xCF);
    OLED_WriteCmd(0xD9); OLED_WriteCmd(0xF1);
    OLED_WriteCmd(0xDB); OLED_WriteCmd(0x40);
    OLED_WriteCmd(0xA4);
    OLED_WriteCmd(0xA6);
    OLED_WriteCmd(0xAF); // Display On
}

void OLED_SetPos(uint8_t x, uint8_t y)
{
    OLED_WriteCmd(0xB0 + y);
    OLED_WriteCmd(((x & 0xF0) >> 4) | 0x10);
    OLED_WriteCmd(x & 0x0F);
}

void OLED_Clear(void)
{
    for (uint8_t y = 0; y < 8; y++) {
        OLED_SetPos(0, y);
        for (uint8_t x = 0; x < 128; x++) {
            OLED_WriteData(0x00);
        }
    }
}

/* 简易6x8字库显示(实际项目建议使用完整字库) */
extern const uint8_t asc2_0608[][6]; // 字库数组(需自行添加)

void OLED_ShowChar(uint8_t x, uint8_t y, char ch)
{
    OLED_SetPos(x, y);
    for (uint8_t i = 0; i < 6; i++) {
        OLED_WriteData(asc2_0608[ch - ' '][i]);
    }
}

void OLED_ShowString(uint8_t x, uint8_t y, const char *str)
{
    while (*str) {
        OLED_ShowChar(x, y, *str++);
        x += 6;
        if (x > 122) { x = 0; y++; }
    }
}

/* ============== 任务1: DHT11采集 ============== */
void vDHT11Task(void *pvParameters)
{
    SensorData_t sensorData;

    for (;;) {
        if (DHT11_Read(&sensorData)) {
            // 发送到队列，等待100ms
            xQueueSend(xTempHumQueue, &sensorData, pdMS_TO_TICKS(100));
            printf("DHT11: T=%.1f H=%.1f\r\n",
                   sensorData.temperature, sensorData.humidity);
        } else {
            printf("DHT11 read failed!\r\n");
        }

        // DHT11采集周期至少2秒
        vTaskDelay(pdMS_TO_TICKS(2000));
    }
}

/* ============== 任务2: OLED刷新 ============== */
void vOLEDTask(void *pvParameters)
{
    SensorData_t sensorData;
    char buf[32];

    for (;;) {
        // 从队列接收数据，阻塞等待最多1秒
        if (xQueueReceive(xTempHumQueue, &sensorData, pdMS_TO_TICKS(1000)) == pdTRUE) {
            // 使用互斥量保护I2C
            if (xSemaphoreTake(xI2CMutex, portMAX_DELAY) == pdTRUE) {
                OLED_Clear();
                snprintf(buf, sizeof(buf), "Temp: %.1f C", sensorData.temperature);
                OLED_ShowString(0, 0, buf);
                snprintf(buf, sizeof(buf), "Humi: %.1f %%", sensorData.humidity);
                OLED_ShowString(0, 2, buf);
                snprintf(buf, sizeof(buf), "AlarmT: %.1f C", tempThreshold);
                OLED_ShowString(0, 4, buf);
                snprintf(buf, sizeof(buf), "AlarmH: %.1f %%", humThreshold);
                OLED_ShowString(0, 6, buf);
                xSemaphoreGive(xI2CMutex);
            }
        }

        vTaskDelay(pdMS_TO_TICKS(500));
    }
}

/* ============== 任务3: 串口命令 ============== */
void vUartCmdTask(void *pvParameters)
{
    uint8_t rxByte;
    char cmdBuf[64];
    uint8_t idx = 0;

    for (;;) {
        // 简易串口接收(中断方式更优，此处简化)
        if (HAL_UART_Receive(&huart1, &rxByte, 1, pdMS_TO_TICKS(100)) == HAL_OK) {
            if (rxByte == '\r' || rxByte == '\n') {
                if (idx > 0) {
                    cmdBuf[idx] = '\0';
                    // 解析命令: SETT 30.0 或 SETH 80.0
                    float val;
                    if (sscanf(cmdBuf, "SETT %f", &val) == 1) {
                        taskENTER_CRITICAL();
                        tempThreshold = val;
                        taskEXIT_CRITICAL();
                        printf("Temp threshold set to %.1f\r\n", val);
                    } else if (sscanf(cmdBuf, "SETH %f", &val) == 1) {
                        taskENTER_CRITICAL();
                        humThreshold = val;
                        taskEXIT_CRITICAL();
                        printf("Humi threshold set to %.1f\r\n", val);
                    } else if (strcmp(cmdBuf, "STATUS") == 0) {
                        printf("Thresholds: T=%.1f H=%.1f\r\n",
                               tempThreshold, humThreshold);
                    } else {
                        printf("Unknown cmd: %s\r\n", cmdBuf);
                        printf("Usage: SETT <val> | SETH <val> | STATUS\r\n");
                    }
                    idx = 0;
                }
            } else {
                if (idx < sizeof(cmdBuf) - 1) {
                    cmdBuf[idx++] = rxByte;
                }
            }
        }

        vTaskDelay(pdMS_TO_TICKS(10));
    }
}

/* ============== 任务4: 报警判断 ============== */
void vAlarmTask(void *pvParameters)
{
    SensorData_t sensorData;
    uint8_t alarmActive = 0;

    for (;;) {
        // 从队列接收(非阻塞窥视或用单独队列)
        // 这里用另一个队列接收（数据会被消耗）
        // 实际做法：DHT11任务同时发给两个队列
        if (xQueueReceive(xTempHumQueue, &sensorData, pdMS_TO_TICKS(3000)) == pdTRUE) {
            if (sensorData.temperature > tempThreshold ||
                sensorData.humidity > humThreshold) {
                if (!alarmActive) {
                    alarmActive = 1;
                    HAL_GPIO_WritePin(BUZZER_PORT, BUZZER_PIN, GPIO_PIN_RESET); // 响
                    printf("!!! ALARM: T=%.1f H=%.1f !!!\r\n",
                           sensorData.temperature, sensorData.humidity);
                }
            } else {
                if (alarmActive) {
                    alarmActive = 0;
                    HAL_GPIO_WritePin(BUZZER_PORT, BUZZER_PIN, GPIO_PIN_SET); // 停
                    printf("Alarm cleared.\r\n");
                }
            }
        }

        vTaskDelay(pdMS_TO_TICKS(500));
    }
}

/* ============== 主函数 ============== */
int main(void)
{
    HAL_Init();
    SystemClock_Config();
    MX_GPIO_Init();
    MX_USART1_UART_Init();
    MX_I2C1_Init();

    printf("FreeRTOS Environment Monitor Station\r\n");
    printf("====================================\r\n");

    /* 初始化OLED */
    OLED_Init();
    OLED_Clear();
    OLED_ShowString(0, 0, "System Init...");
    HAL_Delay(500);

    /* 创建队列: 深度5, 每项sizeof(SensorData_t) */
    xTempHumQueue = xQueueCreate(5, sizeof(SensorData_t));

    /* 创建互斥量 */
    xI2CMutex = xSemaphoreCreateMutex();

    /* 创建信号量 */
    xAlarmSemaphore = xSemaphoreCreateBinary();

    /* 创建任务 */
    osThreadNew(vDHT11Task,  NULL, &(osThreadAttr_t){
        .name = "DHT11Task", .stack_size = 256, .priority = osPriorityAboveNormal
    });
    osThreadNew(vOLEDTask,   NULL, &(osThreadAttr_t){
        .name = "OLEDTask",  .stack_size = 512, .priority = osPriorityNormal
    });
    osThreadNew(vUartCmdTask,NULL, &(osThreadAttr_t){
        .name = "UartTask",  .stack_size = 256, .priority = osPriorityBelowNormal
    });
    osThreadNew(vAlarmTask,  NULL, &(osThreadAttr_t){
        .name = "AlarmTask", .stack_size = 256, .priority = osPriorityHigh
    });

    /* 启动调度器 */
    osKernelStart();

    while (1) {}
}

❓ 常见问题排查

⚠️ 问题1：任务栈溢出
现象：程序跑飞、HardFault、变量值异常
原因：任务栈太小，局部变量或函数调用层次过深
解决：
1. 使用 uxTaskGetStackHighWaterMark() 查看剩余栈空间
2. 如果high water mark < 20 words，增大stack_size
3. 开启 configCHECK_FOR_STACK_OVERFLOW = 2，溢出时进入钩子函数
4. printf非常消耗栈空间（约200+ bytes），含printf的任务至少分配512字节栈
⚠️ 问题2：优先级反转
现象：高优先级任务被低优先级任务长时间阻塞
原因：低优先级任务持有互斥量，中等优先级任务抢占低优先级，导致高优先级等待
解决：
1. FreeRTOS互斥量自带优先级继承，确保使用 xSemaphoreCreateMutex() 而非 xSemaphoreCreateBinary()
2. 尽量缩短持锁时间
3. 避免在持锁期间调用可能阻塞的API
⚠️ 问题3：I2C总线冲突
现象：OLED显示乱码、I2C总线挂死
原因：多个任务同时操作I2C，时序被破坏
解决：
1. 所有I2C操作前获取互斥量：xSemaphoreTake(xI2CMutex, ...)
2. 操作完成后立即释放：xSemaphoreGive(xI2CMutex)
3. 给互斥量设置超时，避免死锁
⚠️ 问题4：任务饿死
现象：低优先级任务长时间无法运行
原因：高优先级任务持续就绪，低优先级得不到CPU时间
解决：
1. 高优先级任务必须有阻塞点（vTaskDelay/等待队列/等待信号量）
2. 使用 vTaskDelay(pdMS_TO_TICKS(x)) 而非空循环延时
3. 监控每个任务的运行时间：vTaskGetRunTimeStats()
⚠️ 问题5：DHT11读取失败
现象：一直读到0或校验和错误
原因：时序不对、上拉电阻缺失、采集频率过高
解决：
1. 确认DATA引脚有4.7K上拉电阻到VCC
2. 采集间隔必须 ≥ 2秒
3. 微秒延时函数需要校准
4. DHT11不要放在发热芯片旁边
💡 调试技巧：
1. 使用 vTaskList() 打印所有任务状态
2. 开启 configGENERATE_RUN_TIME_STATS 统计各任务CPU占用率
3. 使用SEGGER SystemView可视化FreeRTOS运行状态
4. 每个任务入口打印一条信息，确认任务已启动

❓ 常见问题 FAQ

Q: 任务间数据怎么传？

A: 1)队列(xQueueSend/Receive)最安全 2)全局变量+互斥量 3)事件组(EventGroup)适合同步 4)流缓冲区(StreamBuffer)适合DMA数据

Q: 系统跑飞？

A: 1)栈溢出（增大任务栈） 2)优先级设置不合理 3)中断中调用了非ISR安全API 4)看门狗超时
