【DHT11接线要点】
1. DATA引脚必须接4.7kΩ上拉电阻到VCC
2. DHT11供电3.3V-5V均可
3. 上拉电阻确保空闲时DATA线为高电平
4. 有些DHT11模块已内置上拉电阻

【DHT11引脚排列】（面向栅格面，从左到右）
  ┌─────────────┐
  │  DHT11      │
  │  (栅格面)   │
  ├─1──2──3──4──┤
  │ VCC DATA NC GND
  └─────────────┘
  引脚1: VCC (3.3-5V)
  引脚2: DATA (数据)
  引脚3: NC (不接)
  引脚4: GND (地)
⚠️ 注意：DHT11对时序要求极高！起始信号和读取时序必须精确到微秒级别，否则读取失败。建议使用DWT或定时器实现微秒延时。

二、DHT11通信协议详解

2.1 单总线时序图

DHT11通信时序（一次完整读取）：

  MCU主机起始信号:              DHT11响应信号:              40位数据传输:
        _______                        ___                   ____________________
  _____/       \_____    _____/       \   \___    _____/  1  \_____0_____  ___

  │←18ms低→│←20-40→│   │←80us→│←80us→│       │←50us→│←bit→│←50us→│←bit→│

  步骤说明：
  1. MCU拉低18ms以上 → 起始信号
  2. MCU释放总线 → 等待DHT11响应
  3. DHT11拉低80us → 响应信号开始
  4. DHT11拉高80us → 响应信号结束
  5. 传输40位数据：湿度整数+湿度小数+温度整数+温度小数+校验和

  每一位数据：
  - 50us低电平（同步信号）
  - 高电平持续时间决定0或1：
    · 26-28us高电平 = "0"
    · 70us高电平   = "1"

2.2 数据格式

40位数据 = 5字节：

  Byte1: 湿度整数部分 (RH_Int)
  Byte2: 湿度小数部分 (RH_Dec)   ← DHT11小数部分始终为0
  Byte3: 温度整数部分 (T_Int)
  Byte4: 温度小数部分 (T_Dec)     ← DHT11小数部分始终为0
  Byte5: 校验和 = Byte1 + Byte2 + Byte3 + Byte4

  校验： (RH_Int + RH_Dec + T_Int + T_Dec) & 0xFF == Byte5

  示例： 25 00 28 00 53
    湿度 = 25.0%
    温度 = 28.0°C
    校验 = 25+0+28+0 = 53 ✓

2.3 GPIO模式切换

DHT11通信需要MCU的DATA引脚在"输出"和"输入"之间切换：

  发送起始信号时 → 输出模式（推挽输出）
  读取DHT11数据时 → 输入模式（浮空输入或上拉输入）

  HAL库切换代码：
  /* 切换为输出模式 */
  GPIO_InitTypeDef GPIO_InitStruct = {0};
  GPIO_InitStruct.Pin = DHT11_PIN;
  GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;  // 推挽输出
  GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_HIGH;
  HAL_GPIO_Init(DHT11_PORT, &GPIO_InitStruct);

  /* 切换为输入模式 */
  GPIO_InitStruct.Pin = DHT11_PIN;
  GPIO_InitStruct.Mode = GPIO_MODE_INPUT;  // 浮空输入
  GPIO_InitStruct.Pull = GPIO_PULLUP;       // 或 GPIO_NOPULL
  HAL_GPIO_Init(DHT11_PORT, &GPIO_InitStruct);

4.2 dht11.c 驱动文件

/**
  ******************************************************************************
  * @file    dht11.c
  * @brief   DHT11温湿度传感器驱动程序
  ******************************************************************************
  *
  * 驱动原理：
  * 1. MCU发送起始信号（拉低18ms，再拉高）
  * 2. DHT11响应（拉低80us，再拉高80us）
  * 3. DHT11发送40位数据（湿度8+8 + 温度8+8 + 校验8）
  * 4. 每位数据：50us低电平 + 高电平(26-28us=0, 70us=1)
  *
  * 关键：精确的微秒级延时！
  *
  ******************************************************************************
  */

#include "dht11.h"
#include "tim.h"

/* 外部定时器句柄 */
extern TIM_HandleTypeDef htim3;

/**
  * @brief  微秒级延时（使用TIM3定时器）
  * @param  us: 延时微秒数
  * @retval None
  *
  * @note   TIM3配置：72分频，1MHz计数频率
  *         每次计数 = 1us
  */
void DHT11_Delay_us(uint32_t us)
{
  /* 重置计数器 */
  __HAL_TIM_SET_COUNTER(&htim3, 0);

  /* 启动定时器 */
  __HAL_TIM_ENABLE(&htim3);

  /* 等待计数到达 */
  while (__HAL_TIM_GET_COUNTER(&htim3) < us);

  /* 停止定时器 */
  __HAL_TIM_DISABLE(&htim3);
}

/**
  * @brief  DHT11初始化
  * @param  None
  * @retval DHT11_OK
  */
uint8_t DHT11_Init(void)
{
  /* 设置DATA引脚为输出模式 */
  DHT11_Set_Output();

  /* 释放总线（高电平） */
  HAL_GPIO_WritePin(DHT11_PORT, DHT11_PIN, GPIO_PIN_SET);

  /* 等待DHT11上电稳定 */
  HAL_Delay(1000);

  return DHT11_OK;
}

/**
  * @brief  设置DATA引脚为输出模式
  * @param  None
  * @retval None
  */
void DHT11_Set_Output(void)
{
  GPIO_InitTypeDef GPIO_InitStruct = {0};
  GPIO_InitStruct.Pin = DHT11_PIN;
  GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;  /* 推挽输出 */
  GPIO_InitStruct.Pull = GPIO_NOPULL;
  GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_HIGH;
  HAL_GPIO_Init(DHT11_PORT, &GPIO_InitStruct);
}

/**
  * @brief  设置DATA引脚为输入模式
  * @param  None
  * @retval None
  */
void DHT11_Set_Input(void)
{
  GPIO_InitTypeDef GPIO_InitStruct = {0};
  GPIO_InitStruct.Pin = DHT11_PIN;
  GPIO_InitStruct.Mode = GPIO_MODE_INPUT;    /* 输入模式 */
  GPIO_InitStruct.Pull = GPIO_PULLUP;         /* 上拉 */
  GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_HIGH;
  HAL_GPIO_Init(DHT11_PORT, &GPIO_InitStruct);
}

/**
  * @brief  读取DHT11数据
  * @param  data: 数据结构体指针
  * @retval DHT11_OK=成功, 其他=失败
  *
  * @note   DHT11读取间隔至少1秒！
  */
uint8_t DHT11_Read(DHT11_Data_t *data)
{
  uint8_t buffer[5] = {0};
  uint8_t i, j;
  uint16_t timeout;

  /* ===== 步骤1: MCU发送起始信号 ===== */

  /* 设置为输出模式 */
  DHT11_Set_Output();

  /* 拉低18ms以上 */
  HAL_GPIO_WritePin(DHT11_PORT, DHT11_PIN, GPIO_PIN_RESET);
  HAL_Delay(20);  /* 20ms，略大于18ms最低要求 */

  /* 拉高，等待DHT11响应 */
  HAL_GPIO_WritePin(DHT11_PORT, DHT11_PIN, GPIO_PIN_SET);

  /* 切换为输入模式 */
  DHT11_Set_Input();

  /* ===== 步骤2: 等待DHT11响应 ===== */

  /* 等待DHT11拉低（响应信号开始） */
  timeout = 0;
  while (HAL_GPIO_ReadPin(DHT11_PORT, DHT11_PIN) == GPIO_PIN_SET)
  {
    DHT11_Delay_us(1);
    timeout++;
    if (timeout > 100)  /* 超时100us */
    {
      return DHT11_ERR_CONNECT;
    }
  }

  /* DHT11拉低约80us */
  timeout = 0;
  while (HAL_GPIO_ReadPin(DHT11_PORT, DHT11_PIN) == GPIO_PIN_RESET)
  {
    DHT11_Delay_us(1);
    timeout++;
    if (timeout > 100)  /* 超时100us */
    {
      return DHT11_ERR_TIMEOUT;
    }
  }

  /* DHT11拉高约80us */
  timeout = 0;
  while (HAL_GPIO_ReadPin(DHT11_PORT, DHT11_PIN) == GPIO_PIN_SET)
  {
    DHT11_Delay_us(1);
    timeout++;
    if (timeout > 100)
    {
      return DHT11_ERR_TIMEOUT;
    }
  }

  /* ===== 步骤3: 读取40位数据 ===== */

  for (i = 0; i < 5; i++)
  {
    for (j = 0; j < 8; j++)
    {
      /* 等待50us低电平结束 */
      timeout = 0;
      while (HAL_GPIO_ReadPin(DHT11_PORT, DHT11_PIN) == GPIO_PIN_RESET)
      {
        DHT11_Delay_us(1);
        timeout++;
        if (timeout > 100)
        {
          return DHT11_ERR_TIMEOUT;
        }
      }

      /* 延时30us后判断：
       * 如果还是高电平 → "1"（高电平持续70us）
       * 如果变成低电平 → "0"（高电平持续26-28us）
       */
      DHT11_Delay_us(30);

      if (HAL_GPIO_ReadPin(DHT11_PORT, DHT11_PIN) == GPIO_PIN_SET)
      {
        /* 数据位为"1" */
        buffer[i] |= (1 << (7 - j));

        /* 等待高电平结束 */
        timeout = 0;
        while (HAL_GPIO_ReadPin(DHT11_PORT, DHT11_PIN) == GPIO_PIN_SET)
        {
          DHT11_Delay_us(1);
          timeout++;
          if (timeout > 100)
          {
            return DHT11_ERR_TIMEOUT;
          }
        }
      }
      /* 如果是"0"，不需要等待，低电平已经开始了 */
    }
  }

  /* ===== 步骤4: 数据校验 ===== */

  if (buffer[0] + buffer[1] + buffer[2] + buffer[3] == buffer[4])
  {
    /* 校验通过 */
    data->humidity_int = buffer[0];
    data->humidity_dec = buffer[1];
    data->temperature_int = buffer[2];
    data->temperature_dec = buffer[3];
    data->checksum = buffer[4];

    /* 转换为浮点值 */
    data->humidity = (float)buffer[0] + (float)buffer[1] / 10.0f;
    data->temperature = (float)buffer[2] + (float)buffer[3] / 10.0f;

    return DHT11_OK;
  }
  else
  {
    return DHT11_ERR_CHECKSUM;
  }
}

4.3 使用DWT实现微秒延时（备选方案）

/**
  ******************************************************************************
  * @brief   使用DWT(Data Watchpoint Trigger)实现微秒延时
  *          比TIM方式更精确，不占用定时器资源
  ******************************************************************************
  */

/* dwt_delay.h */
#ifndef __DWT_DELAY_H__
#define __DWT_DELAY_H__

#include "main.h"

void DWT_Init(void);
void DWT_Delay_us(uint32_t us);
uint32_t DWT_Get_Tick(void);

#endif

/* dwt_delay.c */
#include "dwt_delay.h"

#define DWT_CYCCNT  ((volatile uint32_t *)0xE0001004)

static uint32_t dwt_ticks_per_us = 0;

/**
  * @brief  DWT初始化
  * @param  None
  * @retval None
  */
void DWT_Init(void)
{
  /* 使能DWT */
  CoreDebug->DEMCR &= ~CoreDebug_DEMCR_TRCENA_Msk;
  CoreDebug->DEMCR |=  CoreDebug_DEMCR_TRCENA_Msk;

  /* 清零计数器 */
  *DWT_CYCCNT = 0;

  /* 使能计数器 */
  DWT->CTRL &= ~DWT_CTRL_CYCCNTENA_Msk;
  DWT->CTRL |=  DWT_CTRL_CYCCNTENA_Msk;

  /* 计算每微秒的时钟周期数 */
  dwt_ticks_per_us = SystemCoreClock / 1000000;
}

/**
  * @brief  微秒级延时
  * @param  us: 延时微秒数
  * @retval None
  */
void DWT_Delay_us(uint32_t us)
{
  uint32_t start = *DWT_CYCCNT;
  uint32_t ticks = us * dwt_ticks_per_us;

  while ((*DWT_CYCCNT - start) < ticks);
}

/**
  * @brief  获取当前DWT计数值
  * @param  None
  * @retval 当前计数值
  */
uint32_t DWT_Get_Tick(void)
{
  return *DWT_CYCCNT;
}

五、主函数代码（main.c）

/* USER CODE BEGIN Header */
/**
  ******************************************************************************
  * @file           : main.c
  * @brief          : DHT11温湿度监测 + OLED显示 + 串口打印
  ******************************************************************************
  */
/* USER CODE END Header */

#include "main.h"
#include "dht11.h"
#include "ssd1306.h"
#include "usart.h"
#include "i2c.h"
#include "tim.h"
#include "gpio.h"

#include <stdio.h>
#include <string.h>

/* 报警阈值定义 */
#define TEMP_ALARM_HIGH    35.0f   /* 高温报警阈值 (°C) */
#define TEMP_ALARM_LOW      5.0f   /* 低温报警阈值 (°C) */
#define HUMI_ALARM_HIGH    80.0f   /* 高湿报警阈值 (%) */
#define HUMI_ALARM_LOW     20.0f   /* 低湿报警阈值 (%) */

/* LED指示灯定义 */
#define LED_PIN         GPIO_PIN_13
#define LED_PORT        GPIOC

/* 报警状态 */
typedef struct {
  uint8_t temp_high_alarm : 1;
  uint8_t temp_low_alarm  : 1;
  uint8_t humi_high_alarm : 1;
  uint8_t humi_low_alarm  : 1;
} Alarm_State_t;

Alarm_State_t alarm_state = {0};

/* DHT11数据 */
DHT11_Data_t dht11_data;

/* 函数声明 */
void Display_On_OLED(DHT11_Data_t *data, Alarm_State_t *alarm);
void Print_To_UART(DHT11_Data_t *data, Alarm_State_t *alarm);
void Check_Alarm(DHT11_Data_t *data, Alarm_State_t *alarm);

/**
  * @brief  主函数
  */
int main(void)
{
  /* HAL库初始化 */
  HAL_Init();

  /* 系统时钟配置 */
  SystemClock_Config();

  /* 外设初始化 */
  MX_GPIO_Init();
  MX_USART1_UART_Init();
  MX_I2C1_Init();
  MX_TIM3_Init();

  /* 启动TIM3（用于微秒延时） */
  HAL_TIM_Base_Start(&htim3);

  /* 初始化OLED */
  ssd1306_Init();
  ssd1306_Fill(SSD1306_COLOR_BLACK);

  /* 初始化DHT11 */
  DHT11_Init();

  /* 打印启动信息 */
  printf("\r\n========================================\r\n");
  printf("  DHT11 Temperature & Humidity Monitor\r\n");
  printf("  System Clock: %lu MHz\r\n", HAL_RCC_GetHCLKFreq() / 1000000);
  printf("========================================\r\n");

  /* 显示启动画面 */
  ssd1306_Fill(SSD1306_COLOR_BLACK);
  ssd1306_DrawRect(0, 0, 127, 63, SSD1306_COLOR_WHITE);
  ssd1306_UpdateScreen();
  HAL_Delay(1000);

  /* 读取计数器 */
  uint32_t read_count = 0;
  uint32_t fail_count = 0;

  /* 主循环 */
  while (1)
  {
    /* 每次读取间隔至少1秒 */
    static uint32_t last_read_time = 0;

    if (HAL_GetTick() - last_read_time >= 2000)  /* 2秒读取一次 */
    {
      last_read_time = HAL_GetTick();
      read_count++;

      /* 读取DHT11数据 */
      uint8_t result = DHT11_Read(&dht11_data);

      if (result == DHT11_OK)
      {
        fail_count = 0;

        /* 检查报警 */
        Check_Alarm(&dht11_data, &alarm_state);

        /* OLED显示 */
        Display_On_OLED(&dht11_data, &alarm_state);

        /* 串口打印 */
        Print_To_UART(&dht11_data, &alarm_state);

        /* 报警指示灯 */
        if (alarm_state.temp_high_alarm || alarm_state.temp_low_alarm ||
            alarm_state.humi_high_alarm || alarm_state.humi_low_alarm)
        {
          /* 报警时LED闪烁 */
          HAL_GPIO_TogglePin(LED_PORT, LED_PIN);
        }
        else
        {
          /* 正常时LED常灭 */
          HAL_GPIO_WritePin(LED_PORT, LED_PIN, GPIO_PIN_SET);
        }
      }
      else
      {
        fail_count++;

        /* 读取失败处理 */
        printf("[ERROR] DHT11读取失败! 错误码: %d, 连续失败: %lu\r\n",
               result, fail_count);

        /* OLED显示错误 */
        ssd1306_Fill(SSD1306_COLOR_BLACK);
        ssd1306_PrintString("DHT11 Error!", SSD1306_COLOR_WHITE);

        char err_buf[16];
        sprintf(err_buf, "Code: %d", result);
        ssd1306_PrintString(err_buf, SSD1306_COLOR_WHITE);

        ssd1306_UpdateScreen();

        /* 连续失败5次以上，LED快闪 */
        if (fail_count > 5)
        {
          HAL_GPIO_TogglePin(LED_PORT, LED_PIN);
        }
      }
    }

    /* 主循环中可以做其他事情 */
    HAL_Delay(10);
  }
}

/**
  * @brief  检查温湿度是否超限
  * @param  data: DHT11数据
  * @param  alarm: 报警状态
  */
void Check_Alarm(DHT11_Data_t *data, Alarm_State_t *alarm)
{
  /* 温度报警检查 */
  alarm->temp_high_alarm = (data->temperature >= TEMP_ALARM_HIGH) ? 1 : 0;
  alarm->temp_low_alarm  = (data->temperature <= TEMP_ALARM_LOW)  ? 1 : 0;

  /* 湿度报警检查 */
  alarm->humi_high_alarm = (data->humidity >= HUMI_ALARM_HIGH) ? 1 : 0;
  alarm->humi_low_alarm  = (data->humidity <= HUMI_ALARM_LOW)  ? 1 : 0;
}

/**
  * @brief  在OLED上显示温湿度数据
  * @param  data: DHT11数据
  * @param  alarm: 报警状态
  */
void Display_On_OLED(DHT11_Data_t *data, Alarm_State_t *alarm)
{
  char buf[32];

  /* 清空显示 */
  ssd1306_Fill(SSD1306_COLOR_BLACK);

  /* 第一行：标题 */
  ssd1306_PrintString("Temp & Humi Monitor", SSD1306_COLOR_WHITE);

  /* 第二行：温度 */
  sprintf(buf, "Temp:%.1f C", data->temperature);
  if (alarm->temp_high_alarm)
  {
    /* 高温报警 - 显示警告标记 */
    ssd1306_PrintString(buf, SSD1306_COLOR_WHITE);
    ssd1306_PrintString(" HIGH!", SSD1306_COLOR_WHITE);
  }
  else if (alarm->temp_low_alarm)
  {
    ssd1306_PrintString(buf, SSD1306_COLOR_WHITE);
    ssd1306_PrintString(" LOW!", SSD1306_COLOR_WHITE);
  }
  else
  {
    ssd1306_PrintString(buf, SSD1306_COLOR_WHITE);
  }

  /* 第三行：湿度 */
  sprintf(buf, "Humi:%.1f %%", data->humidity);
  if (alarm->humi_high_alarm)
  {
    ssd1306_PrintString(buf, SSD1306_COLOR_WHITE);
    ssd1306_PrintString(" HIGH!", SSD1306_COLOR_WHITE);
  }
  else if (alarm->humi_low_alarm)
  {
    ssd1306_PrintString(buf, SSD1306_COLOR_WHITE);
    ssd1306_PrintString(" LOW!", SSD1306_COLOR_WHITE);
  }
  else
  {
    ssd1306_PrintString(buf, SSD1306_COLOR_WHITE);
  }

  /* 第四行：状态 */
  if (alarm->temp_high_alarm || alarm->temp_low_alarm ||
      alarm->humi_high_alarm || alarm->humi_low_alarm)
  {
    ssd1306_PrintString("!! ALARM !!", SSD1306_COLOR_WHITE);
  }
  else
  {
    ssd1306_PrintString("Status: OK", SSD1306_COLOR_WHITE);
  }

  /* 更新显示 */
  ssd1306_UpdateScreen();
}

/**
  * @brief  串口打印温湿度数据
  * @param  data: DHT11数据
  * @param  alarm: 报警状态
  */
void Print_To_UART(DHT11_Data_t *data, Alarm_State_t *alarm)
{
  /* 打印原始数据 */
  printf("[%lu] T:%.1fC H:%.1f%%", HAL_GetTick() / 1000, data->temperature, data->humidity);

  /* 打印报警状态 */
  if (alarm->temp_high_alarm) printf(" TEMP_HIGH!");
  if (alarm->temp_low_alarm)  printf(" TEMP_LOW!");
  if (alarm->humi_high_alarm) printf(" HUMI_HIGH!");
  if (alarm->humi_low_alarm)  printf(" HUMI_LOW!");

  printf("\r\n");
}

/**
  * @brief  系统时钟配置（72MHz）
  */
void SystemClock_Config(void)
{
  RCC_OscInitTypeDef RCC_OscInitStruct = {0};
  RCC_ClkInitTypeDef RCC_ClkInitStruct = {0};

  RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_HSE;
  RCC_OscInitStruct.HSEState = RCC_HSE_ON;
  RCC_OscInitStruct.HSEPredivValue = RCC_HSE_PREDIV_DIV1;
  RCC_OscInitStruct.PLL.PLLState = RCC_PLL_ON;
  RCC_OscInitStruct.PLL.PLLSource = RCC_PLLSOURCE_HSE;
  RCC_OscInitStruct.PLL.PLLMUL = RCC_PLL_MUL9;
  if (HAL_RCC_OscConfig(&RCC_OscInitStruct) != HAL_OK)
  {
    Error_Handler();
  }

  RCC_ClkInitStruct.ClockType = RCC_CLOCKTYPE_HCLK | RCC_CLOCKTYPE_SYSCLK
                              | RCC_CLOCKTYPE_PCLK1 | RCC_CLOCKTYPE_PCLK2;
  RCC_ClkInitStruct.SYSCLKSource = RCC_SYSCLKSOURCE_PLLCLK;
  RCC_ClkInitStruct.AHBCLKDivider = RCC_SYSCLK_DIV1;
  RCC_ClkInitStruct.APB1CLKDivider = RCC_HCLK_DIV2;
  RCC_ClkInitStruct.APB2CLKDivider = RCC_HCLK_DIV1;

  if (HAL_RCC_ClockConfig(&RCC_ClkInitStruct, FLASH_LATENCY_2) != HAL_OK)
  {
    Error_Handler();
  }
}

void Error_Handler(void)
{
  while(1)
  {
    HAL_GPIO_TogglePin(LED_PORT, LED_PIN);
    HAL_Delay(100);
  }
}

六、常见问题排查

6.1 读取失败

原因1：时序不准确 
DHT11时序要求非常严格，常见问题：

1. HAL_Delay精度不够
   - HAL_Delay精度为1ms，而DHT11需要us级精度
   - 必须使用TIM或DWT实现微秒延时

2. 中断干扰
   - 读取DHT11期间，如果被其他中断打断，时序会偏移
   - 解决：读取期间关闭中断

   __disable_irq();    // 关闭中断
   DHT11_Read(&data);  // 读取数据
   __enable_irq();     // 恢复中断

3. 用逻辑分析仪检查
   - 连接逻辑分析仪到DATA引脚
   - 测量起始信号和数据的时序
   - 与规格书对比，找出偏差
原因2：上拉电阻问题 
1. 缺少上拉电阻
   - DATA线需要4.7kΩ上拉到VCC
   - 没有上拉电阻，信号会不稳定

2. 上拉电阻值不对
   - 推荐值：4.7kΩ
   - 短距离(＜20cm)：4.7kΩ-10kΩ
   - 长距离(＞20cm)：4.7kΩ或更小

3. 有些DHT11模块已内置上拉
   - 检查模块背面是否有上拉电阻
   - 如果已有，无需额外添加

6.2 温湿度返回0

温湿度返回0的可能原因：

1. 起始信号太短
   - 最短需要18ms低电平
   - 建议20ms以上
   - 检查：HAL_Delay(20) 是否真的延时了20ms

2. 起始信号太长
   - 起始信号超过30ms可能导致DHT11复位
   - 不要使用过长的起始信号

3. DHT11未上电
   - 检查VCC和GND连接
   - DHT11上电后需要1-2秒稳定

4. 读取频率过高
   - DHT11最小读取间隔1秒
   - 建议间隔2秒

5. 代码逻辑错误
   - 检查buffer[0]-buffer[4]的值
   - 确认校验和是否通过
   - 加调试打印：
     printf("Raw: %d %d %d %d %d\r\n",
            buffer[0], buffer[1], buffer[2], buffer[3], buffer[4]);

💡 DHT22读取区别：DHT22的数据格式不同，温度使用16位补码（最高位为1表示负温度），湿度也有小数部分。驱动代码需要相应修改。

6.4 DHT22数据解析代码（参考）

/**
  * @brief  DHT22数据解析（与DHT11的区别）
  */
uint8_t DHT22_Read(DHT11_Data_t *data)
{
  uint8_t buffer[5] = {0};

  /* ... 读取40位数据（同DHT11）... */

  /* DHT22数据解析 */
  /* 湿度：16位，高8位在前 */
  uint16_t humi_raw = (buffer[0] << 8) | buffer[1];
  data->humidity = (float)humi_raw / 10.0f;

  /* 温度：16位，最高位为符号位 */
  uint16_t temp_raw = (buffer[2] << 8) | buffer[3];
  if (temp_raw & 0x8000)  /* 负温度 */
  {
    temp_raw &= 0x7FFF;  /* 去掉符号位 */
    data->temperature = -((float)temp_raw / 10.0f);
  }
  else
  {
    data->temperature = (float)temp_raw / 10.0f;
  }

  /* 校验 */
  if ((buffer[0] + buffer[1] + buffer[2] + buffer[3]) == buffer[4])
  {
    return DHT11_OK;
  }
  return DHT11_ERR_CHECKSUM;
}

6.5 逻辑分析仪调试技巧

使用逻辑分析仪调试DHT11的步骤：

1. 连接逻辑分析仪到DATA引脚
2. 设置采样率 ≥ 1MHz（推荐2MHz）
3. 触发条件：下降沿
4. 捕获一次完整读取过程
