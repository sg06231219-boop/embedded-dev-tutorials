【接线要点】
1. STM32的TX接CH340的RX（TX→RX）
2. STM32的RX接CH340的TX（RX→TX）
3. GND必须连接（共地！）
4. CH340的VCC接3.3V或5V（根据模块选择）

【CH340G模块电压选择】
- 如果模块有电压选择跳针：
  · 3.3V侧短路 → 输出3.3V
  · 5V侧短路   → 输出5V
- STM32F103使用3.3V电平，直接接3.3V
⚠️ 注意：STM32的PA9/PA10是TTL电平（0-3.3V），不是RS232（±12V）！不要直接接RS232串口。

1.3 串口参数

参数值说明

数据位8位标准配置
停止位1位标准配置
校验位无不使用校验
流控制无不使用RTS/CTS

二、CubeMX配置步骤

2.1 新建工程

  1. 打开STM32CubeMX，点击 File → New Project
  2. 搜索 STM32F103C8T6，选择芯片
  3. 点击Start Project

2.2 配置时钟（RCC）

  1. 点击 RCC
  2. HSE 选择 Crystal/Ceramic Resonator

2.3 配置USART1（异步模式）

  1. 点击芯片上的 PA9 引脚 → 选择 USART1_TX
  2. 点击芯片上的 PA10 引脚 → 选择 USART1_RX
  3. 左侧 Connectivity → USART1： 
    Mode = Asynchronous (异步模式)
Hardware Flow Control = Disable
(其他保持默认)

2.4 配置USART参数

  1. 在USART1的Configuration → Parameter Settings： 
    Baud Rate = 115200
Word Length = 8 Bits
Parity = None
Stop Bits = 1
Data Direction = Receive and Transmit
Over Sampling = 16 Samples

2.5 配置NVIC中断（中断接收用）

  1. 点击USART1的NVIC Settings
  2. 勾选 USART1 global interrupt
  3. 优先级设置为较低值（如1,0）

2.6 （可选）配置DMA用于高速接收

  1. 左侧 Connectivity → USART1 → DMA Settings
  2. 点击 Add 添加DMA通道
  3. 选择 USART1_RX： 
    Mode = Circular (循环模式)
Priority = Low
Memory Size = Byte
Peripheral Size = Byte

2.7 配置时钟树

  1. 点击左侧Clock Configuration
  2. HSE = 8MHz
  3. PLL配置：HSE × 9 = 72MHz
  4. APB2 USART1 = 72MHz（可分频但一般不分）

2.8 生成代码

  1. Project Manager → 项目名 uart_hello
  2. Toolchain/IDE 选择 MDK-ARM
  3. 点击GENERATE CODE

#define RX_BUFFER_SIZE  64

/* USER CODE BEGIN PV */

/* 中断接收缓冲区 */
uint8_t rx_buffer[RX_BUFFER_SIZE];

uint8_t rx_complete = 0;

/* DMA接收缓冲区 */
uint8_t dma_rx_buffer[RX_BUFFER_SIZE];

/**
  * @brief  主函数
  * @param  None
  * @retval None
  */
int main(void)
{
  /* 初始化HAL库 */
  HAL_Init();

  /* 配置系统时钟 */
  SystemClock_Config();

  /* 初始化外设 */
  MX_GPIO_Init();
  MX_USART1_UART_Init();
  /* DMA初始化（可选） */
  MX_USART1_UART_DMA_Init();

  /* 启动DMA接收 */
  HAL_UART_Receive_DMA(&huart1, dma_rx_buffer, RX_BUFFER_SIZE);

  /* 打印启动信息 */
  printf("\r\n");
  printf("========================================\r\n");
  printf("  STM32 UART Demo v1.0\r\n");
  printf("  System Clock: %lu Hz\r\n", HAL_RCC_GetHCLKFreq());
  printf("========================================\r\n");
  printf("\r\n");

  /* 打印帮助信息 */
  printf("可用命令:\r\n");
  printf("  ON   - 点亮PC13 LED\r\n");
  printf("  OFF  - 熄灭PC13 LED\r\n");
  printf("  HELP - 显示帮助信息\r\n");
  printf("  INFO - 显示系统信息\r\n");
  printf("\r\n> ");

  /* 启动中断接收 */
  HAL_UART_Receive_IT(&huart1, &rx_buffer[rx_index], 1);

  /* 主循环 */
  while (1)
  {
    /* 方法1：中断接收处理（单字符接收） */
    if (rx_complete == 1)
    {
      rx_complete = 0;

      /* 打印收到的命令回显 */
      printf("收到: %s\r\n", rx_buffer);

      /* 命令解析 */
      if (command_parser(rx_buffer, rx_index) == 0)
      {
        printf("未知命令: %s\r\n", rx_buffer);
        printf("输入 HELP 查看可用命令\r\n");
      }

      /* 清空缓冲区 */
      memset(rx_buffer, 0, RX_BUFFER_SIZE);
      rx_index = 0;

      printf("\r\n> ");
    }

    /* 方法2：DMA接收处理（循环模式） */
    static uint32_t last_dma_check = 0;
    if (HAL_GetTick() - last_dma_check > 100)  // 每100ms检查一次
    {
      last_dma_check = HAL_GetTick();

      /* DMA循环接收，自动检测新数据 */
      /* 注意：DMA模式下需要手动判断数据有效性 */
    }

    /* 方法3：定时打印温湿度（模拟数据） */
    static uint32_t last_sensor_print = 0;
    if (HAL_GetTick() - last_sensor_print > 2000)  // 每2秒打印一次
    {
      last_sensor_print = HAL_GetTick();

      /* 模拟传感器数据 */
      static float temp = 25.0f;
      static float humi = 60.0f;

      temp += 0.5f;
      humi -= 0.3f;
      if (temp > 40.0f) temp = 20.0f;
      if (humi < 30.0f) humi = 80.0f;

      printf("[传感器] 温度: %.1f°C  湿度: %.1f%%\r\n", temp, humi);
    }

    /* 正常非阻塞运行 */
    HAL_Delay(10);
  }
}

/**
  * @brief  命令解析函数
  * @param  cmd: 命令字符串
  * @param  len: 命令长度
  * @retval 0=未识别命令, 1=已处理
  */
int command_parser(uint8_t *cmd, uint16_t len)
{
  /* 去除尾部换行符 */
  while (len > 0 && (cmd[len-1] == '\r' || cmd[len-1] == '\n' || cmd[len-1] == ' '))
  {
    cmd[len-1] = '\0';
    len--;
  }

  if (len == 0) return 1;

  /* ON命令 - 点亮LED */
  if (strcmp((char*)cmd, "ON") == 0 || strcmp((char*)cmd, "on") == 0 || strcmp((char*)cmd, "On") == 0)
  {
    LED_Control(1);
    printf("LED 已点亮\r\n");
    return 1;
  }

  /* OFF命令 - 熄灭LED */
  if (strcmp((char*)cmd, "OFF") == 0 || strcmp((char*)cmd, "off") == 0 || strcmp((char*)cmd, "Off") == 0)
  {
    LED_Control(0);
    printf("LED 已熄灭\r\n");
    return 1;
  }

  /* HELP命令 */
  if (strcmp((char*)cmd, "HELP") == 0 || strcmp((char*)cmd, "help") == 0 || strcmp((char*)cmd, "?") == 0)
  {
    printf("\r\n========== 帮助 ==========\r\n");
    printf("ON   - 点亮PC13 LED\r\n");
    printf("OFF  - 熄灭PC13 LED\r\n");
    printf("TOGGLE - 翻转LED状态\r\n");
    printf("INFO - 显示系统信息\r\n");
    printf("RESET - 软件复位\r\n");
    printf("========================\r\n");
    return 1;
  }

  /* INFO命令 */
  if (strcmp((char*)cmd, "INFO") == 0 || strcmp((char*)cmd, "info") == 0)
  {
    printf("\r\n========== 系统信息 ==========\r\n");
    printf("芯片: STM32F103C8T6\r\n");
    printf("主频: %lu MHz\r\n", HAL_RCC_GetHCLKFreq() / 1000000);
    printf("UART: 115200-8-N-1\r\n");
    printf("编译时间: %s %s\r\n", __DATE__, __TIME__);
    printf("================================\r\n");
    return 1;
  }

  /* TOGGLE命令 */
  if (strcmp((char*)cmd, "TOGGLE") == 0 || strcmp((char*)cmd, "toggle") == 0)
  {
    HAL_GPIO_TogglePin(LED_PORT, LED_PIN);
    GPIO_PinState state = HAL_GPIO_ReadPin(LED_PORT, LED_PIN);
    printf("LED状态: %s\r\n", state == GPIO_PIN_SET ? "熄灭" : "点亮");
    return 1;
  }

  /* RESET命令 */
  if (strcmp((char*)cmd, "RESET") == 0 || strcmp((char*)cmd, "reset") == 0)
  {
    printf("系统将在3秒后复位...\r\n");
    HAL_Delay(3000);
    HAL_NVIC_SystemReset();
  }

  return 0;
}

/**
  * @brief  LED控制函数
  * @param  state: 1=点亮, 0=熄灭
  * @retval None
  */
void LED_Control(uint8_t state)
{
  if (state == 1)
  {
    HAL_GPIO_WritePin(LED_PORT, LED_PIN, GPIO_PIN_RESET);  // 低电平点亮
  }
  else
  {
    HAL_GPIO_WritePin(LED_PORT, LED_PIN, GPIO_PIN_SET);    // 高电平熄灭
  }
}

/**
  * @brief  系统时钟配置
  * @param  None
  * @retval None
  */
void SystemClock_Config(void)
{
  RCC_OscInitTypeDef RCC_OscInitStruct = {0};
  RCC_ClkInitTypeDef RCC_ClkInitStruct = {0};

  RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_HSE;
  RCC_OscInitStruct.HSEState = RCC_HSE_ON;
  RCC_OscInitStruct.HSEPredivValue = RCC_HSE_PREDIV_DIV1;
  RCC_OscInitStruct.PLL.PLLState = RCC_PLL_ON;
  RCC_OscInitStruct.PLL.PLLSource = RCC_PLLSOURCE_HSE;
  RCC_OscInitStruct.PLL.PLLMUL = RCC_PLL_MUL9;
  if (HAL_RCC_OscConfig(&RCC_OscInitStruct) != HAL_OK)
  {
    Error_Handler();
  }

  RCC_ClkInitStruct.ClockType = RCC_CLOCKTYPE_HCLK | RCC_CLOCKTYPE_SYSCLK
                              | RCC_CLOCKTYPE_PCLK1 | RCC_CLOCKTYPE_PCLK2;
  RCC_ClkInitStruct.SYSCLKSource = RCC_SYSCLKSOURCE_PLLCLK;
  RCC_ClkInitStruct.AHBCLKDivider = RCC_SYSCLK_DIV1;
  RCC_ClkInitStruct.APB1CLKDivider = RCC_HCLK_DIV2;
  RCC_ClkInitStruct.APB2CLKDivider = RCC_HCLK_DIV1;

  if (HAL_RCC_ClockConfig(&RCC_ClkInitStruct, FLASH_LATENCY_2) != HAL_OK)
  {
    Error_Handler();
  }
}

/**
  * @brief  串口接收完成回调函数（中断方式）
  * @param  UartHandle: 串口句柄
  * @retval None
  *
  * @note   每接收到1个字节触发一次
  */
void HAL_UART_RxCpltCallback(UART_HandleTypeDef *UartHandle)
{
  if (UartHandle == &huart1)
  {
    /* 处理接收到的字符 */
    uint8_t ch = rx_buffer[rx_index];

    /* 回显字符 */
    HAL_UART_Transmit(&huart1, &ch, 1, 100);

    /* 判断是否结束（回车或换行） */
    if (ch == '\r' || ch == '\n')
    {
      /* 命令结束，设置完成标志 */
      rx_complete = 1;

      /* 发送换行 */
      uint8_t crlf[2] = {'\r', '\n'};
      HAL_UART_Transmit(&huart1, crlf, 2, 100);
    }
    else
    {
      /* 继续接收下一个字符 */
      rx_index++;

      /* 缓冲区溢出保护 */
      if (rx_index >= RX_BUFFER_SIZE - 1)
      {
        rx_index = 0;
        rx_complete = 1;
      }
    }

    /* 继续启动下一次接收中断 */
    HAL_UART_Receive_IT(&huart1, &rx_buffer[rx_index], 1);
  }
}

4.2 阻塞发送 vs 中断发送 vs DMA发送

方式函数特点适用场景
阻塞HAL_UART_Transmit()等待发送完成简单场景，小数据量
中断HAL_UART_Transmit_IT()发送时CPU可做其他事中等数据量
DMAHAL_UART_Transmit_DMA()几乎不占用CPU大数据量，高速传输

五、常见问题排查

5.1 乱码问题

可能原因1：波特率不匹配 
检查项：
1. 电脑串口软件波特率 = 115200
2. CubeMX配置波特率 = 115200
3. 确保两端完全一致！

常见波特率：9600, 19200, 38400, 57600, 115200, 230400, 460800
可能原因2：时钟频率不对 
如果时钟配置错误，实际波特率会偏差：
  期望波特率 = 115200
  实际波特率 = 115200 × (实际时钟 / 期望时钟)

解决：确保PLL配置正确，系统时钟 = 72MHz
可能原因3：电平不匹配 
STM32F103使用3.3V TTL电平
CH340G支持3.3V和5V（通过跳针选择）

❌ 不要用RS232电平（±12V）直接连接！
✅ 确保使用USB-TTL模块，且电压匹配

5.2 printf无输出

原因：未勾选Use MicroLib
  1. 打开Keil工程选项 Options for Target
  2. 点击 Target 标签页
  3. 勾选 Use MicroLib
  4. 重新编译下载

原因：重定向函数写错 
❌ 错误：函数名拼写错误
int _write(int fd, char *pdata, int len)  // 正确
int _write(int fd, uint8_t *pdata, int len)  // 错误：类型不匹配

5.3 DMA接收丢数据

DMA循环模式工作原理：

┌──────────────────────────────────────────────────┐
│                                                  │
│  DMA循环接收缓冲区 (假设16字节)                  │
│  ┌──┬──┬──┬──┬──┬──┬──┬──┬──┬──┬──┬──┬──┬──┬──┬──┐ │
│  │  │  │  │  │  │  │  │  │  │  │  │  │  │  │  │  │
│  └──┴──┴──┴──┴──┴──┴──┴──┴──┴──┴──┴──┴──┴──┴──┴──┘ │
│    ▲                                              │
│    │  DMA自动循环写入，永不停止                   │
│    │  旧数据会被新数据覆盖                         │
│    │                                              │
│    读取位置需要自己记录！                        │
│                                                  │
└──────────────────────────────────────────────────┘

解决方案：
1. 记录上次读取位置
2. 定时检测缓冲区变化
3. 使用带IDLE中断+DMA的方案（推荐）

5.4 DMA + IDLE中断完美方案

/* 最佳接收方案：DMA + IDLE中断 */
void USART1_IRQHandler(void)
{
  if (__HAL_UART_GET_FLAG(&huart1, UART_FLAG_IDLE))
  {
    __HAL_UART_CLEAR_IDLEFLAG(&huart1);

    /* 计算接收长度 */
    uint16_t len = RX_BUFFER_SIZE - __HAL_DMA_GET_COUNTER(huart1.hdmarx);

    /* 处理数据 */
    if (len > 0)
    {
      memcpy(rx_buffer, dma_rx_buffer, len);
      rx_buffer[len] = '\0';
      rx_complete = 1;
    }
  }
  HAL_UART_IRQHandler(&huart1);
}
